which

A Rust equivalent of Unix command which(1).

Example:

To find which rustc executable binary is using:

use which::which;
use std::path::PathBuf;

let result = which("rustc").unwrap();
assert_eq!(result, PathBuf::from("/usr/bin/rustc"));
