170    /// Get a instruction with it's name
171    /// Compares against all instructions after self, and self.
172    pub fn get_instruction_with_name(&self, name: &str) -> Option<InstructionValue<'ctx>> {
173        if let Some(ins_name) = self.get_name() {
174            if ins_name.to_str() == Ok(name) {
175                return Some(*self);
176            }
177        }
178        self.get_next_instruction()?.get_instruction_with_name(name)
179    }

426    // SubTypes: Only apply to memory access and alloca instructions
427    /// Sets alignment on a memory access instruction or alloca.
428    pub fn set_alignment(self, alignment: u32) -> Result<(), &'static str> {
429        #[cfg(any(feature = "llvm11-0", feature = "llvm12-0"))]
430        {
431            if alignment == 0 {
432                return Err("Alignment cannot be 0");
433            }
434        }
435        //The alignment = 0 check above covers LLVM >= 11, the != 0 check here keeps older versions compatible
436        if !alignment.is_power_of_two() && alignment != 0 {
437            return Err("Alignment is not a power of 2!");
438        }
439        if !self.is_a_alloca_inst() && !self.is_a_load_inst() && !self.is_a_store_inst() {
440            return Err("Value is not an alloca, load or store.");
441        }
442        unsafe { LLVMSetAlignment(self.as_value_ref(), alignment) };
443        Ok(())
444    }

455    // SubTypes: Only apply to memory access instructions
456    /// Sets atomic ordering on a memory access instruction.
457    pub fn set_atomic_ordering(self, ordering: AtomicOrdering) -> Result<(), &'static str> {
458        // Although fence and atomicrmw both have an ordering, the LLVM C API
459        // does not support them. The cmpxchg instruction has two orderings and
460        // does not work with this API.
461        if !self.is_a_load_inst() && !self.is_a_store_inst() {
462            return Err("Value is not a load or store instruction.");
463        }
464        match ordering {
465            AtomicOrdering::Release if self.is_a_load_inst() => {
466                return Err("The release ordering is not valid on load instructions.")
467            },
468            AtomicOrdering::AcquireRelease => {
469                return Err("The acq_rel ordering is not valid on load or store instructions.")
470            },
471            AtomicOrdering::Acquire if self.is_a_store_inst() => {
472                return Err("The acquire ordering is not valid on store instructions.")
473            },
474            _ => {},
475        };
476        unsafe { LLVMSetOrdering(self.as_value_ref(), ordering.into()) };
477        Ok(())
478    }

480    /// Obtains the number of operands an `InstructionValue` has.
481    /// An operand is a `BasicValue` used in an IR instruction.
482    ///
483    /// The following example,
484    ///
485    /// ```no_run
486    /// use inkwell::AddressSpace;
487    /// use inkwell::context::Context;
488    ///
489    /// let context = Context::create();
490    /// let module = context.create_module("ivs");
491    /// let builder = context.create_builder();
492    /// let void_type = context.void_type();
493    /// let f32_type = context.f32_type();
494    /// #[cfg(feature = "typed-pointers")]
495    /// let f32_ptr_type = f32_type.ptr_type(AddressSpace::default());
496    /// #[cfg(not(feature = "typed-pointers"))]
497    /// let f32_ptr_type = context.ptr_type(AddressSpace::default());
498    /// let fn_type = void_type.fn_type(&[f32_ptr_type.into()], false);
499    ///
500    /// let function = module.add_function("take_f32_ptr", fn_type, None);
501    /// let basic_block = context.append_basic_block(function, "entry");
502    ///
503    /// builder.position_at_end(basic_block);
504    ///
505    /// let arg1 = function.get_first_param().unwrap().into_pointer_value();
506    /// let f32_val = f32_type.const_float(std::f64::consts::PI);
507    /// let store_instruction = builder.build_store(arg1, f32_val).unwrap();
508    /// let free_instruction = builder.build_free(arg1).unwrap();
509    /// let return_instruction = builder.build_return(None).unwrap();
510    ///
511    /// assert_eq!(store_instruction.get_num_operands(), 2);
512    /// assert_eq!(free_instruction.get_num_operands(), 2);
513    /// assert_eq!(return_instruction.get_num_operands(), 0);
514    /// ```
515    ///
516    /// will generate LLVM IR roughly like (varying slightly across LLVM versions):
517    ///
518    /// ```ir
519    /// ; ModuleID = 'ivs'
520    /// source_filename = "ivs"
521    ///
522    /// define void @take_f32_ptr(float* %0) {
523    /// entry:
524    ///   store float 0x400921FB60000000, float* %0
525    ///   %1 = bitcast float* %0 to i8*
526    ///   tail call void @free(i8* %1)
527    ///   ret void
528    /// }
529    ///
530    /// declare void @free(i8*)
531    /// ```
532    ///
533    /// which makes the number of instruction operands clear:
534    /// 1) Store has two: a const float and a variable float pointer %0
535    /// 2) Bitcast has one: a variable float pointer %0
536    /// 3) Function call has two: i8 pointer %1 argument, and the free function itself
537    /// 4) Void return has zero: void is not a value and does not count as an operand
538    ///    even though the return instruction can take values.
539    pub fn get_num_operands(self) -> u32 {
540        unsafe { LLVMGetNumOperands(self.as_value_ref()) as u32 }
541    }

590    /// define void @take_f32_ptr(float* %0) {
591    /// entry:
592    ///   store float 0x400921FB60000000, float* %0
593    ///   %1 = bitcast float* %0 to i8*
594    ///   tail call void @free(i8* %1)
595    ///   ret void
596    /// }
597    ///
598    /// declare void @free(i8*)
599    /// ```
600    ///
601    /// which makes the instruction operands clear:
602    /// 1) Store has two: a const float and a variable float pointer %0
603    /// 2) Bitcast has one: a variable float pointer %0
604    /// 3) Function call has two: i8 pointer %1 argument, and the free function itself
605    /// 4) Void return has zero: void is not a value and does not count as an operand
606    ///    even though the return instruction can take values.
607    pub fn get_operand(self, index: u32) -> Option<Either<BasicValueEnum<'ctx>, BasicBlock<'ctx>>> {
608        let num_operands = self.get_num_operands();
609
610        if index >= num_operands {
611            return None;
612        }
613
614        unsafe { self.get_operand_unchecked(index) }
615    }

617    /// Get the operand of an `InstructionValue`.
618    ///
619    /// # Safety
620    ///
621    /// The index must be less than [InstructionValue::get_num_operands].
622    pub unsafe fn get_operand_unchecked(self, index: u32) -> Option<Either<BasicValueEnum<'ctx>, BasicBlock<'ctx>>> {
623        let operand = unsafe { LLVMGetOperand(self.as_value_ref(), index) };
624
625        if operand.is_null() {
626            return None;
627        }
628
629        let is_basic_block = unsafe { !LLVMIsABasicBlock(operand).is_null() };
630
631        if is_basic_block {
632            let bb = unsafe { BasicBlock::new(LLVMValueAsBasicBlock(operand)) };
633
634            Some(Right(bb.expect("BasicBlock should always be valid")))
635        } else {
636            Some(Left(unsafe { BasicValueEnum::new(operand) }))
637        }
638    }

646        }
647    }

649    /// Sets the operand an `InstructionValue` has at a given index if possible.
650    /// An operand is a `BasicValue` used in an IR instruction.
651    ///
652    /// ```no_run
653    /// use inkwell::AddressSpace;
654    /// use inkwell::context::Context;
655    ///
656    /// let context = Context::create();
657    /// let module = context.create_module("ivs");
658    /// let builder = context.create_builder();
659    /// let void_type = context.void_type();
660    /// let f32_type = context.f32_type();
661    /// #[cfg(feature = "typed-pointers")]
662    /// let f32_ptr_type = f32_type.ptr_type(AddressSpace::default());
663    /// #[cfg(not(feature = "typed-pointers"))]
664    /// let f32_ptr_type = context.ptr_type(AddressSpace::default());
665    /// let fn_type = void_type.fn_type(&[f32_ptr_type.into()], false);
666    ///
667    /// let function = module.add_function("take_f32_ptr", fn_type, None);
668    /// let basic_block = context.append_basic_block(function, "entry");
669    ///
670    /// builder.position_at_end(basic_block);
671    ///
672    /// let arg1 = function.get_first_param().unwrap().into_pointer_value();
673    /// let f32_val = f32_type.const_float(std::f64::consts::PI);
674    /// let store_instruction = builder.build_store(arg1, f32_val).unwrap();
675    /// let free_instruction = builder.build_free(arg1).unwrap();
676    /// let return_instruction = builder.build_return(None).unwrap();
677    ///
678    /// // This will produce invalid IR:
679    /// free_instruction.set_operand(0, f32_val);
680    ///
681    /// assert_eq!(free_instruction.get_operand(0).unwrap().left().unwrap(), f32_val);
682    /// ```
683    pub fn set_operand<BV: BasicValue<'ctx>>(self, index: u32, val: BV) -> bool {
684        let num_operands = self.get_num_operands();
685
686        if index >= num_operands {
687            return false;
688        }
689
690        unsafe { LLVMSetOperand(self.as_value_ref(), index, val.as_value_ref()) }
691
692        true
693    }

797    /// Gets the predicate of an `ICmp` `InstructionValue`.
798    /// For instance, in the LLVM instruction
799    /// `%3 = icmp slt i32 %0, %1`
800    /// this gives the `slt`.
801    ///
802    /// If the instruction is not an `ICmp`, this returns None.
803    pub fn get_icmp_predicate(self) -> Option<IntPredicate> {
804        // REVIEW: this call to get_opcode() can be inefficient;
805        // what happens if we don't perform this check, and just call
806        // LLVMGetICmpPredicate() regardless?
807        if self.get_opcode() == InstructionOpcode::ICmp {
808            let pred = unsafe { LLVMGetICmpPredicate(self.as_value_ref()) };
809            Some(IntPredicate::new(pred))
810        } else {
811            None
812        }
813    }

850    /// Determines whether or not this `Instruction` has any associated metadata
851    /// `kind_id`.
852    pub fn set_metadata(self, metadata: MetadataValue<'ctx>, kind_id: u32) -> Result<(), &'static str> {
853        if !metadata.is_node() {
854            return Err("metadata is expected to be a node.");
855        }
856
857        unsafe {
858            LLVMSetMetadata(self.instruction_value.value, kind_id, metadata.as_value_ref());
859        }
860

877/// Iterate over all the operands of an instruction value.
878#[derive(Debug)]
879pub struct OperandIter<'ctx> {
880    iv: InstructionValue<'ctx>,
881    i: u32,
882    count: u32,
883}
884
885impl<'ctx> Iterator for OperandIter<'ctx> {
886    type Item = Option<Either<BasicValueEnum<'ctx>, BasicBlock<'ctx>>>;
887
888    fn next(&mut self) -> Option<Self::Item> {
889        if self.i < self.count {
890            let result = unsafe { self.iv.get_operand_unchecked(self.i) };
891            self.i += 1;
892            Some(result)
893        } else {
894            None
895        }
896    }
897}
