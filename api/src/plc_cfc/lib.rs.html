1//! CFC/FBD XML -> Structured Text AST transpiler.
2//!
3//! [`parse_file`] is the driver entry point for `.cfc` sources: it deserializes
4//! the document, resolves the network into ordered assignments, validates them,
5//! and transpiles the result into a [`CompilationUnit`].
6

21pub fn parse_file(
22    source: &SourceCode,
23    _: LinkageType,
24    id_provider: IdProvider,
25    diagnostician: &mut Diagnostician,
26) -> Result<CompilationUnit, Diagnostic> {
27    // Register the source so diagnostics can render snippets from it.
28    diagnostician.register_file(source.get_location_str().to_string(), source.source.clone());
29
30    // Deserialize the document; a malformed file aborts before any analysis.
31    let pou = match Pou::parse(&source.source) {
32        Ok(pou) => pou,
33        Err(err) => {
34            diagnostician.handle(&[Diagnostic::new(format!("Unable to parse CFC file: {err}"))]);
35            return Err(Diagnostic::new("Compilation aborted due to CFC parse errors"));
36        }
37    };
38
39    // Resolve the network, then validate and lower it, gathering every diagnostic.
40    let resolved = resolve::resolve(pou.content().network());
41    let mut diagnostics = validation::validate(&resolved, source, id_provider.clone());
42    let (unit, transpile_diagnostics) = transpile::transpile(&pou, &resolved, source, id_provider);
43    diagnostics.extend(transpile_diagnostics);
44
45    // An error aborts; warnings still yield a usable unit.
46    if diagnostician.handle(&diagnostics) == Severity::Error {
47        Err(Diagnostic::new("Compilation aborted due to CFC parse errors"))
48    } else {
49        Ok(unit)
50    }
51}

57    use plc_ast::ast::LinkageType;

64    /// Load the `.cfc` of a fixture addressed relative to `fixtures/`, e.g.
65    /// `"variables/valid/simple_assignment"`.
66    pub(crate) fn fixture_source(fixture: &str) -> SourceCode {
67        let name = fixture.rsplit('/').next().unwrap();
68        let disk = format!("{}/fixtures/{fixture}/{name}.cfc", env!("CARGO_MANIFEST_DIR"));
69        // Label the source with a stable, machine-independent name so diagnostic
70        // snapshots don't embed an absolute path.
71        SourceCode {
72            source: std::fs::read_to_string(&disk).unwrap(),
73            path: Some(format!("{name}.cfc").into()),
74        }
75    }
76
77    /// Run the CFC pipeline over a fixture. `Ok` is the transpiled POU serialized
78    /// back to Structured Text; `Err` is the diagnostics rendered as the console
79    /// shows them (returned when compilation aborts).
80    pub(crate) fn transpile_project(fixture: &str) -> Result<String, String> {
81        let source = fixture_source(fixture);
82        let mut diagnostician = Diagnostician::buffered();
83        match crate::parse_file(&source, LinkageType::Internal, IdProvider::default(), &mut diagnostician) {
84            Ok(unit) => Ok(AstSerializer::from_unit(&unit)),
85            Err(_) => Err(diagnostician.buffer().unwrap_or_default()),
86        }
87    }
88
89    /// Render every diagnostic a fixture produces, whether or not it aborts
90    /// compilation. Used to snapshot warnings on otherwise-valid fixtures.
91    pub(crate) fn diagnostics(fixture: &str) -> String {
92        let source = fixture_source(fixture);
93        let mut diagnostician = Diagnostician::buffered();
94        let _ = crate::parse_file(&source, LinkageType::Internal, IdProvider::default(), &mut diagnostician);
95        diagnostician.buffer().unwrap_or_default()
96    }
97}
