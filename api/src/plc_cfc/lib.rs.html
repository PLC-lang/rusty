1//! CFC to AST transpiler
2//!
3//! A `.cfc` document is deserialized into the raw `model` types, then resolved
4//! into a `network::Network`: the `Resolver` surveys the diagram, traces the
5//! wiring, orders and validates the statements — every decision and diagnostic
6//! in one stage — so the `Transpiler` renders the final `CompilationUnit`
7//! without making any decisions of its own.
8//!
9//! The driver enters twice: [`parse_file`] during the parse step, yielding an
10//! interface-only unit so the POU's signature reaches the index, and
11//! [`transpile_file`] post-index, lowering the network with that index in hand.
12

32pub fn parse_file(
33    source: &SourceCode,
34    _: LinkageType,
35    ids: IdProvider,
36    diagnostician: &mut Diagnostician,
37) -> Result<CompilationUnit, Diagnostic> {
38    // Register the source so diagnostics can render snippets from it.
39    diagnostician.register_file(source.get_location_str().to_string(), source.source.clone());
40
41    // Deserialize the document; a malformed file aborts before any analysis.
42    let pou = match Pou::parse(&source.source) {
43        Ok(pou) => pou,
44        Err(err) => {
45            diagnostician.handle(&[Diagnostic::new(format!("Unable to parse CFC file: {err}"))]);
46            return Err(Diagnostic::new("Compilation aborted due to CFC parse errors"));
47        }
48    };
49
50    let (unit, diagnostics) = st::parse_interface(&pou, source, ids);
51
52    // An error aborts; warnings still yield a usable unit.
53    if diagnostician.handle(&diagnostics) == Severity::Error {
54        Err(Diagnostic::new("Compilation aborted due to CFC parse errors"))
55    } else {
56        Ok(unit)
57    }
58}
59
60pub fn transpile_file(
61    source: &SourceCode,
62    index: &Index,
63    ids: IdProvider,
64) -> Result<(CompilationUnit, Vec<Diagnostic>), Diagnostic> {
65    let pou = Pou::parse(&source.source)
66        .map_err(|err| Diagnostic::new(format!("Unable to parse CFC file: {err}")))?;
67
68    // The parse step already reported the interface's diagnostics.
69    let (unit, _) = st::parse_interface(&pou, source, ids.clone());
70
71    let (network, diagnostics) = Resolver::new(ids.clone(), source, index).resolve(pou.content().network());
72    let unit = Transpiler::new(ids).transpile(unit, network);
73
74    Ok((unit, diagnostics))
75}

87    // Indexes every POU a fixture involves: the interface, any companion `.st`
88    // beside the `.cfc` declaring the fixture's callees, and the builtin types.
89    pub(crate) fn fixture_index(
90        fixture: &str,
91        interface: &plc_ast::ast::CompilationUnit,
92    ) -> plc::index::Index {
93        let mut index = plc::index::indexer::index(interface);
94        for data_type in plc::typesystem::get_builtin_types() {
95            index.register_type(data_type);
96        }
97
98        let dir = format!("{}/fixtures/{fixture}", env!("CARGO_MANIFEST_DIR"));
99        for entry in std::fs::read_dir(dir).unwrap() {
100            let path = entry.unwrap().path();
101            if path.extension().is_none_or(|extension| extension != "st") {
102                continue;
103            }
104
105            let source = std::fs::read_to_string(&path).unwrap();
106            let factory = plc_source::source_location::SourceLocationFactory::internal(&source);
107            let session = plc::lexer::lex_with_ids(&source, IdProvider::default(), factory);
108            let (unit, _) = plc::parser::parse(session, LinkageType::Internal, "<callees>");
109            index.import(plc::index::indexer::index(&unit));
110        }
111
112        index
113    }

115    pub(crate) fn fixture_source(fixture: &str) -> SourceCode {
116        let name = fixture.rsplit('/').next().unwrap();
117        let disk = format!("{}/fixtures/{fixture}/{name}.cfc", env!("CARGO_MANIFEST_DIR"));
118        // A stable name, so snapshots don't embed absolute paths.
119        SourceCode {
120            source: std::fs::read_to_string(&disk).unwrap(),
121            path: Some(format!("{name}.cfc").into()),
122        }
123    }
124
125    pub(crate) fn transpile_project(fixture: &str) -> Result<String, String> {
126        let source = fixture_source(fixture);
127        let ids = IdProvider::default();
128        let mut diagnostician = Diagnostician::buffered();
129        let abort = |diagnostician: &mut Diagnostician| diagnostician.buffer().unwrap_or_default();
130
131        // Mimic the driver: the parse step's unit seeds the index.
132        let interface = crate::parse_file(&source, LinkageType::Internal, ids.clone(), &mut diagnostician)
133            .map_err(|_| abort(&mut diagnostician))?;
134        let index = fixture_index(fixture, &interface);
135
136        let (unit, diagnostics) =
137            crate::transpile_file(&source, &index, ids).map_err(|err| err.get_message().to_string())?;
138        match diagnostician.handle(&diagnostics) {
139            Severity::Error => Err(abort(&mut diagnostician)),
140            _ => Ok(AstSerializer::from_unit(&unit)),
141        }
142    }
143
144    // Renders every diagnostic a fixture produces, whether or not it aborts.
145    pub(crate) fn diagnostics(fixture: &str) -> String {
146        let source = fixture_source(fixture);
147        let ids = IdProvider::default();
148        let mut diagnostician = Diagnostician::buffered();
149
150        let Ok(interface) =
151            crate::parse_file(&source, LinkageType::Internal, ids.clone(), &mut diagnostician)
152        else {
153            return diagnostician.buffer().unwrap_or_default();
154        };
155        let index = fixture_index(fixture, &interface);
156
157        if let Ok((_, diagnostics)) = crate::transpile_file(&source, &index, ids) {
158            diagnostician.handle(&diagnostics);
159        }
160        diagnostician.buffer().unwrap_or_default()
161    }
