1//! Structural resolution of an FBD network: classify each element, link every
2//! sink and return to the concrete source that feeds it, and order the
3//! statements by evaluation priority.
4//!
5//! Connector/continuation pairs are spliced out during linking: they carry no
6//! statement of their own, they only reroute what a consumer ultimately reads.
7
8use std::collections::hash_map::Entry;
9use std::collections::{HashMap, HashSet};
10
11use crate::model::{FbdObject, Network, Pin};
12
13pub(crate) struct Resolved<'model> {
14    /// Network statements in evaluation-priority order.
15    pub statements: Vec<Statement<'model>>,
16    pub unconnected: Vec<&'model FbdObject>,
17    /// Connectors that repeat a label already claimed by an earlier one.
18    pub duplicate_connectors: Vec<&'model FbdObject>,
19    /// Connectors/continuations where a consumed chain never reached a source.
20    pub broken: Vec<Broken<'model>>,
21}
22
23pub(crate) enum Statement<'model> {
24    Assignment { sink: &'model FbdObject, source: Source<'model> },
25    Return { object: &'model FbdObject, condition: Option<Source<'model>> },
26    Call { block: &'model FbdObject, arguments: Vec<Argument<'model>> },
27}
28
29/// A block input/in_out pin paired with the value that feeds it.
30pub(crate) struct Argument<'model> {
31    pub pin: &'model Pin,
32    pub source: Source<'model>,
33}
34
35/// The value a consumer reads: a plain variable/literal source, or one output
36/// pin of a block — a block output is read back as a member of its instance.
37pub(crate) enum Source<'model> {
38    Variable(&'model FbdObject),
39    Output { block: &'model FbdObject, pin: &'model Pin },
40}
41
42pub(crate) struct Broken<'model> {
43    pub element: &'model FbdObject,
44    pub reason: BrokenReason,
45}
46
47pub(crate) enum BrokenReason {
48    OpenConnector,
49    DanglingContinuation,
50}
51
52enum Role {
53    Source,
54    Sink,
55    Return,
56    Block,
57    Connector,
58    Continuation,
59    Unconnected,
60    Other,
61}
62
63enum Trace<'model> {
64    Reached(Source<'model>),
65    DeadEnd(Broken<'model>),
66    Unwired,
67}
68
69/// The producers a consumer's wire can land on, indexed by output-pin id.
70struct Producers<'model> {
71    /// Every output pin (data source, continuation, or block) back to its owner,
72    /// used to walk a wire to its origin.
73    by_pin: HashMap<usize, &'model FbdObject>,
74    /// Block output pins only, so a landed wire recovers the exact pin (its
75    /// parameter name and inversion), not just the owning block.
76    block_output: HashMap<usize, &'model Pin>,
77    connector_by_label: HashMap<&'model str, &'model FbdObject>,
78}
79
80impl<'model> Statement<'model> {
81    /// The element carrying this statement's priority and diagram location.
82    fn anchor(&self) -> &'model FbdObject {
83        match self {
84            Statement::Assignment { sink, .. } => sink,
85            Statement::Return { object, .. } => object,
86            Statement::Call { block, .. } => block,
87        }
88    }
89}
90
91impl<'model> Source<'model> {
92    /// The underlying data source, when this is a plain variable/literal rather
93    /// than a block output. Used by validation to vet free-text identifiers.
94    pub(crate) fn variable(&self) -> Option<&'model FbdObject> {
95        match self {
96            Source::Variable(object) => Some(object),
97            Source::Output { .. } => None,
98        }
99    }
100}

102pub(crate) fn resolve(network: &Network) -> Resolved<'_> {
103    let (producers, duplicate_connectors) = index(network);
104
105    // Link each consumer to its source: a sink becomes an assignment, a return
106    // a (guarded) return, a block a call over its wired inputs. A chain that
107    // never reaches a source is set aside for validation; connectors and
108    // continuations are pure routing and emit nothing.
109    let mut statements = Vec::new();
110    let mut unconnected = Vec::new();
111    let mut broken = Vec::new();
112    for object in network.elements() {
113        match role(object) {
114            Role::Sink => match trace(source_pin(object), &producers) {
115                Trace::Reached(source) => statements.push(Statement::Assignment { sink: object, source }),
116                Trace::DeadEnd(at) => broken.push(at),
117                Trace::Unwired => {} // an unwired sink assigns nothing; drop it
118            },
119
120            Role::Return => match trace(source_pin(object), &producers) {
121                Trace::Reached(source) => {
122                    statements.push(Statement::Return { object, condition: Some(source) })
123                }
124                Trace::DeadEnd(at) => broken.push(at),
125                // No wire means no guard; validation flags the bare return (E085).
126                Trace::Unwired => statements.push(Statement::Return { object, condition: None }),
127            },
128
129            // A block always emits its call so its state advances; only its wired
130            // inputs and in_outs become arguments (an unwired input keeps last
131            // cycle's value). Outputs are read back separately, as members.
132            Role::Block => {
133                let mut arguments = Vec::new();
134                for pin in object.input_pins().iter().chain(object.inout_pins()) {
135                    match trace(pin.source_pin(), &producers) {
136                        Trace::Reached(source) => arguments.push(Argument { pin, source }),
137                        Trace::DeadEnd(at) => broken.push(at),
138                        Trace::Unwired => {}
139                    }
140                }
141                statements.push(Statement::Call { block: object, arguments });
142            }
143
144            Role::Unconnected => unconnected.push(object),
145            Role::Source | Role::Connector | Role::Continuation | Role::Other => {}
146        }
147    }
148
149    // Statements run in evaluation-priority order; a break fanned out across
150    // several consumers points at one element, so report it just once.
151    statements.sort_by_key(|statement| statement.anchor().priority().unwrap_or(usize::MAX));
152    let mut seen = HashSet::new();
153    broken.retain(|at| seen.insert(at.element.global_id));
154
155    Resolved { statements, unconnected, duplicate_connectors, broken }
156}

158/// Index the network so the linking pass is pure lookup: map every output pin to
159/// its producer (and, for blocks, the exact pin), and claim connector labels.
160fn index(network: &Network) -> (Producers<'_>, Vec<&FbdObject>) {
161    let mut by_pin = HashMap::new();
162    let mut block_output = HashMap::new();
163    let mut connector_by_label = HashMap::new();
164    let mut duplicate_connectors = Vec::new();
165
166    for object in network.elements() {
167        // A data source or continuation exposes a single output pin; a block
168        // exposes one per output parameter. Map each back so a consumer wired to
169        // the pin can find what drives it.
170        if let Some(out) = &object.connection_out {
171            by_pin.insert(out.id, object);
172        }
173        if matches!(role(object), Role::Block) {
174            for pin in object.output_pins() {
175                if let Some(id) = pin.output_pin() {
176                    by_pin.insert(id, object);
177                    block_output.insert(id, pin);
178                }
179            }
180        }
181
182        // A connector names a source via its label; the first to claim a label
183        // owns it, and any later connector reusing it is a duplicate (E081).
184        if matches!(role(object), Role::Connector) {
185            if let Some(label) = object.label() {
186                match connector_by_label.entry(label) {
187                    Entry::Vacant(entry) => {
188                        entry.insert(object);
189                    }
190                    Entry::Occupied(_) => duplicate_connectors.push(object),
191                }
192            }
193        }
194    }
195
196    (Producers { by_pin, block_output, connector_by_label }, duplicate_connectors)
197}

199fn role(object: &FbdObject) -> Role {
200    match object.kind.as_str() {
201        "ppx:DataSource" => Role::Source,
202        "ppx:DataSink" => Role::Sink,
203        "ppx:Return" => Role::Return,
204        "ppx:Block" => Role::Block,
205        "ppx:Connector" => Role::Connector,
206        "ppx:Continuation" => Role::Continuation,
207        "ppx:Unconnected" => Role::Unconnected,
208        _ => Role::Other,
209    }
210}

212/// The producer pin a consumer's single incoming wire references.
213fn source_pin(consumer: &FbdObject) -> Option<usize> {
214    consumer
215        .connection_in
216        .as_ref()
217        .and_then(|it| it.connections.first())
218        .map(|connection| connection.ref_out_id)
219}

221/// Follow a wire from a producer pin to the concrete source that feeds it,
222/// hopping transparently through any connector/continuation pairs on the way.
223/// Landing on a block output pin ends the walk — a block output is a real
224/// producer, so this is what stops a feedback loop from chasing itself.
225fn trace<'model>(start: Option<usize>, producers: &Producers<'model>) -> Trace<'model> {
226    let Some(mut pin) = start else { return Trace::Unwired };
227    let Some(mut element) = producers.by_pin.get(&pin).copied() else { return Trace::Unwired };
228
229    // A continuation is a stand-in for its connector's input, so replace it with
230    // that input and keep going until the wire lands on a real producer.
231    let mut visited = HashSet::new();
232    while matches!(role(element), Role::Continuation) {
233        // Every way a continuation fails to resolve is a dangling read.
234        let dangling =
235            |element| Trace::DeadEnd(Broken { element, reason: BrokenReason::DanglingContinuation });
236
237        // Resolve the continuation's label to the connector that owns it; a
238        // missing label, a cycle, or a label no connector claims all dangle.
239        let Some(label) = element.label() else { return dangling(element) };
240        if !visited.insert(label) {
241            return dangling(element);
242        }
243        let Some(connector) = producers.connector_by_label.get(label).copied() else {
244            return dangling(element);
245        };
246
247        // Step onto whatever feeds that connector; a connector with no input is
248        // an open connector and produces no value.
249        match source_pin(connector).and_then(|id| producers.by_pin.get(&id).copied().map(|next| (id, next))) {
250            Some((id, next)) => (pin, element) = (id, next),
251            None => {
252                return Trace::DeadEnd(Broken { element: connector, reason: BrokenReason::OpenConnector })
253            }
254        }
255    }
256
257    // A landed wire is a block output pin (read as a member) or a plain source.
258    match producers.block_output.get(&pin).copied() {
259        Some(output) => Trace::Reached(Source::Output { block: element, pin: output }),
260        None => Trace::Reached(Source::Variable(element)),
261    }
262}

275    /// A source rendered as its diagram text and the global id it originates at.
276    fn render_source(source: &Source) -> (String, usize) {
277        match source {
278            Source::Variable(object) => (object.identifier().unwrap_or("?").to_string(), object.global_id),
279            Source::Output { block, pin } => {
280                (format!("{}.{}", block.instance().unwrap_or("?"), pin.parameter_name), block.global_id)
281            }
282        }
283    }
284
285    fn render(resolved: &Resolved) -> String {
286        let mut out = String::new();
287        for statement in &resolved.statements {
288            let line = match statement {
289                Statement::Assignment { sink, source } => {
290                    let (text, id) = render_source(source);
291                    format!(
292                        "{} := {text}   [sink {} <- source {id}]\n",
293                        sink.identifier().unwrap_or("?"),
294                        sink.global_id
295                    )
296                }
297                Statement::Return { object, condition: Some(source) } => {
298                    let (text, id) = render_source(source);
299                    format!(
300                        "RETURN {}{text}   [return {} <- source {id}]\n",
301                        if object.negated() { "NOT " } else { "" },
302                        object.global_id,
303                    )
304                }
305                Statement::Return { object, condition: None } => {
306                    format!("RETURN <disconnected>   [return {}]\n", object.global_id)
307                }
308                Statement::Call { block, arguments } => {
309                    let args = arguments
310                        .iter()
311                        .map(|arg| {
312                            let (text, _) = render_source(&arg.source);
313                            let negate = if arg.pin.negated { "NOT " } else { "" };
314                            format!("{} := {negate}{text}", arg.pin.parameter_name)
315                        })
316                        .collect::<Vec<_>>()
317                        .join(", ");
318                    let target = block.call_target().unwrap_or_default();
319                    format!("{target}({args})   [call {}]\n", block.global_id)
320                }
321            };
322            out.push_str(&line);
323        }
324        let unconnected: Vec<_> =
325            resolved.unconnected.iter().map(|it| it.identifier().unwrap_or("?")).collect();
326        match unconnected.is_empty() {
327            true => out.push_str("unconnected: (none)"),
328            false => out.push_str(&format!("unconnected: {}", unconnected.join(", "))),
329        }
330        out
331    }
