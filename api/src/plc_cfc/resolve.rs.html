1//! Structural resolution of an FBD network: classify each element, link every
2//! sink to its single source, and order the sinks by evaluation priority.
3
4use std::collections::HashMap;
5
6use crate::model::{FbdObject, Network};
7
8pub(crate) struct Resolved<'model> {
9    /// Sinks paired with their source, in evaluation-priority order.
10    pub assignments: Vec<Assignment<'model>>,
11    pub unconnected: Vec<&'model FbdObject>,
12}
13
14pub(crate) struct Assignment<'model> {
15    pub sink: &'model FbdObject,
16    pub source: &'model FbdObject,
17}
18
19enum Role {
20    Source,
21    Sink,
22    Unconnected,
23    Other,
24}
25
26fn role(object: &FbdObject) -> Role {
27    match object.kind.as_str() {
28        "ppx:DataSource" => Role::Source,
29        "ppx:DataSink" => Role::Sink,
30        "ppx:Unconnected" => Role::Unconnected,
31        _ => Role::Other,
32    }
33}
34
35pub(crate) fn resolve(network: &Network) -> Resolved<'_> {
36    // Index every output pin so a sink can find its source in one hop.
37    let mut source_by_pin: HashMap<usize, &FbdObject> = HashMap::new();
38    for object in &network.objects {
39        if let Some(out) = &object.connection_out {
40            source_by_pin.insert(out.id, object);
41        }
42    }
43
44    let mut assignments = Vec::new();
45    let mut unconnected = Vec::new();
46
47    for object in &network.objects {
48        match role(object) {
49            Role::Sink => {
50                let source = object
51                    .connection_in
52                    .as_ref()
53                    .and_then(|it| it.connections.first())
54                    .and_then(|connection| source_by_pin.get(&connection.ref_out_id).copied());
55
56                if let Some(source) = source {
57                    assignments.push(Assignment { sink: object, source });
58                }
59            }
60            Role::Unconnected => unconnected.push(object),
61            Role::Source | Role::Other => {}
62        }
63    }
64
65    assignments.sort_by_key(|it| it.sink.priority().unwrap_or(usize::MAX));
66
67    Resolved { assignments, unconnected }
68}

75    fn resolve_project(fixture: &str) -> String {
76        let source = crate::test_utils::fixture_source(fixture);
77        let pou = Pou::parse(&source.source).unwrap();
78        render(&resolve(pou.content().network()))
79    }
80
81    fn render(resolved: &Resolved) -> String {
82        let mut out = String::new();
83        for it in &resolved.assignments {
84            let (sink, source) = (it.sink, it.source);
85            let line = format!(
86                "{} := {}   [sink {} <- source {}]\n",
87                sink.identifier().unwrap_or("?"),
88                source.identifier().unwrap_or("?"),
89                sink.global_id,
90                source.global_id,
91            );
92            out.push_str(&line);
93        }
94        let unconnected: Vec<_> =
95            resolved.unconnected.iter().map(|it| it.identifier().unwrap_or("?")).collect();
96        match unconnected.is_empty() {
97            true => out.push_str("unconnected: (none)"),
98            false => out.push_str(&format!("unconnected: {}", unconnected.join(", "))),
99        }
100        out
101    }
