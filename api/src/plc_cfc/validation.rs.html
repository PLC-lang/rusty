1//! Semantic validation of a resolved CFC network.
2
3use plc_ast::provider::IdProvider;
4use plc_diagnostics::diagnostics::Diagnostic;
5use plc_source::source_location::SourceLocationFactory;
6use plc_source::SourceCode;
7
8use crate::model::FbdObject;
9use crate::resolve::Resolved;
10use crate::transpile::helper::parse_identifier;
11
12pub(crate) fn validate(resolved: &Resolved, source: &SourceCode, ids: IdProvider) -> Vec<Diagnostic> {
13    let mut diagnostics = Vec::new();
14    diagnostics.extend(validate_expressions(resolved, source, ids));
15    diagnostics.extend(validate_unconnected(resolved, source));
16    diagnostics
17}
18
19fn validate_expressions(resolved: &Resolved, source: &SourceCode, ids: IdProvider) -> Vec<Diagnostic> {
20    let factory = SourceLocationFactory::for_source(source);
21    let mut diagnostics = Vec::new();
22
23    let mut check = |object: &FbdObject| {
24        let Some(text) = object.identifier() else { return };
25        if !helper::is_supported(&parse_identifier(text, ids.clone()).stmt) {
26            let location = factory.create_block_location(object.global_id);
27            diagnostics.push(Diagnostic::unsupported_cfc_expression(text, location));
28        }
29    };
30
31    for assignment in &resolved.assignments {
32        check(assignment.sink);
33        check(assignment.source);
34    }
35
36    diagnostics
37}
38
39fn validate_unconnected(resolved: &Resolved, source: &SourceCode) -> Vec<Diagnostic> {
40    let factory = SourceLocationFactory::for_source(source);
41    resolved
42        .unconnected
43        .iter()
44        .map(|object| {
45            let location = factory.create_block_location(object.global_id);
46            Diagnostic::unconnected_element(object.identifier().unwrap_or("<unnamed>"), location)
47        })
48        .collect()
49}
50
51mod helper {
52    use plc_ast::ast::AstStatement;
53
54    pub(super) fn is_supported(statement: &AstStatement) -> bool {
55        match statement {
56            // See through parentheses, e.g. `(foo)` or `((5))`.
57            AstStatement::ParenExpression(inner) => is_supported(&inner.stmt),
58            AstStatement::Literal(_) | AstStatement::ReferenceExpr(_) => true,
59            _ => false,
60        }
61    }
62}
