3use plc_ast::provider::IdProvider;
4use plc_diagnostics::diagnostics::Diagnostic;
5use plc_source::source_location::SourceLocationFactory;
6use plc_source::SourceCode;
7
8use crate::model::FbdObject;
9use crate::resolve::{BrokenReason, Resolved, Statement};
10use crate::transpile::helper::parse_identifier;

12pub(crate) fn validate(resolved: &Resolved, source: &SourceCode, ids: IdProvider) -> Vec<Diagnostic> {
13    let mut diagnostics = Vec::new();
14    diagnostics.extend(validate_expressions(resolved, source, ids));
15    diagnostics.extend(validate_returns(resolved, source));
16    diagnostics.extend(validate_connectors(resolved, source));
17    diagnostics.extend(validate_unconnected(resolved, source));
18    diagnostics
19}
20
21fn validate_expressions(resolved: &Resolved, source: &SourceCode, ids: IdProvider) -> Vec<Diagnostic> {
22    let factory = SourceLocationFactory::for_source(source);
23    let mut diagnostics = Vec::new();
24
25    // Flag any identifier that isn't a plain reference or literal (E083); a data
26    // source/sink can't carry a call or a compound expression.
27    let mut check = |object: &FbdObject| {
28        let Some(text) = object.identifier() else { return };
29        if !helper::is_supported(&parse_identifier(text, ids.clone()).stmt) {
30            let location = factory.create_block_location(object.global_id);
31            diagnostics.push(Diagnostic::unsupported_cfc_expression(text, location));
32        }
33    };
34
35    // A return's condition is left to the main pipeline's validator, which sees
36    // the full interface; here we only have the untyped model. Block outputs are
37    // synthesized member references, so only free-text data sources are vetted.
38    for statement in &resolved.statements {
39        match statement {
40            Statement::Assignment { sink, source } => {
41                check(sink);
42                if let Some(object) = source.variable() {
43                    check(object);
44                }
45            }
46            Statement::Call { arguments, .. } => {
47                for argument in arguments {
48                    if let Some(object) = argument.source.variable() {
49                        check(object);
50                    }
51                }
52            }
53            Statement::Return { .. } => {}
54        }
55    }
56
57    diagnostics
58}

60fn validate_returns(resolved: &Resolved, source: &SourceCode) -> Vec<Diagnostic> {
61    let factory = SourceLocationFactory::for_source(source);
62    resolved
63        .statements
64        .iter()
65        .filter_map(|statement| match statement {
66            Statement::Return { object, condition: None } => {
67                Some(Diagnostic::disconnected_return(factory.create_block_location(object.global_id)))
68            }
69            _ => None,
70        })
71        .collect()
72}
73
74fn validate_connectors(resolved: &Resolved, source: &SourceCode) -> Vec<Diagnostic> {
75    let factory = SourceLocationFactory::for_source(source);
76    let mut diagnostics = Vec::new();
77
78    // Two connectors claiming one label make the named source ambiguous (E081).
79    for connector in &resolved.duplicate_connectors {
80        let location = factory.create_block_location(connector.global_id);
81        diagnostics.push(Diagnostic::duplicate_connector(connector.label().unwrap_or_default(), location));
82    }
83
84    // A consumed chain that never reached a source: an input-less connector
85    // (E086) or a continuation whose label no connector claims (E082).
86    for broken in &resolved.broken {
87        let location = factory.create_block_location(broken.element.global_id);
88        let label = broken.element.label().unwrap_or_default();
89        diagnostics.push(match broken.reason {
90            BrokenReason::OpenConnector => Diagnostic::open_connector(label, location),
91            BrokenReason::DanglingContinuation => Diagnostic::dangling_continuation(label, location),
92        });
93    }
94
95    diagnostics
96}
97
98fn validate_unconnected(resolved: &Resolved, source: &SourceCode) -> Vec<Diagnostic> {
99    let factory = SourceLocationFactory::for_source(source);
100    resolved
101        .unconnected
102        .iter()
103        .map(|object| {
104            let location = factory.create_block_location(object.global_id);
105            Diagnostic::unconnected_element(object.identifier().unwrap_or("<unnamed>"), location)
106        })
107        .collect()
108}

110mod helper {
111    use plc_ast::ast::AstStatement;
112
113    pub(super) fn is_supported(statement: &AstStatement) -> bool {
114        match statement {
115            // See through parentheses, e.g. `(foo)` or `((5))`.
116            AstStatement::ParenExpression(inner) => is_supported(&inner.stmt),
117            AstStatement::Literal(_) | AstStatement::ReferenceExpr(_) => true,
118            _ => false,
119        }
120    }
121}
