14pub(crate) fn validate(resolved: &Resolved, source: &SourceCode, ids: IdProvider) -> Vec<Diagnostic> {
15    let mut diagnostics = Vec::new();
16    diagnostics.extend(validate_expressions(resolved, source, ids));
17    diagnostics.extend(validate_returns(resolved, source));
18    diagnostics.extend(validate_jumps(resolved, source));
19    diagnostics.extend(validate_connectors(resolved, source));
20    diagnostics.extend(validate_unconnected(resolved, source));
21    diagnostics
22}

24fn validate_expressions(resolved: &Resolved, source: &SourceCode, ids: IdProvider) -> Vec<Diagnostic> {
25    let factory = SourceLocationFactory::for_source(source);
26    let mut diagnostics = Vec::new();
27
28    // Flag any identifier that isn't a plain reference or literal (E083); a data
29    // source/sink can't carry a call or a compound expression.
30    let mut check = |object: &FbdObject| {
31        let Some(text) = object.identifier() else { return };
32        if !helper::is_supported(&parse_identifier(text, ids.clone()).stmt) {
33            let location = factory.create_block_location(object.global_id);
34            diagnostics.push(Diagnostic::unsupported_cfc_expression(text, location));
35        }
36    };
37
38    // A return's condition is left to the main pipeline's validator, which sees
39    // the full interface; here we only have the untyped model. Block outputs are
40    // synthesized member references, so only free-text data sources are vetted.
41    for statement in &resolved.statements {
42        match statement {
43            Statement::Assignment { sink, source } => {
44                check(sink);
45                if let Some(object) = source.variable() {
46                    check(object);
47                }
48            }
49            Statement::Call { arguments, .. } => {
50                for argument in arguments {
51                    if let Some(object) = argument.source.variable() {
52                        check(object);
53                    }
54                }
55            }
56            // A return/jump condition is left to the main pipeline, like any
57            // other typed expression; a label carries no data source.
58            Statement::Return { .. } | Statement::Jump { .. } | Statement::Label { .. } => {}
59        }
60    }
61
62    diagnostics
63}

65fn validate_returns(resolved: &Resolved, source: &SourceCode) -> Vec<Diagnostic> {
66    let factory = SourceLocationFactory::for_source(source);
67    resolved
68        .statements
69        .iter()
70        .filter_map(|statement| match statement {
71            Statement::Return { object, condition: None } => {
72                Some(Diagnostic::disconnected_return(factory.create_block_location(object.global_id)))
73            }
74            _ => None,
75        })
76        .collect()
77}

79fn validate_jumps(resolved: &Resolved, source: &SourceCode) -> Vec<Diagnostic> {
80    let factory = SourceLocationFactory::for_source(source);
81    let mut diagnostics = Vec::new();
82
83    // The labels a jump can land on, and the targets some jump names. Both drive
84    // the cross-checks below; a label and its jump exist as separate elements.
85    let labels: HashSet<&str> = resolved
86        .statements
87        .iter()
88        .filter_map(|it| match it {
89            Statement::Label { object } => object.label(),
90            _ => None,
91        })
92        .collect();
93    let targets: HashSet<&str> = resolved
94        .statements
95        .iter()
96        .filter_map(|it| match it {
97            Statement::Jump { object, .. } => object.target_label(),
98            _ => None,
99        })
100        .collect();
101
102    for statement in &resolved.statements {
103        match statement {
104            // A jump with no wired condition can never be taken; the FALSE guard
105            // it lowers to is valid ST, so only this pass can flag it (E145).
106            Statement::Jump { object, condition } => {
107                let location = factory.create_block_location(object.global_id);
108                if condition.is_none() {
109                    diagnostics.push(Diagnostic::disconnected_jump(location.clone()));
110                }
111                // A jump to a name no label defines has nowhere to land (E142).
112                let target = object.target_label().unwrap_or_default();
113                if !labels.contains(target) {
114                    diagnostics.push(Diagnostic::undefined_jump_target(target, location));
115                }
116            }
117            // A label no jump names is dead routing — kept, but flagged (E143).
118            Statement::Label { object } => {
119                let name = object.label().unwrap_or_default();
120                if !targets.contains(name) {
121                    let location = factory.create_block_location(object.global_id);
122                    diagnostics.push(Diagnostic::unused_label(name, location));
123                }
124            }
125            _ => {}
126        }
127    }
128
129    // Two labels sharing a name make a jump to it ambiguous (E144).
130    for label in &resolved.duplicate_labels {
131        let location = factory.create_block_location(label.global_id);
132        diagnostics.push(Diagnostic::duplicate_label(label.label().unwrap_or_default(), location));
133    }
134
135    diagnostics
136}

138fn validate_connectors(resolved: &Resolved, source: &SourceCode) -> Vec<Diagnostic> {
139    let factory = SourceLocationFactory::for_source(source);
140    let mut diagnostics = Vec::new();
141
142    // Two connectors claiming one label make the named source ambiguous (E081).
143    for connector in &resolved.duplicate_connectors {
144        let location = factory.create_block_location(connector.global_id);
145        diagnostics.push(Diagnostic::duplicate_connector(connector.label().unwrap_or_default(), location));
146    }
147
148    // A consumed chain that never reached a source: an input-less connector
149    // (E086) or a continuation whose label no connector claims (E082).
150    for broken in &resolved.broken {
151        let location = factory.create_block_location(broken.element.global_id);
152        let label = broken.element.label().unwrap_or_default();
153        diagnostics.push(match broken.reason {
154            BrokenReason::OpenConnector => Diagnostic::open_connector(label, location),
155            BrokenReason::DanglingContinuation => Diagnostic::dangling_continuation(label, location),
156        });
157    }
158
159    diagnostics
160}
161
162fn validate_unconnected(resolved: &Resolved, source: &SourceCode) -> Vec<Diagnostic> {
163    let factory = SourceLocationFactory::for_source(source);
164    resolved
165        .unconnected
166        .iter()
167        .map(|object| {
168            let location = factory.create_block_location(object.global_id);
169            Diagnostic::unconnected_element(object.identifier().unwrap_or("<unnamed>"), location)
170        })
171        .collect()
172}

174mod helper {
175    use plc_ast::ast::AstStatement;
176
177    pub(super) fn is_supported(statement: &AstStatement) -> bool {
178        match statement {
179            // See through parentheses, e.g. `(foo)` or `((5))`.
180            AstStatement::ParenExpression(inner) => is_supported(&inner.stmt),
181            AstStatement::Literal(_) | AstStatement::ReferenceExpr(_) => true,
182            _ => false,
183        }
184    }
185}
