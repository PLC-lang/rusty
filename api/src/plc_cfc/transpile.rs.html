1//! Lowering of a resolved CFC network into a Structured Text AST.
2//!
3//! The POU interface comes from the textual declaration, parsed by the ST
4//! parser; the body statements are built directly as AST nodes, each anchored
5//! to its element's `globalId` so diagnostics can point back into the diagram.
6
7use std::collections::HashSet;
8
9use plc_ast::ast::{AstFactory, AstNode, CompilationUnit, VariableBlock};
10use plc_ast::provider::IdProvider;
11use plc_diagnostics::diagnostics::Diagnostic;
12use plc_source::source_location::{SourceLocation, SourceLocationFactory};
13use plc_source::SourceCode;
14
15use crate::lowering;
16use crate::model::{FbdObject, Pou};
17use crate::resolve::{Argument, Resolved, Source, Statement};

19pub(crate) fn transpile(
20    pou: &Pou,
21    resolved: &Resolved,
22    source: &SourceCode,
23    mut ids: IdProvider,
24) -> (CompilationUnit, Vec<Diagnostic>) {
25    // The interface — signature and VAR blocks — comes from the declaration.
26    let (mut unit, diagnostics) = helper::parse_interface(pou, source, ids.clone());
27
28    // A stateless function's outputs are read through temporaries; `resolve`
29    // already recorded which output pins a consumer reads, so only those are
30    // captured.
31    let factory = SourceLocationFactory::for_source(source);
32    let consumed = &resolved.consumed_outputs;
33
34    // Build the body: one AST node per resolved statement, already in order. A
35    // disconnected return carries no condition and contributes nothing.
36    let statements = resolved
37        .statements
38        .iter()
39        .filter_map(|statement| match statement {
40            Statement::Assignment { sink, source } => {
41                Some(transpile_assignment(sink, source, &factory, &mut ids))
42            }
43            Statement::Return { object, condition } => {
44                condition.as_ref().map(|condition| transpile_return(object, condition, &factory, &mut ids))
45            }
46            Statement::Jump { object, condition } => {
47                Some(transpile_jump(object, condition.as_ref(), &factory, &mut ids))
48            }
49            Statement::Label { object } => Some(transpile_label(object, &factory, &mut ids)),
50            Statement::Call { block, arguments } => {
51                Some(transpile_call(block, arguments, consumed, &factory, &mut ids))
52            }
53        })
54        .collect();
55
56    if let Some(implementation) = unit.implementations.first_mut() {
57        implementation.statements = statements;
58    }
59
60    // Declare the captured function outputs as persistent temporaries; their
61    // placeholder types are resolved after annotation.
62    let temporaries = lowering::temporaries(resolved, consumed, &factory);
63    if let (Some(pou), false) = (unit.pous.first_mut(), temporaries.is_empty()) {
64        pou.variable_blocks.push(VariableBlock::default().with_variables(temporaries));
65    }
66
67    (unit, diagnostics)
68}

70fn transpile_assignment(
71    sink: &FbdObject,
72    source: &Source,
73    factory: &SourceLocationFactory,
74    ids: &mut IdProvider,
75) -> AstNode {
76    let location = factory.create_block_location(sink.global_id);
77
78    // Anchor both sides to the sink's diagram block so diagnostics point at the
79    // element, not the synthetic identifier text.
80    let mut left = helper::parse_identifier(sink.identifier().unwrap_or_default(), ids.clone());
81    left.location = location.clone();
82    let right = render_source(source, &location, ids);
83
84    AstFactory::create_assignment(left, right, ids.next_id())
85}

87fn transpile_return(
88    object: &FbdObject,
89    condition: &Source,
90    factory: &SourceLocationFactory,
91    ids: &mut IdProvider,
92) -> AstNode {
93    let location = factory.create_block_location(object.global_id);
94
95    // The wired input guards the return; a negated return element inverts it.
96    let condition = render_source(condition, &location, ids);
97    let condition = match object.negated() {
98        true => AstFactory::create_not_expression(condition, location.clone(), ids.next_id()),
99        false => condition,
100    };
101
102    AstFactory::create_return_statement(Some(condition), location, ids.next_id())
103}

105fn transpile_jump(
106    object: &FbdObject,
107    condition: Option<&Source>,
108    factory: &SourceLocationFactory,
109    ids: &mut IdProvider,
110) -> AstNode {
111    let location = factory.create_block_location(object.global_id);
112
113    let condition = match condition {
114        Some(source) => {
115            let condition = render_source(source, &location, ids);
116            match object.negated() {
117                true => AstFactory::create_not_expression(condition, location.clone(), ids.next_id()),
118                false => condition,
119            }
120        }
121        None => {
122            // No wired condition → a FALSE guard the jump can never pass.
123            let mut condition = helper::parse_identifier("FALSE", ids.clone());
124            condition.location = location.clone();
125            condition
126        }
127    };
128
129    let mut target = helper::parse_identifier(object.target_label().unwrap_or_default(), ids.clone());
130    target.location = location.clone();
131
132    AstFactory::create_jump_statement(Box::new(condition), Box::new(target), location, ids.next_id())
133}

135/// A label is lowered to the compiler's native label statement, a bare target
136/// for a jump.
137fn transpile_label(object: &FbdObject, factory: &SourceLocationFactory, ids: &mut IdProvider) -> AstNode {
138    let location = factory.create_block_location(object.global_id);
139    AstFactory::create_label_statement(
140        object.label().unwrap_or_default().to_string(),
141        location,
142        ids.next_id(),
143    )
144}

146fn transpile_call(
147    block: &FbdObject,
148    arguments: &[Argument],
149    consumed: &HashSet<usize>,
150    factory: &SourceLocationFactory,
151    ids: &mut IdProvider,
152) -> AstNode {
153    let location = factory.create_block_location(block.global_id);
154
155    let target = block.call_target().unwrap_or_default();
156    let mut operator = helper::parse_identifier(&target, ids.clone());
157    operator.location = location.clone();
158
159    // A stateful block passes only its wired inputs (an unwired one keeps last
160    // cycle); a function must supply every parameter, so it fills the gaps.
161    let parameters = match block.is_function() {
162        true => function_arguments(block, arguments, consumed, &location, ids),
163        false => arguments.iter().map(|argument| input_argument(argument, &location, ids)).collect(),
164    };
165
166    let parameters = (!parameters.is_empty())
167        .then(|| AstFactory::create_expression_list(parameters, location.clone(), ids.next_id()));
168    let call = AstFactory::create_call_statement(operator, parameters, ids.next_id(), location.clone());
169
170    // A consumed return pin turns the call into `temp := fn(..)`.
171    match block.is_function().then(|| block.return_pin()).flatten() {
172        Some(pin) if lowering::is_consumed(pin, consumed) => {
173            let mut left = helper::parse_identifier(&lowering::temp_name(block, pin), ids.clone());
174            left.location = location.clone();
175            AstFactory::create_assignment(left, call, ids.next_id())
176        }
177        _ => call,
178    }
179}

181/// A function call must supply every parameter (the compiler rejects a partial
182/// list). Each input/in_out takes its wired value or an empty argument
183/// (`param := `); each non-return output is captured into its temporary when
184/// read, or discarded through an empty sink (`param => `).
185fn function_arguments(
186    block: &FbdObject,
187    arguments: &[Argument],
188    consumed: &HashSet<usize>,
189    location: &SourceLocation,
190    ids: &mut IdProvider,
191) -> Vec<AstNode> {
192    let mut parameters = Vec::new();
193
194    for pin in block.input_pins().iter().chain(block.inout_pins()) {
195        parameters.push(match arguments.iter().find(|argument| std::ptr::eq(argument.pin, pin)) {
196            Some(argument) => input_argument(argument, location, ids),
197            None => empty_argument(&pin.parameter_name, location, ids),
198        });
199    }
200
201    for pin in block.output_pins() {
202        if block.is_return_pin(pin) {
203            continue;
204        }
205        let mut left = helper::parse_identifier(&pin.parameter_name, ids.clone());
206        left.location = location.clone();
207        let right = match lowering::is_consumed(pin, consumed) {
208            true => {
209                let mut temp = helper::parse_identifier(&lowering::temp_name(block, pin), ids.clone());
210                temp.location = location.clone();
211                temp
212            }
213            false => AstFactory::create_empty_statement(location.clone(), ids.next_id()),
214        };
215        parameters.push(AstFactory::create_output_assignment(left, right, ids.next_id()));
216    }
217
218    parameters
219}
220
221/// A wired `param := value` association; a negated input pin inverts its value.
222fn input_argument(argument: &Argument, location: &SourceLocation, ids: &mut IdProvider) -> AstNode {
223    let mut left = helper::parse_identifier(&argument.pin.parameter_name, ids.clone());
224    left.location = location.clone();
225    let mut right = render_source(&argument.source, location, ids);
226    if argument.pin.negated {
227        right = AstFactory::create_not_expression(right, location.clone(), ids.next_id());
228    }
229    AstFactory::create_assignment(left, right, ids.next_id())
230}
231
232/// An empty input argument (`param := `) for a parameter with no wired value.
233fn empty_argument(parameter: &str, location: &SourceLocation, ids: &mut IdProvider) -> AstNode {
234    let mut left = helper::parse_identifier(parameter, ids.clone());
235    left.location = location.clone();
236    let right = AstFactory::create_empty_statement(location.clone(), ids.next_id());
237    AstFactory::create_assignment(left, right, ids.next_id())
238}

240/// Build the expression a consumer reads: a plain source is its identifier; a
241/// block output is read through its instance member (stateful block) or its
242/// generated temporary (stateless function), inverted when the pin is negated.
243/// The result is anchored to the consumer's diagram block.
244fn render_source(source: &Source, location: &SourceLocation, ids: &mut IdProvider) -> AstNode {
245    match source {
246        Source::Variable(object) => {
247            let mut expression =
248                helper::parse_identifier(object.identifier().unwrap_or_default(), ids.clone());
249            expression.location = location.clone();
250            expression
251        }
252        Source::Output { block, pin } => {
253            let mut expression = helper::parse_identifier(&helper::output_read(block, pin), ids.clone());
254            expression.location = location.clone();
255            match pin.negated {
256                true => AstFactory::create_not_expression(expression, location.clone(), ids.next_id()),
257                false => expression,
258            }
259        }
260    }
261}

263pub(crate) mod helper {
264    use plc::lexer;
265    use plc::parser::{self, expressions_parser::parse_expression};
266    use plc_ast::ast::{AstNode, CompilationUnit, LinkageType};
267    use plc_ast::provider::IdProvider;
268    use plc_diagnostics::diagnostics::Diagnostic;
269    use plc_source::source_location::SourceLocationFactory;
270    use plc_source::{SourceCode, SourceContainer};
271
272    use crate::lowering;
273    use crate::model::{FbdObject, Pin, Pou, PouKind};
274
275    /// How a block output is read: a stateful block exposes it as an instance
276    /// member; a stateless function exposes it only through its temporary.
277    pub(super) fn output_read(block: &FbdObject, pin: &Pin) -> String {
278        match block.is_function() {
279            true => lowering::temp_name(block, pin),
280            false => format!("{}.{}", block.instance().unwrap_or_default(), pin.parameter_name),
281        }
282    }
283
284    /// Parse an identifier field into an expression. Its own text locations are
285    /// discarded by the caller in favour of a block location.
286    pub(crate) fn parse_identifier(text: &str, ids: IdProvider) -> AstNode {
287        let factory = SourceLocationFactory::internal(text);
288        let mut session = lexer::lex_with_ids(text, ids, factory);
289        parse_expression(&mut session)
290    }
291
292    pub(super) fn parse_interface(
293        pou: &Pou,
294        source: &SourceCode,
295        ids: IdProvider,
296    ) -> (CompilationUnit, Vec<Diagnostic>) {
297        // The declaration omits its closing keyword; re-attach it so the ST
298        // parser sees a complete POU with an empty body.
299        let end_keyword = match pou.kind() {
300            PouKind::Function => "END_FUNCTION",
301            PouKind::FunctionBlock => "END_FUNCTION_BLOCK",
302            PouKind::Program => "END_PROGRAM",
303        };
304        let declaration = format!("{}\n{end_keyword}", pou.content().declaration().unwrap_or_default());
305
306        let declaration = SourceCode { source: declaration, path: source.path.clone() };
307        let factory = SourceLocationFactory::for_source(&declaration);
308        let session = lexer::lex_with_ids(&declaration.source, ids, factory);
309        parser::parse(session, LinkageType::Internal, source.get_location_str())
310    }
311}
