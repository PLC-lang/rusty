1//! Function-output temporaries and the resolution of their types.
2//!
3//! A stateless function's outputs don't persist, so `transpile` captures each
4//! consumed output into a persistent temporary typed with a placeholder that
5//! names the callee (`return@<fn>` / `output@<fn>@<pin>`). The concrete type is
6//! unknown until the whole project is indexed; [`resolve`] then rewrites each
7//! placeholder to the callee's real return/output type.
8
9use std::collections::HashSet;
10
11use plc::index::Index;
12use plc_ast::ast::{CompilationUnit, DataTypeDeclaration, Variable};
13use plc_source::source_location::SourceLocationFactory;
14
15use crate::model::{FbdObject, Pin};
16use crate::resolve::{Resolved, Statement};
17
18/// A parsed temporary placeholder: which callee, and which of its outputs.
19enum Placeholder<'a> {
20    Return { function: &'a str },
21    Output { function: &'a str, pin: &'a str },
22}
23
24/// One persistent temporary per consumed function output, ordered by the call
25/// then its output pins — matching the order the outputs are captured.
26pub(crate) fn temporaries(
27    resolved: &Resolved,
28    consumed: &HashSet<usize>,
29    factory: &SourceLocationFactory,
30) -> Vec<Variable> {
31    let mut temporaries = Vec::new();
32    for statement in &resolved.statements {
33        let Statement::Call { block, .. } = statement else { continue };
34        if !block.is_function() {
35            continue;
36        }
37        for pin in block.output_pins().iter().filter(|pin| is_consumed(pin, consumed)) {
38            temporaries.push(temp_variable(block, pin, factory));
39        }
40    }
41    temporaries
42}
43
44/// Rewrite every function-output temporary's placeholder type to the callee's
45/// real return/output type, now that the project is indexed. Returns whether any
46/// placeholder was rewritten, so the caller can skip a re-index when none was.
47pub fn resolve(units: &mut [CompilationUnit], index: &Index) -> bool {
48    let mut rewritten = false;
49    for unit in units {
50        let variables = unit
51            .pous
52            .iter_mut()
53            .flat_map(|pou| &mut pou.variable_blocks)
54            .flat_map(|block| &mut block.variables);
55        for variable in variables {
56            let DataTypeDeclaration::Reference { referenced_type, .. } = &mut variable.data_type_declaration
57            else {
58                continue;
59            };
60
61            // A placeholder no callee/output resolves is left untouched;
62            // annotation then reports the unknown type at the temporary itself.
63            // TODO: emit a dedicated CFC diagnostic rather than leaning on that.
64            if let Some(resolved) = parse(referenced_type).and_then(|placeholder| placeholder.lookup(index)) {
65                *referenced_type = resolved;
66                rewritten = true;
67            }
68        }
69    }
70    rewritten
71}

73pub(crate) fn temp_name(block: &FbdObject, pin: &Pin) -> String {
74    format!("__out_{}_{}", pin.parameter_name, block.global_id)
75}
76
77pub(crate) fn is_consumed(pin: &Pin, consumed: &HashSet<usize>) -> bool {
78    pin.output_pin().is_some_and(|id| consumed.contains(&id))
79}
80
81fn temp_variable(block: &FbdObject, pin: &Pin, factory: &SourceLocationFactory) -> Variable {
82    let location = factory.create_block_location(block.global_id);
83    Variable {
84        name: temp_name(block, pin),
85        data_type_declaration: DataTypeDeclaration::Reference {
86            referenced_type: placeholder(block, pin),
87            location: location.clone(),
88        },
89        initializer: None,
90        address: None,
91        location,
92    }
93}

95/// The placeholder type a temporary carries until [`resolve`] rewrites it; it
96/// names the callee so the concrete type can be looked up post-index.
97// TODO: A type-inference marker (e.g. `__AUTO`) would be more elegant than
98//       threading the callee's name through a textual placeholder.
99fn placeholder(block: &FbdObject, pin: &Pin) -> String {
100    let function = block.type_name().unwrap_or_default();
101    match block.is_return_pin(pin) {
102        true => format!("return@{function}"),
103        false => format!("output@{function}@{}", pin.parameter_name),
104    }
105}
106
107/// Parse a temporary's placeholder type; `None` for any ordinary type name.
108fn parse(referenced_type: &str) -> Option<Placeholder<'_>> {
109    if let Some(function) = referenced_type.strip_prefix("return@") {
110        return Some(Placeholder::Return { function });
111    }
112    let (function, pin) = referenced_type.strip_prefix("output@")?.split_once('@')?;
113    Some(Placeholder::Output { function, pin })
114}
115
116impl Placeholder<'_> {
117    /// The callee's declared type name for this output, looked up in the index.
118    /// A function's output is passed by reference, so its member type is an
119    /// auto-deref pointer; the temporary needs the pointed-to (inner) type.
120    fn lookup(&self, index: &Index) -> Option<String> {
121        let variable = match self {
122            Placeholder::Return { function } => index.find_return_variable(function),
123            Placeholder::Output { function, pin } => index.find_member(function, pin),
124        };
125        let type_name = variable?.get_type_name();
126        Some(index.get_type_information_or_void(type_name).get_inner_name().to_string())
127    }
128}
