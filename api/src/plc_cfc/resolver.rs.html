73    pub fn resolve(mut self, network: &model::Network) -> (Network, Vec<Diagnostic>) {
74        let mut survey = self.survey(network);
75        survey.consumed = consumed(network, &survey);
76
77        let mut statements = Vec::new();
78        let mut temporaries = Vec::new();
79        let mut broken = Vec::new();
80        for object in network.elements() {
81            match Role::from(object) {
82                Role::Sink => match trace(consumes(object), &survey) {
83                    // The traced source becomes the sink's assigned value.
84                    Trace::Reached(source) => {
85                        let location = self.factory.create_block_location(object.global_id);
86                        let sink = self.expression(object, &location);
87                        let source = self.value(&source, &location);
88
89                        let statement = Statement::Assignment { sink, source };
90                        statements.push((object.priority(), statement));
91                    }
92
93                    // A consumed chain that never reached a source; reported below.
94                    Trace::DeadEnd(at) => broken.push(at),
95
96                    // An unwired sink assigns nothing; drop it.
97                    Trace::Unwired => {}
98                },
99
100                Role::Return => match trace(consumes(object), &survey) {
101                    // The traced source guards the return.
102                    Trace::Reached(source) => {
103                        let location = self.factory.create_block_location(object.global_id);
104                        let condition = self.condition(object, &source, &location);
105
106                        let statement = Statement::Return { condition, location };
107                        statements.push((object.priority(), statement));
108                    }
109
110                    // A consumed chain that never reached a source; reported below.
111                    Trace::DeadEnd(at) => broken.push(at),
112
113                    // A return without a condition is rejected; drop it.
114                    Trace::Unwired => {
115                        let location = self.factory.create_block_location(object.global_id);
116                        self.diagnostics.push(Diagnostic::disconnected_return(location));
117                    }
118                },
119
120                Role::Jump => {
121                    let location = self.factory.create_block_location(object.global_id);
122                    let target = object.target_label().unwrap_or_default().to_string();
123
124                    // A jump to a name no label defines has nowhere to land.
125                    if !survey.labels.contains(target.as_str()) {
126                        self.diagnostics.push(Diagnostic::undefined_jump_target(&target, location.clone()));
127                    }
128
129                    match trace(consumes(object), &survey) {
130                        // The traced source guards the jump.
131                        Trace::Reached(source) => {
132                            let condition = Some(self.condition(object, &source, &location));
133
134                            let statement = Statement::Jump { condition, target, location };
135                            statements.push((object.priority(), statement));
136                        }
137
138                        // A consumed chain that never reached a source; reported below.
139                        Trace::DeadEnd(at) => broken.push(at),
140
141                        // An unwired jump is legal but can never be taken; warn and keep it.
142                        Trace::Unwired => {
143                            self.diagnostics.push(Diagnostic::disconnected_jump(location.clone()));
144
145                            let statement = Statement::Jump { condition: None, target, location };
146                            statements.push((object.priority(), statement));
147                        }
148                    }
149                }
150
151                Role::Label => {
152                    let location = self.factory.create_block_location(object.global_id);
153                    let name = object.label().unwrap_or_default().to_string();
154
155                    // A label no jump targets is dead routing; keep it, but flag it.
156                    if !survey.targets.contains(name.as_str()) {
157                        self.diagnostics.push(Diagnostic::unused_label(&name, location.clone()));
158                    }
159
160                    let statement = Statement::Label { name, location };
161                    statements.push((object.priority(), statement));
162                }
163
164                // A callee the index doesn't know can't be classified; reject it.
165                Role::Block if block::is_unknown(object, self.index) => {
166                    let location = self.factory.create_block_location(object.global_id);
167                    let name = object.type_name().unwrap_or_default();
168
169                    self.diagnostics.push(Diagnostic::unknown_block_type(name, location));
170                }
171
172                // Stateless blocks (functions and sort of methods); temporaries for outputs
173                Role::Block if block::is_stateless(object, self.index) => {
174                    let location = self.factory.create_block_location(object.global_id);
175
176                    let mut arguments = Vec::new();
177                    for pin in block::inputs(object) {
178                        let argument = match trace(pin.source_pin(), &survey) {
179                            // The traced source becomes the pin's passed value.
180                            Trace::Reached(source) => self.input(pin, &source, &location),
181
182                            // A consumed chain that never reached a source; reported below.
183                            Trace::DeadEnd(at) => {
184                                broken.push(at);
185                                self.empty_input(pin, &location)
186                            }
187
188                            // An unwired pin takes the callee's declared default.
189                            Trace::Unwired => self.empty_input(pin, &location),
190                        };
191
192                        arguments.push(argument);
193                    }
194
195                    // Consumed outputs are captured into temporaries
196                    let mut capture = None;
197                    for pin in object.output_pins() {
198                        let consumed = pin.output_pin().is_some_and(|id| survey.consumed.contains(&id));
199                        let name = consumed.then(|| block::temp_name(object, pin));
200
201                        if let Some(name) = &name {
202                            if let Some(temporary) = self.temporary(name, object, pin, &location) {
203                                temporaries.push((object.priority(), temporary));
204                            }
205                        }
206
207                        match block::is_return_pin(object, pin) {
208                            // The return value is received by assigning the call itself.
209                            true => capture = name,
210
211                            // Any other output is received inside the parentheses.
212                            false => arguments.push(Argument::Output {
213                                parameter: pin.parameter_name.clone(),
214                                capture: name,
215                            }),
216                        }
217                    }
218
219                    let statement =
220                        Statement::Call { target: block::call_target(object), arguments, capture, location };
221                    statements.push((object.priority(), statement));
222                }
223
224                // A stateful call passes only its wired inputs; outputs read back as members.
225                Role::Block => {
226                    let location = self.factory.create_block_location(object.global_id);
227
228                    let mut arguments = Vec::new();
229                    for pin in block::inputs(object) {
230                        match trace(pin.source_pin(), &survey) {
231                            Trace::Reached(source) => arguments.push(self.input(pin, &source, &location)),
232
233                            // A consumed chain that never reached a source; reported below.
234                            Trace::DeadEnd(at) => broken.push(at),
235
236                            // An unwired input passes nothing; the callee keeps last cycle's value.
237                            Trace::Unwired => {}
238                        }
239                    }
240
241                    let statement = Statement::Call {
242                        target: block::call_target(object),
243                        arguments,
244                        capture: None,
245                        location,
246                    };
247                    statements.push((object.priority(), statement));
248                }
249
250                // Placed but never wired; warn and ignore.
251                Role::Unconnected => {
252                    let location = self.factory.create_block_location(object.global_id);
253                    let name = object.identifier().unwrap_or("<unnamed>");
254
255                    self.diagnostics.push(Diagnostic::unconnected_element(name, location));
256                }
257
258                // Read-only (sources) or pure routing; no statements.
259                Role::Source | Role::Connector | Role::Continuation | Role::Other => {}
260            }
261        }
262
263        // Report a broken element once, no matter how many consumers hit it.
264        broken.sort_by_key(|at| at.element.global_id);
265        broken.dedup_by_key(|at| at.element.global_id);
266        for at in broken {
267            self.diagnostics.push(at.diagnostic(&self.factory));
268        }
269
270        // Statements and temporaries run in evaluation-priority order.
271        statements.sort_by_key(|(priority, _)| priority.unwrap_or(usize::MAX));
272        let statements = statements.into_iter().map(|(_, statement)| statement).collect();
273        temporaries.sort_by_key(|(priority, _)| priority.unwrap_or(usize::MAX));
274        let temporaries = temporaries.into_iter().map(|(_, temporary)| temporary).collect();
275
276        (Network { statements, temporaries }, self.diagnostics)
277    }

279    // `parameter := value`, negated by the pin's inversion bubble.
280    fn input(&mut self, pin: &Pin, source: &Source, location: &SourceLocation) -> Argument {
281        let value = self.value(source, location);
282        let value = self.negate_if(value, location, pin.negated);
283
284        Argument::Input { parameter: pin.parameter_name.clone(), value: Box::new(value) }
285    }

287    // `parameter := `; an empty argument for an unwired parameter.
288    fn empty_input(&mut self, pin: &Pin, location: &SourceLocation) -> Argument {
289        let value = AstFactory::create_empty_statement(location.clone(), self.ids.next_id());
290        Argument::Input { parameter: pin.parameter_name.clone(), value: Box::new(value) }
291    }

293    fn temporary(
294        &mut self,
295        name: &str,
296        block: &FbdObject,
297        pin: &Pin,
298        location: &SourceLocation,
299    ) -> Option<Temporary> {
300        let Some(data_type) = block::temp_type(block, pin, self.index) else {
301            self.diagnostics.push(Diagnostic::undeclared_block_output(
302                &pin.parameter_name,
303                block.type_name().unwrap_or_default(),

305            ));

307            return None;
308        };
309

311    }

313    fn survey<'model>(&mut self, network: &'model model::Network) -> Survey<'model> {
314        let mut by_pin = HashMap::new();
315        let mut block_output = HashMap::new();
316        let mut connector_by_label = HashMap::new();
317        let mut labels = HashSet::new();
318        let mut targets = HashSet::new();
319
320        for object in network.elements() {
321            // A label name claimed twice makes its jump target ambiguous.
322            if matches!(Role::from(object), Role::Label) {
323                if let Some(label) = object.label() {
324                    if !labels.insert(label) {
325                        let location = self.factory.create_block_location(object.global_id);
326                        self.diagnostics.push(Diagnostic::duplicate_label(label, location));
327                    }
328                }
329            }
330
331            if matches!(Role::from(object), Role::Jump) {
332                if let Some(target) = object.target_label() {
333                    targets.insert(target);
334                }
335            }
336
337            // Register every output pin an incoming wire could reference; a block exposes one per output parameter.
338            if let Some(out) = &object.connection_out {
339                by_pin.insert(out.id, object);
340            }
341
342            if matches!(Role::from(object), Role::Block) {
343                for pin in object.output_pins() {
344                    if let Some(id) = pin.output_pin() {
345                        by_pin.insert(id, object);
346                        block_output.insert(id, pin);
347                    }
348                }
349            }
350
351            if matches!(Role::from(object), Role::Connector) {
352                if let Some(label) = object.label() {
353                    match connector_by_label.entry(label) {
354                        // The first connector to claim a label owns it.
355                        Entry::Vacant(entry) => {
356                            entry.insert(object);
357                        }
358
359                        // A later one reusing the label is a duplicate.
360                        Entry::Occupied(_) => {
361                            let location = self.factory.create_block_location(object.global_id);
362                            self.diagnostics.push(Diagnostic::duplicate_connector(label, location));
363                        }
364                    }
365                }
366            }
367        }
368
369        Survey { by_pin, block_output, connector_by_label, labels, targets, consumed: HashSet::new() }
370    }

372    fn expression(&mut self, object: &FbdObject, location: &SourceLocation) -> AstNode {
373        let node = self.parse(object, location);
374
375        // Expressions other than literals and references defined in the text declaration of a CFC graph node
376        // are unsupported. A variable node element should only contain a literal or reference, not a call
377        // for example (calls must use a block element instead).
378        if !is_supported(&node.stmt) {
379            let location = self.factory.create_block_location(object.global_id);
380            let text = object.identifier().unwrap_or_default();
381            self.diagnostics.push(Diagnostic::unsupported_cfc_expression(text, location));
382        }
383
384        node
385    }

387    // The value a source produces: a vetted plain expression, or a block-output member read.
388    fn value(&mut self, source: &Source, location: &SourceLocation) -> AstNode {
389        match source {
390            Source::Variable(object) => self.expression(object, location),
391            Source::Output { block, pin } => self.member_read(block, pin, location),
392        }
393    }
394
395    // Creates an AST node, wrapped in a NOT expression if the consumer is negated in the graph.
396    fn condition(&mut self, consumer: &FbdObject, source: &Source, location: &SourceLocation) -> AstNode {
397        // Unvetted; a condition may be any ST expression.
398        let node = match source {
399            Source::Variable(object) => self.parse(object, location),
400            Source::Output { block, pin } => self.member_read(block, pin, location),
401        };
402
403        self.negate_if(node, location, consumer.negated())
404    }
405
406    fn member_read(&mut self, block: &FbdObject, pin: &Pin, location: &SourceLocation) -> AstNode {
407        let mut node = st::parse_expression(&block::read_target(block, pin, self.index), self.ids.clone());
408        node.location = location.clone();
409
410        self.negate_if(node, location, pin.negated)
411    }
412
413    fn negate_if(&mut self, node: AstNode, location: &SourceLocation, negated: bool) -> AstNode {
414        match negated {
415            true => AstFactory::create_not_expression(node, location.clone(), self.ids.next_id()),
416            false => node,
417        }
418    }
419
420    fn parse(&mut self, object: &FbdObject, location: &SourceLocation) -> AstNode {
421        let mut node = st::parse_expression(object.identifier().unwrap_or_default(), self.ids.clone());
422        node.location = location.clone();
423        node
424    }

444impl Broken<'_> {
445    fn diagnostic(&self, factory: &SourceLocationFactory) -> Diagnostic {
446        let location = factory.create_block_location(self.element.global_id);
447        let label = self.element.label().unwrap_or_default();
448
449        match self.reason {
450            Reason::OpenConnector => Diagnostic::open_connector(label, location),
451            Reason::DanglingContinuation => Diagnostic::dangling_continuation(label, location),
452        }
453    }
454}

456fn trace<'model>(start: Option<usize>, survey: &Survey<'model>) -> Trace<'model> {
457    let Some(mut pin) = start else { return Trace::Unwired };
458    let Some(mut element) = survey.by_pin.get(&pin).copied() else { return Trace::Unwired };
459
460    // Trace through the connections until we hit a producer (cycle-guarded).
461    let mut visited = HashSet::new();
462    while matches!(Role::from(element), Role::Continuation) {
463        let dangling = |element| Trace::DeadEnd(Broken { element, reason: Reason::DanglingContinuation });
464
465        // An unlabeled continuation, or one revisiting a label (a cycle), dangles.
466        let Some(label) = element.label() else { return dangling(element) };
467        if !visited.insert(label) {
468            return dangling(element);
469        }
470
471        // Resolve the label to the connector owning it; an unclaimed label dangles.
472        let Some(connector) = survey.connector_by_label.get(label).copied() else {
473            return dangling(element);
474        };
475
476        // Step onto whatever feeds that connector; an input-less one is open.
477        let open = Broken { element: connector, reason: Reason::OpenConnector };
478        let Some(wire) = consumes(connector) else { return Trace::DeadEnd(open) };
479        let Some(producer) = survey.by_pin.get(&wire).copied() else { return Trace::DeadEnd(open) };
480
481        pin = wire;
482        element = producer;
483    }
484
485    // A landed wire is a block output pin (read as a member) or a plain source.
486    match survey.block_output.get(&pin).copied() {
487        Some(output) => Trace::Reached(Source::Output { block: element, pin: output }),
488        None => Trace::Reached(Source::Variable(element)),
489    }
490}

492fn is_supported(statement: &AstStatement) -> bool {
493    match statement {
494        // See through parentheses, e.g. `(foo)` or `((5))`.
495        AstStatement::ParenExpression(inner) => is_supported(&inner.stmt),
496        AstStatement::Literal(_) | AstStatement::ReferenceExpr(_) => true,
497        _ => false,
498    }
499}

507    pub(super) fn is_unknown(block: &FbdObject, index: &Index) -> bool {
508        block.type_name().and_then(|name| index.find_pou(name)).is_none()
509    }
510
511    pub(super) fn is_stateless(block: &FbdObject, index: &Index) -> bool {
512        block.instance_name.is_none()
513            && block.type_name().and_then(|name| index.find_pou(name)).is_some_and(|pou| pou.is_function())
514    }
515
516    pub(super) fn is_return_pin(block: &FbdObject, pin: &Pin) -> bool {
517        block.type_name() == Some(pin.parameter_name.as_str())
518    }
519
520    pub(super) fn temp_name(block: &FbdObject, pin: &Pin) -> String {
521        format!("__out_{}_{}", pin.parameter_name, block.global_id)
522    }
523
524    pub(super) fn inputs(block: &FbdObject) -> impl Iterator<Item = &Pin> {
525        block.input_pins().iter().chain(block.inout_pins())
526    }
527
528    // The captured temporary (stateless) or instance member (stateful).
529    pub(super) fn read_target(block: &FbdObject, pin: &Pin, index: &Index) -> String {
530        match is_stateless(block, index) {
531            true => temp_name(block, pin),
532            false => format!("{}.{}", instance(block).unwrap_or_default(), pin.parameter_name),
533        }
534    }
535
536    // The output's declared type; `None` for a pin the callee doesn't declare.
537    // Outputs pass by reference, so unwrap the auto-deref pointer's inner type.
538    pub(super) fn temp_type(block: &FbdObject, pin: &Pin, index: &Index) -> Option<String> {
539        let function = block.type_name().unwrap_or_default();
540        let variable = match is_return_pin(block, pin) {
541            true => index.find_return_variable(function),
542            false => index.find_member(function, &pin.parameter_name),
543        }?;
544
545        let information = index.get_type_information_or_void(variable.get_type_name());
546        Some(information.get_inner_name().to_string())
547    }
548
549    // The call operator: `inst`, or `inst.act` for an action.
550    pub(super) fn call_target(block: &FbdObject) -> String {
551        let instance = instance(block).unwrap_or_default();
552        match action(block) {
553            Some(action) => format!("{instance}.{action}"),
554            None => instance.to_string(),
555        }
556    }
557
558    // What outputs are read through: the declared instance, or the owner POU.
559    fn instance(block: &FbdObject) -> Option<&str> {
560        block.instance_name.as_deref().or_else(|| owner(block))
561    }
562
563    // Everything before an action suffix (`MyFb.act` -> `MyFb`).
564    fn owner(block: &FbdObject) -> Option<&str> {
565        block.type_name().map(|name| name.rsplit_once('.').map_or(name, |(owner, _)| owner))
566    }
567
568    // The suffix of a qualified `typeName` (`owner.action`), if any.
569    fn action(block: &FbdObject) -> Option<&str> {
570        block.type_name().and_then(|name| name.rsplit_once('.').map(|(_, action)| action))
571    }
572}

574// Which block output pins some consumer actually reads.
575fn consumed(network: &model::Network, survey: &Survey) -> HashSet<usize> {
576    let mut consumed = HashSet::new();
577    let mut record = |wire: Option<usize>| {
578        if let Trace::Reached(Source::Output { pin, .. }) = trace(wire, survey) {
579            if let Some(id) = pin.output_pin() {
580                consumed.insert(id);
581            }
582        }
583    };
584
585    for object in network.elements() {
586        match Role::from(object) {
587            Role::Sink | Role::Return | Role::Jump => record(consumes(object)),
588            Role::Block => {
589                for pin in block::inputs(object) {
590                    record(pin.source_pin());
591                }
592            }
593            _ => {}
594        }
595    }
596
597    consumed
598}
599
600fn consumes(consumer: &FbdObject) -> Option<usize> {
601    consumer
602        .connection_in
603        .as_ref()
604        .and_then(|it| it.connections.first())
605        .map(|connection| connection.ref_out_id)
606}

629    fn render(network: &Network) -> String {
630        network
631            .statements
632            .iter()
633            .map(|statement| match statement {
634                Statement::Assignment { sink, source } => {
635                    format!("{} := {}", AstSerializer::format(sink), AstSerializer::format(source))
636                }
637                Statement::Return { condition, .. } => {
638                    format!("RETURN {}", AstSerializer::format(condition))
639                }
640                Statement::Jump { condition: Some(condition), target, .. } => {
641                    format!("JMP {target} IF {}", AstSerializer::format(condition))
642                }
643                Statement::Jump { condition: None, target, .. } => {
644                    format!("JMP {target} <disconnected>")
645                }
646                Statement::Label { name, .. } => format!("LABEL {name}"),
647                Statement::Call { target, arguments, capture, .. } => {
648                    let arguments = arguments
649                        .iter()
650                        .map(|argument| match argument {
651                            Argument::Input { parameter, value } => {
652                                format!("{parameter} := {}", AstSerializer::format(value))
653                            }
654                            Argument::Output { parameter, capture: Some(capture) } => {
655                                format!("{parameter} => {capture}")
656                            }
657                            Argument::Output { parameter, capture: None } => format!("{parameter} => "),
658                        })
659                        .collect::<Vec<_>>()
660                        .join(", ");
661                    match capture {
662                        Some(capture) => format!("{capture} := {target}({arguments})"),
663                        None => format!("{target}({arguments})"),
664                    }
665                }
666            })
667            .collect::<Vec<_>>()
668            .join("\n")
669    }
