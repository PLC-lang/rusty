1/// Utility macro to return the name of the current function.
2#[doc(hidden)]
3#[macro_export]
4macro_rules! _function_name {
5    () => {{
6        fn f() {}
7        fn type_name_of_val<T>(_: T) -> &'static str {
8            std::any::type_name::<T>()
9        }
10        let mut name = type_name_of_val(f).strip_suffix("::f").unwrap_or("");
11        while let Some(rest) = name.strip_suffix("::{{closure}}") {
12            name = rest;
13        }
14        name
15    }};
16}

90/// Asserts a [`serde::Serialize`] snapshot in YAML format.
91///
92/// **Feature:** `yaml`
93///
94/// The value needs to implement the [`serde::Serialize`] trait and the snapshot
95/// will be serialized in YAML format.  This does mean that unlike the debug
96/// snapshot variant the type of the value does not appear in the output.
97/// You can however use the [`assert_ron_snapshot!`](crate::assert_ron_snapshot!) macro to dump out
98/// the value in [RON](https://github.com/ron-rs/ron/) format which retains some
99/// type information for more accurate comparisons.
100///
101/// Example:
102///
103/// ```no_run
104/// # use insta::*;
105/// assert_yaml_snapshot!(vec![1, 2, 3]);
106/// ```
107///
108/// Unlike the [`assert_debug_snapshot!`](crate::assert_debug_snapshot!)
109/// macro, this one has a secondary mode where redactions can be defined.
110///
111/// The third argument to the macro can be an object expression for redaction.
112/// It's in the form `{ selector => replacement }` or `match .. { selector => replacement }`.
113/// For more information about redactions refer to the [redactions feature in
114/// the guide](https://insta.rs/docs/redactions/).
115///
116/// Example:
117///
118#[cfg_attr(feature = "redactions", doc = " ```no_run")]
119#[cfg_attr(not(feature = "redactions"), doc = " ```ignore")]
120/// # use insta::*; use serde::Serialize;
121/// # #[derive(Serialize)] struct Value; let value = Value;
122/// assert_yaml_snapshot!(value, {
123///     ".key.to.redact" => "[replacement value]",
124///     ".another.key.*.to.redact" => 42
125/// });
126/// ```
127///
128/// The replacement value can be a string, integer or any other primitive value.
129///
130/// For inline usage the format is `(expression, @reference_value)` where the
131/// reference value must be a string literal.  If you make the initial snapshot
132/// just use an empty string (`@""`).
133///
134/// The snapshot name is optional but can be provided as first argument.
135#[cfg(feature = "yaml")]
136#[cfg_attr(docsrs, doc(cfg(feature = "yaml")))]
137#[macro_export]
138macro_rules! assert_yaml_snapshot {
139    ($($arg:tt)*) => {
140        $crate::_assert_serialized_snapshot!(format=Yaml, $($arg)*);
141    };
142}

235// This macro handles optional trailing commas.
236#[doc(hidden)]
237#[macro_export]
238macro_rules! _assert_serialized_snapshot {
239    // If there are redaction expressions, capture the redactions expressions
240    // and pass to `_assert_snapshot_base`
241    (format=$format:ident, $value:expr, $(match ..)? {$($k:expr => $v:expr),* $(,)?} $($arg:tt)*) => {{
242        let transform = |value| {
243            let (_, value) = $crate::_prepare_snapshot_for_redaction!(value, {$($k => $v),*}, $format);
244            value
245        };
246        $crate::_assert_snapshot_base!(transform=transform, $value $($arg)*);
247    }};
248    // If there's a name and redaction expressions, capture and pass to `_assert_snapshot_base`
249    (format=$format:ident, $name:expr, $value:expr, $(match ..)? {$($k:expr => $v:expr),* $(,)?} $(,)?) => {{
250        let transform = |value| {
251            let (_, value) = $crate::_prepare_snapshot_for_redaction!(value, {$($k => $v),*}, $format);
252            value
253        };
254        $crate::_assert_snapshot_base!(transform=transform, $name, $value);
255    }};
256    // Capture serialization function and pass to `_assert_snapshot_base`
257    //
258    (format=$format:ident, $($arg:tt)*) => {{
259        let transform = |value| {$crate::_macro_support::serialize_value(
260            &value,
261            $crate::_macro_support::SerializationFormat::$format,
262        )};
263        $crate::_assert_snapshot_base!(transform = transform, $($arg)*);
264    }};
265}

267#[cfg(feature = "redactions")]
268#[doc(hidden)]
269#[macro_export]
270macro_rules! _prepare_snapshot_for_redaction {
271    ($value:expr, {$($k:expr => $v:expr),*}, $format:ident) => {
272        {
273            let vec = std::vec![
274                $((
275                    $crate::_macro_support::Selector::parse($k).unwrap(),
276                    $crate::_macro_support::Redaction::from($v)
277                ),)*
278            ];
279            let value = $crate::_macro_support::serialize_value_redacted(
280                &$value,
281                &vec,
282                $crate::_macro_support::SerializationFormat::$format,
283            );
284            (vec, value)
285        }
286    }
287}

328// A helper macro which takes a closure as `transform`, and runs the closure on
329// the value. This allows us to implement other macros with a small wrapper. All
330// snapshot macros eventually call this macro.
331//
332// This macro handles optional trailing commas.
333#[doc(hidden)]
334#[macro_export]
335macro_rules! _assert_snapshot_base {
336    // If there's an inline literal value, wrap the literal in a
337    // `ReferenceValue::Inline`, call self.
338    (transform=$transform:expr, $($arg:expr),*, @$snapshot:literal $(,)?) => {
339        $crate::_assert_snapshot_base!(
340            transform = $transform,
341            #[allow(clippy::needless_raw_string_hashes)]
342            $crate::_macro_support::InlineValue($snapshot),
343            $($arg),*
344        )
345    };
346    // If there's no debug_expr, use the stringified value, call self.
347    (transform=$transform:expr, $name:expr, $value:expr $(,)?) => {
348        $crate::_assert_snapshot_base!(transform = $transform, $name, $value, stringify!($value))
349    };
350    // If there's no name (and necessarily no debug expr), auto generate the
351    // name, call self.
352    (transform=$transform:expr, $value:expr $(,)?) => {
353        $crate::_assert_snapshot_base!(
354            transform = $transform,
355            $crate::_macro_support::AutoName,
356            $value
357        )
358    };
359    // The main macro body — every call to this macro should end up here.
360    (transform=$transform:expr, $name:expr, $value:expr, $debug_expr:expr $(,)?) => {
361        $crate::_macro_support::assert_snapshot(
362            (
363                $name,
364                #[allow(clippy::redundant_closure_call)]
365                $transform(&$value).as_str(),
366            ).into(),
367            $crate::_get_workspace_root!().as_path(),
368            $crate::_function_name!(),
369            module_path!(),
370            file!(),
371            line!(),
372            $debug_expr,
373        )
374        .unwrap()
375    };
376}

457/// Settings configuration macro.
458///
459/// This macro lets you bind some [`Settings`](crate::Settings) temporarily.  The first argument
460/// takes key value pairs that should be set, the second is the block to
461/// execute.  All settings can be set (`sort_maps => value` maps to `set_sort_maps(value)`).
462/// The exception are redactions which can only be set to a vector this way.
463///
464/// This example:
465///
466/// ```rust
467/// insta::with_settings!({sort_maps => true}, {
468///     // run snapshot test here
469/// });
470/// ```
471///
472/// Is equivalent to the following:
473///
474/// ```rust
475/// # use insta::Settings;
476/// let mut settings = Settings::clone_current();
477/// settings.set_sort_maps(true);
478/// settings.bind(|| {
479///     // run snapshot test here
480/// });
481/// ```
482///
483/// Note: before insta 0.17 this macro used
484/// [`Settings::new`](crate::Settings::new) which meant that original settings
485/// were always reset rather than extended.
486#[macro_export]
487macro_rules! with_settings {
488    ({$($k:ident => $v:expr),*$(,)?}, $body:block) => {{
489        let mut settings = $crate::Settings::clone_current();
490        $(
491            settings._private_inner_mut().$k($v);
492        )*
493        settings.bind(|| $body)
494    }}
495}

497/// Executes a closure for all input files matching a glob.
498///
499/// The closure is passed the path to the file.  You can use [`std::fs::read_to_string`]
500/// or similar functions to load the file and process it.
501///
502/// ```
503/// # use insta::{assert_snapshot, glob, Settings};
504/// # let mut settings = Settings::clone_current();
505/// # settings.set_allow_empty_glob(true);
506/// # let _dropguard = settings.bind_to_scope();
507/// use std::fs;
508///
509/// glob!("inputs/*.txt", |path| {
510///     let input = fs::read_to_string(path).unwrap();
511///     assert_snapshot!(input.to_uppercase());
512/// });
513/// ```
514///
515/// The `INSTA_GLOB_FILTER` environment variable can be set to only execute certain files.
516/// The format of the filter is a semicolon separated filter.  For instance by setting
517/// `INSTA_GLOB_FILTER` to `foo-*txt;bar-*.txt` only files starting with `foo-` or `bar-`
518/// end ending in `.txt` will be executed.  When using `cargo-insta` the `--glob-filter`
519/// option can be used instead.
520///
521/// Another effect of the globbing system is that snapshot failures within the glob macro
522/// are deferred until the end of of it.  In other words this means that each snapshot
523/// assertion within the `glob!` block are reported.  It can be disabled by setting
524/// `INSTA_GLOB_FAIL_FAST` environment variable to `1`.
525///
526/// A three-argument version of this macro allows specifying a base directory
527/// for the glob to start in. This allows globbing in arbitrary directories,
528/// including parent directories:
529///
530/// ```
531/// # use insta::{assert_snapshot, glob, Settings};
532/// # let mut settings = Settings::clone_current();
533/// # settings.set_allow_empty_glob(true);
534/// # let _dropguard = settings.bind_to_scope();
535/// use std::fs;
536///
537/// glob!("../test_data", "inputs/*.txt", |path| {
538///     let input = fs::read_to_string(path).unwrap();
539///     assert_snapshot!(input.to_uppercase());
540/// });
541/// ```
542#[cfg(feature = "glob")]
543#[cfg_attr(docsrs, doc(cfg(feature = "glob")))]
544#[macro_export]
545macro_rules! glob {
546    // TODO: I think we could remove the three-argument version of this macro
547    // and just support a pattern such as
548    // `glob!("../test_data/inputs/*.txt"...`.
549    ($base_path:expr, $glob:expr, $closure:expr) => {{
550        use std::path::Path;
551
552        let base = $crate::_get_workspace_root!()
553            .join(Path::new(file!()).parent().unwrap())
554            .join($base_path)
555            .to_path_buf();
556
557        // we try to canonicalize but on some platforms (eg: wasm) that might not work, so
558        // we instead silently fall back.
559        let base = base.canonicalize().unwrap_or_else(|_| base);
560        $crate::_macro_support::glob_exec(
561            $crate::_get_workspace_root!().as_path(),
562            &base,
563            $glob,
564            $closure,
565        );
566    }};
567
568    ($glob:expr, $closure:expr) => {{
569        insta::glob!(".", $glob, $closure)
570    }};
571}

573/// Utility macro to permit a multi-snapshot run where all snapshots match.
574///
575/// Within this block, insta will allow an assertion to be run more than once
576/// (even inline) without generating another snapshot.  Instead it will assert
577/// that snapshot expressions visited more than once are matching.
578///
579/// ```rust
580/// insta::allow_duplicates! {
581///     for x in (0..10).step_by(2) {
582///         let is_even = x % 2 == 0;
583///         insta::assert_debug_snapshot!(is_even, @"true");
584///     }
585/// }
586/// ```
587///
588/// The first snapshot assertion will be used as a gold master and every further
589/// assertion will be checked against it.  If they don't match the assertion will
590/// fail.
591#[macro_export]
592macro_rules! allow_duplicates {
593    ($($x:tt)*) => {
594        $crate::_macro_support::with_allow_duplicates(|| {
595            $($x)*
596        })
597    }
