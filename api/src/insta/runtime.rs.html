38// Writes to stderr and also to a warnings file (if INSTA_WARNINGS_FILE is set).
39// The warnings file allows cargo-insta to display warnings after tests complete,
40// since test runners like nextest suppress output from passing tests by default.
41#[macro_export]
42macro_rules! elog {
43    ($($arg:tt)*) => {{
44        use std::io::Write as _;
45        let msg = format!($($arg)*);
46        let _ = writeln!(std::io::stderr(), "{}", msg);
47        $crate::env::memoize_warning(&msg);
48    }};
49}

52    ($fail_fast:expr, $($tokens:tt)*) => {{
53        if (!$fail_fast) {
54            eprintln!($($tokens)*);
55            eprintln!();
56        } else {
57            panic!($($tokens)*);
58        }
59    }}

193fn detect_snapshot_name(function_name: &str, module_path: &str) -> Result<String, &'static str> {
194    // clean test name first
195    let name = function_name.rsplit("::").next().unwrap();
196
197    let (name, test_prefixed) = if let Some(stripped) = name.strip_prefix("test_") {
198        (stripped, true)
199    } else {
200        (name, false)
201    };
202
203    // next check if we need to add a suffix
204    let name = add_suffix_to_snapshot_name(Cow::Borrowed(name));
205    let key = format!("{}::{}", module_path.replace("::", "__"), name);
206
207    // because fn foo and fn test_foo end up with the same snapshot name we
208    // make sure we detect this here and raise an error.
209    let mut name_clash_detection = TEST_NAME_CLASH_DETECTION
210        .lock()
211        .unwrap_or_else(|x| x.into_inner());
212    match name_clash_detection.get(&key) {
213        None => {
214            name_clash_detection.insert(key.clone(), test_prefixed);
215        }
216        Some(&was_test_prefixed) => {
217            if was_test_prefixed != test_prefixed {
218                panic!(
219                    "Insta snapshot name clash detected between '{name}' \
220                     and 'test_{name}' in '{module_path}'. Rename one function."
221                );
222            }
223        }
224    }
225
226    // The rest of the code just deals with duplicates, which we in some
227    // cases do not want to guard against.
228    if allow_duplicates() {
229        return Ok(name.to_string());
230    }
231
232    // if the snapshot name clashes we need to increment a counter.
233    // we really do not care about poisoning here.
234    let mut counters = TEST_NAME_COUNTERS.lock().unwrap_or_else(|x| x.into_inner());
235    let test_idx = counters.get(&key).cloned().unwrap_or(0) + 1;
236    let rv = if test_idx == 1 {
237        name.to_string()
238    } else {
239        format!("{name}-{test_idx}")
240    };
241    counters.insert(key, test_idx);
242
243    Ok(rv)
244}

246/// If there is a suffix on the settings, append it to the snapshot name.
247fn add_suffix_to_snapshot_name(name: Cow<'_, str>) -> Cow<'_, str> {
248    Settings::with(|settings| {
249        settings
250            .snapshot_suffix()
251            .map(|suffix| Cow::Owned(format!("{name}@{suffix}")))
252            .unwrap_or_else(|| name)
253    })
254}

256fn get_snapshot_filename(
257    module_path: &str,
258    assertion_file: &str,
259    snapshot_name: &str,
260    cargo_workspace: &Path,
261    is_doctest: bool,
262) -> PathBuf {
263    let root = Path::new(cargo_workspace);
264    let base = Path::new(assertion_file);
265    Settings::with(|settings| {
266        root.join(base.parent().unwrap())
267            .join(settings.snapshot_path())
268            .join({
269                use std::fmt::Write;
270                let mut f = String::new();
271                if settings.prepend_module_to_snapshot() {
272                    if is_doctest {
273                        write!(
274                            &mut f,
275                            "doctest_{}__",
276                            base.file_name()
277                                .unwrap()
278                                .to_string_lossy()
279                                .replace('.', "_")
280                        )
281                        .unwrap();
282                    } else {
283                        write!(&mut f, "{}__", module_path.replace("::", "__")).unwrap();
284                    }
285                }
286                write!(
287                    &mut f,
288                    "{}.snap",
289                    snapshot_name.replace(&['/', '\\'][..], "__")
290                )
291                .unwrap();
292                f
293            })
294    })

298/// (but not including the generated value). Responsible for saving the

300#[derive(Debug)]
301struct SnapshotAssertionContext<'a> {
302    tool_config: Arc<ToolConfig>,
303    workspace: &'a Path,
304    module_path: &'a str,
305    snapshot_name: Option<Cow<'a, str>>,
306    snapshot_file: Option<PathBuf>,
307    duplication_key: Option<String>,
308    old_snapshot: Option<Snapshot>,
309    pending_snapshots_path: Option<PathBuf>,
310    assertion_file: &'a str,
311    assertion_line: u32,
312    is_doctest: bool,
313    snapshot_kind: SnapshotKind,
314}

316impl<'a> SnapshotAssertionContext<'a> {
317    fn prepare(
318        new_snapshot_value: &SnapshotValue<'a>,
319        workspace: &'a Path,
320        function_name: &'a str,
321        module_path: &'a str,
322        assertion_file: &'a str,
323        assertion_line: u32,
324    ) -> Result<SnapshotAssertionContext<'a>, Box<dyn Error>> {
325        let tool_config = get_tool_config(workspace);
326        let snapshot_name;
327        let mut duplication_key = None;
328        let mut snapshot_file = None;
329        let mut old_snapshot = None;
330        let mut pending_snapshots_path = None;
331        let is_doctest = is_doctest(function_name);
332
333        match new_snapshot_value {
334            SnapshotValue::FileText { name, .. } | SnapshotValue::Binary { name, .. } => {
335                let name = match &name {
336                    Some(name) => add_suffix_to_snapshot_name(name.clone()),
337                    None => {
338                        if is_doctest {
339                            panic!("Cannot determine reliable names for snapshot in doctests.  Please use explicit names instead.");
340                        }
341                        detect_snapshot_name(function_name, module_path)
342                            .unwrap()
343                            .into()
344                    }
345                };
346                if allow_duplicates() {
347                    duplication_key = Some(format!("named:{module_path}|{name}"));
348                }
349                let file = get_snapshot_filename(
350                    module_path,
351                    assertion_file,
352                    &name,
353                    workspace,
354                    is_doctest,
355                );
356                if fs::metadata(&file).is_ok() {
357                    match Snapshot::from_file(&file) {
358                        Ok(snapshot) => {
359                            old_snapshot = Some(snapshot);
360                        }
361                        Err(err) => {
362                            // If we can't parse the snapshot (e.g., invalid YAML,
363                            // merge conflicts, truncated file), log a warning and
364                            // proceed. The test will generate a new pending snapshot.
365                            elog!(
366                                "{}: Failed to parse snapshot file; \
367                                 a new snapshot will be generated: {}\n  Error: {}",
368                                style("warning").yellow().bold(),
369                                file.display(),
370                                err
371                            );
372                        }
373                    }
374                }
375                snapshot_name = Some(name);
376                snapshot_file = Some(file);
377            }
378            SnapshotValue::InlineText {
379                reference_content: contents,
380                ..
381            } => {
382                if allow_duplicates() {
383                    duplication_key = Some(format!(
384                        "inline:{function_name}|{assertion_file}|{assertion_line}"
385                    ));
386                } else {
387                    prevent_inline_duplicate(function_name, assertion_file, assertion_line);
388                }
389                snapshot_name = detect_snapshot_name(function_name, module_path)
390                    .ok()
391                    .map(Cow::Owned);
392                let mut pending_file = workspace.join(assertion_file);
393                pending_file.set_file_name(format!(
394                    ".{}.pending-snap",
395                    pending_file
396                        .file_name()
397                        .expect("no filename")
398                        .to_str()
399                        .expect("non unicode filename")
400                ));
401                pending_snapshots_path = Some(pending_file);
402                old_snapshot = Some(Snapshot::from_components(
403                    module_path.replace("::", "__"),
404                    None,
405                    MetaData::default(),
406                    SnapshotContents::Text(TextSnapshotContents::from_inline_literal(contents)),
407                ));
408            }
409        };
410
411        let snapshot_type = match new_snapshot_value {
412            SnapshotValue::FileText { .. } | SnapshotValue::InlineText { .. } => SnapshotKind::Text,
413            &SnapshotValue::Binary { extension, .. } => SnapshotKind::Binary {
414                extension: extension.to_string(),
415            },
416        };
417
418        Ok(SnapshotAssertionContext {
419            tool_config,
420            workspace,
421            module_path,
422            snapshot_name,
423            snapshot_file,
424            old_snapshot,
425            pending_snapshots_path,
426            assertion_file,
427            assertion_line,
428            duplication_key,
429            is_doctest,
430            snapshot_kind: snapshot_type,
431        })
432    }

434    /// Given a path returns the local path within the workspace.
435    pub fn localize_path(&self, p: &Path) -> Option<PathBuf> {
436        let workspace = self.workspace.canonicalize().ok()?;
437        let p = self.workspace.join(p).canonicalize().ok()?;
438        p.strip_prefix(&workspace).ok().map(|x| x.to_path_buf())
439    }

441    /// Creates the new snapshot from input values.
442    pub fn new_snapshot(&self, contents: SnapshotContents, expr: &str) -> Snapshot {
443        assert_eq!(
444            contents.is_binary(),
445            matches!(self.snapshot_kind, SnapshotKind::Binary { .. })
446        );
447
448        Snapshot::from_components(
449            self.module_path.replace("::", "__"),
450            self.snapshot_name.as_ref().map(|x| x.to_string()),
451            Settings::with(|settings| MetaData {
452                source: {
453                    let source_path = Path::new(self.assertion_file);
454                    // We need to compute a relative path from the workspace to the source file.
455                    // This is necessary for workspace setups where the project is not a direct
456                    // child of the workspace root (e.g., when workspace and project are siblings).
457                    // We canonicalize paths first to properly handle symlinks.
458                    let canonicalized_base = self.workspace.canonicalize().ok();
459                    let canonicalized_path = source_path.canonicalize().ok();
460
461                    let relative = if let (Some(base), Some(path)) =
462                        (canonicalized_base, canonicalized_path)
463                    {
464                        path_relative_from(&path, &base)
465                            .unwrap_or_else(|| source_path.to_path_buf())
466                    } else {
467                        // If canonicalization fails, try with original paths
468                        path_relative_from(source_path, self.workspace)
469                            .unwrap_or_else(|| source_path.to_path_buf())
470                    };
471                    Some(path_to_storage(&relative))
472                },
473                assertion_line: Some(self.assertion_line),
474                description: settings.description().map(Into::into),
475                expression: if settings.omit_expression() {
476                    None
477                } else {
478                    Some(expr.to_string())
479                },
480                info: settings.info().map(ToOwned::to_owned),
481                input_file: settings
482                    .input_file()
483                    .and_then(|x| self.localize_path(x))
484                    .map(|x| path_to_storage(&x)),
485                snapshot_kind: self.snapshot_kind.clone(),
486            }),
487            contents,
488        )
489    }

491    /// Cleanup logic for passing snapshots.
492    pub fn cleanup_passing(&self) -> Result<(), Box<dyn Error>> {
493        // let's just make sure there are no more pending files lingering
494        // around.
495        if let Some(ref snapshot_file) = self.snapshot_file {
496            let target_path = pending_snapshot_path(self.workspace, snapshot_file);
497            let new_file = target_path.with_extension("snap.new");
498            fs::remove_file(new_file).ok();
499        }
500
501        // and add a null pending snapshot to a pending snapshot file if needed
502        if let Some(ref pending_snapshots) = self.pending_snapshots_path {
503            let target_path = pending_snapshot_path(self.workspace, pending_snapshots);
504            if fs::metadata(&target_path).is_ok() {
505                PendingInlineSnapshot::new(None, None, self.assertion_line).save(&target_path)?;
506            }
507        }
508        Ok(())
509    }

511    /// Removes any old .snap.new.* files that belonged to previous pending snapshots. This should
512    /// only ever remove maximum one file because we do this every time before we create a new
513    /// pending snapshot.
514    pub fn cleanup_previous_pending_binary_snapshots(&self) -> Result<(), Box<dyn Error>> {
515        if let Some(ref path) = self.snapshot_file {
516            // Use pending directory if set
517            let target_path = pending_snapshot_path(self.workspace, path);
518
519            // The file name to compare against has to be valid utf-8 as it is generated by this crate
520            // out of utf-8 strings.
521            let file_name_prefix = format!(
522                "{}.new.",
523                target_path.file_name().unwrap().to_str().unwrap()
524            );
525
526            let read_dir = target_path.parent().unwrap().read_dir();
527
528            match read_dir {
529                Err(e) if e.kind() == ErrorKind::NotFound => return Ok(()),
530                _ => (),
531            }
532
533            // We have to loop over where whole directory here because there is no filesystem API
534            // for getting files by prefix.
535            for entry in read_dir? {
536                let entry = entry?;
537                let entry_file_name = entry.file_name();
538
539                // We'll just skip over files with non-utf-8 names. The assumption being that those
540                // would not have been generated by this crate.
541                if entry_file_name
542                    .to_str()
543                    .map(|f| f.starts_with(&file_name_prefix))
544                    .unwrap_or(false)
545                {
546                    std::fs::remove_file(entry.path())?;
547                }
548            }
549        }
550
551        Ok(())
552    }

554    /// Writes the changes of the snapshot back.
555    pub fn update_snapshot(
556        &self,
557        new_snapshot: Snapshot,
558    ) -> Result<SnapshotUpdateBehavior, Box<dyn Error>> {
559        // TODO: this seems to be making `unseen` be true when there is an
560        // existing snapshot file; which seems wrong??
561        let unseen = self
562            .snapshot_file
563            .as_ref()
564            .map_or(false, |x| fs::metadata(x).is_ok());
565        let should_print = self.tool_config.output_behavior() != OutputBehavior::Nothing;
566        let snapshot_update = snapshot_update_behavior(&self.tool_config, unseen);
567
568        // If snapshot_update is `InPlace` and we have an inline snapshot, then
569        // use `NewFile`, since we can't use `InPlace` for inline. `cargo-insta`
570        // then accepts all snapshots at the end of the test.
571        let snapshot_update =
572            // TODO: could match on the snapshot kind instead of whether snapshot_file is None
573            if snapshot_update == SnapshotUpdateBehavior::InPlace && self.snapshot_file.is_none() {
574                SnapshotUpdateBehavior::NewFile
575            } else {
576                snapshot_update
577            };
578
579        match snapshot_update {
580            SnapshotUpdateBehavior::InPlace => {
581                if let Some(ref snapshot_file) = self.snapshot_file {
582                    new_snapshot.save(snapshot_file)?;
583                    if should_print {
584                        elog!(
585                            "{} {}",
586                            style("updated snapshot").green(),
587                            style(snapshot_file.display()).cyan().underlined(),
588                        );
589                    }
590                } else {
591                    // Checked self.snapshot_file.is_none() above
592                    unreachable!()
593                }
594            }
595            SnapshotUpdateBehavior::NewFile => {
596                if let Some(ref snapshot_file) = self.snapshot_file {
597                    // File snapshot - use pending directory if set
598                    let target_path = pending_snapshot_path(self.workspace, snapshot_file);
599                    let new_path = new_snapshot.save_new(&target_path)?;
600                    if should_print {
601                        elog!(
602                            "{} {}",
603                            style("stored new snapshot").green(),
604                            style(new_path.display()).cyan().underlined(),
605                        );
606                    }
607                } else if self.is_doctest {
608                    if should_print {
609                        elog!(
610                            "{}",
611                            style("warning: cannot update inline snapshots in doctests")
612                                .red()
613                                .bold(),
614                        );
615                    }
616                } else {
617                    // Inline snapshot - use pending directory if set
618                    let pending_path = self.pending_snapshots_path.as_ref().unwrap();
619                    let target_path = pending_snapshot_path(self.workspace, pending_path);
620                    PendingInlineSnapshot::new(
621                        Some(new_snapshot),
622                        self.old_snapshot.clone(),
623                        self.assertion_line,
624                    )
625                    .save(&target_path)?;
626                }
627            }
628            SnapshotUpdateBehavior::NoUpdate => {}
629        }
630
631        Ok(snapshot_update)
632    }

654    /// Finalizes the assertion when the snapshot comparison fails, potentially
655    /// panicking to fail the test
656    fn finalize(&self, update_result: SnapshotUpdateBehavior) {
657        // if we are in glob mode, we want to adjust the finalization
658        // so that we do not show the hints immediately.
659        let fail_fast = {
660            #[cfg(feature = "glob")]
661            {
662                if let Some(top) = crate::glob::GLOB_STACK.lock().unwrap().last() {
663                    top.fail_fast
664                } else {
665                    true
666                }
667            }
668            #[cfg(not(feature = "glob"))]
669            {
670                true
671            }
672        };
673
674        if fail_fast
675            && update_result == SnapshotUpdateBehavior::NewFile
676            && self.tool_config.output_behavior() != OutputBehavior::Nothing
677            && !self.is_doctest
678        {
679            println!(
680                "{hint}",
681                hint = style("To update snapshots run `cargo insta review`").dim(),
682            );
683        }
684
685        if update_result != SnapshotUpdateBehavior::InPlace && !self.tool_config.force_pass() {
686            if fail_fast && self.tool_config.output_behavior() != OutputBehavior::Nothing {
687                let msg = if env::var("INSTA_CARGO_INSTA") == Ok("1".to_string()) {
688                    "Stopped on the first failure."
689                } else {
690                    "Stopped on the first failure. Run `cargo insta test` to run all snapshots."
691                };
692                println!("{hint}", hint = style(msg).dim(),);
693            }
694
695            // if we are in glob mode, count the failures and print the
696            // errors instead of panicking.  The glob will then panic at
697            // the end.
698            #[cfg(feature = "glob")]
699            {
700                let mut stack = crate::glob::GLOB_STACK.lock().unwrap();
701                if let Some(glob_collector) = stack.last_mut() {
702                    glob_collector.failed += 1;
703                    if update_result == SnapshotUpdateBehavior::NewFile
704                        && self.tool_config.output_behavior() != OutputBehavior::Nothing
705                    {
706                        glob_collector.show_insta_hint = true;
707                    }
708
709                    print_or_panic!(
710                        fail_fast,
711                        "snapshot assertion from glob for '{}' failed in line {}",
712                        self.snapshot_name.as_deref().unwrap_or("unnamed snapshot"),
713                        self.assertion_line
714                    );
715                    return;
716                }
717            }
718
719            panic!(
720                "snapshot assertion for '{}' failed in line {}",
721                self.snapshot_name.as_deref().unwrap_or("unnamed snapshot"),
722                self.assertion_line
723            );
724        }
725    }
726}

728/// Computes a relative path from `base` to `path`, returning a path with `../` components
729/// if necessary.
730///
731/// This function is vendored from the old Rust standard library implementation
732/// (pre-1.0, removed in RFC 474) and is distributed under the same terms as the
733/// Rust project (MIT/Apache-2.0 dual license).
734///
735/// Unlike `Path::strip_prefix`, this function can handle cases where `path` is not
736/// a descendant of `base`, making it suitable for finding relative paths between
737/// arbitrary directories (e.g., between sibling directories in a workspace).
738fn path_relative_from(path: &Path, base: &Path) -> Option<PathBuf> {
739    use std::path::Component;
740
741    if path.is_absolute() != base.is_absolute() {
742        if path.is_absolute() {
743            Some(PathBuf::from(path))
744        } else {
745            None
746        }
747    } else {
748        let mut ita = path.components();
749        let mut itb = base.components();
750        let mut comps: Vec<Component> = vec![];
751        loop {
752            match (ita.next(), itb.next()) {
753                (None, None) => break,
754                (Some(a), None) => {
755                    comps.push(a);
756                    comps.extend(ita.by_ref());
757                    break;
758                }
759                (None, _) => comps.push(Component::ParentDir),
760                (Some(a), Some(b)) if comps.is_empty() && a == b => {}
761                (Some(a), Some(_b)) => {
762                    comps.push(Component::ParentDir);
763                    for _ in itb {
764                        comps.push(Component::ParentDir);
765                    }
766                    comps.push(a);
767                    comps.extend(ita.by_ref());
768                    break;
769                }
770            }
771        }
772        Some(comps.iter().map(|c| c.as_os_str()).collect())
773    }
774}

776fn prevent_inline_duplicate(function_name: &str, assertion_file: &str, assertion_line: u32) {
777    let key = format!("{function_name}|{assertion_file}|{assertion_line}");
778    let mut set = INLINE_DUPLICATES.lock().unwrap();
779    if set.contains(&key) {
780        // drop the lock so we don't poison it
781        drop(set);
782        panic!(
783            "Insta does not allow inline snapshot assertions in loops. \
784            Wrap your assertions in allow_duplicates! to change this."
785        );
786    }
787    set.insert(key);
788}
789
790fn record_snapshot_duplicate(
791    results: &mut BTreeMap<String, Snapshot>,
792    snapshot: &Snapshot,
793    ctx: &SnapshotAssertionContext,
794) {
795    let key = ctx.duplication_key.as_deref().unwrap();
796    if let Some(prev_snapshot) = results.get(key) {
797        if prev_snapshot.contents() != snapshot.contents() {
798            println!("Snapshots in allow-duplicates block do not match.");
799            let mut printer = SnapshotPrinter::new(ctx.workspace, Some(prev_snapshot), snapshot);
800            printer.set_line(Some(ctx.assertion_line));
801            printer.set_snapshot_file(ctx.snapshot_file.as_deref());
802            printer.set_title(Some("Differences in Block"));
803            printer.set_snapshot_hints("previous assertion", "current assertion");
804            if ctx.tool_config.output_behavior() == OutputBehavior::Diff {
805                printer.set_show_diff(true);
806            }
807            printer.print();
808            panic!(
809                "snapshot assertion for '{}' failed in line {}. Result \
810                    does not match previous snapshot in allow-duplicates block.",
811                ctx.snapshot_name.as_deref().unwrap_or("unnamed snapshot"),
812                ctx.assertion_line
813            );
814        }
815    } else {
816        results.insert(key.to_string(), snapshot.clone());
817    }
818}

820/// Do we allow recording of duplicates?
821fn allow_duplicates() -> bool {
822    RECORDED_DUPLICATES.with(|x| !x.borrow().is_empty())
823}
824
825/// Helper function to support perfect duplicate detection.
826pub fn with_allow_duplicates<R, F>(f: F) -> R
827where
828    F: FnOnce() -> R,
829{
830    RECORDED_DUPLICATES.with(|x| x.borrow_mut().push(BTreeMap::new()));
831    let rv = std::panic::catch_unwind(std::panic::AssertUnwindSafe(f));
832    RECORDED_DUPLICATES.with(|x| x.borrow_mut().pop().unwrap());
833    match rv {
834        Ok(rv) => rv,
835        Err(payload) => std::panic::resume_unwind(payload),
836    }
837}

839/// This function is invoked from the macros to run the main assertion logic.
840///
841/// This will create the assertion context, run the main logic to assert
842/// on snapshots and write changes to the pending snapshot files.  It will
843/// also print the necessary bits of information to the output and fail the
844/// assertion with a panic if needed.
845#[allow(clippy::too_many_arguments)]
846pub fn assert_snapshot(
847    snapshot_value: SnapshotValue<'_>,
848    workspace: &Path,
849    function_name: &str,
850    module_path: &str,
851    assertion_file: &str,
852    assertion_line: u32,
853    expr: &str,
854) -> Result<(), Box<dyn Error>> {
855    let ctx = SnapshotAssertionContext::prepare(
856        &snapshot_value,
857        workspace,
858        function_name,
859        module_path,
860        assertion_file,
861        assertion_line,
862    )?;
863
864    ctx.cleanup_previous_pending_binary_snapshots()?;
865
866    let content = match snapshot_value {
867        SnapshotValue::FileText { content, .. } | SnapshotValue::InlineText { content, .. } => {
868            // apply filters if they are available
869            #[cfg(feature = "filters")]
870            let content = Settings::with(|settings| settings.filters().apply_to(content));
871
872            let kind = match ctx.snapshot_file {
873                Some(_) => TextSnapshotKind::File,
874                None => TextSnapshotKind::Inline,
875            };
876
877            TextSnapshotContents::new(content.into(), kind).into()
878        }
879        SnapshotValue::Binary {
880            content, extension, ..
881        } => {
882            assert!(
883                extension != "new",
884                "'.new' is not allowed as a file extension"
885            );
886            assert!(
887                !extension.starts_with("new."),
888                "file extensions starting with 'new.' are not allowed",
889            );
890
891            SnapshotContents::Binary(Rc::new(content))
892        }
893    };
894
895    let new_snapshot = ctx.new_snapshot(content, expr);
896
897    // memoize the snapshot file if requested, as part of potentially removing unreferenced snapshots
898    if let Some(ref snapshot_file) = ctx.snapshot_file {
899        memoize_snapshot_file(snapshot_file);
900    }
901
902    // If we allow assertion with duplicates, we record the duplicate now.  This will
903    // in itself fail the assertion if the previous visit of the same assertion macro
904    // did not yield the same result.
905    RECORDED_DUPLICATES.with(|x| {
906        if let Some(results) = x.borrow_mut().last_mut() {
907            record_snapshot_duplicate(results, &new_snapshot, &ctx);
908        }
909    });
910
911    let pass = ctx
912        .old_snapshot
913        .as_ref()
914        .map(|x| {
915            if ctx.tool_config.require_full_match() {
916                x.matches_fully(&new_snapshot)
917            } else {
918                x.matches(&new_snapshot)
919            }
920        })
921        .unwrap_or(false);
922
923    if pass {
924        ctx.cleanup_passing()?;
925
926        if matches!(
927            ctx.tool_config.snapshot_update(),
928            crate::env::SnapshotUpdate::Force
929        ) {
930            ctx.update_snapshot(new_snapshot)?;
931        }
932    // otherwise print information and update snapshots.
933    } else {
934        ctx.print_snapshot_info(&new_snapshot);
935        let update_result = ctx.update_snapshot(new_snapshot)?;
936        ctx.finalize(update_result);
937    }
938
939    Ok(())
940}
