3const NANOS_PER_MILLISECOND: i64 = 1_000 * 1_000;
4const NANOS_PER_SECOND: i64 = 1_000 * 1_000 * 1_000;
5const SECONDS_PER_DAY: u32 = 60 * 60 * 24;

7/// .
8/// Converts DT to DATE
9///
10#[allow(non_snake_case)]
11#[no_mangle]
12pub extern "C" fn DATE_AND_TIME_TO_DATE(input: u32) -> u32 {
13    let input_seconds = input as i64;
14    let date_time = chrono::Utc.timestamp_opt(input_seconds, 0).single().expect("Out of range DT value");
15
16    let midnight_seconds = date_time
17        .date_naive()
18        .and_hms_opt(0, 0, 0)
19        .expect("Cannot create date time from date")
20        .and_utc()
21        .timestamp();
22
23    midnight_seconds as u32
24}

26/// .
27/// Converts LDT to LDATE
28///
29#[allow(non_snake_case)]
30#[no_mangle]
31pub extern "C" fn LDATE_AND_TIME_TO_LDATE(input: i64) -> i64 {
32    let date_time = chrono::Utc.timestamp_nanos(input);
33
34    let new_date_time =
35        date_time.date_naive().and_hms_opt(0, 0, 0).expect("Cannot create date time from date");
36
37    new_date_time.and_utc().timestamp_nanos_opt().expect("Out of range, cannot create DATE")
38}

40/// .
41/// Converts DT to TOD
42///
43#[allow(non_snake_case)]
44#[no_mangle]
45pub extern "C" fn DATE_AND_TIME_TO_TIME_OF_DAY(input: u32) -> u32 {
46    let input_seconds = input as i64;
47    let date_time = chrono::Utc.timestamp_opt(input_seconds, 0).single().expect("Out of range DT value");
48
49    let midnight = chrono::NaiveDate::from_ymd_opt(1970, 1, 1)
50        .and_then(|date| date.and_hms_opt(date_time.hour(), date_time.minute(), date_time.second()))
51        .expect("Cannot create date time from given parameters")
52        .and_utc();
53
54    midnight.timestamp_millis() as u32
55}

57/// .
58/// Converts LDT to LTOD
59///
60#[allow(non_snake_case)]
61#[no_mangle]
62pub extern "C" fn LDATE_AND_TIME_TO_LTIME_OF_DAY(input: i64) -> i64 {
63    let date_time = chrono::Utc.timestamp_nanos(input);
64    let hour = date_time.hour();
65    let min = date_time.minute();
66    let sec = date_time.second();
67    let nano = date_time.timestamp_subsec_nanos();
68
69    let new_date_time = chrono::NaiveDate::from_ymd_opt(1970, 1, 1)
70        .and_then(|date| date.and_hms_nano_opt(hour, min, sec, nano))
71        .expect("Cannot create date time from given parameters");
72
73    new_date_time.and_utc().timestamp_nanos_opt().expect("Out of range, cannot create TOD")
74}
