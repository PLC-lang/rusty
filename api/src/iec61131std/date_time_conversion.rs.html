3/// .
4/// Converts DT/LDT to DATE
5///
6#[allow(non_snake_case)]
7#[no_mangle]
8pub extern "C" fn DATE_AND_TIME_TO_DATE(input: i64) -> i64 {
9    let date_time = chrono::Utc.timestamp_nanos(input);
10
11    let new_date_time =
12        date_time.date_naive().and_hms_opt(0, 0, 0).expect("Cannot create date time from date");
13
14    new_date_time.and_utc().timestamp_nanos_opt().expect("Out of range, cannot create DATE")
15}

17/// .
18/// Converts DT/LDT to TOD/LTOD
19///
20#[allow(non_snake_case)]
21#[no_mangle]
22pub extern "C" fn DATE_AND_TIME_TO_TIME_OF_DAY(input: i64) -> i64 {
23    let date_time = chrono::Utc.timestamp_nanos(input);
24    let hour = date_time.hour();
25    let min = date_time.minute();
26    let sec = date_time.second();
27    let nano = date_time.timestamp_subsec_nanos();
28
29    let new_date_time = chrono::NaiveDate::from_ymd_opt(1970, 1, 1)
30        .and_then(|date| date.and_hms_nano_opt(hour, min, sec, nano))
31        .expect("Cannot create date time from given parameters");
32
33    new_date_time.and_utc().timestamp_nanos_opt().expect("Out of range, cannot create TOD")
34}
