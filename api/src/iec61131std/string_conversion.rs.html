44/// .
45/// Converts WCHAR to CHAR
46///
47#[allow(non_snake_case)]
48#[no_mangle]
49pub extern "C" fn WCHAR_TO_CHAR(input: u16) -> u8 {
50    let u16_arr = [input];
51    let mut res_iter = char::decode_utf16(u16_arr).map(|r| r.unwrap_or(std::char::REPLACEMENT_CHARACTER));
52    let mut res_arr = [u8::MAX; 80];
53    if let Some(res) = res_iter.next() {
54        if res_iter.next().is_none() {
55            res.encode_utf8(&mut res_arr);
56        }
57    }
58    res_arr[0]
59}
