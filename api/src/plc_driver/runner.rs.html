24///
25/// Compiles and runs the given sources
26/// Sources must be `Compilable`, default implementations include `String` and `&str`
27/// An implementation is also provided for `Vec<SourceContainer>`
28///
29pub fn compile<T: Compilable>(codegen_context: &CodegenContext, source: T) -> GeneratedModule<'_> {
30    let source = source.containers();
31    let project = Project::new("TestProject".to_string()).with_sources(source);
32    let context = GlobalContext::new().with_source(project.get_sources(), None).unwrap();
33    let diagnostician = Diagnostician::null_diagnostician();
34    let mut pipeline = BuildPipeline {
35        context,
36        project,
37        diagnostician,
38        compile_parameters: None,
39        linker: plc::linker::LinkerType::Internal,
40        mutable_participants: Default::default(),
41        participants: Default::default(),
42        module_name: Some("<internal>".to_string()),
43    };
44
45    pipeline.register_default_participants();
46
47    let project = pipeline.parse().unwrap();
48    let project = pipeline.index(project).unwrap();
49    let project = pipeline.annotate(project).unwrap();
50
51    let compile_options = CompileOptions {
52        optimization: plc::OptimizationLevel::None,
53        debug_level: plc::DebugLevel::None,
54        ..Default::default()
55    };
56
57    match project.generate_single_module(codegen_context, &compile_options, None) {
58        Ok(res) => res.unwrap(),
59        Err(e) => panic!("{e}"),
60    }
61}

63///
64/// A Convenience method to compile and then run the given source
65///
66pub fn compile_and_run<T, U, S: Compilable>(source: S, params: &mut T) -> U {
67    let context: CodegenContext = CodegenContext::create();
68    let module = compile(&context, source);
69    module.print_to_stderr();
70    module.run::<T, U>("main", params)
71}

73///
74/// A Convenience method to compile and then run the given source
75/// without external parameters
76///
77pub fn compile_and_run_no_params<U, S: Compilable>(source: S) -> U {
78    let context: CodegenContext = CodegenContext::create();
79    let module = compile(&context, source);
80    module.print_to_stderr();
81    module.run_no_param::<U>("main")
