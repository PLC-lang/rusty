31/// A Build particitpant for different steps in the pipeline
32/// Implementors can decide parse the Ast and project information
33/// to do actions like validation or logging
34pub trait PipelineParticipant: Sync + Send {
35    /// Implement this to access the project before it gets indexed
36    /// This happens directly after parsing
37    fn pre_index(&mut self, _parsed_project: &ParsedProject) {}
38    /// Implement this to access the project after it got indexed
39    /// This happens directly after the index returns
40    fn post_index(&self, _indexed_project: &IndexedProject) {}
41    /// Implement this to access the project before it gets annotated
42    /// This happens after indexing
43    fn pre_annotate(&self, _indexed_project: &IndexedProject) {}
44    /// Implement this to access the project after it got annotated
45    /// This happens directly after annotations
46    fn post_annotate(&self, _annotated_project: &AnnotatedProject) {}
47    /// Implement this to access the project before it gets generated
48    /// This happens after annotation
49    fn pre_generate(&mut self, _annotated_project: &AnnotatedProject) -> Result<(), Diagnostic> {
50        Ok(())
51    }
52    /// Implement this to get access to the module generation section of the codegen
53    /// This is useful if generating multiple modules to hook into single module generation
54    fn generate(&self, _generated_module: &GeneratedModule) -> Result<(), Diagnostic> {
55        Ok(())
56    }
57    /// Implement this to access the project after it got generated
58    /// This happens after codegen
59    fn post_generate(&self) -> Result<(), Diagnostic> {
60        Ok(())
61    }
62}
63
64/// A Mutating Build particitpant for different steps in the pipeline
65/// Implementors can decide to modify the AST, project and generated code,
66/// for example for de-sugaring/lowering/pre-processing the AST
67/// If a previous step is being modified, such as the AST or index,
68/// the caller is responsible for calling the previous steps
69pub trait PipelineParticipantMut {
70    /// Implement this to access the project before it gets indexed
71    /// This happens directly after parsing
72    fn pre_index(&mut self, parsed_project: ParsedProject) -> ParsedProject {
73        parsed_project
74    }
75    /// Implement this to access the project after it got indexed
76    /// This happens directly after the index returns
77    fn post_index(&mut self, indexed_project: IndexedProject) -> IndexedProject {
78        indexed_project
79    }
80    /// Implement this to access the project before it gets annotated
81    /// This happens directly after the constants are evaluated
82    fn pre_annotate(&mut self, indexed_project: IndexedProject) -> IndexedProject {
83        indexed_project
84    }
85    /// Implement this to access the project after it got annotated
86    /// This happens directly after annotations
87    fn post_annotate(&mut self, annotated_project: AnnotatedProject) -> AnnotatedProject {
88        annotated_project
89    }
90}

92pub struct CodegenParticipant<T: SourceContainer> {
93    pub compile_options: crate::CompileOptions,
94    pub link_options: crate::LinkOptions,
95    pub target: Target,
96    pub got_layout: Mutex<HashMap<String, u64>>,
97    pub compile_dirs: HashMap<Target, PathBuf>,
98    pub objects: Arc<RwLock<GeneratedProject>>,
99    pub libraries: Vec<LibraryInformation<T>>,
100}
101
102impl<T: SourceContainer> CodegenParticipant<T> {
103    /// Ensures the directores for the various targets have been created
104    fn ensure_compile_dirs(&mut self) -> Result<(), Diagnostic> {
105        let compile_directory = self.compile_options.build_location.clone().unwrap_or_else(|| {
106            let tempdir = tempfile::tempdir().expect("Could not create tempdir");
107            tempdir.keep()
108        });
109        if let Some(name) = self.target.try_get_name() {
110            let dir = compile_directory.join(name);
111            fs::create_dir_all(&dir)?;
112            self.compile_dirs.insert(self.target.clone(), dir);
113        } else {
114            self.compile_dirs.insert(self.target.clone(), compile_directory);
115        }
116        Ok(())
117    }
118    pub fn read_got_layout(location: &str, format: ConfigFormat) -> Result<HashMap<String, u64>, Diagnostic> {
119        let path = Path::new(location);
120        if !path.is_file() {
121            // Assume if the file doesn't exist that there is no existing GOT layout yet. write_got_layout will handle
122            // creating our file when we want to.
123            return Ok(HashMap::new());
124        }
125
126        let s = fs::read_to_string(location)
127            .map_err(|_| Diagnostic::new("GOT layout could not be read from file"))?;
128        match format {
129            ConfigFormat::JSON => serde_json::from_str(&s)
130                .map_err(|_| Diagnostic::new("Could not deserialize GOT layout from JSON")),
131            ConfigFormat::TOML => toml::de::from_str(&s)
132                .map_err(|_| Diagnostic::new("Could not deserialize GOT layout from TOML")),
133        }
134    }
135}

137impl<T: SourceContainer + Send> PipelineParticipant for CodegenParticipant<T> {
138    fn pre_generate(&mut self, _annotated_project: &AnnotatedProject) -> Result<(), Diagnostic> {
139        self.ensure_compile_dirs()?;
140
141        let got_layout =
142            if let OnlineChange::Enabled { file_name, format } = &self.compile_options.online_change {
143                Self::read_got_layout(file_name, *format)?
144            } else {
145                HashMap::default()
146            };
147        self.got_layout = Mutex::new(got_layout);
148        Ok(())
149    }
150
151    fn generate(&self, module: &GeneratedModule) -> Result<(), Diagnostic> {
152        let current_dir = env::current_dir()?;
153        let current_dir = self.compile_options.root.as_deref().unwrap_or(&current_dir);
154        let unit_location = module.get_unit_location();
155        let unit_location =
156            if unit_location.exists() { fs::canonicalize(unit_location)? } else { unit_location.into() };
157        let output_name = if unit_location.starts_with(current_dir) {
158            unit_location.strip_prefix(current_dir).map_err(|it| {
159                Diagnostic::new(format!("Could not strip prefix for {}", current_dir.to_string_lossy()))
160                    .with_internal_error(it.into())
161            })?
162        } else if unit_location.has_root() {
163            let root = unit_location.ancestors().last().expect("Should exist?");
164            unit_location.strip_prefix(root).expect("The root directory should exist")
165        } else {
166            unit_location.as_path()
167        };
168
169        let output_name = match self.compile_options.output_format {
170            FormatOption::IR => format!("{}.ll", output_name.to_string_lossy()),
171            FormatOption::Bitcode => format!("{}.bc", output_name.to_string_lossy()),
172            _ => format!("{}.o", output_name.to_string_lossy()),
173        };
174
175        let target = &self.target;
176        let compile_directory = self.compile_dirs.get(target).expect("Required dir");
177        let object = module
178            .persist(
179                Some(compile_directory),
180                &output_name,
181                self.compile_options.output_format,
182                target,
183                self.compile_options.optimization,
184            )
185            .map(Into::into)
186            .map(|it: Object| it.with_target(target))?;
187        self.objects.write().expect("Failed to aquire read write lock").objects.push(object);
188        Ok(())
189    }
190
191    fn post_generate(&self) -> Result<(), Diagnostic> {
192        let output_name = &self.compile_options.output;
193
194        let _objects = self.objects.read().expect("Failed to aquire read lock for objects").link(
195            &[], //Original project objects embedded in participant
196            self.link_options.build_location.as_deref(),
197            self.link_options.lib_location.as_deref(),
198            output_name,
199            self.link_options.clone(),
200        )?;
201        if let Some(lib_location) = &self.link_options.lib_location {
202            for library in self.libraries.iter().filter(|it| it.should_copy()).map(|it| it.get_compiled_lib())
203            {
204                for obj in library.get_objects() {
205                    let path = obj.get_path();
206                    if let Some(name) = path.file_name() {
207                        std::fs::copy(path, lib_location.join(name))?;
208                    }
209                }
210            }
211        }
212        Ok(())
213    }
214}

227impl PipelineParticipantMut for InitParticipant {
228    fn pre_annotate(&mut self, indexed_project: IndexedProject) -> IndexedProject {
229        // Create a new init lowerer
230        let IndexedProject { project: ParsedProject { units }, index, .. } = indexed_project;
231        let mut resulting_units = vec![];
232        let index = Rc::new(index);
233        for unit in units {
234            let initializer = Initializer::new(self.id_provider.clone(), self.generate_externals);
235            let unit = initializer.apply_initialization(unit, index.clone());
236            resulting_units.push(unit);
237        }
238        // Append new units and constructor to the ast and re-index
239        let project = ParsedProject { units: resulting_units };
240        project.index(self.id_provider.clone())
241    }
242}

244impl PipelineParticipantMut for InheritanceLowerer {
245    fn pre_index(&mut self, parsed_project: ParsedProject) -> ParsedProject {
246        let ParsedProject { mut units } = parsed_project;
247        units.iter_mut().for_each(|unit| self.visit_unit(unit));
248        ParsedProject { units }
249    }
250
251    fn post_annotate(&mut self, annotated_project: AnnotatedProject) -> AnnotatedProject {
252        let AnnotatedProject { mut units, index, annotations } = annotated_project;
253        self.annotations = Some(Box::new(annotations));
254        self.index = Some(index);
255        units.iter_mut().for_each(|unit| self.visit_unit(&mut unit.unit));
256        let index = self.index.take().expect("Index should be present");
257        // re-resolve
258        IndexedProject {
259            project: ParsedProject {
260                units: units.into_iter().map(|AnnotatedUnit { unit, .. }| unit).collect(),
261            },
262            index,
263            _unresolvables: vec![],
264        }
265        .annotate(self.provider())
266    }
267}

269impl PipelineParticipantMut for AggregateTypeLowerer {
270    fn post_index(&mut self, indexed_project: IndexedProject) -> IndexedProject {
271        let IndexedProject { mut project, index, .. } = indexed_project;
272        self.index = Some(index);
273        self.visit(&mut project.units);
274
275        //Re-index
276        //TODO: it would be nice if we could unimport
277        project.index(self.id_provider.clone())
278    }
279
280    fn post_annotate(&mut self, annotated_project: AnnotatedProject) -> AnnotatedProject {
281        let AnnotatedProject { units, index, annotations } = annotated_project;
282        self.index = Some(index);
283        self.annotation = Some(Box::new(annotations));
284
285        let units = units
286            .into_iter()
287            .map(|AnnotatedUnit { mut unit, .. }| {
288                self.visit_unit(&mut unit);
289                unit
290            })
291            .collect();
292        let indexed_project = IndexedProject {
293            project: ParsedProject { units },
294            index: self.index.take().expect("Index"),
295            _unresolvables: vec![],
296        };
297        indexed_project.annotate(self.id_provider.clone())
298    }
299}

301impl PipelineParticipantMut for VirtualTableGenerator {
302    fn post_index(&mut self, indexed_project: IndexedProject) -> IndexedProject {
303        let IndexedProject { mut project, index, .. } = indexed_project;
304
305        let mut gen = VirtualTableGenerator::new(self.ids.clone(), self.generate_external_constructors);
306        gen.generate(&index, &mut project.units);
307
308        project.index(self.ids.clone())
309    }

328            index: self.index.take().expect("Index"),
329            _unresolvables: vec![],

331
332        indexed_project.annotate(self.ids.clone())
