27/// Glob represents a single glob in a gitignore file.
28///
29/// This is used to report information about the highest precedent glob that
30/// matched in one or more gitignore files.
31#[derive(Clone, Debug)]
32pub struct Glob {
33    /// The file path that this glob was extracted from.
34    from: Option<PathBuf>,
35    /// The original glob string.
36    original: String,
37    /// The actual glob string used to convert to a regex.
38    actual: String,
39    /// Whether this is a whitelisted glob or not.
40    is_whitelist: bool,
41    /// Whether this glob should only match directories or not.
42    is_only_dir: bool,
43}
44
45impl Glob {
46    /// Returns the file path that defined this glob.
47    pub fn from(&self) -> Option<&Path> {
48        self.from.as_ref().map(|p| &**p)
49    }
50
51    /// The original glob as it was defined in a gitignore file.
52    pub fn original(&self) -> &str {
53        &self.original
54    }
55
56    /// The actual glob that was compiled to respect gitignore
57    /// semantics.
58    pub fn actual(&self) -> &str {
59        &self.actual
60    }
61
62    /// Whether this was a whitelisted glob or not.
63    pub fn is_whitelist(&self) -> bool {
64        self.is_whitelist
65    }
66
67    /// Whether this glob must match a directory or not.
68    pub fn is_only_dir(&self) -> bool {
69        self.is_only_dir
70    }
71
72    /// Returns true if and only if this glob has a `**/` prefix.
73    fn has_doublestar_prefix(&self) -> bool {
74        self.actual.starts_with("**/") || self.actual == "**"
75    }
76}

78/// Gitignore is a matcher for the globs in one or more gitignore files
79/// in the same directory.
80#[derive(Clone, Debug)]
81pub struct Gitignore {
82    set: GlobSet,
83    root: PathBuf,
84    globs: Vec<Glob>,
85    num_ignores: u64,
86    num_whitelists: u64,
87    matches: Option<Arc<Pool<Vec<usize>>>>,
88}

90impl Gitignore {
91    /// Creates a new gitignore matcher from the gitignore file path given.
92    ///
93    /// If it's desirable to include multiple gitignore files in a single
94    /// matcher, or read gitignore globs from a different source, then
95    /// use `GitignoreBuilder`.
96    ///
97    /// This always returns a valid matcher, even if it's empty. In particular,
98    /// a Gitignore file can be partially valid, e.g., when one glob is invalid
99    /// but the rest aren't.
100    ///
101    /// Note that I/O errors are ignored. For more granular control over
102    /// errors, use `GitignoreBuilder`.
103    pub fn new<P: AsRef<Path>>(
104        gitignore_path: P,
105    ) -> (Gitignore, Option<Error>) {
106        let path = gitignore_path.as_ref();
107        let parent = path.parent().unwrap_or(Path::new("/"));
108        let mut builder = GitignoreBuilder::new(parent);
109        let mut errs = PartialErrorBuilder::default();
110        errs.maybe_push_ignore_io(builder.add(path));
111        match builder.build() {
112            Ok(gi) => (gi, errs.into_error_option()),
113            Err(err) => {
114                errs.push(err);
115                (Gitignore::empty(), errs.into_error_option())
116            }
117        }
118    }
119
120    /// Creates a new gitignore matcher from the global ignore file, if one
121    /// exists.
122    ///
123    /// The global config file path is specified by git's `core.excludesFile`
124    /// config option.
125    ///
126    /// Git's config file location is `$HOME/.gitconfig`. If `$HOME/.gitconfig`
127    /// does not exist or does not specify `core.excludesFile`, then
128    /// `$XDG_CONFIG_HOME/git/ignore` is read. If `$XDG_CONFIG_HOME` is not
129    /// set or is empty, then `$HOME/.config/git/ignore` is used instead.
130    pub fn global() -> (Gitignore, Option<Error>) {
131        match std::env::current_dir() {
132            Ok(cwd) => GitignoreBuilder::new(cwd).build_global(),
133            Err(err) => (Gitignore::empty(), Some(err.into())),
134        }
135    }

180    /// Returns whether the given path (file or directory) matched a pattern in
181    /// this gitignore matcher.
182    ///
183    /// `is_dir` should be true if the path refers to a directory and false
184    /// otherwise.
185    ///
186    /// The given path is matched relative to the path given when building
187    /// the matcher. Specifically, before matching `path`, its prefix (as
188    /// determined by a common suffix of the directory containing this
189    /// gitignore) is stripped. If there is no common suffix/prefix overlap,
190    /// then `path` is assumed to be relative to this matcher.
191    pub fn matched<P: AsRef<Path>>(
192        &self,
193        path: P,
194        is_dir: bool,
195    ) -> Match<&Glob> {
196        if self.is_empty() {
197            return Match::None;
198        }
199        self.matched_stripped(self.strip(path.as_ref()), is_dir)
200    }
201
202    /// Returns whether the given path (file or directory, and expected to be
203    /// under the root) or any of its parent directories (up to the root)
204    /// matched a pattern in this gitignore matcher.
205    ///
206    /// NOTE: This method is more expensive than walking the directory hierarchy
207    /// top-to-bottom and matching the entries. But, is easier to use in cases
208    /// when a list of paths are available without a hierarchy.
209    ///
210    /// `is_dir` should be true if the path refers to a directory and false
211    /// otherwise.
212    ///
213    /// The given path is matched relative to the path given when building
214    /// the matcher. Specifically, before matching `path`, its prefix (as
215    /// determined by a common suffix of the directory containing this
216    /// gitignore) is stripped. If there is no common suffix/prefix overlap,
217    /// then `path` is assumed to be relative to this matcher.
218    ///
219    /// # Panics
220    ///
221    /// This method panics if the given file path is not under the root path
222    /// of this matcher.
223    pub fn matched_path_or_any_parents<P: AsRef<Path>>(
224        &self,
225        path: P,
226        is_dir: bool,
227    ) -> Match<&Glob> {
228        if self.is_empty() {
229            return Match::None;
230        }
231        let mut path = self.strip(path.as_ref());
232        assert!(!path.has_root(), "path is expected to be under the root");
233
234        match self.matched_stripped(path, is_dir) {
235            Match::None => (), // walk up
236            a_match => return a_match,
237        }
238        while let Some(parent) = path.parent() {
239            match self.matched_stripped(parent, /* is_dir */ true) {
240                Match::None => path = parent, // walk up
241                a_match => return a_match,
242            }
243        }
244        Match::None
245    }
246
247    /// Like matched, but takes a path that has already been stripped.
248    fn matched_stripped<P: AsRef<Path>>(
249        &self,
250        path: P,
251        is_dir: bool,
252    ) -> Match<&Glob> {
253        if self.is_empty() {
254            return Match::None;
255        }
256        let path = path.as_ref();
257        let mut matches = self.matches.as_ref().unwrap().get();
258        let candidate = Candidate::new(path);
259        self.set.matches_candidate_into(&candidate, &mut *matches);
260        for &i in matches.iter().rev() {
261            let glob = &self.globs[i];
262            if !glob.is_only_dir() || is_dir {
263                return if glob.is_whitelist() {
264                    Match::Whitelist(glob)
265                } else {
266                    Match::Ignore(glob)
267                };
268            }
269        }
270        Match::None
271    }
272
273    /// Strips the given path such that it's suitable for matching with this
274    /// gitignore matcher.
275    fn strip<'a, P: 'a + AsRef<Path> + ?Sized>(
276        &'a self,
277        path: &'a P,
278    ) -> &'a Path {
279        let mut path = path.as_ref();
280        // A leading ./ is completely superfluous. We also strip it from
281        // our gitignore root path, so we need to strip it from our candidate
282        // path too.
283        if let Some(p) = strip_prefix("./", path) {
284            path = p;
285        }
286        // Strip any common prefix between the candidate path and the root
287        // of the gitignore, to make sure we get relative matching right.
288        // BUT, a file name might not have any directory components to it,
289        // in which case, we don't want to accidentally strip any part of the
290        // file name.
291        //
292        // As an additional special case, if the root is just `.`, then we
293        // shouldn't try to strip anything, e.g., when path begins with a `.`.
294        if self.root != Path::new(".") && !is_file_name(path) {
295            if let Some(p) = strip_prefix(&self.root, path) {
296                path = p;
297                // If we're left with a leading slash, get rid of it.
298                if let Some(p) = strip_prefix("/", path) {
299                    path = p;
300                }
301            }
302        }
303        path
304    }
305}

307/// Builds a matcher for a single set of globs from a .gitignore file.
308#[derive(Clone, Debug)]
309pub struct GitignoreBuilder {
310    builder: GlobSetBuilder,
311    root: PathBuf,
312    globs: Vec<Glob>,
313    case_insensitive: bool,
314    allow_unclosed_class: bool,
315}

317impl GitignoreBuilder {
318    /// Create a new builder for a gitignore file.
319    ///
320    /// The path given should be the path at which the globs for this gitignore
321    /// file should be matched. Note that paths are always matched relative
322    /// to the root path given here. Generally, the root path should correspond
323    /// to the *directory* containing a `.gitignore` file.
324    pub fn new<P: AsRef<Path>>(root: P) -> GitignoreBuilder {
325        let root = root.as_ref();
326        GitignoreBuilder {
327            builder: GlobSetBuilder::new(),
328            root: strip_prefix("./", root).unwrap_or(root).to_path_buf(),
329            globs: vec![],
330            case_insensitive: false,
331            allow_unclosed_class: true,
332        }
333    }
334
335    /// Builds a new matcher from the globs added so far.
336    ///
337    /// Once a matcher is built, no new globs can be added to it.
338    pub fn build(&self) -> Result<Gitignore, Error> {
339        let nignore = self.globs.iter().filter(|g| !g.is_whitelist()).count();
340        let nwhite = self.globs.iter().filter(|g| g.is_whitelist()).count();
341        let set = self
342            .builder
343            .build()
344            .map_err(|err| Error::Glob { glob: None, err: err.to_string() })?;
345        Ok(Gitignore {
346            set,
347            root: self.root.clone(),
348            globs: self.globs.clone(),
349            num_ignores: nignore as u64,
350            num_whitelists: nwhite as u64,
351            matches: Some(Arc::new(Pool::new(|| vec![]))),
352        })
353    }
354
355    /// Build a global gitignore matcher using the configuration in this
356    /// builder.
357    ///
358    /// This consumes ownership of the builder unlike `build` because it
359    /// must mutate the builder to add the global gitignore globs.
360    ///
361    /// Note that this ignores the path given to this builder's constructor
362    /// and instead derives the path automatically from git's global
363    /// configuration.
364    pub fn build_global(mut self) -> (Gitignore, Option<Error>) {
365        match gitconfig_excludes_path() {
366            None => (Gitignore::empty(), None),
367            Some(path) => {
368                if !path.is_file() {
369                    (Gitignore::empty(), None)
370                } else {
371                    let mut errs = PartialErrorBuilder::default();
372                    errs.maybe_push_ignore_io(self.add(path));
373                    match self.build() {
374                        Ok(gi) => (gi, errs.into_error_option()),
375                        Err(err) => {
376                            errs.push(err);
377                            (Gitignore::empty(), errs.into_error_option())
378                        }
379                    }
380                }
381            }
382        }
383    }

385    /// Add each glob from the file path given.
386    ///
387    /// The file given should be formatted as a `gitignore` file.
388    ///
389    /// Note that partial errors can be returned. For example, if there was
390    /// a problem adding one glob, an error for that will be returned, but
391    /// all other valid globs will still be added.
392    pub fn add<P: AsRef<Path>>(&mut self, path: P) -> Option<Error> {
393        let path = path.as_ref();
394        let file = match File::open(path) {
395            Err(err) => return Some(Error::Io(err).with_path(path)),
396            Ok(file) => file,
397        };
398        log::debug!("opened gitignore file: {}", path.display());
399        let rdr = BufReader::new(file);
400        let mut errs = PartialErrorBuilder::default();
401        for (i, line) in rdr.lines().enumerate() {
402            let lineno = (i + 1) as u64;
403            let line = match line {
404                Ok(line) => line,
405                Err(err) => {
406                    errs.push(Error::Io(err).tagged(path, lineno));
407                    break;
408                }
409            };
410
411            // Match Git's handling of .gitignore files that begin with the Unicode BOM
412            const UTF8_BOM: &str = "\u{feff}";
413            let line =
414                if i == 0 { line.trim_start_matches(UTF8_BOM) } else { &line };
415
416            if let Err(err) = self.add_line(Some(path.to_path_buf()), &line) {
417                errs.push(err.tagged(path, lineno));
418            }
419        }
420        errs.into_error_option()
421    }

423    /// Add each glob line from the string given.
424    ///
425    /// If this string came from a particular `gitignore` file, then its path
426    /// should be provided here.
427    ///
428    /// The string given should be formatted as a `gitignore` file.
429    #[cfg(test)]
430    fn add_str(
431        &mut self,
432        from: Option<PathBuf>,
433        gitignore: &str,
434    ) -> Result<&mut GitignoreBuilder, Error> {
435        for line in gitignore.lines() {
436            self.add_line(from.clone(), line)?;
437        }
438        Ok(self)
439    }

441    /// Add a line from a gitignore file to this builder.
442    ///
443    /// If this line came from a particular `gitignore` file, then its path
444    /// should be provided here.
445    ///
446    /// If the line could not be parsed as a glob, then an error is returned.
447    pub fn add_line(
448        &mut self,
449        from: Option<PathBuf>,
450        mut line: &str,
451    ) -> Result<&mut GitignoreBuilder, Error> {
452        #![allow(deprecated)]
453
454        if line.starts_with("#") {
455            return Ok(self);
456        }
457        if !line.ends_with("\\ ") {
458            line = line.trim_right();
459        }
460        if line.is_empty() {
461            return Ok(self);
462        }
463        let mut glob = Glob {
464            from,
465            original: line.to_string(),
466            actual: String::new(),
467            is_whitelist: false,
468            is_only_dir: false,
469        };
470        let mut is_absolute = false;
471        if line.starts_with("\\!") || line.starts_with("\\#") {
472            line = &line[1..];
473            is_absolute = line.chars().nth(0) == Some('/');
474        } else {
475            if line.starts_with("!") {
476                glob.is_whitelist = true;
477                line = &line[1..];
478            }
479            if line.starts_with("/") {
480                // `man gitignore` says that if a glob starts with a slash,
481                // then the glob can only match the beginning of a path
482                // (relative to the location of gitignore). We achieve this by
483                // simply banning wildcards from matching /.
484                line = &line[1..];
485                is_absolute = true;
486            }
487        }
488        // If it ends with a slash, then this should only match directories,
489        // but the slash should otherwise not be used while globbing.
490        if line.as_bytes().last() == Some(&b'/') {
491            glob.is_only_dir = true;
492            line = &line[..line.len() - 1];
493            // If the slash was escaped, then remove the escape.
494            // See: https://github.com/BurntSushi/ripgrep/issues/2236
495            if line.as_bytes().last() == Some(&b'\\') {
496                line = &line[..line.len() - 1];
497            }
498        }
499        glob.actual = line.to_string();
500        // If there is a literal slash, then this is a glob that must match the
501        // entire path name. Otherwise, we should let it match anywhere, so use
502        // a **/ prefix.
503        if !is_absolute && !line.chars().any(|c| c == '/') {
504            // ... but only if we don't already have a **/ prefix.
505            if !glob.has_doublestar_prefix() {
506                glob.actual = format!("**/{}", glob.actual);
507            }
508        }
509        // If the glob ends with `/**`, then we should only match everything
510        // inside a directory, but not the directory itself. Standard globs
511        // will match the directory. So we add `/*` to force the issue.
512        if glob.actual.ends_with("/**") {
513            glob.actual = format!("{}/*", glob.actual);
514        }
515        let parsed = GlobBuilder::new(&glob.actual)
516            .literal_separator(true)
517            .case_insensitive(self.case_insensitive)
518            .backslash_escape(true)
519            .allow_unclosed_class(self.allow_unclosed_class)
520            .build()
521            .map_err(|err| Error::Glob {
522                glob: Some(glob.original.clone()),
523                err: err.kind().to_string(),
524            })?;
525        self.builder.add(parsed);
526        self.globs.push(glob);
527        Ok(self)
528    }

530    /// Toggle whether the globs should be matched case insensitively or not.
531    ///
532    /// When this option is changed, only globs added after the change will be
533    /// affected.
534    ///
535    /// This is disabled by default.
536    pub fn case_insensitive(
537        &mut self,
538        yes: bool,
539    ) -> Result<&mut GitignoreBuilder, Error> {
540        // TODO: This should not return a `Result`. Fix this in the next semver
541        // release.
542        self.case_insensitive = yes;
543        Ok(self)
544    }

546    /// Toggle whether unclosed character classes are allowed. When allowed,
547    /// a `[` without a matching `]` is treated literally instead of resulting
548    /// in a parse error.
549    ///
550    /// For example, if this is set then the glob `[abc` will be treated as the
551    /// literal string `[abc` instead of returning an error.
552    ///
553    /// By default, this is true in order to match established `gitignore`
554    /// semantics. Generally speaking, enabling this leads to worse failure
555    /// modes since the glob parser becomes more permissive. You might want to
556    /// enable this when compatibility (e.g., with POSIX glob implementations)
557    /// is more important than good error messages.
558    pub fn allow_unclosed_class(
559        &mut self,
560        yes: bool,
561    ) -> &mut GitignoreBuilder {
562        self.allow_unclosed_class = yes;
563        self
564    }

567/// Return the file path of the current environment's global gitignore file.
568///
569/// Note that the file path returned may not exist.
570pub fn gitconfig_excludes_path() -> Option<PathBuf> {
571    // git supports $HOME/.gitconfig and $XDG_CONFIG_HOME/git/config. Notably,
572    // both can be active at the same time, where $HOME/.gitconfig takes
573    // precedent. So if $HOME/.gitconfig defines a `core.excludesFile`, then
574    // we're done.
575    match gitconfig_home_contents().and_then(|x| parse_excludes_file(&x)) {
576        Some(path) => return Some(path),
577        None => {}
578    }
579    match gitconfig_xdg_contents().and_then(|x| parse_excludes_file(&x)) {
580        Some(path) => return Some(path),
581        None => {}
582    }
583    excludes_file_default()
584}

586/// Returns the file contents of git's global config file, if one exists, in
587/// the user's home directory.
588fn gitconfig_home_contents() -> Option<Vec<u8>> {
589    let home = match home_dir() {
590        None => return None,
591        Some(home) => home,
592    };
593    let mut file = match File::open(home.join(".gitconfig")) {
594        Err(_) => return None,
595        Ok(file) => BufReader::new(file),
596    };
597    let mut contents = vec![];
598    file.read_to_end(&mut contents).ok().map(|_| contents)
599}

601/// Returns the file contents of git's global config file, if one exists, in
602/// the user's XDG_CONFIG_HOME directory.
603fn gitconfig_xdg_contents() -> Option<Vec<u8>> {
604    let path = std::env::var_os("XDG_CONFIG_HOME")
605        .and_then(|x| if x.is_empty() { None } else { Some(PathBuf::from(x)) })
606        .or_else(|| home_dir().map(|p| p.join(".config")))
607        .map(|x| x.join("git/config"));
608    let mut file = match path.and_then(|p| File::open(p).ok()) {
609        None => return None,
610        Some(file) => BufReader::new(file),
611    };
612    let mut contents = vec![];
613    file.read_to_end(&mut contents).ok().map(|_| contents)
614}

616/// Returns the default file path for a global .gitignore file.
617///
618/// Specifically, this respects XDG_CONFIG_HOME.
619fn excludes_file_default() -> Option<PathBuf> {
620    std::env::var_os("XDG_CONFIG_HOME")
621        .and_then(|x| if x.is_empty() { None } else { Some(PathBuf::from(x)) })
622        .or_else(|| home_dir().map(|p| p.join(".config")))
623        .map(|x| x.join("git/ignore"))

626/// Extract git's `core.excludesfile` config setting from the raw file contents
627/// given.
628fn parse_excludes_file(data: &[u8]) -> Option<PathBuf> {
629    use std::sync::OnceLock;
630
631    use regex_automata::{meta::Regex, util::syntax};
632
633    // N.B. This is the lazy approach, and isn't technically correct, but
634    // probably works in more circumstances. I guess we would ideally have
635    // a full INI parser. Yuck.
636    static RE: OnceLock<Regex> = OnceLock::new();
637    let re = RE.get_or_init(|| {
638        Regex::builder()
639            .configure(Regex::config().utf8_empty(false))
640            .syntax(syntax::Config::new().utf8(false))
641            .build(r#"(?im-u)^\s*excludesfile\s*=\s*"?\s*(\S+?)\s*"?\s*$"#)
642            .unwrap()
643    });
644    // We don't care about amortizing allocs here I think. This should only
645    // be called ~once per traversal or so? (Although it's not guaranteed...)
646    let mut caps = re.create_captures();
647    re.captures(data, &mut caps);
648    let span = caps.get_group(1)?;
649    let candidate = &data[span];
650    std::str::from_utf8(candidate).ok().map(|s| PathBuf::from(expand_tilde(s)))
651}

653/// Expands ~ in file paths to the value of $HOME.
654fn expand_tilde(path: &str) -> String {
655    let home = match home_dir() {
656        None => return path.to_string(),
657        Some(home) => home.to_string_lossy().into_owned(),
658    };
659    path.replace("~", &home)
660}
661
662/// Returns the location of the user's home directory.
663fn home_dir() -> Option<PathBuf> {
664    // We're fine with using std::env::home_dir for now. Its bugs are, IMO,
665    // pretty minor corner cases.
666    #![allow(deprecated)]
667    std::env::home_dir()
668}
