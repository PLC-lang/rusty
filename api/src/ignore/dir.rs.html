1// This module provides a data structure, `Ignore`, that connects "directory
2// traversal" with "ignore matchers." Specifically, it knows about gitignore
3// semantics and precedence, and is organized based on directory hierarchy.
4// Namely, every matcher logically corresponds to ignore rules from a single
5// directory, and points to the matcher for its corresponding parent directory.
6// In this sense, `Ignore` is a *persistent* data structure.
7//
8// This design was specifically chosen to make it possible to use this data
9// structure in a parallel directory iterator.
10//
11// My initial intention was to expose this module as part of this crate's
12// public API, but I think the data structure's public API is too complicated
13// with non-obvious failure modes. Alas, such things haven't been documented
14// well.
15

34/// IgnoreMatch represents information about where a match came from when using
35/// the `Ignore` matcher.
36#[derive(Clone, Debug)]
37#[allow(dead_code)]
38pub(crate) struct IgnoreMatch<'a>(IgnoreMatchInner<'a>);
39
40/// IgnoreMatchInner describes precisely where the match information came from.
41/// This is private to allow expansion to more matchers in the future.
42#[derive(Clone, Debug)]
43#[allow(dead_code)]
44enum IgnoreMatchInner<'a> {
45    Override(overrides::Glob<'a>),
46    Gitignore(&'a gitignore::Glob),
47    Types(types::Glob<'a>),
48    Hidden,
49}
50
51impl<'a> IgnoreMatch<'a> {
52    fn overrides(x: overrides::Glob<'a>) -> IgnoreMatch<'a> {
53        IgnoreMatch(IgnoreMatchInner::Override(x))
54    }
55
56    fn gitignore(x: &'a gitignore::Glob) -> IgnoreMatch<'a> {
57        IgnoreMatch(IgnoreMatchInner::Gitignore(x))
58    }
59
60    fn types(x: types::Glob<'a>) -> IgnoreMatch<'a> {
61        IgnoreMatch(IgnoreMatchInner::Types(x))
62    }
63
64    fn hidden() -> IgnoreMatch<'static> {
65        IgnoreMatch(IgnoreMatchInner::Hidden)
66    }
67}

174    /// Create a new `Ignore` matcher with the parent directories of `dir`.
175    ///
176    /// Note that this can only be called on an `Ignore` matcher with no
177    /// parents (i.e., `is_root` returns `true`). This will panic otherwise.
178    pub(crate) fn add_parents<P: AsRef<Path>>(
179        &self,
180        path: P,
181    ) -> (Ignore, Option<Error>) {
182        if !self.0.opts.parents
183            && !self.0.opts.git_ignore
184            && !self.0.opts.git_exclude
185            && !self.0.opts.git_global
186        {
187            // If we never need info from parent directories, then don't do
188            // anything.
189            return (self.clone(), None);
190        }
191        if !self.is_root() {
192            panic!("Ignore::add_parents called on non-root matcher");
193        }
194        let absolute_base = match path.as_ref().canonicalize() {
195            Ok(path) => Arc::new(path),
196            Err(_) => {
197                // There's not much we can do here, so just return our
198                // existing matcher. We drop the error to be consistent
199                // with our general pattern of ignoring I/O errors when
200                // processing ignore files.
201                return (self.clone(), None);
202            }
203        };
204        // List of parents, from child to root.
205        let mut parents = vec![];
206        let mut path = &**absolute_base;
207        while let Some(parent) = path.parent() {
208            parents.push(parent);
209            path = parent;
210        }
211        let mut errs = PartialErrorBuilder::default();
212        let mut ig = self.clone();
213        for parent in parents.into_iter().rev() {
214            let mut compiled = self.0.compiled.write().unwrap();
215            if let Some(weak) = compiled.get(parent.as_os_str()) {
216                if let Some(prebuilt) = weak.upgrade() {
217                    ig = Ignore(prebuilt);
218                    continue;
219                }
220            }
221            let (mut igtmp, err) = ig.add_child_path(parent);
222            errs.maybe_push(err);
223            igtmp.is_absolute_parent = true;
224            igtmp.absolute_base = Some(absolute_base.clone());
225            igtmp.has_git =
226                if self.0.opts.require_git && self.0.opts.git_ignore {
227                    parent.join(".git").exists() || parent.join(".jj").exists()
228                } else {
229                    false
230                };
231            let ig_arc = Arc::new(igtmp);
232            ig = Ignore(ig_arc.clone());
233            compiled.insert(
234                parent.as_os_str().to_os_string(),
235                Arc::downgrade(&ig_arc),
236            );
237        }
238        (ig, errs.into_error_option())
239    }

257    /// Like add_child, but takes a full path and returns an IgnoreInner.
258    fn add_child_path(&self, dir: &Path) -> (IgnoreInner, Option<Error>) {
259        let check_vcs_dir = self.0.opts.require_git
260            && (self.0.opts.git_ignore || self.0.opts.git_exclude);
261        let git_type = if check_vcs_dir {
262            dir.join(".git").metadata().ok().map(|md| md.file_type())
263        } else {
264            None
265        };
266        let has_git =
267            check_vcs_dir && (git_type.is_some() || dir.join(".jj").exists());
268
269        let mut errs = PartialErrorBuilder::default();
270        let custom_ig_matcher = if self.0.custom_ignore_filenames.is_empty() {
271            Gitignore::empty()
272        } else {
273            let (m, err) = create_gitignore(
274                &dir,
275                &dir,
276                &self.0.custom_ignore_filenames,
277                self.0.opts.ignore_case_insensitive,
278            );
279            errs.maybe_push(err);
280            m
281        };
282        let ig_matcher = if !self.0.opts.ignore {
283            Gitignore::empty()
284        } else {
285            let (m, err) = create_gitignore(
286                &dir,
287                &dir,
288                &[".ignore"],
289                self.0.opts.ignore_case_insensitive,
290            );
291            errs.maybe_push(err);
292            m
293        };
294        let gi_matcher = if !self.0.opts.git_ignore {
295            Gitignore::empty()
296        } else {
297            let (m, err) = create_gitignore(
298                &dir,
299                &dir,
300                &[".gitignore"],
301                self.0.opts.ignore_case_insensitive,
302            );
303            errs.maybe_push(err);
304            m
305        };
306
307        let gi_exclude_matcher = if !self.0.opts.git_exclude {
308            Gitignore::empty()
309        } else {
310            match resolve_git_commondir(dir, git_type) {
311                Ok(git_dir) => {
312                    let (m, err) = create_gitignore(
313                        &dir,
314                        &git_dir,
315                        &["info/exclude"],
316                        self.0.opts.ignore_case_insensitive,
317                    );
318                    errs.maybe_push(err);
319                    m
320                }
321                Err(err) => {
322                    errs.maybe_push(err);
323                    Gitignore::empty()
324                }
325            }
326        };
327        let ig = IgnoreInner {
328            compiled: self.0.compiled.clone(),
329            dir: dir.to_path_buf(),
330            overrides: self.0.overrides.clone(),
331            types: self.0.types.clone(),
332            parent: Some(self.clone()),
333            is_absolute_parent: false,
334            absolute_base: self.0.absolute_base.clone(),
335            global_gitignores_relative_to: self
336                .0
337                .global_gitignores_relative_to
338                .clone(),
339            explicit_ignores: self.0.explicit_ignores.clone(),
340            custom_ignore_filenames: self.0.custom_ignore_filenames.clone(),
341            custom_ignore_matcher: custom_ig_matcher,
342            ignore_matcher: ig_matcher,
343            git_global_matcher: self.0.git_global_matcher.clone(),
344            git_ignore_matcher: gi_matcher,
345            git_exclude_matcher: gi_exclude_matcher,
346            has_git,
347            opts: self.0.opts,
348        };
349        (ig, errs.into_error_option())
350    }

379    /// Returns a match indicating whether the given file path should be
380    /// ignored or not.
381    ///
382    /// The match contains information about its origin.
383    fn matched<'a, P: AsRef<Path>>(
384        &'a self,
385        path: P,
386        is_dir: bool,
387    ) -> Match<IgnoreMatch<'a>> {
388        // We need to be careful with our path. If it has a leading ./, then
389        // strip it because it causes nothing but trouble.
390        let mut path = path.as_ref();
391        if let Some(p) = strip_prefix("./", path) {
392            path = p;
393        }
394        // Match against the override patterns. If an override matches
395        // regardless of whether it's whitelist/ignore, then we quit and
396        // return that result immediately. Overrides have the highest
397        // precedence.
398        if !self.0.overrides.is_empty() {
399            let mat = self
400                .0
401                .overrides
402                .matched(path, is_dir)
403                .map(IgnoreMatch::overrides);
404            if !mat.is_none() {
405                return mat;
406            }
407        }
408        let mut whitelisted = Match::None;
409        if self.has_any_ignore_rules() {
410            let mat = self.matched_ignore(path, is_dir);
411            if mat.is_ignore() {
412                return mat;
413            } else if mat.is_whitelist() {
414                whitelisted = mat;
415            }
416        }
417        if !self.0.types.is_empty() {
418            let mat =
419                self.0.types.matched(path, is_dir).map(IgnoreMatch::types);
420            if mat.is_ignore() {
421                return mat;
422            } else if mat.is_whitelist() {
423                whitelisted = mat;
424            }
425        }
426        whitelisted
427    }

429    /// Performs matching only on the ignore files for this directory and
430    /// all parent directories.
431    fn matched_ignore<'a>(
432        &'a self,
433        path: &Path,
434        is_dir: bool,
435    ) -> Match<IgnoreMatch<'a>> {
436        let (
437            mut m_custom_ignore,
438            mut m_ignore,
439            mut m_gi,
440            mut m_gi_exclude,
441            mut m_explicit,
442        ) = (Match::None, Match::None, Match::None, Match::None, Match::None);
443        let any_git =
444            !self.0.opts.require_git || self.parents().any(|ig| ig.0.has_git);
445        let mut saw_git = false;
446        for ig in self.parents().take_while(|ig| !ig.0.is_absolute_parent) {
447            if m_custom_ignore.is_none() {
448                m_custom_ignore =
449                    ig.0.custom_ignore_matcher
450                        .matched(path, is_dir)
451                        .map(IgnoreMatch::gitignore);
452            }
453            if m_ignore.is_none() {
454                m_ignore =
455                    ig.0.ignore_matcher
456                        .matched(path, is_dir)
457                        .map(IgnoreMatch::gitignore);
458            }
459            if any_git && !saw_git && m_gi.is_none() {
460                m_gi =
461                    ig.0.git_ignore_matcher
462                        .matched(path, is_dir)
463                        .map(IgnoreMatch::gitignore);
464            }
465            if any_git && !saw_git && m_gi_exclude.is_none() {
466                m_gi_exclude =
467                    ig.0.git_exclude_matcher
468                        .matched(path, is_dir)
469                        .map(IgnoreMatch::gitignore);
470            }
471            saw_git = saw_git || ig.0.has_git;
472        }
473        if self.0.opts.parents {
474            if let Some(abs_parent_path) = self.absolute_base() {
475                // What we want to do here is take the absolute base path of
476                // this directory and join it with the path we're searching.
477                // The main issue we want to avoid is accidentally duplicating
478                // directory components, so we try to strip any common prefix
479                // off of `path`. Overall, this seems a little ham-fisted, but
480                // it does fix a nasty bug. It should do fine until we overhaul
481                // this crate.
482                let path = abs_parent_path.join(
483                    self.parents()
484                        .take_while(|ig| !ig.0.is_absolute_parent)
485                        .last()
486                        .map_or(path, |ig| {
487                            // This is a weird special case when ripgrep users
488                            // search with just a `.`, as some tools do
489                            // automatically (like consult). In this case, if
490                            // we don't bail out now, the code below will strip
491                            // a leading `.` from `path`, which might mangle
492                            // a hidden file name!
493                            if ig.0.dir.as_path() == Path::new(".") {
494                                return path;
495                            }
496                            let without_dot_slash =
497                                strip_if_is_prefix("./", ig.0.dir.as_path());
498                            let relative_base =
499                                strip_if_is_prefix(without_dot_slash, path);
500                            strip_if_is_prefix("/", relative_base)
501                        }),
502                );
503
504                for ig in
505                    self.parents().skip_while(|ig| !ig.0.is_absolute_parent)
506                {
507                    if m_custom_ignore.is_none() {
508                        m_custom_ignore =
509                            ig.0.custom_ignore_matcher
510                                .matched(&path, is_dir)
511                                .map(IgnoreMatch::gitignore);
512                    }
513                    if m_ignore.is_none() {
514                        m_ignore =
515                            ig.0.ignore_matcher
516                                .matched(&path, is_dir)
517                                .map(IgnoreMatch::gitignore);
518                    }
519                    if any_git && !saw_git && m_gi.is_none() {
520                        m_gi =
521                            ig.0.git_ignore_matcher
522                                .matched(&path, is_dir)
523                                .map(IgnoreMatch::gitignore);
524                    }
525                    if any_git && !saw_git && m_gi_exclude.is_none() {
526                        m_gi_exclude =
527                            ig.0.git_exclude_matcher
528                                .matched(&path, is_dir)
529                                .map(IgnoreMatch::gitignore);
530                    }
531                    saw_git = saw_git || ig.0.has_git;
532                }
533            }
534        }
535        for gi in self.0.explicit_ignores.iter().rev() {
536            if !m_explicit.is_none() {
537                break;
538            }
539            m_explicit = gi.matched(&path, is_dir).map(IgnoreMatch::gitignore);
540        }
541        let m_global = if any_git {
542            self.0
543                .git_global_matcher
544                .matched(&path, is_dir)
545                .map(IgnoreMatch::gitignore)
546        } else {
547            Match::None
548        };
549
550        m_custom_ignore
551            .or(m_ignore)
552            .or(m_gi)
553            .or(m_gi_exclude)
554            .or(m_global)
555            .or(m_explicit)
556    }

570/// An iterator over all parents of an ignore matcher, including itself.
571///
572/// The lifetime `'a` refers to the lifetime of the initial `Ignore` matcher.
573pub(crate) struct Parents<'a>(Option<&'a Ignore>);
574
575impl<'a> Iterator for Parents<'a> {
576    type Item = &'a Ignore;
577
578    fn next(&mut self) -> Option<&'a Ignore> {
579        match self.0.take() {
580            None => None,
581            Some(ig) => {
582                self.0 = ig.0.parent.as_ref();
583                Some(ig)
584            }
585        }
586    }

651    /// Builds a new `Ignore` matcher using the given CWD directory.
652    ///
653    /// The matcher returned won't match anything until ignore rules from
654    /// directories are added to it.
655    pub(crate) fn build_with_cwd(&self, cwd: Option<PathBuf>) -> Ignore {
656        let global_gitignores_relative_to =
657            cwd.or_else(|| self.global_gitignores_relative_to.clone());
658        let git_global_matcher = if !self.opts.git_global {
659            Gitignore::empty()
660        } else if let Some(ref cwd) = global_gitignores_relative_to {
661            let mut builder = GitignoreBuilder::new(cwd);
662            builder
663                .case_insensitive(self.opts.ignore_case_insensitive)
664                .unwrap();
665            let (gi, err) = builder.build_global();
666            if let Some(err) = err {
667                log::debug!("{}", err);
668            }
669            gi
670        } else {
671            log::debug!(
672                "ignoring global gitignore file because CWD is not known"
673            );
674            Gitignore::empty()
675        };
676
677        Ignore(Arc::new(IgnoreInner {
678            compiled: Arc::new(RwLock::new(HashMap::new())),
679            dir: self.dir.clone(),
680            overrides: self.overrides.clone(),
681            types: self.types.clone(),
682            parent: None,
683            is_absolute_parent: true,
684            absolute_base: None,
685            global_gitignores_relative_to,
686            explicit_ignores: Arc::new(self.explicit_ignores.clone()),
687            custom_ignore_filenames: Arc::new(
688                self.custom_ignore_filenames.clone(),
689            ),
690            custom_ignore_matcher: Gitignore::empty(),
691            ignore_matcher: Gitignore::empty(),
692            git_global_matcher: Arc::new(git_global_matcher),
693            git_ignore_matcher: Gitignore::empty(),
694            git_exclude_matcher: Gitignore::empty(),
695            has_git: false,
696            opts: self.opts,
697        }))
698    }

839/// Creates a new gitignore matcher for the directory given.
840///
841/// The matcher is meant to match files below `dir`.
842/// Ignore globs are extracted from each of the file names relative to
843/// `dir_for_ignorefile` in the order given (earlier names have lower
844/// precedence than later names).
845///
846/// I/O errors are ignored.
847pub(crate) fn create_gitignore<T: AsRef<OsStr>>(
848    dir: &Path,
849    dir_for_ignorefile: &Path,
850    names: &[T],
851    case_insensitive: bool,
852) -> (Gitignore, Option<Error>) {
853    let mut builder = GitignoreBuilder::new(dir);
854    let mut errs = PartialErrorBuilder::default();
855    builder.case_insensitive(case_insensitive).unwrap();
856    for name in names {
857        let gipath = dir_for_ignorefile.join(name.as_ref());
858        // This check is not necessary, but is added for performance. Namely,
859        // a simple stat call checking for existence can often be just a bit
860        // quicker than actually trying to open a file. Since the number of
861        // directories without ignore files likely greatly exceeds the number
862        // with ignore files, this check generally makes sense.
863        //
864        // However, until demonstrated otherwise, we speculatively do not do
865        // this on Windows since Windows is notorious for having slow file
866        // system operations. Namely, it's not clear whether this analysis
867        // makes sense on Windows.
868        //
869        // For more details: https://github.com/BurntSushi/ripgrep/pull/1381
870        if cfg!(windows) || gipath.exists() {
871            errs.maybe_push_ignore_io(builder.add(gipath));
872        }
873    }
874    let gi = match builder.build() {
875        Ok(gi) => gi,
876        Err(err) => {
877            errs.push(err);
878            GitignoreBuilder::new(dir).build().unwrap()
879        }
880    };
881    (gi, errs.into_error_option())
882}

884/// Find the GIT_COMMON_DIR for the given git worktree.
885///
886/// This is the directory that may contain a private ignore file
887/// "info/exclude". Unlike git, this function does *not* read environment
888/// variables GIT_DIR and GIT_COMMON_DIR, because it is not clear how to use
889/// them when multiple repositories are searched.
890///
891/// Some I/O errors are ignored.
892fn resolve_git_commondir(
893    dir: &Path,
894    git_type: Option<FileType>,
895) -> Result<PathBuf, Option<Error>> {
896    let git_dir_path = || dir.join(".git");
897    let git_dir = git_dir_path();
898    if !git_type.map_or(false, |ft| ft.is_file()) {
899        return Ok(git_dir);
900    }
901    let file = match File::open(git_dir) {
902        Ok(file) => io::BufReader::new(file),
903        Err(err) => {
904            return Err(Some(Error::Io(err).with_path(git_dir_path())));
905        }
906    };
907    let dot_git_line = match file.lines().next() {
908        Some(Ok(line)) => line,
909        Some(Err(err)) => {
910            return Err(Some(Error::Io(err).with_path(git_dir_path())));
911        }
912        None => return Err(None),
913    };
914    if !dot_git_line.starts_with("gitdir: ") {
915        return Err(None);
916    }
917    let real_git_dir = PathBuf::from(&dot_git_line["gitdir: ".len()..]);
918    let git_commondir_file = || real_git_dir.join("commondir");
919    let file = match File::open(git_commondir_file()) {
920        Ok(file) => io::BufReader::new(file),
921        Err(_) => return Err(None),
922    };
923    let commondir_line = match file.lines().next() {
924        Some(Ok(line)) => line,
925        Some(Err(err)) => {
926            return Err(Some(Error::Io(err).with_path(git_commondir_file())));
927        }
928        None => return Err(None),
929    };
930    let commondir_abs = if commondir_line.starts_with(".") {
931        real_git_dir.join(commondir_line) // relative commondir
932    } else {
933        PathBuf::from(commondir_line)
934    };
935    Ok(commondir_abs)
936}

945}
