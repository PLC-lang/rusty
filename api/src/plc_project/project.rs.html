124impl Project<PathBuf> {
125    /// Retrieve a project for compilation from a json description
126    pub fn from_config(config: &Path) -> Result<Self> {
127        let project_config = ProjectConfig::from_file(config)?;
128        let libraries = project_config
129            .libraries
130            .into_iter()
131            .map(|conf| {
132                let lib_path = config.parent().map(|it| it.join(&conf.path)).unwrap_or_else(|| conf.path);
133                let linkage: Linkage = conf.package.into();
134                // Use the linkage type to find the library from the given name
135                // TODO: We should allow for a fix name in the configuration if the library does not follow the unix convention
136                // TODO: We should also allow a way to define objects based on the architecture
137                let object_name = match linkage {
138                    Linkage::Static => format! {"lib{}.a", &conf.name},
139                    Linkage::Shared(_) => format! {"lib{}.so", &conf.name},
140                };
141
142                let lib_file = lib_path.join(object_name);
143                let mut objects = vec![];
144                if lib_file.exists() {
145                    objects.push(lib_file.into());
146                }
147                let compiled_library = CompiledLibrary {
148                    objects,
149                    headers: resolve_file_paths(Some(&lib_path), conf.include_path)?,
150                };
151                Ok(LibraryInformation {
152                    name: conf.name,
153                    location: Some(lib_path),
154                    linkage: conf.package.into(),
155                    library: Library::Compiled(compiled_library),
156                })
157            })
158            .collect::<Result<Vec<_>>>()?;
159
160        let current_dir = env::current_dir()?;
161        let location = config.parent().map(Path::to_path_buf).or(Some(current_dir));
162        let sources = resolve_file_paths(location.as_deref(), project_config.files)?;
163        Ok(Project {
164            name: project_config.name,
165            location,
166            sources,
167            libraries,
168            format: project_config.compile_type,
169            output: project_config.output,
170            includes: vec![],
171            objects: vec![],
172            library_paths: vec![],
173        })
174    }
175
176    pub fn with_file_paths(self, files: Vec<PathBuf>) -> Self {
177        let mut proj = self;
178        let files = resolve_file_paths(proj.get_location(), files).unwrap();
179        for file in files {
180            if matches!(file.get_type(), SourceType::Unknown) {
181                let obj = file.into();
182                proj.objects.push(obj);
183            } else {
184                proj.sources.push(file);
185            }
186        }
187        proj
188    }
189
190    pub fn with_include_paths(self, files: Vec<PathBuf>) -> Self {
191        let mut proj = self;
192        proj.includes = resolve_file_paths(proj.get_location(), files).unwrap();
193        proj
194    }

309fn resolve_file_paths(location: Option<&Path>, inputs: Vec<PathBuf>) -> Result<Vec<PathBuf>> {
310    let mut sources = Vec::new();
311    //Ensure we are working with a directory
312    let location = location.and_then(|it| if it.is_file() { it.parent() } else { Some(it) });
313    for input in &inputs {
314        let input = location.map(|it| it.join(input)).unwrap_or(input.to_path_buf());
315        let path = &input.to_string_lossy();
316        let paths = glob(path).context(format!("Failed to read glob pattern {path}"))?;
317
318        for p in paths {
319            let path = p.context("Illegal Path")?;
320            sources.push(path);
321        }
322    }
323    Ok(sources)
324}

326impl From<LinkageInfo> for Linkage {
