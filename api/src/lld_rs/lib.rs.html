35impl LldResult {
36    pub fn ok(self) -> Result<(), String> {
37        if self.success {
38            Ok(())
39        } else {
40            Err(self.messages)
41        }
42    }

45/// Invokes LLD of the given flavor with the specified arguments.
46pub fn link(target: LldFlavor, args: &[String]) -> LldResult {
47    // Prepare arguments
48    let c_args = args
49        .iter()
50        .map(|arg| CString::new(arg.as_bytes()).unwrap())
51        .collect::<Vec<CString>>();
52    let args: Vec<*const c_char> = c_args.iter().map(|arg| arg.as_ptr()).collect();
53
54    // Invoke LLD
55    let mut lld_result = unsafe { mun_lld_link(target, args.len() as c_int, args.as_ptr()) };
56
57    // Get the messages from the invocation
58    let messages = if !lld_result.messages.is_null() {
59        unsafe {
60            CStr::from_ptr(lld_result.messages)
61                .to_string_lossy()
62                .to_string()
63        }
64    } else {
65        String::new()
66    };
67
68    // Construct the result
69    let result = LldResult {
70        success: lld_result.success,
71        messages,
72    };
73
74    // Release the result
75    unsafe { mun_link_free_result(&mut lld_result as *mut LldInvokeResult) };
76    drop(lld_result);
77
78    result
79}
