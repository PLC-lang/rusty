20/// a wrapper for an unresolvable const-expression with the reason
21/// why it could not be resolved
22#[derive(PartialEq, Debug)]
23pub struct UnresolvableConstant {
24    pub id: ConstId,
25    pub kind: Option<UnresolvableKind>,
26    //location
27    //source-file
28}
29
30impl UnresolvableConstant {
31    pub fn new(id: ConstId, reason: &str) -> Self {
32        UnresolvableConstant { id, kind: Some(UnresolvableKind::Misc(reason.into())) }
33    }
34
35    pub fn with_kind(self, kind: UnresolvableKind) -> Self {
36        UnresolvableConstant { id: self.id, kind: Some(kind) }
37    }
38
39    pub fn incomplete_initialzation(id: &ConstId) -> Self {
40        UnresolvableConstant::new(*id, "Incomplete initialization - cannot evaluate const expressions")
41    }
42
43    pub fn no_initial_value(id: &ConstId) -> Self {
44        UnresolvableConstant::new(*id, "No initial value")
45    }
46
47    pub fn get_reason(&self) -> Option<&str> {
48        self.kind.as_ref().map(|it| it.get_reason())
49    }

52/// returns the resolved constants index and a Vec of qualified names of constants that could not be resolved.
53pub fn evaluate_constants(mut index: Index) -> (Index, Vec<UnresolvableConstant>) {
54    let mut unresolvable: Vec<UnresolvableConstant> = Vec::new();
55    let constants = index.get_const_expressions();
56
57    //todo should these be references?
58    let mut remaining_constants: VecDeque<ConstId> = constants.into_iter().map(|(id, _)| id).collect();
59
60    //count how many consecutive resolve-attempts failed
61    let mut failed_tries = 0;
62    //if we need more tries than entries we cannot solve the issue
63    //TODO is can be more efficient
64    // - we can know when retries are smart
65    // - with recursion, we can remove all of a recursion ring
66    while failed_tries < remaining_constants.len() {
67        if let Some(candidate) = remaining_constants.pop_front() {
68            if let (Some(const_expr), target_type) = (
69                index.get_const_expressions().find_const_expression(&candidate),
70                index.get_const_expressions().find_expression_target_type(&candidate),
71            ) {
72                let candidates_type = target_type
73                    .and_then(|type_name| index.find_effective_type_by_name(type_name))
74                    .map(DataType::get_type_information);
75
76                if candidates_type
77                    .map(|it| (it.is_struct() || it.is_array()) && const_expr.is_default())
78                    .unwrap_or(false)
79                {
80                    // we skip structs and arrays with default-initializers since they cannot be used inside expressions of other consts.
81                    // we leave generating the default value to the llvm-index later.
82                    // And we resolve it so we dont get a validation problem
83                    let expr_clone = const_expr.get_statement().clone();
84                    do_resolve_candidate(&mut index, candidate, expr_clone);
85                    continue;
86                }
87
88                let initial_value_literal = evaluate_with_target_hint(
89                    const_expr.get_statement(),
90                    const_expr.get_qualifier(),
91                    &index,
92                    target_type,
93                    const_expr.get_lhs(),
94                );
95
96                match (initial_value_literal, &candidates_type) {
97                    //we found an Int-Value and we found the const's datatype to be an unsigned Integer type (e.g. WORD)
98                    (
99                        Ok(Some(AstNode {
100                            stmt: AstStatement::Literal(AstLiteral::Integer(i)),
101                            id,
102                            location,
103                            ..
104                        })),
105                        Some(DataTypeInformation::Integer { size, signed: false, .. }),
106                    ) => {
107                        // since we store literal-ints as i128 we need to truncate all of them down to their
108                        // original size to avoid negative numbers
109                        let mask = 2_i128.pow(*size) - 1; // bitmask for this type's size
110                        let masked_value = i & mask; //delete all bits > size of data_type
111
112                        index
113                            .get_mut_const_expressions()
114                            .mark_resolved(
115                                &candidate,
116                                AstNode::new_literal(AstLiteral::new_integer(masked_value), id, location),
117                            )
118                            .expect("unknown id for const-expression"); //panic if we dont know the id
119                        failed_tries = 0;
120                    }
121
122                    // we were able to evaluate a valid statement
123                    (Ok(Some(literal)), _) => {
124                        let literal = cast_if_necessary(
125                            literal,
126                            &index.get_const_expressions().find_expression_target_type(&candidate),
127                            &index,
128                        );
129
130                        do_resolve_candidate(&mut index, candidate, literal);
131                        failed_tries = 0;
132                    }
133
134                    // we could not evaluate a valid statement - maybe later?
135                    (Ok(None), _) => {
136                        failed_tries += 1;
137                        remaining_constants.push_back(candidate) //try again later
138                    }
139
140                    // there was an error during evaluation
141                    (Err(kind), _) => {
142                        //error during resolving
143                        unresolvable.push(
144                            UnresolvableConstant::new(candidate, kind.get_reason()).with_kind(kind.clone()),
145                        );
146                        index
147                            .get_mut_const_expressions()
148                            .mark_unresolvable(&candidate, kind)
149                            .expect("unknown id for const-expression"); //panic if we dont know the id
150                    }
151                }
152            } else {
153                //no initial value in a const ... well
154                unresolvable.push(UnresolvableConstant::no_initial_value(&candidate));
155            }
156        }
157    }
158
159    //import all constants that were note resolved in the loop above
160    unresolvable.extend(remaining_constants.iter().map(UnresolvableConstant::incomplete_initialzation));
161
162    (index, unresolvable)
163}

165fn do_resolve_candidate(index: &mut Index, candidate: ConstId, new_statement: AstNode) {
166    index
167        .get_mut_const_expressions()
168        .mark_resolved(&candidate, new_statement)
169        .expect("unknown id for const-expression");
170}

172/// returns true, if the given expression needs to be evaluated.
173/// literals must not be further evaluated and can be known at
174/// compile time
175fn needs_evaluation(expr: &AstNode) -> bool {
176    match expr.get_stmt() {
177        AstStatement::Literal(kind) => match &kind {
178            &AstLiteral::Array(Array { elements: Some(elements), .. }) => match &elements.get_stmt() {
179                AstStatement::ExpressionList(expressions) => expressions.iter().any(needs_evaluation),
180                _ => needs_evaluation(elements.as_ref()),
181            },
182
183            // We want to check if literals will overflow, hence they'll need to be evaluated
184            AstLiteral::Integer(_) | AstLiteral::Real(_) => true,
185
186            _ => false,
187        },
188        AstStatement::Assignment(data) => needs_evaluation(data.right.as_ref()),
189        AstStatement::ExpressionList(expressions) => expressions.iter().any(needs_evaluation),
190        AstStatement::RangeStatement(data) => needs_evaluation(&data.start) || needs_evaluation(&data.end),
191        _ => true,
192    }
193}

195/// generates an ast-statement that initializes the given type with the registered default values
196fn get_default_initializer(
197    id: AstId,
198    target_type: &str,
199    index: &Index,
200    location: &SourceLocation,
201) -> Result<Option<AstNode>, UnresolvableKind> {
202    if let Some(init) = index.get_initial_value_for_type(target_type) {
203        evaluate(init, None, index, None) //TODO do we ave a scope here?
204    } else {
205        let dt = index.get_type_information_or_void(target_type);
206        let init = match dt {
207            DataTypeInformation::Pointer { .. } => {
208                Some(AstFactory::create_literal(AstLiteral::Null, location.clone(), id))
209            }
210            DataTypeInformation::Integer { .. } => {
211                Some(AstFactory::create_literal(AstLiteral::new_integer(0), location.clone(), id))
212            }
213            DataTypeInformation::Enum { name, variants, .. } => variants
214                .first()
215                .and_then(|default_enum| index.find_enum_variant(name, default_enum.get_name()))
216                .and_then(|enum_element| enum_element.initial_value)
217                .and_then(|initial_val| {
218                    index.get_const_expressions().get_resolved_constant_statement(&initial_val)
219                })
220                .cloned(),
221            DataTypeInformation::Float { .. } => Some(AstFactory::create_literal(
222                AstLiteral::new_real("0.0".to_string()),
223                location.clone(),
224                id,
225            )),
226            DataTypeInformation::String { encoding, .. } => Some(AstFactory::create_literal(
227                AstLiteral::new_string("".to_string(), encoding == &StringEncoding::Utf16),
228                location.clone(),
229                id,
230            )),
231            DataTypeInformation::SubRange { referenced_type, .. }
232            | DataTypeInformation::Alias { referenced_type, .. } => {
233                return get_default_initializer(id, referenced_type, index, location)
234            }
235            _ => None,
236        };
237        Ok(init)
238    }
239}

241/// transforms the given literal to better fit the datatype of the candidate
242/// effectively this casts an IntLiteral to a RealLiteral if necessary
243fn cast_if_necessary(statement: AstNode, target_type_name: &Option<&str>, index: &Index) -> AstNode {
244    let Some(dti) = target_type_name.and_then(|it| index.find_effective_type_info(it)) else {
245        return statement;
246    };
247
248    if let AstStatement::Literal(literal) = statement.get_stmt() {
249        let (id, location) = (statement.get_id(), statement.get_location());
250        match literal {
251            AstLiteral::Integer(value) if dti.is_float() => {
252                return AstNode::new_real(value.to_string(), id, location)
253            }
254
255            AstLiteral::String(StringValue { value, is_wide: true }) if dti.is_string_utf8() => {
256                return AstNode::new_string(value, false, id, location)
257            }
258
259            AstLiteral::String(StringValue { value, is_wide: false }) if dti.is_string_utf16() => {
260                return AstNode::new_string(value, true, id, location)
261            }
262
263            _ => (),
264        }
265    };
266
267    statement
268}

270/// Checks if a literal integer or float overflows based on its value, and if so returns true.
271fn does_overflow(literal: &AstNode, dti: Option<&DataTypeInformation>) -> bool {
272    let Some(dti) = dti else { return false };
273    let AstStatement::Literal(kind) = literal.get_stmt() else { return false };
274
275    if !matches!(kind, AstLiteral::Integer(_) | AstLiteral::Real(_)) {
276        return false;
277    };
278
279    match &dti {
280        DataTypeInformation::Integer { signed, size, semantic_size, .. } => {
281            let size = semantic_size.as_ref().unwrap_or(size).to_owned();
282            match (kind, signed, size) {
283                // Signed
284                (AstLiteral::Integer(value), true, 8) => i8::try_from(*value).is_err(),
285                (AstLiteral::Integer(value), true, 16) => i16::try_from(*value).is_err(),
286                (AstLiteral::Integer(value), true, 32) => i32::try_from(*value).is_err(),
287                (AstLiteral::Integer(value), true, 64) => i64::try_from(*value).is_err(),
288
289                // Unsigned
290                (AstLiteral::Integer(value), false, 8) => u8::try_from(*value).is_err(),
291                (AstLiteral::Integer(value), false, 16) => u16::try_from(*value).is_err(),
292                (AstLiteral::Integer(value), false, 32) => u32::try_from(*value).is_err(),
293                (AstLiteral::Integer(value), false, 64) => u64::try_from(*value).is_err(),
294
295                // We are dealing with e.g. booleans if we have a semantic size != 8, 16, 32 or 64
296                (AstLiteral::Integer(value), _, _) => {
297                    let min = if *signed { -(2_i128.pow(size - 1)) } else { 0 };
298                    let max = 2_i128.pow(size) - 1;
299
300                    *value < min || *value > max
301                }
302
303                // XXX: Returning an error might be a better idea, as we'll be catching edge-cases we missed?
304                _ => false,
305            }
306        }
307
308        // Only interested in integers. Floats can be infinite or nan which is _technically_ not
309        // an overflow.
310        _ => false,
311    }

323/// evaluates the given Syntax-Tree `initial` to a `LiteralValue` if possible
324/// ## Arguments
325/// - `initial` the constant expression to resolve
326/// - `scope` an optional qualifier to be used when resolving references
327/// - `index` the global symbol-table
328/// ## Returns
329/// - returns an Err if resolving caused an internal error (e.g. number parsing)
330/// - returns None if the initializer cannot be resolved  (e.g. missing value)
331fn evaluate_with_target_hint(
332    initial: &AstNode,
333    scope: Option<&str>,
334    index: &Index,
335    target_type: Option<&str>,
336    lhs: Option<&str>,
337) -> Result<Option<AstNode>, UnresolvableKind> {
338    if !needs_evaluation(initial) {
339        return Ok(Some(initial.clone())); // TODO hmm ...
340    }
341    let (id, location) = (initial.get_id(), initial.get_location());
342    let literal = match initial.get_stmt() {
343        AstStatement::Literal(kind) => match kind {
344            AstLiteral::Array(Array { elements: Some(elements) }) => {
345                let tt = target_type
346                    .and_then(|it| index.find_effective_type_info(it))
347                    .and_then(|it| it.get_inner_array_type_name())
348                    .or(target_type);
349
350                let inner_elements = AstNode::get_as_list(elements)
351                    .iter()
352                    .map(|e| evaluate_with_target_hint(e, scope, index, tt, lhs))
353                    .collect::<Result<Vec<Option<AstNode>>, UnresolvableKind>>()?
354                    .into_iter()
355                    .collect::<Option<Vec<AstNode>>>();
356
357                inner_elements.map(|ie| {
358                    AstFactory::create_literal(
359                        AstLiteral::new_array(Some(Box::new(AstFactory::create_expression_list(
360                            ie,
361                            location.clone(),
362                            id,
363                        )))),
364                        location.clone(),
365                        id,
366                    )
367                })
368            }
369
370            AstLiteral::Integer(_) | AstLiteral::Real(_) => {
371                let dti = target_type.and_then(|it| index.find_effective_type_info(it));
372                if does_overflow(initial, dti) {
373                    return Err(UnresolvableKind::Overflow(
374                        format!("This will overflow for type {}", dti.unwrap().get_name()),
375                        initial.get_location(),
376                    ));
377                }
378
379                return Ok(Some(initial.clone()));
380            }
381
382            _ => return Ok(Some(initial.clone())),
383        },
384
385        AstStatement::DefaultValue(_) => {
386            return get_default_initializer(
387                initial.get_id(),
388                target_type.unwrap_or(VOID_TYPE),
389                index,
390                &location,
391            )
392        }
393        AstStatement::ReferenceExpr(ReferenceExpr {
394            access: ReferenceAccess::Cast(target),
395            base: Some(type_name),
396        }) => {
397            let dti = type_name
398                .get_flat_reference_name()
399                .and_then(|type_name| index.find_effective_type_info(type_name));
400            match dti {
401                Some(DataTypeInformation::Enum { name: enum_name, .. }) => {
402                    if let AstStatement::Identifier(ref_name) = target.get_stmt() {
403                        return index
404                            .find_enum_variant(enum_name, ref_name)
405                            .ok_or_else(|| {
406                                UnresolvableKind::Misc(format!(
407                                    "Cannot resolve constant enum {enum_name}#{ref_name}."
408                                ))
409                            })
410                            .and_then(|v| {
411                                resolve_const_reference(v, ref_name, index, target_type, scope, lhs)
412                            });
413                    } else {
414                        return Err(UnresolvableKind::Misc("Cannot resolve unknown constant.".to_string()));
415                    }
416                }
417                Some(dti) => {
418                    evaluate_with_target_hint(target, scope, index, Some(dti.get_name()), lhs)?;
419                    Some(get_cast_statement_literal(target, dti.get_name(), scope, index, lhs)?)
420                }
421                None => return Err(UnresolvableKind::Misc("Cannot resolve unknown Type-Cast.".to_string())),
422            }
423        }
424        AstStatement::ReferenceExpr(ReferenceExpr { access: ReferenceAccess::Member(reference), base }) => {
425            if let Some(name) = reference.get_flat_reference_name() {
426                index
427                    .find_variable(
428                        base.as_ref().and_then(|it| it.get_flat_reference_name()).or(scope),
429                        std::slice::from_ref(&name),
430                    )
431                    .map(|variable| resolve_const_reference(variable, name, index, target_type, scope, lhs))
432                    .transpose()?
433                    .flatten()
434            } else {
435                None
436            }
437        }
438        AstStatement::BinaryExpression(BinaryExpression { left, right, operator }) => {
439            let eval_left = evaluate(left, scope, index, lhs)?;
440            let eval_right = evaluate(right, scope, index, lhs)?;
441
442            if let Some((left, right)) = eval_left.zip(eval_right).as_ref() {
443                let evalualted = match operator {
444                    Operator::Plus => arithmetic_expression!(left, +, right, "+", id)?,
445                    Operator::Minus => arithmetic_expression!(left, -, right, "-", id)?,
446                    Operator::Multiplication => arithmetic_expression!(left, *, right, "*", id)?,
447                    Operator::Division if !(right.is_real() || left.is_real()) && right.is_zero() => {
448                        return Err(UnresolvableKind::Misc("Attempt to divide by zero".to_string()))
449                    }
450                    Operator::Division => arithmetic_expression!(left, /, right, "/", id)?,
451                    Operator::Modulo if right.is_zero() => {
452                        return Err(UnresolvableKind::Misc(
453                            "Attempt to calculate the remainder with a divisor of zero".to_string(),
454                        ))
455                    }
456                    Operator::Modulo => arithmetic_expression!(left, %, right, "MOD", id)?,
457                    Operator::Equal => compare_expression!(left, ==, right, "=", id)?,
458                    Operator::NotEqual => compare_expression!(left, !=, right, "<>", id)?,
459                    Operator::Greater => compare_expression!(left, >, right, ">", id)?,
460                    Operator::GreaterOrEqual => compare_expression!(left, >=, right, ">=", id)?,
461                    Operator::Less => compare_expression!(left, <, right, "<", id)?,
462                    Operator::LessOrEqual => compare_expression!(left, <=, right, "<=", id)?,
463                    Operator::And => bitwise_expression!(left, & , right, "AND", id)?,
464                    Operator::Or => bitwise_expression!(left, | , right, "OR", id)?,
465                    Operator::Xor => bitwise_expression!(left, ^, right, "XOR", id)?,
466                    _ => {
467                        return Err(UnresolvableKind::Misc(format!(
468                            "Cannot resolve operator {operator:?} in constant evaluation"
469                        )))
470                    }
471                };
472
473                // We have to re-evaluate to detect overflows
474                evaluate_with_target_hint(&evalualted, scope, index, target_type, lhs)?
475            } else {
476                None //not all operators can be resolved
477            }
478        }
479
480        // NOT x
481        AstStatement::UnaryExpression(UnaryExpression { operator: Operator::Not, value }) => {
482            let eval = evaluate(value, scope, index, lhs)?;
483
484            match eval.as_ref() {
485                Some(AstNode { stmt: AstStatement::Literal(AstLiteral::Bool(v)), id, location, .. }) => {
486                    Some(AstFactory::create_literal(AstLiteral::Bool(!v), location.clone(), *id))
487                }
488                Some(AstNode {
489                    stmt: AstStatement::Literal(AstLiteral::Integer(v)), id, location, ..
490                }) => {
491                    evaluate_with_target_hint(eval.as_ref().unwrap(), scope, index, target_type, lhs)?;
492                    Some(AstFactory::create_literal(AstLiteral::Integer(!v), location.clone(), *id))
493                }
494                None => {
495                    None //not yet resolvable
496                }
497                _ => return Err(UnresolvableKind::Misc(format!("Cannot resolve constant Not {value:?}"))),
498            }
499        }
500        // - x
501        AstStatement::UnaryExpression(UnaryExpression { operator: Operator::Minus, value }) => {
502            match evaluate(value, scope, index, lhs)? {
503                Some(AstNode {
504                    stmt: AstStatement::Literal(AstLiteral::Integer(v)), id, location, ..
505                }) => Some(AstNode::new(AstStatement::Literal(AstLiteral::Integer(-v)), id, location)),
506                Some(AstNode { stmt: AstStatement::Literal(AstLiteral::Real(v)), id, location, .. }) => {
507                    let lit = AstNode::new(
508                        AstStatement::Literal(AstLiteral::new_real(format!(
509                            "{:}",
510                            -(v.parse::<f64>())
511                                .map_err(|err| UnresolvableKind::Misc(format!("{err:}: {v:}")))?
512                        ))),
513                        id,
514                        location,
515                    );
516                    evaluate_with_target_hint(&lit, scope, index, target_type, lhs)?
517                }
518                None => {
519                    None //not yet resolvable
520                }
521                _ => return Err(UnresolvableKind::Misc(format!("Cannot resolve constant Minus {value:?}"))),
522            }
523        }
524        AstStatement::UnaryExpression(UnaryExpression { operator: Operator::Plus, value }) => {
525            evaluate(value, scope, index, lhs)?
526        }
527        AstStatement::ExpressionList(expressions) => {
528            let inner_elements = expressions
529                .iter()
530                .map(|e| evaluate(e, scope, index, lhs))
531                .collect::<Result<Vec<Option<AstNode>>, UnresolvableKind>>()?
532                .into_iter()
533                .collect::<Option<Vec<AstNode>>>();
534
535            //return a new array, or return none if one was not resolvable
536            inner_elements.map(|ie| AstNode::new(AstStatement::ExpressionList(ie), id, location))
537        }
538        AstStatement::MultipliedStatement(MultipliedStatement { element, multiplier }) => {
539            let inner_elements = AstNode::get_as_list(element.as_ref())
540                .iter()
541                .map(|e| evaluate(e, scope, index, lhs))
542                .collect::<Result<Vec<Option<AstNode>>, UnresolvableKind>>()?
543                .into_iter()
544                .collect::<Option<Vec<AstNode>>>();
545
546            //return a new array, or return none if one was not resolvable
547            inner_elements.map(|ie| {
548                if let [ie] = ie.as_slice() {
549                    AstFactory::create_multiplied_statement(*multiplier, ie.clone(), location.clone(), id)
550                } else {
551                    AstFactory::create_multiplied_statement(
552                        *multiplier,
553                        AstFactory::create_expression_list(ie, location.clone(), id),
554                        location.clone(),
555                        id,
556                    )
557                }
558            })
559        }
560        AstStatement::Assignment(data) => {
561            //Right needs evaluation
562            match evaluate(&data.right, scope, index, lhs) {
563                Ok(Some(value)) => Ok(Some(AstFactory::create_assignment(*data.left.clone(), value, id))),
564                Ok(None) => Ok(Some(initial.clone())),
565                Err(UnresolvableKind::Address(mut init)) => {
566                    init.initializer.replace(initial.clone());
567                    Err(UnresolvableKind::Address(init))
568                }
569                Err(why) => Err(why),
570            }?
571        }
572        AstStatement::RangeStatement(data) => {
573            let start = evaluate(&data.start, scope, index, lhs)?.unwrap_or_else(|| *data.start.to_owned());
574            let end = evaluate(&data.end, scope, index, lhs)?.unwrap_or_else(|| *data.end.to_owned());
575
576            Some(AstFactory::create_range_statement(start, end, id))
577        }
578        AstStatement::ParenExpression(expr) => {
579            match evaluate_with_target_hint(expr, scope, index, target_type, lhs) {
580                Ok(init) => Ok(init),
581                Err(UnresolvableKind::Address(mut init)) => {
582                    init.lhs = lhs.map(str::to_string);
583                    init.initializer.replace(initial.clone());
584                    Err(UnresolvableKind::Address(init))
585                }
586                Err(why) => Err(why),
587            }?
588        }
589        AstStatement::CallStatement(plc_ast::ast::CallStatement { operator, .. }) => {
590            if let Some(pou) = operator.as_ref().get_flat_reference_name().and_then(|it| index.find_pou(it)) {
591                if !(pou.is_constant() && index.get_builtin_function(pou.get_name()).is_some()) {
592                    return Err(UnresolvableKind::Misc(format!(
593                        "Call-statement '{}' in initializer is not constant.",
594                        pou.get_name()
595                    )));
596                }
597            } else {
598                // POU not found
599                return Err(UnresolvableKind::Misc(format!("Cannot resolve constant: {:#?}", initial)));
600            };
601
602            return Err(UnresolvableKind::Address(InitData::new(Some(initial), target_type, scope, lhs)));
603        }
604        _ => return Err(UnresolvableKind::Misc(format!("Cannot resolve constant: {initial:#?}"))),
605    };
606    Ok(literal)
607}

609/// attempts to resolve the inital value of this reference's target
610/// may return Ok(None) if the variable's initial value can not be
611/// resolved yet
612fn resolve_const_reference(
613    variable: &crate::index::VariableIndexEntry,
614    name: &str,
615    index: &Index,
616    target_type: Option<&str>,
617    scope: Option<&str>,
618    lhs: Option<&str>,
619) -> Result<Option<AstNode>, UnresolvableKind> {
620    if !variable.is_constant() {
621        if !target_type
622            .is_some_and(|it| index.find_effective_type_by_name(it).is_some_and(|it| it.is_pointer()))
623        {
624            return Err(UnresolvableKind::Misc(format!("`{name}` is no const reference")));
625        } else {
626            return Err(UnresolvableKind::Address(InitData::new(None, target_type, scope, lhs)));
627        }
628    }
629
630    if let Some(ConstExpression::Resolved(statement)) =
631        variable.initial_value.as_ref().and_then(|it| index.get_const_expressions().find_const_expression(it))
632    {
633        Ok(Some(statement.clone()))
634    } else {
635        Ok(None) //not resolved yet
636    }
637}

639/// Transforms the given `cast_statement` into a literal. For example `WORD#FFFF` will be a
640/// [`AstLiteral::Integer`] with value `65_535` whereas `INT#FFFF` will not evaluate because it overflows
641/// (see also [`does_overflow`] and [`evaluate_with_target_hint`]).
642fn get_cast_statement_literal(
643    cast_statement: &AstNode,
644    type_name: &str,
645    scope: Option<&str>,
646    index: &Index,
647    lhs: Option<&str>,
648) -> Result<AstNode, UnresolvableKind> {
649    let dti = index.find_effective_type_info(type_name);
650    match dti {
651        Some(&DataTypeInformation::Integer { .. }) => {
652            let evaluated_initial =
653                evaluate_with_target_hint(cast_statement, scope, index, Some(type_name), lhs)?
654                    .as_ref()
655                    .map(|v| {
656                        if let AstStatement::Literal(AstLiteral::Integer(value)) = v.get_stmt() {
657                            Ok(*value)
658                        } else {
659                            Err(UnresolvableKind::Misc(format!("Expected integer value, found {v:?}")))
660                        }
661                    })
662                    .transpose()?;
663
664            if let Some(value) = evaluated_initial {
665                return Ok(AstNode::new(
666                    AstStatement::Literal(AstLiteral::new_integer(value)),
667                    cast_statement.get_id(),
668                    cast_statement.get_location(),
669                ));
670            }
671
672            Err(UnresolvableKind::Misc(format!("Cannot resolve constant: {type_name}#{cast_statement:?}")))
673        }
674
675        Some(DataTypeInformation::Float { .. }) => {
676            let evaluated = evaluate(cast_statement, scope, index, lhs)?;
677            let value = match evaluated.as_ref().map(|it| it.get_stmt()) {
678                Some(AstStatement::Literal(AstLiteral::Integer(value))) => Some(*value as f64),
679                Some(AstStatement::Literal(AstLiteral::Real(value))) => value.parse::<f64>().ok(),
680                _ => {
681                    return Err(UnresolvableKind::Misc(format!(
682                        "Expected floating point type, got: {evaluated:?}"
683                    )))
684                }
685            };
686
687            let Some(value) = value else {
688                return Err(UnresolvableKind::Misc(format!(
689                    "cannot resolve constant: {type_name}#{cast_statement:?}"
690                )));
691            };
692
693            Ok(AstNode::new(
694                AstStatement::Literal(AstLiteral::new_real(value.to_string())),
695                cast_statement.get_id(),
696                cast_statement.get_location(),
697            ))
698        }
699
700        _ => Err(UnresolvableKind::Misc(format!("Cannot resolve constant: {type_name}#{cast_statement:?}"))),
701    }
702}

704macro_rules! cannot_eval_error {
705    ($left:expr, $op_text:expr, $right:expr) => {
706        Err(UnresolvableKind::Misc(format!("Cannot evaluate {:?} {:} {:?}", $left, $op_text, $right)))
707    };

711macro_rules! arithmetic_expression {
712    ($left:expr, $op:tt, $right:expr, $op_text:expr, $resulting_id:expr) => {
713        {
714        let loc_left = $left.get_location();
715        let loc_right = $right.get_location();
716        match ($left.get_stmt(), $right.get_stmt()) {
717            (   AstStatement::Literal(AstLiteral::Integer(lvalue)),
718                AstStatement::Literal(AstLiteral::Integer(rvalue))) => {
719                    Ok(AstStatement::Literal(AstLiteral::new_integer(lvalue $op rvalue)))
720            },
721
722            (   AstStatement::Literal(AstLiteral::Integer(lvalue)),
723                AstStatement::Literal(AstLiteral::Real(rvalue))) => {
724                    let rvalue = rvalue.parse::<f64>()
725                        .map_err(|err| UnresolvableKind::Misc(err.to_string()))?;
726                    Ok(AstStatement::Literal(
727                        AstLiteral::new_real((*lvalue as f64 $op rvalue).to_string())))
728            },
729
730            (   AstStatement::Literal(AstLiteral::Real(lvalue)),
731                AstStatement::Literal(AstLiteral::Integer(rvalue))) => {
732                    let lvalue = lvalue.parse::<f64>()
733                        .map_err(|err| UnresolvableKind::Misc(err.to_string()))?;
734                    Ok(AstStatement::Literal(AstLiteral::new_real((lvalue $op *rvalue as f64).to_string())))
735            },
736
737            (   AstStatement::Literal(AstLiteral::Real(lvalue)),
738                AstStatement::Literal(AstLiteral::Real(rvalue))) => {
739                    let lvalue = lvalue.parse::<f64>()
740                        .map_err(|err| UnresolvableKind::Misc(err.to_string()))?;
741                    let rvalue = rvalue.parse::<f64>()
742                        .map_err(|err| UnresolvableKind::Misc(err.to_string()))?;
743                    Ok(AstStatement::Literal(
744                        AstLiteral::new_real((lvalue $op rvalue).to_string()),
745                    ))
746            },
747            _ => cannot_eval_error!($left, $op_text, $right),
748        }.map(|it| AstNode::new(it, $resulting_id, loc_left.span(&loc_right)))
749        }
750    }
751}

774macro_rules! compare_expression {
775    ($left:expr, $op:tt, $right:expr, $op_text:expr, $resulting_id:expr) => {{
776        let loc_left = $left.get_location();
777        let loc_right = $right.get_location();
778        match ($left.get_stmt(), $right.get_stmt()) {
779            (   AstStatement::Literal(AstLiteral::Integer(lvalue)),
780                AstStatement::Literal(AstLiteral::Integer(rvalue))) => {
781                Ok(AstStatement::Literal(
782                    AstLiteral::new_bool(lvalue $op rvalue)))
783            },
784            (   AstStatement::Literal(AstLiteral::Real(..)),

787            },
788            (   AstStatement::Literal(AstLiteral::Bool(lvalue)),
789                AstStatement::Literal(AstLiteral::Bool(rvalue))) => {
790                    Ok(AstStatement::Literal(AstLiteral::new_bool(lvalue $op rvalue)))
791            },
792            _ => cannot_eval_error!($left, $op_text, $right),
793        }.map(|it| AstNode::new(it, $resulting_id, loc_left.span(&loc_right)))
794    }}
795}
