312impl HardwareBinding {
313    pub fn from_statement(index: &mut Index, it: &AstNode, scope: Option<String>) -> Option<Self> {
314        if let AstStatement::HardwareAccess(data) = it.get_stmt() {
315            Some(HardwareBinding {
316                access: data.access,
317                direction: data.direction,
318                entries: data
319                    .address
320                    .iter()
321                    .map(|expr| {
322                        index.constant_expressions.add_constant_expression(
323                            expr.clone(),
324                            typesystem::DINT_SIZE.to_string(),
325                            scope.clone(),
326                            None,
327                        )
328                    })
329                    .collect(),
330                location: it.get_location(),
331            })
332        } else {
333            None
334        }
335    }

526pub struct InterfaceIndexEntry {
527    /// The interface identifier, consisting of its name and name-location
528    pub ident: Identifier,
529
530    /// The location of the interface as a whole
531    pub location: SourceLocation,
532
533    /// A list of qualified names of the methods in this interface; the actual methods are located in
534    /// [`Index::pous`]
535    pub methods: Vec<String>,
536
537    /// A list of other interfaces this interface extends
538    pub extensions: Vec<Identifier>,
539
540    /// A list of the properties declared in this interface
541    pub properties: Vec<PropertyBlock>,
542}
543
544impl InterfaceIndexEntry {
545    pub fn get_name(&self) -> &str {
546        self.ident.name.as_str()
547    }
548
549    pub fn get_name_location(&self) -> &SourceLocation {
550        &self.ident.location
551    }
552
553    pub fn get_location(&self) -> &SourceLocation {
554        &self.location
555    }
556
557    /// Returns a list of methods this interface declared
558    pub fn get_declared_methods<'idx>(&self, index: &'idx Index) -> Vec<&'idx PouIndexEntry> {
559        self.methods
560            .iter()
561            .map(|name| index.find_pou(name).expect("must exist because of present InterfaceIndexEntry"))
562            .collect()
563    }
564
565    /// Returns a list of methods this interface inherited
566    pub fn get_derived_methods<'idx>(&'idx self, index: &'idx Index) -> Vec<&'idx PouIndexEntry> {
567        self.get_derived_methods_recursive(index, &mut FxHashSet::default())
568    }
569
570    /// Returns a list of methods defined in this interface, including inherited methods from derived interfaces
571    pub fn get_methods<'idx>(&'idx self, index: &'idx Index) -> Vec<&'idx PouIndexEntry> {
572        self.get_methods_recursive(index, &mut FxHashSet::default())
573    }
574
575    /// Returns a list of interfaces this interface implements
576    pub fn get_extensions(&self) -> Vec<&Identifier> {
577        self.extensions.iter().collect()
578    }
579
580    /// Returns a list of interfaces this interface inherited directly
581    pub fn get_derived_interfaces<'idx>(
582        &self,
583        index: &'idx Index,
584    ) -> Vec<Result<&'idx InterfaceIndexEntry, Identifier>> {
585        self.extensions
586            .iter()
587            .flat_map(|id| index.find_interface(&id.name).map(Result::Ok).or(Some(Err(id.to_owned()))))
588            .collect()
589    }
590
591    /// Returns a list of ALL existing interfaces this interface inherited directly or indirectly
592    pub fn get_derived_interfaces_recursive<'i>(&self, index: &'i Index) -> Vec<&'i InterfaceIndexEntry> {
593        let mut seen: FxHashSet<&Identifier> = FxHashSet::default();
594        let mut queue: VecDeque<&InterfaceIndexEntry> = VecDeque::new();
595
596        queue.push_back(self);
597        while let Some(interface) = queue.pop_front() {
598            if seen.insert(&interface.ident) {
599                queue.extend(interface.get_derived_interfaces(index).into_iter().flatten());
600            }
601        }
602
603        seen.into_iter().filter_map(|ident| index.find_interface(&ident.name)).collect()
604    }
605
606    fn get_methods_recursive<'idx>(
607        &'idx self,
608        index: &'idx Index,
609        seen: &mut FxHashSet<&'idx str>,
610    ) -> Vec<&'idx PouIndexEntry> {
611        seen.insert(self.get_name());
612        self.get_declared_methods(index)
613            .into_iter()
614            .chain(self.get_derived_methods_recursive(index, seen))
615            .collect_vec()
616    }
617
618    fn get_derived_methods_recursive<'idx>(
619        &'idx self,
620        index: &'idx Index,
621        seen: &mut FxHashSet<&'idx str>,
622    ) -> Vec<&'idx PouIndexEntry> {
623        self.get_derived_interfaces(index)
624            .iter()
625            .filter_map(|it| it.as_ref().ok())
626            .flat_map(|it| {
627                if !seen.contains(it.get_name()) {
628                    it.get_methods_recursive(index, seen)
629                } else {
630                    vec![]
631                }
632            })
633            .collect()
634    }
635}

1204    /// Retrieves the "Effective" type behind this datatype
1205    /// An effective type will be any end type i.e. Structs, Integers, Floats, String and Array
1206    pub fn find_effective_type<'ret>(&'ret self, data_type: &'ret DataType) -> Option<&'ret DataType> {
1207        self.find_effective_type_recursive(data_type, &mut FxHashSet::default())
1208    }
1209
1210    fn find_effective_type_recursive<'ret>(
1211        &'ret self,
1212        data_type: &'ret DataType,
1213        seen: &mut FxHashSet<&'ret str>,
1214    ) -> Option<&'ret DataType> {
1215        match data_type.get_type_information() {
1216            DataTypeInformation::Alias { referenced_type, .. } => {
1217                // Check for cycles in type aliases
1218                if !seen.insert(data_type.get_name()) {
1219                    // Cycle detected, return None to indicate error
1220                    return None;
1221                }
1222
1223                let result = self
1224                    .find_type(referenced_type)
1225                    .and_then(|it| self.find_effective_type_recursive(it, seen));
1226
1227                // Remove from seen set when backtracking
1228                seen.remove(data_type.get_name());
1229                result
1230            }
1231            _ => Some(data_type),
1232        }
1233    }

1279impl Index {
1280    /// imports all entries from the given index into the current index
1281    ///
1282    /// imports all global_variables, types and implementations
1283    /// # Arguments
1284    /// - `other` the other index. The elements are drained from the given index and moved into the current one
1285    pub fn import(&mut self, mut other: Index) {
1286        //global variables
1287        for (name, e) in other.global_variables.drain(..) {
1288            let entries = e
1289                .into_iter()
1290                .map(|it| self.transfer_constants(it, &mut other.constant_expressions))
1291                .collect::<Vec<_>>();
1292            self.global_variables.insert_many(name, entries);
1293        }
1294
1295        //initializers
1296        for (name, elements) in other.global_initializers.drain(..) {
1297            let elements = elements
1298                .into_iter()
1299                .map(|e| self.transfer_constants(e, &mut other.constant_expressions))
1300                .collect::<Vec<_>>();
1301            self.global_initializers.insert_many(name, elements);
1302        }
1303
1304        //types
1305        for (name, mut elements) in other.type_index.types.drain(..) {
1306            for mut e in elements.drain(..) {
1307                //avoid re-importing internally auto-generated types that we already know
1308                if !e.is_internal() || self.type_index.find_effective_type_by_name(name.as_str()).is_none() {
1309                    e.initial_value =
1310                        self.maybe_import_const_expr(&mut other.constant_expressions, &e.initial_value);
1311
1312                    match &mut e.information {
1313                        //import constant expressions in array-type-definitions
1314                        DataTypeInformation::Array { dimensions, .. } => {
1315                            for d in dimensions.iter_mut() {
1316                                d.start_offset =
1317                                    self.import_type_size(&mut other.constant_expressions, &d.start_offset);
1318                                d.end_offset =
1319                                    self.import_type_size(&mut other.constant_expressions, &d.end_offset);
1320                            }
1321                        }
1322                        // import constant expressions in String-size defintions
1323                        DataTypeInformation::String { size, .. } => {
1324                            *size = self.import_type_size(&mut other.constant_expressions, size);
1325                        }
1326                        DataTypeInformation::Struct { members, .. } => {
1327                            let mut variables = members
1328                                .drain(..)
1329                                .map(|variable| {
1330                                    self.transfer_constants(variable, &mut other.constant_expressions)
1331                                })
1332                                .collect::<Vec<_>>();
1333                            members.append(&mut variables);
1334                        }
1335                        DataTypeInformation::Enum { variants, .. } => {
1336                            let mut variables = variants
1337                                .drain(..)
1338                                .map(|variable| {
1339                                    self.transfer_constants(variable, &mut other.constant_expressions)
1340                                })
1341                                .collect::<Vec<_>>();
1342
1343                            for e in variables.iter() {
1344                                self.enum_global_variables.insert(e.get_name().to_lowercase(), e.clone());
1345                            }
1346                            variants.append(&mut variables);
1347                        }
1348                        // import constant expressions in SubRange definitions
1349                        DataTypeInformation::SubRange { sub_range, .. } => {
1350                            sub_range.start =
1351                                self.import_type_size(&mut other.constant_expressions, &sub_range.start);
1352                            sub_range.end =
1353                                self.import_type_size(&mut other.constant_expressions, &sub_range.end);
1354                        }
1355                        _ => {}
1356                    }
1357
1358                    self.type_index.types.insert(name.clone(), e)
1359                }
1360            }
1361        }
1362
1363        //pou_types
1364        for (name, mut elements) in other.type_index.pou_types.drain(..) {
1365            elements.iter_mut().for_each(|e| {
1366                self.maybe_import_const_expr(&mut other.constant_expressions, &e.initial_value);
1367
1368                if let DataTypeInformation::Struct { members, .. } = &mut e.information {
1369                    let mut variables = members
1370                        .drain(..)
1371                        .map(|variable| self.transfer_constants(variable, &mut other.constant_expressions))
1372                        .collect::<Vec<_>>();
1373                    members.append(&mut variables);
1374                }
1375            });
1376            self.type_index.pou_types.insert_many(name, elements)
1377        }
1378
1379        //implementations
1380        self.implementations.extend(other.implementations);
1381
1382        // interfaces
1383        self.interfaces.extend(other.interfaces);
1384
1385        // properties
1386        self.properties.extend(other.properties);
1387
1388        //pous

1740        for &name in chain.iter().rev() {

1742                self.get_pou_members(name).iter().filter(|var| var.is_parameter() && !var.is_variadic()),

2086    /// Fixes up enum types to set their default initial values.
2087    /// This must be called after constant resolution, as it needs to evaluate
2088    /// constant expressions to determine which variant is zero.
2089    ///
2090    /// For each enum without an explicit initializer, this sets the initial_value to:
2091    /// 1. The zero-variant (if one exists), or
2092    /// 2. The first variant (as fallback)
2093    pub fn finalize_enum_defaults(&mut self) {
2094        // Process all types and update enum defaults
2095        let mut fixed_types = Vec::new();
2096
2097        for (name, mut datatypes) in self.type_index.types.drain(..) {
2098            for mut datatype in datatypes.drain(..) {
2099                if let DataTypeInformation::Enum { variants, .. } = &datatype.information {
2100                    // Only process if there's no explicit initializer
2101                    if datatype.initial_value.is_none() && !variants.is_empty() {
2102                        let mut zero_variant_id: Option<ConstId> = None;
2103                        let mut first_variant_id: Option<ConstId> = None;
2104
2105                        // Look for a variant that evaluates to zero, or use the first one
2106                        for (idx, variant) in variants.iter().enumerate() {
2107                            if let Some(variant_init) = variant.initial_value {
2108                                if idx == 0 {
2109                                    first_variant_id = Some(variant_init);
2110                                }
2111
2112                                if let Ok(0) =
2113                                    self.constant_expressions.get_constant_int_statement_value(&variant_init)
2114                                {
2115                                    zero_variant_id = Some(variant_init);
2116                                    break;
2117                                }
2118                            }
2119                        }
2120
2121                        // Prefer zero variant, fall back to first variant
2122                        let default_value = zero_variant_id.or(first_variant_id);
2123                        if let Some(const_id) = default_value {
2124                            datatype.initial_value = Some(const_id);
2125                        }
2126                    }
2127                }
2128
2129                fixed_types.push((name.clone(), datatype));
2130            }
2131        }
2132
2133        // Re-insert all types
2134        for (name, datatype) in fixed_types {
2135            self.type_index.types.insert(name, datatype);
2136        }
2137    }

2331        &'b self,
2332        container: &str,
2333        current_methods: Vec<&'b PouIndexEntry>,
2334        seen: &mut FxHashSet<&'b str>,

2340                .filter(|it| it.is_method())

2342                .filter(|it| !current_methods.iter().any(|m| m.get_call_name() == it.get_call_name()))
2343                .collect::<Vec<_>>();
2344            res.extend(current_methods);

2349                self.get_methods_recursive(super_class, res, seen)

2358    /// Returns all methods defined in a container, including methods from super "classes". Thereby the result
2359    /// is in fixed traversal order, meaning that the methods of the super class are always positioned before
2360    /// the methods of any child class. This ordering is neccessary for virtual tables, where bitcasting them
2361    /// from one type to another requires such an order to ensure that the correct method is called.
2362    ///
2363    /// For example, if class `A` has a method `foo` and class `B` inherits from `A` but adds another method
2364    /// `bar` then the virtual table must have the form [`A.foo`] and [`B.foo`, `B.bar`] such that upcasting
2365    /// `B` to `A` will still call method `foo` rather than `bar`. If not, e.g. [`B.bar`, `B.foo`] is used,
2366    /// the upcasting to `A` would result calling `B.bar` when we have a call such as `reInstance^.foo()`
2367    pub fn get_methods_in_fixed_order(&self, container: &str) -> Vec<&PouIndexEntry> {
2368        let res = self.get_methods_recursive_in_fixed_order(
2369            container,
2370            FxIndexMap::default(),
2371            &mut FxHashSet::default(),
2372        );
2373        res.into_values().collect()
2374    }

2376    /// See [`Index::get_methods_in_fixed_order`]
2377    fn get_methods_recursive_in_fixed_order<'b>(
2378        &'b self,
2379        container: &str,
2380        mut collected: FxIndexMap<&'b str, &'b PouIndexEntry>,
2381        seen: &mut FxHashSet<&'b str>,
2382    ) -> FxIndexMap<&'b str, &'b PouIndexEntry> {
2383        if let Some(pou) = self.find_pou(container) {
2384            if let Some(super_class) = pou.get_super_class() {
2385                if !seen.insert(super_class) {
2386                    return collected;
2387                }
2388
2389                // We want to recursively climb up the inheritance chain before collecting methods
2390                collected = self.get_methods_recursive_in_fixed_order(super_class, collected, seen);
2391            }
2392
2393            let methods = self
2394                .get_pous()
2395                .values()
2396                .filter(|pou| pou.is_method())
2397                .filter(|pou| pou.get_parent_pou_name().is_some_and(|opt| opt == container));
2398
2399            for method in methods {
2400                let name = method.get_name().split_once('.').unwrap().1;
2401                collected.insert(name, method);
2402            }
2403        }
2404
2405        collected
2406    }
