299    pub direction: HardwareAccessType,
300    /// The datatype (size) of the binding
301    pub access: DirectAccessType,
302    /// A list of entries that form this binding
303    pub entries: Vec<ConstId>,

305    pub location: SourceLocation,

307
308impl HardwareBinding {
309    pub fn from_statement(index: &mut Index, it: &AstNode, scope: Option<String>) -> Option<Self> {
310        if let AstStatement::HardwareAccess(data) = it.get_stmt() {
311            Some(HardwareBinding {
312                access: data.access,
313                direction: data.direction,
314                entries: data
315                    .address
316                    .iter()
317                    .map(|expr| {
318                        index.constant_expressions.add_constant_expression(
319                            expr.clone(),
320                            typesystem::DINT_SIZE.to_string(),
321                            scope.clone(),
322                            None,
323                        )
324                    })
325                    .collect(),
326                location: it.get_location(),
327            })
328        } else {
329            None
330        }
331    }

519#[derive(Eq, PartialEq, Hash)]
520pub struct InterfaceIndexEntry {
521    /// The interface identifier, consisting of its name and name-location
522    pub ident: Identifier,
523
524    /// The location of the interface as a whole
525    pub location: SourceLocation,
526
527    /// A list of qualified names of the methods in this interface; the actual methods are located in
528    /// [`Index::pous`]
529    pub methods: Vec<String>,
530
531    /// A list of other interfaces this interface extends
532    pub extensions: Vec<Identifier>,
533
534    /// A list of the properties declared in this interface
535    pub properties: Vec<PropertyBlock>,
536}
537
538impl InterfaceIndexEntry {
539    pub fn get_name(&self) -> &str {
540        self.ident.name.as_str()
541    }
542
543    pub fn get_name_location(&self) -> &SourceLocation {
544        &self.ident.location
545    }
546
547    pub fn get_location(&self) -> &SourceLocation {
548        &self.location
549    }
550
551    /// Returns a list of methods this interface declared
552    pub fn get_declared_methods<'idx>(&self, index: &'idx Index) -> Vec<&'idx PouIndexEntry> {
553        self.methods
554            .iter()
555            .map(|name| index.find_pou(name).expect("must exist because of present InterfaceIndexEntry"))
556            .collect()
557    }
558
559    /// Returns a list of methods this interface inherited
560    pub fn get_derived_methods<'idx>(&'idx self, index: &'idx Index) -> Vec<&'idx PouIndexEntry> {
561        self.get_derived_methods_recursive(index, &mut FxHashSet::default())
562    }
563
564    /// Returns a list of methods defined in this interface, including inherited methods from derived interfaces
565    pub fn get_methods<'idx>(&'idx self, index: &'idx Index) -> Vec<&'idx PouIndexEntry> {
566        self.get_methods_recursive(index, &mut FxHashSet::default())
567    }
568
569    /// Returns a list of interfaces this interface implements
570    pub fn get_extensions(&self) -> Vec<&Identifier> {
571        self.extensions.iter().collect()
572    }
573
574    /// Returns a list of interfaces this interface inherited directly
575    pub fn get_derived_interfaces<'idx>(
576        &self,
577        index: &'idx Index,
578    ) -> Vec<Result<&'idx InterfaceIndexEntry, Identifier>> {
579        self.extensions
580            .iter()
581            .flat_map(|id| index.find_interface(&id.name).map(Result::Ok).or(Some(Err(id.to_owned()))))
582            .collect()
583    }
584
585    /// Returns a list of ALL existing interfaces this interface inherited directly or indirectly
586    pub fn get_derived_interfaces_recursive<'i>(&self, index: &'i Index) -> Vec<&'i InterfaceIndexEntry> {
587        let mut seen: FxHashSet<&Identifier> = FxHashSet::default();
588        let mut queue: VecDeque<&InterfaceIndexEntry> = VecDeque::new();
589
590        queue.push_back(self);
591        while let Some(interface) = queue.pop_front() {
592            if seen.insert(&interface.ident) {
593                queue.extend(interface.get_derived_interfaces(index).into_iter().flatten());
594            }
595        }
596
597        seen.into_iter().filter_map(|ident| index.find_interface(&ident.name)).collect()
598    }
599
600    fn get_methods_recursive<'idx>(
601        &'idx self,
602        index: &'idx Index,
603        seen: &mut FxHashSet<&'idx str>,
604    ) -> Vec<&'idx PouIndexEntry> {
605        seen.insert(self.get_name());
606        self.get_declared_methods(index)
607            .into_iter()
608            .chain(self.get_derived_methods_recursive(index, seen))
609            .collect_vec()
610    }
611
612    fn get_derived_methods_recursive<'idx>(
613        &'idx self,
614        index: &'idx Index,
615        seen: &mut FxHashSet<&'idx str>,
616    ) -> Vec<&'idx PouIndexEntry> {
617        self.get_derived_interfaces(index)
618            .iter()
619            .filter_map(|it| it.as_ref().ok())
620            .flat_map(|it| {
621                if !seen.contains(it.get_name()) {
622                    it.get_methods_recursive(index, seen)
623                } else {
624                    vec![]
625                }
626            })
627            .collect()
628    }
629}

724        }

1197    /// An effective type will be any end type i.e. Structs, Integers, Floats, String and Array
1198    pub fn find_effective_type<'ret>(&'ret self, data_type: &'ret DataType) -> Option<&'ret DataType> {

1201
1202    fn find_effective_type_recursive<'ret>(
1203        &'ret self,
1204        data_type: &'ret DataType,
1205        seen: &mut FxHashSet<&'ret str>,
1206    ) -> Option<&'ret DataType> {
1207        match data_type.get_type_information() {
1208            DataTypeInformation::Alias { referenced_type, .. } => {
1209                // Check for cycles in type aliases
1210                if !seen.insert(data_type.get_name()) {
1211                    // Cycle detected, return None to indicate error
1212                    return None;
1213                }
1214
1215                let result = self
1216                    .find_type(referenced_type)
1217                    .and_then(|it| self.find_effective_type_recursive(it, seen));
1218
1219                // Remove from seen set when backtracking
1220                seen.remove(data_type.get_name());
1221                result
1222            }
1223            _ => Some(data_type),
1224        }
1225    }

1270impl Index {
1271    /// imports all entries from the given index into the current index
1272    ///
1273    /// imports all global_variables, types and implementations
1274    /// # Arguments
1275    /// - `other` the other index. The elements are drained from the given index and moved into the current one
1276    pub fn import(&mut self, mut other: Index) {
1277        //global variables
1278        for (name, e) in other.global_variables.drain(..) {
1279            let entries = e
1280                .into_iter()
1281                .map(|it| self.transfer_constants(it, &mut other.constant_expressions))
1282                .collect::<Vec<_>>();
1283            self.global_variables.insert_many(name, entries);
1284        }
1285
1286        //initializers
1287        for (name, elements) in other.global_initializers.drain(..) {
1288            let elements = elements
1289                .into_iter()
1290                .map(|e| self.transfer_constants(e, &mut other.constant_expressions))
1291                .collect::<Vec<_>>();
1292            self.global_initializers.insert_many(name, elements);
1293        }
1294
1295        //types
1296        for (name, mut elements) in other.type_index.types.drain(..) {
1297            for mut e in elements.drain(..) {
1298                //avoid re-importing internally auto-generated types that we already know
1299                if !e.is_internal() || self.type_index.find_effective_type_by_name(name.as_str()).is_none() {
1300                    e.initial_value =
1301                        self.maybe_import_const_expr(&mut other.constant_expressions, &e.initial_value);
1302
1303                    match &mut e.information {
1304                        //import constant expressions in array-type-definitions
1305                        DataTypeInformation::Array { dimensions, .. } => {
1306                            for d in dimensions.iter_mut() {
1307                                d.start_offset =
1308                                    self.import_type_size(&mut other.constant_expressions, &d.start_offset);
1309                                d.end_offset =
1310                                    self.import_type_size(&mut other.constant_expressions, &d.end_offset);
1311                            }
1312                        }
1313                        // import constant expressions in String-size defintions
1314                        DataTypeInformation::String { size, .. } => {
1315                            *size = self.import_type_size(&mut other.constant_expressions, size);
1316                        }
1317                        DataTypeInformation::Struct { members, .. } => {
1318                            let mut variables = members
1319                                .drain(..)
1320                                .map(|variable| {
1321                                    self.transfer_constants(variable, &mut other.constant_expressions)
1322                                })
1323                                .collect::<Vec<_>>();
1324                            members.append(&mut variables);
1325                        }
1326                        DataTypeInformation::Enum { variants, .. } => {
1327                            let mut variables = variants
1328                                .drain(..)
1329                                .map(|variable| {
1330                                    self.transfer_constants(variable, &mut other.constant_expressions)
1331                                })
1332                                .collect::<Vec<_>>();
1333
1334                            for e in variables.iter() {
1335                                self.enum_global_variables.insert(e.get_name().to_lowercase(), e.clone());
1336                            }
1337                            variants.append(&mut variables);
1338                        }
1339                        _ => {}
1340                    }
1341
1342                    self.type_index.types.insert(name.clone(), e)
1343                }
1344            }
1345        }
1346
1347        //pou_types
1348        for (name, mut elements) in other.type_index.pou_types.drain(..) {
1349            elements.iter_mut().for_each(|e| {
1350                self.maybe_import_const_expr(&mut other.constant_expressions, &e.initial_value);
1351
1352                if let DataTypeInformation::Struct { members, .. } = &mut e.information {
1353                    let mut variables = members
1354                        .drain(..)
1355                        .map(|variable| self.transfer_constants(variable, &mut other.constant_expressions))
1356                        .collect::<Vec<_>>();
1357                    members.append(&mut variables);
1358                }
1359            });
1360            self.type_index.pou_types.insert_many(name, elements)
1361        }
1362
1363        //implementations
1364        self.implementations.extend(other.implementations);
1365
1366        // interfaces
1367        self.interfaces.extend(other.interfaces);
1368
1369        // properties
1370        self.properties.extend(other.properties);
1371
1372        //pous
1373        for (name, elements) in other.pous.drain(..) {
1374            for ele in elements {
1375                // skip automatically generated pou's if they are already in the target index
1376                if !ele.is_auto_generated_function() || !self.pous.contains_key(&name) {
1377                    self.pous.insert(name.clone(), ele);
1378                }
1379            }
1380        }
1381
1382        //labels
1383        self.labels.extend(other.labels);
1384
1385        self.config_variables.extend(other.config_variables);
1386
1387        //Constant expressions are intentionally not imported
1388        // self.constant_expressions.import(other.constant_expressions)
1389    }
1390
1391    fn transfer_constants(
1392        &mut self,
1393        mut variable: VariableIndexEntry,
1394        import_from: &mut ConstExpressions,
1395    ) -> VariableIndexEntry {
1396        variable.initial_value = self.maybe_import_const_expr(import_from, &variable.initial_value);
1397
1398        let binding = if let Some(HardwareBinding { direction, access, entries, location }) =
1399            variable.get_hardware_binding()
1400        {
1401            let mut new_entries = vec![];
1402            for entry in entries {

1409                access: *access,
1410                entries: new_entries,
1411                location: location.clone(),

1415        };
1416        variable.set_hardware_binding(binding)

1431    /// imports the corresponding TypeSize (according to the given initializer-id) from the given ConstExpressions
1432    /// into self's const-expressions and returns the new Id
1433    ///
1434    /// panics if the import fails (e.g. the given TypeSize::ConstExpression(id) does not exist in this Index)
1435    /// this problem would indicate a programming mistake
1436    fn import_type_size(&mut self, import_from: &mut ConstExpressions, type_size: &TypeSize) -> TypeSize {
1437        let ts = match type_size {
1438            TypeSize::LiteralInteger(_) => Some(*type_size),
1439            TypeSize::ConstExpression(id) => import_from
1440                .clone(id)
1441                .map(|(expr, target_type, scope, lhs)| {
1442                    self.get_mut_const_expressions().add_constant_expression(expr, target_type, scope, lhs)
1443                })
1444                .map(TypeSize::from_expression),
1445            TypeSize::Undetermined => Some(*type_size),
1446        };
1447
1448        match ts {
1449            Some(it) => it,
1450            None => {
1451                unreachable!(
1452                    "requested type-size is not part of the given ConstExpressions. Cannot import '{:?}', from {:?}",
1453                    type_size, import_from
1454                );
1455            }
1456        }
1457    }

1601        if segments.is_empty() {
1602            return None;
1603        }
1604
1605        //For the first element, if the context does not contain that element, it is possible that the element is also a global variable

2166    pub fn get_inheritance_chain<'idx>(
2167        &'idx self,
2168        current_gen: &str,
2169        target_gen: &str,
2170    ) -> Vec<&'idx PouIndexEntry> {
2171        //Does the current type exist?
2172        let Some(current_ty) = self.find_pou(current_gen) else {
2173            return vec![];
2174        };
2175
2176        // If the current type matches, add it as return
2177        if current_ty.get_name() == target_gen {
2178            return vec![current_ty];
2179        }
2180
2181        let mut res =
2182            self.get_inheritance_chain(current_ty.get_super_class().unwrap_or_default(), target_gen);

2185        };
2186
2187        res
2188    }

2196    fn get_methods_recursive<'b>(
2197        &'b self,
2198        container: &str,
2199        current_methods: Vec<&'b PouIndexEntry>,
2200        seen: &mut FxHashSet<&'b str>,
2201    ) -> Vec<&'b PouIndexEntry> {
2202        if let Some(pou) = self.find_pou(container) {
2203            let mut res = self
2204                .get_pous()
2205                .values()
2206                .filter(|it| it.is_method())
2207                .filter(|it| it.get_parent_pou_name().is_some_and(|it| it == container))
2208                .filter(|it| {
2209                    !current_methods
2210                        .iter()
2211                        .any(|m| m.get_flat_reference_name() == it.get_flat_reference_name())
2212                })
2213                .collect::<Vec<_>>();
2214            res.extend(current_methods);
2215            if let Some(super_class) = pou.get_super_class() {
2216                if !seen.insert(super_class) {
2217                    return res;
2218                };
2219                self.get_methods_recursive(super_class, res, seen)
2220            } else {
2221                res
2222            }
2223        } else {
2224            current_methods
2225        }
2226    }
