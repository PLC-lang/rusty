16/// A convenience macro to call the `cast` function with fewer parameters.
17///
18/// Generates a cast from the given `value` to the given `target_type` if necessary and returns the casted value. It returns
19/// the original `value` if no cast is necessary or if the provided value is not eligible to be cast (to the target type or at all).
20///
21/// This function provides no additional validation or safeguards for invalid casts, as such validation is expected to be
22/// performed at the validation stage prior to code-gen.
23/// Cast instructions for values other than IntValue, FloatValue and PointerValue will simply be ignored (and the value
24/// returned unchanged). Invalid casting instructions for the above-mentioned values will fail spectacularly instead.
25///
26/// - `generator` the generator to use. must contain the following fields:
27///     - `llvm` the llvm utilities to use for code-generation
28///     - `index` the current Index used for type-lookups
29///     - `llvm_type_index` the type index to lookup llvm generated types
30/// - `target_type` the expected target type of the value
31/// - `value_type` the current type of the given value
32/// - `value` the value to (maybe) cast
33macro_rules! cast_if_needed {
34    ($generator:expr, $target_type:expr, $value_type:expr, $value:expr, $annotation:expr) => {
35        crate::codegen::llvm_typesystem::cast(
36            $generator.llvm,
37            $generator.index,
38            $generator.llvm_index,
39            $target_type,
40            $value_type,
41            $value,
42            $annotation,
43        )
44    };
45}
46
47pub(crate) use cast_if_needed;

49pub fn get_llvm_int_type<'a>(context: &'a Context, size: u32, name: &str) -> IntType<'a> {
50    match size {
51        1 => context.bool_type(),
52        8 => context.i8_type(),
53        16 => context.i16_type(),
54        32 => context.i32_type(),
55        64 => context.i64_type(),
56        128 => context.i128_type(),
57        _ => unreachable!("Invalid size for type : '{name}' at {size}"),
58    }
59}
60
61pub fn get_llvm_float_type<'a>(context: &'a Context, size: u32, name: &str) -> FloatType<'a> {
62    match size {
63        32 => context.f32_type(),
64        64 => context.f64_type(),
65        _ => unreachable!("Invalid size for type : '{name}' at {size}"),
66    }
67}

69/// Generates a cast from the given `value` to the given `target_type` if necessary and returns the casted value. It returns
70/// the original `value` if no cast is necessary or if the provided value is not eligible to be cast (to the target type or at all).
71///
72/// This function provides no additional validation or safeguards for invalid casts, as such validation is expected to be
73/// performed at the validation stage prior to code-gen.
74/// Cast instructions for values other than IntValue, FloatValue and PointerValue will simply be ignored (and the value
75/// returned unchanged). Invalid casting instructions for the above-mentioned values will fail spectacularly instead.
76///
77/// - `llvm` the llvm utilities to use for code-generation
78/// - `index` the current Index used for type-lookups
79/// - `llvm_type_index` the type index to lookup llvm generated types
80/// - `target_type` the expected target type of the value
81/// - `value_type` the current type of the given value
82/// - `value` the value to (maybe) cast
83pub fn cast<'ctx>(
84    llvm: &Llvm<'ctx>,
85    index: &Index,
86    llvm_type_index: &LlvmTypedIndex<'ctx>,
87    target_type: &DataType,
88    value_type: &DataType,
89    value: BasicValueEnum<'ctx>,
90    annotation: Option<&StatementAnnotation>,
91) -> BasicValueEnum<'ctx> {
92    value.cast(&CastInstructionData::new(llvm, index, llvm_type_index, value_type, target_type, annotation))
93}
94
95struct CastInstructionData<'ctx, 'cast> {
96    llvm: &'cast Llvm<'ctx>,
97    index: &'cast Index,
98    llvm_type_index: &'cast LlvmTypedIndex<'ctx>,
99    value_type: &'cast DataTypeInformation,
100    target_type: &'cast DataTypeInformation,
101    annotation: Option<&'cast StatementAnnotation>,
102}
103
104impl<'ctx, 'cast> CastInstructionData<'ctx, 'cast> {
105    fn new(
106        llvm: &'cast Llvm<'ctx>,
107        index: &'cast Index,
108        llvm_type_index: &'cast LlvmTypedIndex<'ctx>,
109        value_type: &DataType,
110        target_type: &DataType,
111        annotation: Option<&'cast StatementAnnotation>,
112    ) -> Self {
113        let target_type = index.get_intrinsic_type_by_name(target_type.get_name()).get_type_information();
114        let value_type = index.get_intrinsic_type_by_name(value_type.get_name()).get_type_information();
115
116        let target_type =
117            if let DataTypeInformation::Pointer { auto_deref: Some(_), inner_type_name, .. } = target_type {
118                // Deref auto-deref pointers before casting
119                index.get_intrinsic_type_by_name(inner_type_name.as_str()).get_type_information()
120            } else {
121                target_type
122            };
123
124        CastInstructionData { llvm, index, llvm_type_index, value_type, target_type, annotation }
125    }
126}

140impl<'ctx, 'cast> Castable<'ctx, 'cast> for BasicValueEnum<'ctx> {
141    fn cast(self, cast_data: &CastInstructionData<'ctx, 'cast>) -> BasicValueEnum<'ctx> {
142        match self {
143            BasicValueEnum::IntValue(val) => val.cast(cast_data),
144            BasicValueEnum::FloatValue(val) => val.cast(cast_data),
145            BasicValueEnum::PointerValue(val) => val.cast(cast_data),
146            BasicValueEnum::ArrayValue(val) => val.cast(cast_data),
147            _ => self,
148        }
149    }
150}

152impl<'ctx, 'cast> Castable<'ctx, 'cast> for IntValue<'ctx> {
153    fn cast(self, cast_data: &CastInstructionData<'ctx, 'cast>) -> BasicValueEnum<'ctx> {
154        let lsize = cast_data.target_type.get_size_in_bits(cast_data.index).unwrap();
155        match cast_data.target_type {
156            DataTypeInformation::Integer { .. } => {
157                //its important to use the real type's size here, because we may have an i1 which is annotated as BOOL (8 bit)
158                let rsize = self.get_type().get_bit_width();
159                if lsize < rsize {
160                    //Truncate
161                    self.truncate(lsize, cast_data)
162                } else {
163                    //Expand
164                    self.promote(lsize, cast_data)
165                }
166            }
167            DataTypeInformation::Float { .. } => {
168                let float_type = get_llvm_float_type(cast_data.llvm.context, lsize, "Float");
169                if cast_data.value_type.is_signed_int() {
170                    cast_data.llvm.builder.build_signed_int_to_float(self, float_type, "").into()
171                } else {
172                    cast_data.llvm.builder.build_unsigned_int_to_float(self, float_type, "").into()
173                }
174            }
175            DataTypeInformation::Pointer { .. } => {
176                let Ok(associated_type) =
177                    cast_data.llvm_type_index.get_associated_type(cast_data.target_type.get_name())
178                else {
179                    unreachable!(
180                        "Target type of cast instruction does not exist: {}",
181                        cast_data.target_type.get_name()
182                    )
183                };
184
185                cast_data.llvm.builder.build_int_to_ptr(self, associated_type.into_pointer_type(), "").into()
186            }
187            _ => unreachable!("Cannot cast integer value to {}", cast_data.target_type.get_name()),
188        }
189    }
190}

192impl<'ctx, 'cast> Castable<'ctx, 'cast> for FloatValue<'ctx> {
193    fn cast(self, cast_data: &CastInstructionData<'ctx, 'cast>) -> BasicValueEnum<'ctx> {
194        let rsize = &cast_data.value_type.get_size_in_bits(cast_data.index).unwrap();
195        match cast_data.target_type {
196            DataTypeInformation::Float { size: lsize, .. } => {
197                if lsize < rsize {
198                    self.truncate(*lsize, cast_data)
199                } else {
200                    self.promote(*lsize, cast_data)
201                }
202            }
203            DataTypeInformation::Integer { signed, size: lsize, .. } => {
204                let int_type = get_llvm_int_type(cast_data.llvm.context, *lsize, "Integer");
205                if *signed {
206                    cast_data.llvm.builder.build_float_to_signed_int(self, int_type, "").into()
207                } else {
208                    cast_data.llvm.builder.build_float_to_unsigned_int(self, int_type, "").into()
209                }
210            }
211            _ => unreachable!("Cannot cast floating-point value to {}", cast_data.target_type.get_name()),
212        }
213    }
214}

216impl<'ctx, 'cast> Castable<'ctx, 'cast> for PointerValue<'ctx> {
217    fn cast(self, cast_data: &CastInstructionData<'ctx, 'cast>) -> BasicValueEnum<'ctx> {
218        match &cast_data.target_type {
219            DataTypeInformation::Integer { size: lsize, .. } => cast_data
220                .llvm
221                .builder
222                .build_ptr_to_int(self, get_llvm_int_type(cast_data.llvm.context, *lsize, ""), "")
223                .into(),
224            DataTypeInformation::Pointer { .. } => {
225                let Ok(target_ptr_type) =
226                    cast_data.llvm_type_index.get_associated_type(cast_data.target_type.get_name())
227                else {
228                    unreachable!(
229                        "Target type of cast instruction does not exist: {}",
230                        cast_data.target_type.get_name()
231                    )
232                };
233                if BasicValueEnum::from(self).get_type() != target_ptr_type {
234                    // bit-cast necessary
235                    cast_data.llvm.builder.build_bitcast(self, target_ptr_type, "")
236                } else {
237                    //this is ok, no cast required
238                    self.into()
239                }
240            }
241            DataTypeInformation::Struct {
242                source: StructSource::Internal(InternalType::VariableLengthArray { .. }),
243                ..
244            } => {
245                // we are dealing with an auto-deref vla parameter. first we have to deref our array and build the fat pointer
246                let struct_val = cast_data.llvm.builder.build_load(self, "auto_deref").cast(cast_data);
247
248                // create a pointer to the generated StructValue
249                let struct_ptr = cast_data.llvm.builder.build_alloca(struct_val.get_type(), "vla_struct_ptr");
250                cast_data.llvm.builder.build_store(struct_ptr, struct_val);
251                struct_ptr.into()
252            }
253            _ => unreachable!("Cannot cast pointer value to {}", cast_data.target_type.get_name()),
254        }
255    }
256}

258impl<'ctx, 'cast> Castable<'ctx, 'cast> for ArrayValue<'ctx> {
259    /// Generates a fat pointer struct for an array if the target type is a VLA,
260    /// otherwise returns the value as is.
261    fn cast(self, cast_data: &CastInstructionData<'ctx, 'cast>) -> BasicValueEnum<'ctx> {
262        if !cast_data.target_type.is_vla() {
263            return self.into();
264        }
265        let builder = &cast_data.llvm.builder;
266        let zero = cast_data.llvm.i32_type().const_zero();
267
268        let Ok(associated_type) =
269            cast_data.llvm_type_index.get_associated_type(cast_data.target_type.get_name())
270        else {
271            unreachable!(
272                "Target type of cast instruction does not exist: {}",
273                cast_data.target_type.get_name()
274            )
275        };
276
277        // Get array annotation from parent POU and get pointer to array
278        let Some(StatementAnnotation::Variable { qualified_name, .. }) = cast_data.annotation else {
279            unreachable!("Undefined reference: {}", cast_data.value_type.get_name())
280        };
281        let array_pointer = cast_data
282            .llvm_type_index
283            .find_loaded_associated_variable_value(qualified_name.as_str())
284            .unwrap_or_else(|| unreachable!("passed array must be in the llvm index"));
285
286        // gep into the original array. the resulting address will be stored in the VLA struct
287        let arr_gep = unsafe { builder.build_in_bounds_gep(array_pointer, &[zero, zero], "outer_arr_gep") };
288
289        // -- Generate struct & arr_ptr --
290        let ty = associated_type.into_struct_type();
291        let vla_struct = builder.build_alloca(ty, "vla_struct");
292
293        let Ok(vla_arr_ptr) = builder.build_struct_gep(vla_struct, 0, "vla_array_gep") else {
294            unreachable!("Must have a valid, GEP-able fat-pointer struct at this stage")
295        };
296
297        let Ok(vla_dimensions_ptr) = builder.build_struct_gep(vla_struct, 1, "vla_dimensions_gep") else {
298            unreachable!("Must have a valid, GEP-able fat-pointer struct at this stage")
299        };
300
301        // -- Generate dimensions --
302        let DataTypeInformation::Array { dimensions, .. } = cast_data.value_type else { unreachable!() };
303        let mut dims = Vec::new();
304        for dim in dimensions {
305            dims.push(dim.start_offset.as_int_value(cast_data.index).unwrap());
306            dims.push(dim.end_offset.as_int_value(cast_data.index).unwrap());
307        }
308
309        // Populate each array element
310        let dimensions =
311            dims.iter().map(|it| cast_data.llvm.i32_type().const_int(*it as u64, true)).collect::<Vec<_>>();
312        let array_value = cast_data.llvm.i32_type().const_array(&dimensions);
313        // FIXME: should be memcopied, but is an rvalue. can only initialize global variables with value types. any other way for alloca'd variables than using store?
314        builder.build_store(vla_dimensions_ptr, array_value);
315
316        builder.build_store(vla_arr_ptr, arr_gep);
317
318        builder.build_load(vla_struct, "")
319    }
320}

322impl<'ctx, 'cast> Promotable<'ctx, 'cast> for IntValue<'ctx> {
323    fn promote(self, lsize: u32, cast_data: &CastInstructionData<'ctx, 'cast>) -> BasicValueEnum<'ctx> {
324        let llvm_int_type = get_llvm_int_type(cast_data.llvm.context, lsize, "Integer");
325        if cast_data.value_type.is_signed_int() {
326            cast_data.llvm.builder.build_int_s_extend_or_bit_cast(self, llvm_int_type, "")
327        } else {
328            cast_data.llvm.builder.build_int_z_extend_or_bit_cast(self, llvm_int_type, "")
329        }
330        .into()
331    }
332}
