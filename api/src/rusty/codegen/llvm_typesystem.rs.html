17/// A convenience macro to call the `cast` function with fewer parameters.
18///
19/// Generates a cast from the given `value` to the given `target_type` if necessary and returns the casted value. It returns
20/// the original `value` if no cast is necessary or if the provided value is not eligible to be cast (to the target type or at all).
21///
22/// This function provides no additional validation or safeguards for invalid casts, as such validation is expected to be
23/// performed at the validation stage prior to code-gen.
24/// Cast instructions for values other than IntValue, FloatValue and PointerValue will simply be ignored (and the value
25/// returned unchanged). Invalid casting instructions for the above-mentioned values will fail spectacularly instead.
26///
27/// - `generator` the generator to use. must contain the following fields:
28///     - `llvm` the llvm utilities to use for code-generation
29///     - `index` the current Index used for type-lookups
30///     - `llvm_type_index` the type index to lookup llvm generated types
31/// - `target_type` the expected target type of the value
32/// - `value_type` the current type of the given value
33/// - `value` the value to (maybe) cast
34macro_rules! cast_if_needed {
35    ($generator:expr, $target_type:expr, $value_type:expr, $value:expr, $annotation:expr) => {
36        crate::codegen::llvm_typesystem::cast(
37            $generator.llvm,
38            $generator.index,
39            $generator.llvm_index,
40            $target_type,
41            $value_type,
42            $value,
43            $annotation,
44        )
45    };
46}
47
48pub(crate) use cast_if_needed;

50pub fn get_llvm_int_type<'a>(context: &'a Context, size: u32, name: &str) -> IntType<'a> {
51    match size {
52        1 => context.bool_type(),
53        8 => context.i8_type(),
54        16 => context.i16_type(),
55        32 => context.i32_type(),
56        64 => context.i64_type(),
57        128 => context.i128_type(),
58        _ => unreachable!("Invalid size for type : '{name}' at {size}"),
59    }
60}
61
62pub fn get_llvm_float_type<'a>(context: &'a Context, size: u32, name: &str) -> FloatType<'a> {
63    match size {
64        32 => context.f32_type(),
65        64 => context.f64_type(),
66        _ => unreachable!("Invalid size for type : '{name}' at {size}"),
67    }
68}

70/// Generates a cast from the given `value` to the given `target_type` if necessary and returns the casted value. It returns
71/// the original `value` if no cast is necessary or if the provided value is not eligible to be cast (to the target type or at all).
72///
73/// This function provides no additional validation or safeguards for invalid casts, as such validation is expected to be
74/// performed at the validation stage prior to code-gen.
75/// Cast instructions for values other than IntValue, FloatValue and PointerValue will simply be ignored (and the value
76/// returned unchanged). Invalid casting instructions for the above-mentioned values will fail spectacularly instead.
77///
78/// - `llvm` the llvm utilities to use for code-generation
79/// - `index` the current Index used for type-lookups
80/// - `llvm_type_index` the type index to lookup llvm generated types
81/// - `target_type` the expected target type of the value
82/// - `value_type` the current type of the given value
83/// - `value` the value to (maybe) cast
84pub fn cast<'ctx>(
85    llvm: &Llvm<'ctx>,
86    index: &Index,
87    llvm_type_index: &LlvmTypedIndex<'ctx>,
88    target_type: &DataType,
89    value_type: &DataType,
90    value: BasicValueEnum<'ctx>,
91    annotation: Option<&StatementAnnotation>,
92) -> Result<BasicValueEnum<'ctx>, CodegenError> {
93    value.cast(&CastInstructionData::new(llvm, index, llvm_type_index, value_type, target_type, annotation))
94}
95
96struct CastInstructionData<'ctx, 'cast> {
97    llvm: &'cast Llvm<'ctx>,
98    index: &'cast Index,
99    llvm_type_index: &'cast LlvmTypedIndex<'ctx>,
100    value_type: &'cast DataTypeInformation,
101    target_type: &'cast DataTypeInformation,
102    annotation: Option<&'cast StatementAnnotation>,
103}
104
105impl<'ctx, 'cast> CastInstructionData<'ctx, 'cast> {
106    fn new(
107        llvm: &'cast Llvm<'ctx>,
108        index: &'cast Index,
109        llvm_type_index: &'cast LlvmTypedIndex<'ctx>,
110        value_type: &DataType,
111        target_type: &DataType,
112        annotation: Option<&'cast StatementAnnotation>,
113    ) -> Self {
114        let target_type = index.get_intrinsic_type_by_name(target_type.get_name()).get_type_information();
115        let value_type = index.get_intrinsic_type_by_name(value_type.get_name()).get_type_information();
116
117        let target_type =
118            if let DataTypeInformation::Pointer { auto_deref: Some(_), inner_type_name, .. } = target_type {
119                // Deref auto-deref pointers before casting
120                index.get_intrinsic_type_by_name(inner_type_name.as_str()).get_type_information()
121            } else {
122                target_type
123            };
124
125        CastInstructionData { llvm, index, llvm_type_index, value_type, target_type, annotation }
126    }
127}

156impl<'ctx, 'cast> Castable<'ctx, 'cast> for BasicValueEnum<'ctx> {
157    fn cast(
158        self,
159        cast_data: &CastInstructionData<'ctx, 'cast>,
160    ) -> Result<BasicValueEnum<'ctx>, CodegenError> {
161        // If the builder is not yet set, ignore casting
162        if cast_data.llvm.builder.get_insert_block().is_none() {
163            log::debug!("The builder is not set, attempting a constant cast");
164            if let Ok(const_val) = self.cast_constant(cast_data) {
165                return Ok(const_val);
166            }
167            log::debug!("Constant cast failed, returning original value");
168            return Ok(self);
169        }
170        Ok(match self {
171            BasicValueEnum::IntValue(val) => val.cast(cast_data)?,
172            BasicValueEnum::FloatValue(val) => val.cast(cast_data)?,
173            BasicValueEnum::PointerValue(val) => val.cast(cast_data)?,
174            BasicValueEnum::ArrayValue(val) => val.cast(cast_data)?,
175            _ => self,
176        })
177    }
178
179    fn cast_constant(
180        self,
181        cast_data: &CastInstructionData<'ctx, 'cast>,
182    ) -> Result<BasicValueEnum<'ctx>, CodegenError> {
183        Ok(match self {
184            BasicValueEnum::IntValue(val) => val.cast_constant(cast_data)?,
185            BasicValueEnum::FloatValue(val) => val.cast_constant(cast_data)?,
186            // Pointers and Arrays cannot be constants in our use-case, so just call the normal
187            // cast
188            BasicValueEnum::PointerValue(val) => val.cast(cast_data)?,
189            BasicValueEnum::ArrayValue(val) => val.cast(cast_data)?,
190            _ => self,
191        })
192    }
193}

195impl<'ctx, 'cast> Castable<'ctx, 'cast> for IntValue<'ctx> {
196    fn cast(
197        self,
198        cast_data: &CastInstructionData<'ctx, 'cast>,
199    ) -> Result<BasicValueEnum<'ctx>, CodegenError> {
200        let lsize = cast_data
201            .target_type
202            .get_size_in_bits(cast_data.index)
203            .map_err(|it| CodegenError::new(it.to_string(), SourceLocation::undefined()))?;
204        match cast_data.target_type {
205            DataTypeInformation::Integer { .. } => {
206                //its important to use the real type's size here, because we may have an i1 which is annotated as BOOL (8 bit)
207                let rsize = self.get_type().get_bit_width();
208                if lsize < rsize {
209                    //Truncate
210                    self.truncate(lsize, cast_data)
211                } else if lsize > rsize {
212                    //Expand
213                    self.promote(lsize, cast_data)
214                } else {
215                    //same size, no cast needed
216                    Ok(self.into())
217                }
218            }
219            DataTypeInformation::Float { .. } => {
220                let float_type = get_llvm_float_type(cast_data.llvm.context, lsize, "Float");
221                if cast_data.value_type.is_signed_int() {
222                    cast_data
223                        .llvm
224                        .builder
225                        .build_signed_int_to_float(self, float_type, "")
226                        .map(Into::into)
227                        .map_err(Into::into)
228                } else {
229                    cast_data
230                        .llvm
231                        .builder
232                        .build_unsigned_int_to_float(self, float_type, "")
233                        .map(Into::into)
234                        .map_err(Into::into)
235                }
236            }
237            DataTypeInformation::Pointer { .. } => {
238                let associated_type =
239                    cast_data.llvm_type_index.get_associated_type(cast_data.target_type.get_name())?;
240                cast_data
241                    .llvm
242                    .builder
243                    .build_int_to_ptr(self, associated_type.into_pointer_type(), "")
244                    .map(Into::into)
245                    .map_err(Into::into)
246            }
247            _ => Err(CodegenError::new(
248                format!("Cannot cast integer value to {}", cast_data.target_type.get_name()),
249                SourceLocation::undefined(),
250            )),
251        }
252    }
253
254    fn cast_constant(
255        self,
256        cast_data: &CastInstructionData<'ctx, 'cast>,
257    ) -> Result<BasicValueEnum<'ctx>, CodegenError> {
258        // If self is not a constant, fail
259        if !self.is_constant_int() {
260            return Err(CodegenError::new(
261                "Value is not a constant".to_string(),
262                SourceLocation::undefined(),
263            ));
264        }
265
266        let mut signed_value = None;
267        let mut unsigned_value = None;
268        if cast_data.value_type.is_signed_int() {
269            signed_value = self.get_sign_extended_constant()
270        } else {
271            unsigned_value = self.get_zero_extended_constant()
272        };
273
274        match cast_data.target_type {
275            DataTypeInformation::Integer { .. } => {
276                let target_type = cast_data
277                    .llvm_type_index
278                    .get_associated_type(cast_data.target_type.get_name())?
279                    .into_int_type();
280                if let Some(sval) = signed_value {
281                    Ok(target_type.const_int(sval as u64, true).into())
282                } else if let Some(uval) = unsigned_value {
283                    Ok(target_type.const_int(uval, false).into())
284                } else {
285                    Err(CodegenError::new("Value is not a constant".to_string(), SourceLocation::undefined()))
286                }
287            }
288            DataTypeInformation::Float { size: lsize, .. } => {
289                let float_type = get_llvm_float_type(cast_data.llvm.context, *lsize, "Float");
290                if let Some(sval) = signed_value {
291                    Ok(float_type.const_float(sval as f64).into())
292                } else if let Some(uval) = unsigned_value {
293                    Ok(float_type.const_float(uval as f64).into())
294                } else {
295                    Err(CodegenError::new("Value is not a constant".to_string(), SourceLocation::undefined()))
296                }
297            }
298            _ => Err(CodegenError::new(
299                format!("Cannot cast integer value to a constant {}", cast_data.target_type.get_name()),
300                SourceLocation::undefined(),
301            )),
302        }
303    }
304}

306impl<'ctx, 'cast> Castable<'ctx, 'cast> for FloatValue<'ctx> {
307    fn cast(
308        self,
309        cast_data: &CastInstructionData<'ctx, 'cast>,
310    ) -> Result<BasicValueEnum<'ctx>, CodegenError> {
311        let rsize = &cast_data.value_type.get_size_in_bits(cast_data.index).unwrap();
312        match cast_data.target_type {
313            DataTypeInformation::Float { size: lsize, .. } => {
314                if lsize < rsize {
315                    self.truncate(*lsize, cast_data)
316                } else if lsize > rsize {
317                    self.promote(*lsize, cast_data)
318                } else {
319                    //same size, no cast needed
320                    Ok(self.into())
321                }
322            }
323            DataTypeInformation::Integer { signed, size: lsize, .. } => {
324                let int_type = get_llvm_int_type(cast_data.llvm.context, *lsize, "Integer");
325                if *signed {
326                    cast_data
327                        .llvm
328                        .builder
329                        .build_float_to_signed_int(self, int_type, "")
330                        .map(Into::into)
331                        .map_err(Into::into)
332                } else {
333                    cast_data
334                        .llvm
335                        .builder
336                        .build_float_to_unsigned_int(self, int_type, "")
337                        .map(Into::into)
338                        .map_err(Into::into)
339                }
340            }
341            _ => Err(CodegenError::new(
342                format!("Cannot cast float value to {}", cast_data.target_type.get_name()),
343                SourceLocation::undefined(),
344            )),
345        }
346    }
347
348    fn cast_constant(
349        self,
350        cast_data: &CastInstructionData<'ctx, 'cast>,
351    ) -> Result<BasicValueEnum<'ctx>, CodegenError> {
352        // If self is not a constant, fail
353        let Some((float_value, _)) = self.get_constant() else {
354            return Err(CodegenError::new(
355                "Value is not a constant".to_string(),
356                SourceLocation::undefined(),
357            ));
358        };
359
360        match cast_data.target_type {
361            DataTypeInformation::Float { size: lsize, .. } => {
362                let target_type = get_llvm_float_type(cast_data.llvm.context, *lsize, "Float");
363                Ok(target_type.const_float(float_value).into())
364            }
365            DataTypeInformation::Integer { signed, size: lsize, .. } => {
366                let int_type = get_llvm_int_type(cast_data.llvm.context, *lsize, "Integer");
367                if *signed {
368                    Ok(int_type.const_int(float_value as i64 as u64, true).into())
369                } else {
370                    Ok(int_type.const_int(float_value as u64, false).into())
371                }
372            }
373            _ => Err(CodegenError::new(
374                format!("Cannot cast float value to a constant {}", cast_data.target_type.get_name()),
375                SourceLocation::undefined(),
376            )),
377        }
378    }
379}

381impl<'ctx, 'cast> Castable<'ctx, 'cast> for PointerValue<'ctx> {
382    fn cast(
383        self,
384        cast_data: &CastInstructionData<'ctx, 'cast>,
385    ) -> Result<BasicValueEnum<'ctx>, CodegenError> {
386        match &cast_data.target_type {
387            DataTypeInformation::Integer { size: lsize, .. } => Ok(cast_data
388                .llvm
389                .builder
390                .build_ptr_to_int(self, get_llvm_int_type(cast_data.llvm.context, *lsize, ""), "")?
391                .into()),
392            DataTypeInformation::Pointer { .. } => {
393                let target_ptr_type =
394                    cast_data.llvm_type_index.get_associated_type(cast_data.target_type.get_name())?;
395                if BasicValueEnum::from(self).get_type() != target_ptr_type {
396                    // bit-cast necessary
397                    cast_data.llvm.builder.build_bit_cast(self, target_ptr_type, "").map_err(Into::into)
398                } else {
399                    //this is ok, no cast required
400                    Ok(self.into())
401                }
402            }
403            DataTypeInformation::Struct {
404                source: StructSource::Internal(InternalType::VariableLengthArray { .. }),
405                ..
406            } => {
407                // we are dealing with an auto-deref vla parameter. first we have to deref our array and build the fat pointer
408                let struct_val = cast_data.llvm.builder.build_load(self, "auto_deref")?.cast(cast_data)?;
409
410                // create a pointer to the generated StructValue
411                let struct_ptr =
412                    cast_data.llvm.builder.build_alloca(struct_val.get_type(), "vla_struct_ptr")?;
413                cast_data.llvm.builder.build_store(struct_ptr, struct_val)?;
414                Ok(struct_ptr.into())
415            }
416            _ => Err(CodegenError::new(
417                format!("Cannot cast pointer value to {}", cast_data.target_type.get_name()),
418                SourceLocation::undefined(),
419            )),
420        }
421    }
422
423    fn cast_constant(
424        self,
425        cast_data: &CastInstructionData<'ctx, 'cast>,
426    ) -> Result<BasicValueEnum<'ctx>, CodegenError> {
427        // Pointers cannot be constants in our use-case, so just call the normal cast
428        self.cast(cast_data)
429    }
430}

432impl<'ctx, 'cast> Castable<'ctx, 'cast> for ArrayValue<'ctx> {
433    /// Generates a fat pointer struct for an array if the target type is a VLA,
434    /// otherwise returns the value as is.
435    fn cast(
436        self,
437        cast_data: &CastInstructionData<'ctx, 'cast>,
438    ) -> Result<BasicValueEnum<'ctx>, CodegenError> {
439        if !cast_data.target_type.is_vla() {
440            return Ok(self.into());
441        }
442        let builder = &cast_data.llvm.builder;
443        let zero = cast_data.llvm.i32_type().const_zero();
444
445        let associated_type =
446            cast_data.llvm_type_index.get_associated_type(cast_data.target_type.get_name())?;
447
448        // Get array annotation from parent POU and get pointer to array
449        let Some(StatementAnnotation::Variable { qualified_name, .. }) = cast_data.annotation else {
450            return Err(CodegenError::new(
451                format!("Undefined reference: {}", cast_data.value_type.get_name()),
452                SourceLocation::undefined(),
453            ));
454        };
455        let array_pointer = cast_data
456            .llvm_type_index
457            .find_loaded_associated_variable_value(qualified_name.as_str())
458            .ok_or_else(|| {
459                CodegenError::new(
460                    format!("Cannot find array pointer for {}", cast_data.value_type.get_name()),
461                    SourceLocation::undefined(),
462                )
463            })?;
464
465        // gep into the original array. the resulting address will be stored in the VLA struct
466        let arr_gep = unsafe { builder.build_in_bounds_gep(array_pointer, &[zero, zero], "outer_arr_gep")? };
467
468        // -- Generate struct & arr_ptr --
469        let ty = associated_type.into_struct_type();
470        let vla_struct = builder.build_alloca(ty, "vla_struct")?;
471
472        let vla_arr_ptr = builder.build_struct_gep(vla_struct, 0, "vla_array_gep")?;
473
474        let vla_dimensions_ptr = builder.build_struct_gep(vla_struct, 1, "vla_dimensions_gep")?;
475
476        // -- Generate dimensions --
477        let DataTypeInformation::Array { dimensions, .. } = cast_data.value_type else { unreachable!() };
478        let mut dims = Vec::new();
479        for dim in dimensions {
480            dims.push(dim.start_offset.as_int_value(cast_data.index).unwrap());
481            dims.push(dim.end_offset.as_int_value(cast_data.index).unwrap());
482        }
483
484        // Populate each array element
485        let dimensions =
486            dims.iter().map(|it| cast_data.llvm.i32_type().const_int(*it as u64, true)).collect::<Vec<_>>();
487        let array_value = cast_data.llvm.i32_type().const_array(&dimensions);
488        // FIXME: should be memcopied, but is an rvalue. can only initialize global variables with value types. any other way for alloca'd variables than using store?
489        builder.build_store(vla_dimensions_ptr, array_value)?;
490
491        builder.build_store(vla_arr_ptr, arr_gep)?;
492
493        builder.build_load(vla_struct, "").map_err(Into::into)
494    }
495
496    fn cast_constant(
497        self,
498        cast_data: &CastInstructionData<'ctx, 'cast>,
499    ) -> Result<BasicValueEnum<'ctx>, CodegenError> {
500        // Arrays cannot be constants in our use-case, so just call the normal cast
501        self.cast(cast_data)
502    }
503}

505impl<'ctx, 'cast> Promotable<'ctx, 'cast> for IntValue<'ctx> {
506    fn promote(
507        self,
508        lsize: u32,
509        cast_data: &CastInstructionData<'ctx, 'cast>,
510    ) -> Result<BasicValueEnum<'ctx>, CodegenError> {
511        let llvm_int_type = get_llvm_int_type(cast_data.llvm.context, lsize, "Integer");
512        Ok(if cast_data.value_type.is_signed_int() {
513            cast_data.llvm.builder.build_int_s_extend_or_bit_cast(self, llvm_int_type, "")?
514        } else {
515            cast_data.llvm.builder.build_int_z_extend_or_bit_cast(self, llvm_int_type, "")?
516        }
517        .into())
518    }
519}
