56/// Creates opaque implementations for all callable items in the index
57/// Returns a Typed index containing the associated implementations.
58/// FIXME: Ignoring the arguments warning for now, should refactor later to pass the options in a struct
59#[allow(clippy::too_many_arguments)]
60pub fn generate_implementation_stubs<'ink>(
61    module: &Module<'ink>,
62    llvm: Llvm<'ink>,
63    dependencies: &FxIndexSet<Dependency>,
64    index: &Index,
65    annotations: &AstAnnotations,
66    types_index: &LlvmTypedIndex<'ink>,
67    debug: &mut DebugBuilderEnum<'ink>,
68    online_change: &OnlineChange,
69) -> Result<LlvmTypedIndex<'ink>, CodegenError> {
70    let mut llvm_index = LlvmTypedIndex::default();
71    let pou_generator = PouGenerator::new(llvm, index, annotations, types_index, online_change);
72    let implementations = dependencies
73        .into_iter()
74        .filter_map(|it| match it {
75            Dependency::Call(name) | Dependency::Datatype(name) => Some(name),
76            _ => None,
77        })
78        .filter_map(|name| index.find_implementation_by_name(name).map(|it| (name, it)))
79        .collect::<FxIndexMap<_, _>>();
80    for (name, implementation) in implementations {
81        if !implementation.is_generic() {
82            let curr_f =
83                pou_generator.generate_implementation_stub(implementation, module, debug, &mut llvm_index)?;
84            llvm_index.associate_implementation(name, curr_f)?;
85        }
86    }
87
88    Ok(llvm_index)
89}

91///Generates a global constant for each initialized pou member
92/// The given constant can then be used to initialize the variable using memcpy without re-evaluating the expression
93/// Retrieves the POUs from the index (implementation)
94/// Returns a new LLVM index to be merged with the parent codegen index.
95pub fn generate_global_constants_for_pou_members<'ink>(
96    module: &Module<'ink>,
97    llvm: &Llvm<'ink>,
98    dependencies: &FxIndexSet<Dependency>,
99    index: &Index,
100    annotations: &AstAnnotations,
101    llvm_index: &LlvmTypedIndex<'ink>,
102    location: &str,
103) -> Result<LlvmTypedIndex<'ink>, CodegenError> {
104    let mut local_llvm_index = LlvmTypedIndex::default();
105    let implementations = dependencies
106        .into_iter()
107        .filter_map(|it| match it {
108            Dependency::Call(name) | Dependency::Datatype(name) => index.find_implementation_by_name(name),
109            _ => None,
110        })
111        .filter(|it| it.is_in_unit(location));
112    for implementation in implementations {
113        let type_name = implementation.get_type_name();
114        if implementation.is_init() {
115            // initializer functions don't need global constants to initialize members
116            continue;
117        }
118        let pou_members = index.get_pou_members(type_name);
119        let variables = pou_members.iter().filter(|it| it.is_local() || it.is_temp()).filter(|it| {
120            let var_type =
121                index.get_effective_type_or_void_by_name(it.get_type_name()).get_type_information();
122            var_type.is_struct() || var_type.is_array() || var_type.is_string()
123        });
124        let exp_gen = ExpressionCodeGenerator::new_context_free(llvm, index, annotations, llvm_index);
125        for variable in variables {
126            let name = index::get_initializer_name(variable.get_qualified_name());
127            let right_stmt =
128                index.get_const_expressions().maybe_get_constant_statement(&variable.initial_value);
129
130            if right_stmt.is_some() && llvm_index.find_global_value(&name).is_none() {
131                let variable_type = llvm_index.get_associated_type(variable.get_type_name())?;
132                let value = if let Some(stmt) = right_stmt {
133                    Some(exp_gen.generate_expression(stmt)?)
134                } else {
135                    llvm_index.find_associated_initial_value(variable.get_qualified_name())
136                };
137                if let Some(value) = value {
138                    let global_value = llvm
139                        .create_global_variable(module, &name, variable_type)
140                        .make_constant()
141                        .set_initial_value(Some(value), variable_type);
142                    local_llvm_index.associate_global(&name, global_value)?;
143                    local_llvm_index.insert_new_got_index(&name)?;
144                }
145            }
146        }
147    }
148    Ok(local_llvm_index)
149}

165    fn mangle_function(&self, implementation: &ImplementationIndexEntry) -> Result<String, CodegenError> {
166        let ctx = SectionMangler::function(implementation.get_call_name_for_ir().to_lowercase());
167
168        let params = self.index.get_declared_parameters(implementation.get_call_name());
169
170        let ctx = params.into_iter().try_fold(ctx, |ctx, param| -> Result<SectionMangler, CodegenError> {
171            let ty = section_names::mangle_type(
172                self.index,
173                self.index.get_effective_type_by_name(&param.data_type_name)?,
174            )?;
175            let parameter = match param.argument_type {
176                // TODO: We need to handle the `VariableType` enum as well - this describes the mode of
177                // argument passing, e.g. inout
178                index::ArgumentType::ByVal(_) => FunctionArgument::ByValue(ty),
179                index::ArgumentType::ByRef(_) => FunctionArgument::ByRef(ty),
180            };
181
182            Ok(ctx.with_parameter(parameter))
183        })?;
184
185        let return_ty = self
186            .index
187            .find_return_type(implementation.get_type_name())
188            .map(|ty| section_names::mangle_type(self.index, ty));
189
190        let ctx = match return_ty {
191            Some(rty) => ctx.with_return_type(rty?),
192            None => ctx,
193        };
194
195        Ok(ctx.mangle())
196    }

198    /// generates an empty llvm function for the given implementation, including all parameters and the return type
199    pub fn generate_implementation_stub(
200        &self,
201        implementation: &ImplementationIndexEntry,
202        module: &Module<'ink>,
203        debug: &mut DebugBuilderEnum<'ink>,
204        new_llvm_index: &mut LlvmTypedIndex<'ink>,
205    ) -> Result<FunctionValue<'ink>, CodegenError> {
206        let declared_parameters = self.index.get_declared_parameters(implementation.get_call_name());
207        let mut parameters = self.collect_parameters_for_implementation(implementation)?;
208        // if we are handling a method, take the first parameter as the instance
209        let instance = if implementation.is_method() { Some(parameters.remove(0)) } else { None };
210        let mut parameters = parameters
211            .iter()
212            .enumerate()
213            .map(|(i, p)| {
214                let param = declared_parameters.get(i);
215                let dti = param.map(|it| self.index.get_type_information_or_void(it.get_type_name()));
216                match param {
217                    Some(v)
218                        if v.is_in_parameter_by_ref() &&
219                        // parameters by ref will always be a pointer
220                        p.into_pointer_type().get_element_type().is_array_type() =>
221                    {
222                        // for by-ref array types we will generate a pointer to the arrays element type
223                        // not a pointer to array
224                        let ty = p
225                            .into_pointer_type()
226                            .get_element_type()
227                            .into_array_type()
228                            .get_element_type()
229                            .ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC));
230
231                        // set the new type for further codegen
232                        let _ = new_llvm_index.associate_type(v.get_type_name(), ty.into());
233
234                        ty.into()
235                    }
236                    _ => {
237                        dti.map(|it| {
238                            if !implementation.get_implementation_type().is_function_method_or_init() {
239                                return *p;
240                            }
241                            // for aggregate function parameters we will generate a pointer instead of the value type.
242                            // it will then later be memcopied into a locally allocated variable
243                            match it {
244                                DataTypeInformation::Struct { .. } => p
245                                    .into_struct_type()
246                                    .ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC))
247                                    .into(),
248                                DataTypeInformation::Array { .. } | DataTypeInformation::String { .. } => p
249                                    .into_array_type()
250                                    .get_element_type()
251                                    .ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC))
252                                    .into(),
253                                _ => *p,
254                            }
255                        })
256                        .unwrap_or(*p)
257                    }
258                }
259            })
260            .collect::<Vec<BasicMetadataTypeEnum>>();
261        // insert the instance as the first parameter
262        if let Some(instance) = instance {
263            parameters.insert(0, instance);
264        }
265
266        let return_type = self
267            .index
268            .find_return_type(implementation.get_type_name())
269            .and_then(|dt| self.index.find_effective_type(dt));
270
271        // see if we need to adapt the parameters list
272        let (return_type_llvm, parameters) = match return_type {
273            // function with an intrinsic return-type
274            Some(r_type) => (Some(self.llvm_index.get_associated_type(r_type.get_name())?), parameters),
275            // no return
276            None => (None, parameters),
277        };
278
279        let variadic = self.index.get_variadic_member(implementation.get_type_name());
280
281        let function_declaration = self.create_llvm_function_type(parameters, variadic, return_type_llvm)?;
282
283        let curr_f: FunctionValue<'_> =
284            module.add_function(&implementation.get_call_name_for_ir(), function_declaration, None);
285
286        let section_name = self.get_section(implementation)?;
287        curr_f.set_section(section_name.as_deref());
288
289        if implementation.get_implementation_type().is_project_init() {
290            self.add_global_constructor(module, curr_f)?;
291        }
292
293        let mut parameter_types = declared_parameters
294            .iter()
295            .map(|v| self.index.get_effective_type_or_void_by_name(v.get_type_name()))
296            .collect::<Vec<&DataType>>();
297
298        // If the implementation is a method, the first parameter is the instance
299        if let Some(class_name) = implementation.get_associated_class_name() {
300            if let Some(class_type) = self.index.find_type(class_name) {
301                parameter_types.insert(0, class_type);
302            }
303        } else if !implementation.get_implementation_type().is_function_method_or_init() {
304            //For non functions or methods, the first parameter is self
305            if let Some(type_name) = self.index.find_type(implementation.get_type_name()) {
306                parameter_types.insert(0, type_name);
307            }
308        }
309
310        let parent_function =
311            implementation.get_associated_class_name().and_then(|it| module.get_function(it));
312        let function_context = FunctionContext {
313            linking_context: implementation,
314            function: curr_f,
315            blocks: FxHashMap::default(),
316        };
317        debug.register_function(
318            (self.index, self.llvm_index),
319            &function_context,
320            return_type,
321            parent_function,
322            parameter_types.as_slice(),
323            implementation.get_location().get_line(),
324        );
325        Ok(curr_f)
326    }

328    /// Generates a global constructors entry
329    /// The entry contains the a call to the initializer function
330    fn add_global_constructor(
331        &self,
332        module: &Module<'ink>,
333        curr_f: FunctionValue<'ink>,
334    ) -> Result<(), CodegenError> {
335        //Create a constructor struct
336        let ctor_str = self.llvm.context.struct_type(
337            &[
338                //Priority
339                self.llvm.context.i32_type().as_basic_type_enum(),
340                // Function pointer
341                curr_f.as_global_value().as_basic_value_enum().get_type(),
342                //Data
343                self.llvm.context.i8_type().ptr_type(AddressSpace::default()).as_basic_type_enum(),
344            ],
345            false,
346        );
347
348        //Create an entry for the global constructor of the project
349        let str_value = ctor_str.const_named_struct(&[
350            self.llvm.context.i32_type().const_zero().as_basic_value_enum(),
351            curr_f.as_global_value().as_basic_value_enum(),
352            self.llvm.context.i8_type().ptr_type(AddressSpace::default()).const_zero().as_basic_value_enum(),
353        ]);
354        //Create an array with the global constructor as an entry
355        let arr = ctor_str.const_array(&[str_value]);
356        //Create the global constructors variable or fetch it and append to it if already
357        //availabe
358        let global_ctors = module.get_global("llvm.global_ctors").unwrap_or_else(|| {
359            module.add_global(arr.get_type().as_basic_type_enum(), None, "llvm.global_ctors")
360        });
361
362        global_ctors.set_initializer(&arr);
363        global_ctors.set_linkage(Linkage::Appending);
364        Ok(())
365    }
366    /// creates and returns all parameters for the given implementation
367    /// for functions, this method creates a full list of parameters, for other POUs
368    /// this method creates a single state-struct parameter
369    fn collect_parameters_for_implementation(
370        &self,
371        implementation: &ImplementationIndexEntry,
372    ) -> Result<Vec<BasicMetadataTypeEnum<'ink>>, CodegenError> {
373        if !implementation.implementation_type.is_function_method_or_init() {
374            let mut parameters = vec![];
375            let instance_struct_type: StructType = self
376                .llvm_index
377                .get_associated_pou_type(implementation.get_type_name())
378                .map(|it| it.into_struct_type())?;
379            parameters.push(instance_struct_type.ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)).into());
380            Ok(parameters)
381        } else {
382            let declared_params = self.index.get_declared_parameters(implementation.get_call_name());
383            //find the function's parameters
384            let mut parameters = declared_params
385                .iter()
386                .map(|v| self.llvm_index.get_associated_type(v.get_type_name()).map(Into::into))
387                .collect::<Result<Vec<BasicMetadataTypeEnum>, _>>()?;
388
389            if implementation.get_implementation_type() == &ImplementationType::Method {
390                let class_name =
391                    implementation.get_associated_class_name().expect("Method needs to have a class-name");
392                let instance_members_struct_type: StructType =
393                    self.llvm_index.get_associated_type(class_name).map(|it| it.into_struct_type())?;
394                parameters.insert(
395                    0,
396                    instance_members_struct_type.ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)).into(),
397                );
398            }
399
400            Ok(parameters)
401        }
402    }

404    /// generates a function for the given pou
405    pub fn generate_implementation(
406        &self,
407        implementation: &Implementation,
408        debug: &DebugBuilderEnum<'ink>,
409    ) -> Result<(), CodegenError> {
410        let context = self.llvm.context;
411        let mut local_index = LlvmTypedIndex::create_child(self.llvm_index);
412
413        let pou_name = &implementation.name;
414
415        let current_function = self.llvm_index.find_associated_implementation(pou_name).ok_or_else(|| {
416            CodegenError::new(
417                format!("Could not find generated stub for {pou_name}"),
418                &implementation.location,
419            )
420        })?;
421
422        //Unset the debug location so we ignore initialization logic
423        debug.unset_debug_location(&self.llvm);
424
425        //generate the body
426        let block = context.append_basic_block(current_function, "entry");
427
428        //Create all labels this function will have
429        let mut blocks = FxHashMap::default();
430        if let Some(labels) = self.index.get_labels(&implementation.name) {
431            for name in labels.keys() {
432                blocks.insert(name.to_string(), self.llvm.context.append_basic_block(current_function, name));
433            }
434        }
435        self.llvm.builder.position_at_end(block);
436        blocks.insert("entry".into(), block);
437
438        let function_context = FunctionContext {
439            linking_context: self.index.find_implementation_by_name(&implementation.name).ok_or_else(
440                || {
441                    CodegenError::new(
442                        format!("Could not find implementation for {}", &implementation.name),
443                        &implementation.location,
444                    )
445                },
446            )?,
447            function: current_function,
448            blocks,
449        };
450
451        if let PouType::Method { .. } = implementation.pou_type {
452            let class_name = implementation.type_name.split('.').collect::<Vec<&str>>()[0];
453            self.generate_local_pou_variable_accessors(
454                &mut local_index,
455                class_name,
456                &function_context,
457                &implementation.location,
458                debug,
459            )?;
460        }
461
462        // generate local variables
463        if implementation.pou_type.is_function_method_or_init() {
464            self.generate_local_function_arguments_accessors(
465                &mut local_index,
466                &implementation.type_name,
467                &function_context,
468                debug,
469            )?;
470        } else {
471            self.generate_local_pou_variable_accessors(
472                &mut local_index,
473                &implementation.type_name,
474                &function_context,
475                &implementation.location,
476                debug,
477            )?;
478        }
479        {
480            let pou_members =
481                self.index.get_pou_members(&implementation.type_name).iter().collect::<Vec<_>>();
482            //if this is a function, we need to initilialize the VAR-variables
483            if matches!(implementation.pou_type, PouType::Function | PouType::Method { .. }) {
484                self.generate_initialization_of_local_vars(
485                    &pou_members,
486                    &local_index,
487                    &function_context,
488                    debug,
489                )?;
490            } else {
491                //Generate temp variables
492                let members = pou_members.into_iter().filter(|it| it.is_temp()).collect::<Vec<_>>();
493                self.generate_initialization_of_local_vars(&members, &local_index, &function_context, debug)?;
494            }
495            let statement_gen = StatementCodeGenerator::new(
496                &self.llvm,
497                self.index,
498                self.annotations,
499                &local_index,
500                &function_context,
501                debug,
502            );
503            let (line, column) = implementation
504                .statements
505                .first()
506                .map(|it| (it.get_location().get_line_plus_one(), it.get_location().get_column()))
507                .unwrap_or_else(|| {
508                    (implementation.location.get_line_plus_one(), implementation.location.get_column())
509                });
510            //Set the debug location to the first statement in the body
511            debug.set_debug_location(&self.llvm, &function_context, line, column);
512            statement_gen.generate_body(&implementation.statements)?;
513            //TODO the return statement should be lowered
514            let line = implementation.end_location.get_line_plus_one();
515            let column = implementation.end_location.get_column();
516            debug.set_debug_location(&self.llvm, &function_context, line, column);
517            statement_gen.generate_return_statement()?;
518        }
519
520        Ok(())
521    }

523    /// TODO llvm.rs
524    /// generates a llvm `FunctionType` that takes the given list of `parameters` and
525    /// returns the given `return_type`
526    fn create_llvm_function_type(
527        &self,
528        parameters: Vec<BasicMetadataTypeEnum<'ink>>,
529        variadic: Option<&'cg VariableIndexEntry>,
530        return_type: Option<BasicTypeEnum<'ink>>,
531    ) -> Result<FunctionType<'ink>, CodegenError> {
532        // sized variadic is not considered a variadic function, but receives 2 extra parameters, size and a pointer
533        let is_var_args = variadic
534            .map(|it| it.get_varargs().map(|it| !it.is_sized()).unwrap_or_default())
535            .unwrap_or_default();
536        let size_and_type = self.get_variadic_size_and_pointer(variadic);
537        let mut params = parameters;
538        if let Some(sized_variadics) = size_and_type {
539            params.extend_from_slice(&sized_variadics);
540        };
541
542        match return_type {
543            Some(enum_type) if enum_type.is_int_type() => {
544                Ok(enum_type.into_int_type().fn_type(&params, is_var_args))
545            }
546            Some(enum_type) if enum_type.is_float_type() => {
547                Ok(enum_type.into_float_type().fn_type(&params, is_var_args))
548            }
549            Some(enum_type) if enum_type.is_array_type() => {
550                Ok(enum_type.into_array_type().fn_type(&params, is_var_args))
551            }
552            Some(enum_type) if enum_type.is_pointer_type() => {
553                Ok(enum_type.into_pointer_type().fn_type(&params, is_var_args))
554            }
555            Some(enum_type) if enum_type.is_struct_type() => {
556                Ok(enum_type.into_struct_type().fn_type(&params, is_var_args))
557            }
558            None => Ok(self.llvm.context.void_type().fn_type(&params, is_var_args)),
559            _ => Err(Diagnostic::codegen_error(
560                format!("Unsupported return type {return_type:?}"),
561                SourceLocation::undefined(),
562            )
563            .into()),
564        }
565    }

567    /// generates a load-statement for the given members of a function
568    fn generate_local_function_arguments_accessors(
569        &self,
570        index: &mut LlvmTypedIndex<'ink>,
571        type_name: &str,
572        function_context: &FunctionContext<'ink, '_>,
573        debug: &DebugBuilderEnum<'ink>,
574    ) -> Result<(), CodegenError> {
575        let members = self.index.get_pou_members(type_name);
576        //Generate reference to parameter
577        // cannot use index from members because return and temp variables may not be considered for index in build_struct_gep
578        // eagerly handle the return-variable
579        let mut params_iter = function_context.function.get_param_iter();
580        // if we are in a method, skip the first parameter (the instance)
581        if matches!(function_context.linking_context.get_implementation_type(), ImplementationType::Method) {
582            params_iter.next();
583        }
584        if let Some(ret_v) = members.iter().find(|it| it.is_return()) {
585            let return_type = index.get_associated_type(ret_v.get_type_name())?;
586            let return_variable = self.llvm.create_local_variable(type_name, &return_type)?;
587            index.associate_loaded_local_variable(type_name, ret_v.get_name(), return_variable)?;
588        }
589
590        // handle all parameters (without return!)
591        for m in members.iter().filter(|it| !(it.is_return() || it.is_var_external())) {
592            let parameter_name = m.get_name();
593
594            let (name, variable) = if m.is_parameter() {
595                let ptr_value = params_iter
596                    .next()
597                    .ok_or_else(|| CodegenError::from(Diagnostic::missing_function(&m.source_location)))?;
598                let member_type_name = m.get_type_name();
599                let type_info = self.index.get_type_information_or_void(member_type_name);
600                let ty = index.get_associated_type(member_type_name)?;
601                let ptr = self.llvm.create_local_variable(m.get_name(), &ty)?;
602                if let Some(block) = self.llvm.builder.get_insert_block() {
603                    debug.add_variable_declaration(
604                        m.get_qualified_name(),
605                        ptr,
606                        function_context,
607                        block,
608                        m.source_location.get_line(),
609                        m.source_location.get_column(),
610                    );
611                }
612
613                if matches!(m.argument_type, ArgumentType::ByVal(VariableType::Input))
614                    && type_info.is_aggregate()
615                {
616                    // a by-value aggregate type => we need to memcpy the data into the local variable
617                    let ty = if ty.is_array_type() {
618                        ty.into_array_type()
619                            .get_element_type()
620                            .ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC))
621                    } else {
622                        ty.into_struct_type().ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC))
623                    };
624                    let bitcast = self.llvm.builder.build_bit_cast(ptr, ty, "bitcast")?.into_pointer_value();
625                    let (size, alignment) = if let DataTypeInformation::String { size, encoding } = type_info
626                    {
627                        // since passed string args might be larger than the local acceptor, we need to first memset the local variable to 0
628                        let size = size
629                            .as_int_value(self.index)
630                            .map_err(|err| CodegenError::new(err.as_str(), &m.source_location))?
631                            as u64;
632                        let char_width = encoding.get_bytes_per_char();
633                        self.llvm.builder.build_memset(
634                            bitcast,
635                            char_width,
636                            self.llvm.context.i8_type().const_zero(),
637                            self.llvm.context.i64_type().const_int(size * char_width as u64, true),
638                        )?;
639                        (
640                            // we then reduce the amount of bytes to be memcopied by the equivalent of one grapheme in bytes to preserve the null-terminator
641                            self.llvm.context.i64_type().const_int((size - 1) * char_width as u64, true),
642                            char_width,
643                        )
644                    } else {
645                        let Some(size) = index.get_associated_type(member_type_name)?.size_of() else {
646                            // XXX: can this still fail at this point? might be `unreachable`
647                            return Err(CodegenError::new(
648                                "Unable to determine type size",
649                                &m.source_location,
650                            ));
651                        };
652                        (size, 1)
653                    };
654                    self.llvm.builder.build_memcpy(
655                        bitcast,
656                        alignment,
657                        ptr_value.into_pointer_value(),
658                        alignment,
659                        size,
660                    )?;
661                } else {
662                    self.llvm.builder.build_store(ptr, ptr_value)?;
663                };
664
665                (parameter_name, ptr)
666            } else {
667                let temp_type = index.get_associated_type(m.get_type_name())?;
668                let value = self.llvm.create_local_variable(parameter_name, &temp_type)?;
669                (parameter_name, value)
670            };
671
672            index.associate_loaded_local_variable(type_name, name, variable)?;
673        }
674
675        Ok(())
676    }

678    /// generates a load-statement for the given members
679    /// for pous that take a struct-state-variable
680    fn generate_local_pou_variable_accessors(
681        &self,
682        index: &mut LlvmTypedIndex<'ink>,
683        type_name: &str,
684        function_context: &FunctionContext<'ink, '_>,
685        location: &SourceLocation,
686        debug: &DebugBuilderEnum<'ink>,
687    ) -> Result<(), CodegenError> {
688        let members = self.index.get_pou_members(type_name);
689        let param_pointer = function_context
690            .function
691            .get_nth_param(0)
692            .map(BasicValueEnum::into_pointer_value)
693            .ok_or_else(|| CodegenError::from(Diagnostic::missing_function(location)))?;
694        //Generate POU struct declaration for debug
695        if let Some(block) = self.llvm.builder.get_insert_block() {
696            debug.add_variable_declaration(
697                type_name,
698                param_pointer,
699                function_context,
700                block,
701                location.get_line(),
702                location.get_column(),
703            );
704        }
705
706        if ((function_context.linking_context.is_method() || function_context.linking_context.is_action())
707            && self.index.find_pou(type_name).is_some_and(|it| it.is_function_block()))
708            || function_context.linking_context.get_implementation_type()
709                == &ImplementationType::FunctionBlock
710        {
711            let alloca = self.llvm.builder.build_alloca(param_pointer.get_type(), "this")?;
712            self.llvm.builder.build_store(alloca, param_pointer)?;
713            index.associate_loaded_local_variable(type_name, "__THIS", alloca)?;
714        }
715
716        //Generate reference to parameter
717        // cannot use index from members because return and temp variables may not be considered for index in build_struct_gep
718        let mut var_count = 0;
719        for m in members.iter().filter(|it| !it.is_var_external()) {
720            let parameter_name = m.get_name();
721            //TODO: this is not creating local variables
722            let (name, variable) = if m.is_temp() || m.is_return() {
723                let temp_type = index.get_associated_type(m.get_type_name())?;
724                (parameter_name, self.llvm.create_local_variable(parameter_name, &temp_type)?)
725            } else {
726                let ptr = self
727                    .llvm
728                    .builder
729                    .build_struct_gep(param_pointer, var_count as u32, parameter_name)
730                    .expect(INTERNAL_LLVM_ERROR);
731
732                var_count += 1;
733
734                (parameter_name, ptr)
735            };
736
737            index.associate_loaded_local_variable(type_name, name, variable)?;
738        }
739
740        Ok(())
741    }

743    /// generates assignment statements for initialized variables in the VAR-block
744    ///
745    /// - `blocks` - all declaration blocks of the current pou
746    fn generate_initialization_of_local_vars(
747        &self,
748        variables: &[&VariableIndexEntry],
749        local_llvm_index: &LlvmTypedIndex,
750        function_context: &FunctionContext,
751        debug: &DebugBuilderEnum<'ink>,
752    ) -> Result<(), CodegenError> {
753        let variables_with_initializers =
754            variables.iter().filter(|it| it.is_local() || it.is_temp() || it.is_return());
755
756        let exp_gen = ExpressionCodeGenerator::new_context_free(
757            &self.llvm,
758            self.index,
759            self.annotations,
760            local_llvm_index,
761        );
762
763        for variable in variables_with_initializers {
764            //get the loaded_ptr for the parameter and store right in it
765            if let Some(left) =
766                local_llvm_index.find_loaded_associated_variable_value(variable.get_qualified_name())
767            {
768                if let Some(block) = self.llvm.builder.get_insert_block() {
769                    debug.add_variable_declaration(
770                        variable.get_qualified_name(),
771                        left,
772                        function_context,
773                        block,
774                        variable.source_location.get_line(),
775                        variable.source_location.get_column(),
776                    );
777                }
778
779                if self
780                    .index
781                    .find_effective_type_by_name(variable.get_type_name())
782                    .map(|it| it.get_type_information())
783                    .is_some_and(|it| it.is_reference_to() || it.is_alias())
784                {
785                    // aliases and reference to variables have special handling for initialization. initialize with a nullpointer
786                    self.llvm.builder.build_store(
787                        left,
788                        left.get_type().get_element_type().into_pointer_type().const_null(),
789                    )?;
790                    continue;
791                };
792                let right_stmt =
793                    self.index.get_const_expressions().maybe_get_constant_statement(&variable.initial_value);
794                self.generate_variable_initializer(variable, left, right_stmt, &exp_gen)?;
795            } else {
796                return Err(Diagnostic::cannot_generate_initializer(
797                    variable.get_qualified_name(),
798                    &variable.source_location,
799                )
800                .into());
801            }
802        }
803        Ok(())
804    }

806    /// initializes the variable represented by `variable` by storing into the given `variable_to_initialize` pointer using either
807    /// the optional `initializer_statement` (hence code like: `variable : type := initializer_statement`), or determine the initial
808    /// value with the help of the `variable`'s index entry by e.g. looking for a default value of the variable's type
809    fn generate_variable_initializer(
810        &self,
811        variable: &&VariableIndexEntry,
812        variable_to_initialize: PointerValue,
813        initializer_statement: Option<&AstNode>,
814        exp_gen: &ExpressionCodeGenerator,
815    ) -> Result<(), CodegenError> {
816        self.llvm.generate_variable_initializer(
817            self.llvm_index,
818            self.index,
819            (variable.get_qualified_name(), variable.get_type_name(), &variable.source_location),
820            variable_to_initialize,
821            initializer_statement,
822            exp_gen,
823        )
824    }

826    fn get_variadic_size_and_pointer(
827        &self,
828        variadic: Option<&'cg VariableIndexEntry>,
829    ) -> Option<[BasicMetadataTypeEnum<'ink>; 2]> {
830        if let (Some(var), Some(VarArgs::Sized(Some(type_name)))) =
831            (variadic, variadic.and_then(VariableIndexEntry::get_varargs))
832        {
833            //Create a size parameter of type i32 (DINT)
834            let size = self.llvm_index.find_associated_type(DINT_TYPE).map(Into::into)?;
835
836            let ty = self
837                .llvm_index
838                .find_associated_type(type_name)
839                .map(|it| {
840                    if it.is_array_type() && var.get_declaration_type().is_by_ref() {
841                        // the declaration for array types passed by ref are generatad as pointer to element type
842                        // we need to match the declaration
843                        it.into_array_type().get_element_type()
844                    } else {
845                        it
846                    }
847                })
848                .map(|it| {
849                    if var.get_declaration_type().is_by_ref() {
850                        // variadic by ref will result in a double pointer
851                        it.ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)).as_basic_type_enum()
852                    } else {
853                        it
854                    }
855                })?
856                .ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC))
857                .into();
858
859            Some([size, ty])
860        } else {
861            None
862        }
863    }
