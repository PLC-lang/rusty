1use crate::index::Index;
2use crate::typesystem::{self, DataTypeInformation, StringEncoding, TypeSize};
3use plc_diagnostics::diagnostics::Diagnostic;
4use section_mangler::{StringEncoding as SectionStringEncoding, Type};
5
6pub fn mangle_type(index: &Index, ty: &typesystem::DataType) -> Result<section_mangler::Type, Diagnostic> {
7    let access_inner = |ty_name| mangle_type(index, index.get_effective_type_by_name(ty_name)?);
8
9    // TODO: This is a bit ugly because we keep dereferencing references to Copy types like
10    // bool, u32, etc, because `DataTypeInformation::Pointer` keeps a `String` which is not
11    // Copy. the alternative is for section_mangle::Type to keep references everywhere, and
12    // have a lifetime generic parameter, e.g. `section_mangler::Type<'a>` - which is also
13    // annoying.
14    let mangled = match ty.get_type_information() {
15        DataTypeInformation::Void => Type::Void,
16        DataTypeInformation::Integer { signed, size, semantic_size, .. } => {
17            Type::Integer { signed: *signed, size: *size, semantic_size: *semantic_size }
18        }
19        DataTypeInformation::Float { size, .. } => Type::Float { size: *size },
20        DataTypeInformation::String { size: TypeSize::LiteralInteger(size), encoding } => {
21            let encoding = match encoding {
22                StringEncoding::Utf8 => SectionStringEncoding::Utf8,
23                StringEncoding::Utf16 => SectionStringEncoding::Utf16,
24            };
25
26            Type::String { size: *size as usize, encoding }
27        }
28        DataTypeInformation::Pointer { inner_type_name, .. } => {
29            Type::Pointer { inner: Box::new(access_inner(inner_type_name)?) }
30        }
31        DataTypeInformation::Enum { referenced_type, variants, .. } => {
32            Type::Enum { referenced_type: Box::new(access_inner(referenced_type)?), elements: variants.len() }
33        }
34        DataTypeInformation::Struct { members, .. } => Type::Struct {
35            members: members.iter().try_fold(Vec::new(), |mut acc, m| -> Result<Vec<Type>, Diagnostic> {
36                let inner = access_inner(m.get_type_name())?;
37
38                acc.push(inner);
39
40                Ok(acc)
41            })?,
42        },
43        DataTypeInformation::Array { inner_type_name, .. } => {
44            Type::Array { inner: Box::new(access_inner(inner_type_name)?) }
45        }
46        // FIXME: Is that correct?
47        // For code generation, the actual range does not matter - it is not a breaking change
48        // if a variable's range changes, at least not for codegen, since the underlying type will stay
49        // the same. Therefore, only encode it as its underlying type.
50        DataTypeInformation::SubRange { referenced_type, .. }
51        // Similarly, we do not care about the alias - only the type which is being codegen'd
52        | DataTypeInformation::Alias { referenced_type, .. } => access_inner(referenced_type)?,
53        DataTypeInformation::Generic { .. } => {
54            // FIXME: Is that correct?
55            unreachable!("generic types should not exist at codegen")
56        }
57         // FIXME: For now, encode all unknown types as "void" since this is not required for
58        // execution. Not doing so (and doing an `unreachable!()` for example) obviously causes
59        // failures, because complex types are already implemented in the compiler.
60        _ => Type::Void,
61    };
62
63    Ok(mangled)
64}
