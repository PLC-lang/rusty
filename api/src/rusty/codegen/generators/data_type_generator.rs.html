46/// generates the llvm-type for the given data-type and registers it at the index
47/// this function may create and register a ...
48/// - Struct type for a STRUCT
49/// - global variables for enum-elements
50/// - an alias index entry for sub-range types
51/// - Array type for arrays
52/// - array type for sized Strings
53pub fn generate_data_types<'ink>(
54    llvm: &Llvm<'ink>,
55    debug: &mut DebugBuilderEnum<'ink>,
56    dependencies: &FxIndexSet<Dependency>,
57    index: &Index,
58    annotations: &AstAnnotations,
59) -> Result<LlvmTypedIndex<'ink>, Diagnostic> {
60    let mut types = vec![];
61    let mut pou_types = vec![];
62
63    // Always add the builtin types
64    let builtins = typesystem::get_builtin_types();
65    for builtin in &builtins {
66        types.push((builtin.get_name(), builtin));
67    }
68
69    for dep in dependencies {
70        if let Dependency::Datatype(name) = dep {
71            if let Some(pou) = index.find_pou(name) {
72                if !pou.is_generic() && !pou.is_action() {
73                    pou_types.push((name.as_str(), pou.get_instance_struct_type_or_void(index)));
74                }
75            } else if let Some(datatype) = index.find_type(name) {
76                if !datatype.get_type_information().is_generic(index) {
77                    types.push((name, datatype))
78                }
79            }
80        }
81    }
82
83    let mut generator =
84        DataTypeGenerator { llvm, debug, index, annotations, types_index: LlvmTypedIndex::default() };
85
86    // first create all STUBs for struct types (empty structs)
87    // and associate them in the llvm index
88    for (name, user_type) in &types {
89        if let DataTypeInformation::Struct { name: struct_name, .. } = user_type.get_type_information() {
90            generator.types_index.associate_type(name, llvm.create_struct_stub(struct_name).into())?;
91        }
92    }
93    // pou_types will always be struct
94    for (name, user_type) in &pou_types {
95        if let DataTypeInformation::Struct { name: struct_name, .. } = user_type.get_type_information() {
96            generator.types_index.associate_pou_type(name, llvm.create_struct_stub(struct_name).into())?;
97        }
98    }
99
100    // now create all other types (enum's, arrays, etc.)
101    for (name, user_type) in &types {
102        let gen_type = generator.create_type(name, user_type)?;
103        generator.types_index.associate_type(name, gen_type)?
104        //Get and associate debug type
105    }
106
107    for (name, user_type) in &pou_types {
108        let gen_type = generator.create_type(name, user_type)?;
109        generator.types_index.associate_pou_type(name, gen_type)?
110    }
111
112    // Combine the types and pou_types into a single Vector
113    let mut types_to_init = VecDeque::new();
114    types_to_init.extend(types);
115    types_to_init.extend(pou_types);
116    // now since all types should be available in the llvm index, we can think about constructing and associating
117    for (_, user_type) in &types_to_init {
118        //Expand all types
119        generator.expand_opaque_types(user_type)?;
120    }
121
122    // Now generate debug information for all types
123    generator.generate_debug_types(&types_to_init)?;
124
125    let mut tries = 0;
126    let mut errors = FxHashMap::default();
127
128    // If the tries are equal to the number of types remaining, it means we failed to resolve
129    // anything
130    while tries < types_to_init.len() {
131        //Take the current element,
132        if let Some((name, user_type)) = types_to_init.pop_front() {
133            errors.remove(name);
134            //try to resolve it
135            match generator.generate_initial_value(user_type) {
136                Err(err) => {
137                    tries += 1;
138                    types_to_init.push_back((name, user_type));
139                    errors.insert(name, err);
140                }
141                Ok(init_value) => {
142                    if let Some(init_value) = init_value {
143                        if let Err(err) = generator.types_index.associate_initial_value(name, init_value) {
144                            //If it fails, push it back into the list
145                            tries += 1;
146                            types_to_init.push_back((name, user_type));
147                            errors.insert(name, err);
148                        } else {
149                            tries = 0;
150                        }
151                    }
152                }
153            }
154        }
155    }
156    //If we didn't resolve anything this cycle, report the remaining issues and exit
157    if !types_to_init.is_empty() {
158        //Report each error as a new diagnostic, add the type's location as related to the error
159        let diags = types_to_init
160            .into_iter()
161            .map(|(name, ty)| {
162                errors
163                    .remove(name)
164                    .map(|diag| diag.with_secondary_location(&ty.location))
165                    .unwrap_or_else(|| Diagnostic::cannot_generate_initializer(name, &ty.location))
166            })
167            .collect::<Vec<_>>();
168        if !diags.is_empty() {
169            //Report the operation failure
170            return Err(Diagnostic::new("Some initial values were not generated")
171                .with_error_code("E075")
172                .with_sub_diagnostics(diags)); // FIXME: these sub-diagnostics aren't printed to the console
173        }
174    }
175    Ok(generator.types_index)
176}

178impl<'ink> DataTypeGenerator<'ink, '_> {
179    /// generates the members of an opaque struct and associates its initial values
180    fn expand_opaque_types(&mut self, data_type: &DataType) -> Result<(), Diagnostic> {
181        let information = data_type.get_type_information();
182        if let DataTypeInformation::Struct { source, members, .. } = information {
183            let members = members
184                .iter()
185                .filter(|it| !(it.is_temp() || it.is_return() || it.is_var_external()))
186                .map(|m| self.types_index.get_associated_type(m.get_type_name()))
187                .collect::<Result<Vec<BasicTypeEnum>, Diagnostic>>()?;
188
189            let struct_type = match source {
190                StructSource::Pou(..) => self.types_index.get_associated_pou_type(data_type.get_name()),
191                StructSource::OriginalDeclaration | StructSource::Internal(_) => {
192                    self.types_index.get_associated_type(data_type.get_name())
193                }
194            }
195            .map(BasicTypeEnum::into_struct_type)?;
196
197            struct_type.set_body(members.as_slice(), false);
198        }
199        Ok(())
200    }

202    /// Creates an llvm type to be associated with the given data type.
203    /// Generates only an opaque type for structs.
204    /// Eagerly generates but does not associate nested array and referenced aliased types
205    fn create_type(&mut self, name: &str, data_type: &DataType) -> Result<AnyTypeEnum<'ink>, Diagnostic> {
206        let information = data_type.get_type_information();
207        match information {
208            DataTypeInformation::Struct { source, .. } => match source {
209                StructSource::Pou(..) => self
210                    .types_index
211                    .get_associated_pou_type(data_type.get_name())
212                    .map(|res| res.as_any_type_enum()),
213                StructSource::OriginalDeclaration => self
214                    .types_index
215                    .get_associated_type(data_type.get_name())
216                    .map(|res| res.as_any_type_enum()),
217                StructSource::Internal(_) => self
218                    .types_index
219                    .get_associated_type(data_type.get_name())
220                    .map(|res| res.as_any_type_enum()),
221            },
222
223            // We distinguish between two types of arrays, normal and variable length ones.
224            // Variable Length Arrays are defined by having unknown sizes at compile time, thus we handle
225            // these two cases based on whether the {start,end}-offset equals `TypeSize::Undetermined`.
226            DataTypeInformation::Array { inner_type_name, dimensions, .. } => {
227                if dimensions.iter().any(|dimension| dimension.is_undetermined()) {
228                    self.create_type(inner_type_name, self.index.get_type(inner_type_name)?)
229                } else {
230                    self.index
231                        .get_effective_type_by_name(inner_type_name)
232                        .and_then(|inner_type| self.create_type(inner_type_name, inner_type))
233                        .and_then(|inner_type| self.create_nested_array_type(inner_type, dimensions))
234                        .map(|it| it.as_any_type_enum())
235                }
236            }
237            DataTypeInformation::Integer { size, .. } => {
238                Ok(get_llvm_int_type(self.llvm.context, *size, name).into())
239            }
240            DataTypeInformation::Enum { name, referenced_type, .. } => {
241                let effective_type = self.index.get_effective_type_or_void_by_name(referenced_type);
242                if let DataTypeInformation::Integer { .. } = effective_type.get_type_information() {
243                    self.create_type(name, effective_type)
244                } else {
245                    Err(Diagnostic::new(format!(
246                        "Invalid type nature for generic argument. {} is no `ANY_INT`.",
247                        effective_type.get_name()
248                    ))
249                    .with_error_code("E062"))
250                }
251            }
252            DataTypeInformation::Float { size, .. } => {
253                Ok(get_llvm_float_type(self.llvm.context, *size, name).into())
254            }
255            DataTypeInformation::String { size, encoding } => {
256                let base_type = if *encoding == StringEncoding::Utf8 {
257                    self.llvm.context.i8_type()
258                } else {
259                    self.llvm.context.i16_type()
260                };
261
262                let string_size = size
263                    .as_int_value(self.index)
264                    .map_err(|it| Diagnostic::codegen_error(it.as_str(), SourceLocation::undefined()))?
265                    as u32;
266                Ok(base_type.array_type(string_size).into())
267            }
268            DataTypeInformation::SubRange { referenced_type, .. }
269            | DataTypeInformation::Alias { referenced_type, .. } => self
270                .index
271                .get_effective_type_by_name(referenced_type)
272                .and_then(|data_type| self.create_type(name, data_type)),
273            DataTypeInformation::Void => Ok(get_llvm_int_type(self.llvm.context, 32, "Void").into()),
274            DataTypeInformation::Pointer { inner_type_name, .. } => {
275                let inner_type = if information.is_function_pointer() {
276                    self.create_function_type(inner_type_name)?.as_any_type_enum()
277                } else {
278                    self.create_type(inner_type_name, self.index.get_type(inner_type_name)?)?
279                };
280
281                Ok(inner_type.create_ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)).into())
282            }
283            DataTypeInformation::Generic { .. } => {
284                unreachable!("Generic types should not be generated")
285            }
286        }
287    }

289    fn create_function_type(&mut self, method_name: &str) -> Result<FunctionType<'ink>, Diagnostic> {
290        let return_type = self
291            .types_index
292            .find_associated_type(self.index.get_return_type_or_void(method_name).get_name())
293            .map(|opt| opt.as_any_type_enum())
294            .unwrap_or(self.llvm.context.void_type().as_any_type_enum());
295
296        let mut parameter_types = Vec::new();
297
298        match self.index.find_pou(method_name) {
299            Some(PouIndexEntry::Method { parent_name, .. }) => {
300                let ty = self.types_index.get_associated_type(parent_name).expect("must exist");
301                let ty_ptr = ty.ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)).into();
302
303                // Methods are defined as functions in the LLVM IR, but carry the underlying POU type as their
304                // first parameter to operate on them, hence push the POU type to the very first position.
305                parameter_types.push(ty_ptr);
306
307                for parameter in self.index.get_declared_parameters(method_name) {
308                    // Instead of relying on the LLVM index, we create data-types on the fly here because some
309                    // types have not yet been visited and as a result may not be in the index. For example at
310                    // the time of writing this the index was not able to find a input parameter of type
311                    // `__auto_pointer_to_DINT`, consequently panicking
312                    let ty = self.create_type(
313                        parameter.get_name(),
314                        self.index.get_type(&parameter.data_type_name).expect("must exist"),
315                    )?;
316
317                    parameter_types.push(ty.try_into().unwrap());
318                }
319            }
320
321            Some(PouIndexEntry::FunctionBlock { name, .. }) => {
322                let ty = self.types_index.get_associated_type(name).expect("must exist");
323                let ty_ptr = ty.ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)).into();
324
325                // Function blocks are a bit "weird" in that they only expect an instance argument even if
326                // they define input, output and/or inout parameters. Effectively, while being methods per-se,
327                // their calling convention differs from regular methods.
328                parameter_types.push(ty_ptr);
329            }
330
331            _ => unreachable!("internal error, invalid method call"),
332        }
333
334        let fn_type = match return_type {
335            AnyTypeEnum::ArrayType(value) => value.fn_type(parameter_types.as_slice(), false),
336            AnyTypeEnum::FloatType(value) => value.fn_type(parameter_types.as_slice(), false),
337            AnyTypeEnum::IntType(value) => value.fn_type(parameter_types.as_slice(), false),
338            AnyTypeEnum::PointerType(value) => value.fn_type(parameter_types.as_slice(), false),
339            AnyTypeEnum::StructType(value) => value.fn_type(parameter_types.as_slice(), false),
340            AnyTypeEnum::VectorType(value) => value.fn_type(parameter_types.as_slice(), false),
341            AnyTypeEnum::VoidType(value) => value.fn_type(parameter_types.as_slice(), false),
342
343            AnyTypeEnum::FunctionType(_) => unreachable!(),
344        };
345
346        Ok(fn_type)
347    }

349    fn generate_initial_value(
350        &mut self,
351        data_type: &DataType,
352    ) -> Result<Option<BasicValueEnum<'ink>>, Diagnostic> {
353        let information = data_type.get_type_information();
354        match information {
355            DataTypeInformation::Struct { source, members, .. } => {
356                let member_names_and_initializers = members
357                    .iter()
358                    .filter(|it| it.get_variable_type() != VariableType::Temp)
359                    .map(|it| {
360                        self.generate_initial_value_for_variable(it).and_then(|v| match v {
361                            Some(v) => Ok((it.get_qualified_name(), v)),
362                            None => self
363                                .types_index
364                                .get_associated_type(it.get_type_name())
365                                .map(get_default_for)
366                                .map(|v| (it.get_qualified_name(), v)),
367                        })
368                    })
369                    .collect::<Result<Vec<(&str, BasicValueEnum)>, Diagnostic>>()?;
370
371                let mut member_values: Vec<BasicValueEnum> = Vec::new();
372                for (name, v) in &member_names_and_initializers {
373                    self.types_index.associate_initial_value(name, *v)?;
374                    member_values.push(*v);
375                }
376
377                let struct_type = match source {
378                    StructSource::Pou(..) => self.types_index.get_associated_pou_type(data_type.get_name()),
379                    StructSource::OriginalDeclaration => {
380                        self.types_index.get_associated_type(data_type.get_name())
381                    }
382                    StructSource::Internal(_) => self.types_index.get_associated_type(data_type.get_name()),
383                }?
384                .into_struct_type();
385
386                Ok(Some(struct_type.const_named_struct(&member_values).as_basic_value_enum()))
387            }
388            DataTypeInformation::Array { .. } => self.generate_array_initializer(
389                data_type,
390                |stmt| matches!(stmt.stmt, AstStatement::Literal(AstLiteral::Array { .. })),
391                "LiteralArray",
392            ),
393            DataTypeInformation::String { .. } => self.generate_array_initializer(
394                data_type,
395                |stmt| matches!(stmt.stmt, AstStatement::Literal(AstLiteral::String { .. })),
396                "LiteralString",
397            ),
398            DataTypeInformation::SubRange { referenced_type, .. } => {
399                self.generate_initial_value_for_type(data_type, referenced_type)
400            }
401            DataTypeInformation::Alias { referenced_type, .. } => {
402                self.generate_initial_value_for_type(data_type, referenced_type)
403            }
404            //all other types (scalars, pointer and void)
405            _ => Ok(None),
406        }
407    }

409    /// generates and returns an optional inital value at the given declared variable
410    /// if no initial value is defined, it returns the initial value of the variable's
411    /// datatype or Ok(None) if the type also has no declared default value
412    fn generate_initial_value_for_variable(
413        &mut self,
414        variable: &VariableIndexEntry,
415    ) -> Result<Option<BasicValueEnum<'ink>>, Diagnostic> {
416        let initializer = variable
417            .initial_value
418            .and_then(|it| self.index.get_const_expressions().get_constant_statement(&it));
419        self.generate_initializer(variable.get_qualified_name(), initializer, variable.get_type_name())
420    }

422    /// generates and returns an optional inital value at the given dataType
423    /// if no initial value is defined, it returns  an optional initial value of
424    /// the aliased type (referenced_type)
425    fn generate_initial_value_for_type(
426        &mut self,
427        data_type: &DataType,
428        referenced_type: &str,
429    ) -> Result<Option<BasicValueEnum<'ink>>, Diagnostic> {
430        self.generate_initializer(
431            data_type.get_name(),
432            self.index.get_const_expressions().maybe_get_constant_statement(&data_type.initial_value),
433            referenced_type,
434        )
435    }

437    /// generates the given initializer-statement if one is present
438    /// if no initializer is present, it returns an optional default value
439    /// of the given datatype.
440    /// Errors will be reported using the given qualified_name
441    fn generate_initializer(
442        &mut self,
443        qualified_name: &str,
444        initializer: Option<&AstNode>,
445        data_type_name: &str,
446    ) -> Result<Option<BasicValueEnum<'ink>>, Diagnostic> {
447        if let Some(initializer) = initializer {
448            let generator = ExpressionCodeGenerator::new_context_free(
449                self.llvm,
450                self.index,
451                self.annotations,
452                &self.types_index,
453            );
454
455            let lhs_type = self.index.get_type_information_or_void(data_type_name);
456            if !initializer.is_literal() && (lhs_type.is_pointer() || lhs_type.is_alias()) {
457                return Ok(self
458                    .types_index
459                    .find_associated_type(lhs_type.get_name())
460                    .map(|it| it.const_zero().as_basic_value_enum()));
461            };
462
463            generator
464                .generate_expression(initializer)
465                .map(Some)
466                .map_err(|_| Diagnostic::cannot_generate_initializer(qualified_name, initializer))
467        } else {
468            // if there's no initializer defined for this alias, we go and check the aliased type for an initial value
469            self.index
470                .get_type(data_type_name)
471                .and_then(|referenced_data_type| self.generate_initial_value(referenced_data_type))
472        }
473    }

475    /// generates and associates the given array-datatype (used for arrays and strings)
476    fn generate_array_initializer(
477        &self,
478        data_type: &DataType,
479        predicate: fn(&AstNode) -> bool,
480        expected_ast: &str,
481    ) -> Result<Option<BasicValueEnum<'ink>>, Diagnostic> {
482        if let Some(initializer) =
483            self.index.get_const_expressions().maybe_get_constant_statement(&data_type.initial_value)
484        {
485            if predicate(initializer) {
486                let generator = ExpressionCodeGenerator::new_context_free(
487                    self.llvm,
488                    self.index,
489                    self.annotations,
490                    &self.types_index,
491                );
492                Ok(Some(generator.generate_literal(initializer)?.get_basic_value_enum()))
493            } else {
494                Err(Diagnostic::codegen_error(
495                    format!("Expected {expected_ast} but found {initializer:?}"),
496                    initializer,
497                ))
498            }
499        } else {
500            Ok(None)
501        }
502    }

504    /// creates the llvm types for a multi-dimensional array
505    ///
506    /// an array with multiple dimensions will be flattened into a long
507    /// 1-dimensional array.
508    /// e.g. `arr: ARRAY[0..2, 0..2] OF INT` produces the same result like
509    /// `arr: ARRAY[0..3] OF INT`.
510    fn create_nested_array_type(
511        &self,
512        inner_type: AnyTypeEnum<'ink>,
513        dimensions: &[Dimension],
514    ) -> Result<AnyTypeEnum<'ink>, Diagnostic> {
515        let len = dimensions
516            .iter()
517            .map(|dimension| {
518                dimension
519                    .get_length(self.index)
520                    .map_err(|it| Diagnostic::codegen_error(it.as_str(), SourceLocation::undefined()))
521            })
522            .collect::<Result<Vec<u32>, Diagnostic>>()?
523            .into_iter()
524            .reduce(|a, b| a * b)
525            .ok_or_else(|| {
526                Diagnostic::codegen_error("Invalid array dimensions", SourceLocation::undefined())
527            })?;
528
529        let result = match inner_type {
530            AnyTypeEnum::IntType(ty) => ty.array_type(len),
531            AnyTypeEnum::FloatType(ty) => ty.array_type(len),
532            AnyTypeEnum::StructType(ty) => ty.array_type(len),
533            AnyTypeEnum::ArrayType(ty) => ty.array_type(len),
534            AnyTypeEnum::PointerType(ty) => ty.array_type(len),
535            AnyTypeEnum::VectorType(ty) => ty.array_type(len),
536
537            AnyTypeEnum::FunctionType(_) => unimplemented!("function types are not supported in arrays"),
538            AnyTypeEnum::VoidType(_) => unimplemented!("void types not supported in arrays"),
539        }
540        .as_any_type_enum();
541
542        Ok(result)
543    }

545    fn generate_debug_types(&mut self, types: &VecDeque<(&str, &DataType)>) -> Result<(), Diagnostic> {
546        for (_, data_type) in types.iter().filter(|(_, data_type)| data_type.is_backed_by_struct()) {
547            self.debug.register_debug_type(data_type.get_name(), data_type, self.index, &self.types_index)?;
548        }
549
550        Ok(())
551    }

554pub fn get_default_for(basic_type: BasicTypeEnum) -> BasicValueEnum {
555    match basic_type {
556        BasicTypeEnum::ArrayType(t) => t.const_zero().into(),
557        BasicTypeEnum::FloatType(t) => t.const_zero().into(),
558        BasicTypeEnum::IntType(t) => t.const_zero().into(),
559        BasicTypeEnum::PointerType(t) => t.const_zero().into(),
560        BasicTypeEnum::StructType(t) => t.const_zero().into(),
561        BasicTypeEnum::VectorType(t) => t.const_zero().into(),
562    }
