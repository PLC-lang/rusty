47/// generates the llvm-type for the given data-type and registers it at the index
48/// this function may create and register a ...
49/// - Struct type for a STRUCT
50/// - global variables for enum-elements
51/// - an alias index entry for sub-range types
52/// - Array type for arrays
53/// - array type for sized Strings
54pub fn generate_data_types<'ink>(
55    llvm: &Llvm<'ink>,
56    debug: &mut DebugBuilderEnum<'ink>,
57    dependencies: &FxIndexSet<Dependency>,
58    index: &Index,
59    annotations: &AstAnnotations,
60) -> Result<LlvmTypedIndex<'ink>, CodegenError> {
61    let mut types = vec![];
62    let mut pou_types = vec![];
63
64    // Always add the builtin types
65    let builtins = typesystem::get_builtin_types();
66    for builtin in &builtins {
67        types.push((builtin.get_name(), builtin));
68    }
69
70    for dep in dependencies {
71        if let Dependency::Datatype(name) = dep {
72            if let Some(pou) = index.find_pou(name) {
73                if !pou.is_generic() && !pou.is_action() {
74                    pou_types.push((name.as_str(), pou.get_instance_struct_type_or_void(index)));
75                }
76            } else if let Some(datatype) = index.find_type(name) {
77                if !datatype.get_type_information().is_generic(index) {
78                    types.push((name, datatype))
79                }
80            }
81        }
82    }
83
84    let mut generator =
85        DataTypeGenerator { llvm, debug, index, annotations, types_index: LlvmTypedIndex::default() };
86
87    // first create all STUBs for struct types (empty structs)
88    // and associate them in the llvm index
89    for (name, user_type) in &types {
90        if let DataTypeInformation::Struct { name: struct_name, .. } = user_type.get_type_information() {
91            generator.types_index.associate_type(name, llvm.create_struct_stub(struct_name).into())?;
92        }
93    }
94    // pou_types will always be struct
95    for (name, user_type) in &pou_types {
96        if let DataTypeInformation::Struct { name: struct_name, .. } = user_type.get_type_information() {
97            generator.types_index.associate_pou_type(name, llvm.create_struct_stub(struct_name).into())?;
98        }
99    }
100
101    // now create all other types (enum's, arrays, etc.)
102    for (name, user_type) in &types {
103        let gen_type = generator.create_type(name, user_type)?;
104        generator.types_index.associate_type(name, gen_type)?
105        //Get and associate debug type
106    }
107
108    for (name, user_type) in &pou_types {
109        let gen_type = generator.create_type(name, user_type)?;
110        generator.types_index.associate_pou_type(name, gen_type)?
111    }
112
113    // Combine the types and pou_types into a single Vector
114    let mut types_to_init = VecDeque::new();
115    types_to_init.extend(types);
116    types_to_init.extend(pou_types);
117    // now since all types should be available in the llvm index, we can think about constructing and associating
118    for (_, user_type) in &types_to_init {
119        //Expand all types
120        generator.expand_opaque_types(user_type)?;
121    }
122
123    // Now generate debug information for all types
124    generator.generate_debug_types(&types_to_init)?;
125
126    let mut tries = 0;
127    let mut errors = FxHashMap::default();
128
129    // If the tries are equal to the number of types remaining, it means we failed to resolve
130    // anything
131    while tries < types_to_init.len() {
132        //Take the current element,
133        if let Some((name, user_type)) = types_to_init.pop_front() {
134            errors.remove(name);
135            //try to resolve it
136            match generator.generate_initial_value(user_type) {
137                Err(err) => {
138                    tries += 1;
139                    types_to_init.push_back((name, user_type));
140                    errors.insert(name, err);
141                }
142                Ok(init_value) => {
143                    if let Some(init_value) = init_value {
144                        if let Err(err) = generator.types_index.associate_initial_value(name, init_value) {
145                            //If it fails, push it back into the list
146                            tries += 1;
147                            types_to_init.push_back((name, user_type));
148                            errors.insert(name, err);
149                        } else {
150                            tries = 0;
151                        }
152                    }
153                }
154            }
155        }
156    }
157    //If we didn't resolve anything this cycle, report the remaining issues and exit
158    if !types_to_init.is_empty() {
159        //Report each error as a new diagnostic, add the type's location as related to the error
160        let diags = types_to_init
161            .into_iter()
162            .map(|(name, ty)| {
163                errors
164                    .remove(name)
165                    .map(|diag| Diagnostic::from(diag).with_secondary_location(&ty.location))
166                    .unwrap_or_else(|| Diagnostic::cannot_generate_initializer(name, &ty.location))
167            })
168            .collect::<Vec<_>>();
169        if !diags.is_empty() {
170            //Report the operation failure
171            return Err(Diagnostic::new("Some initial values were not generated")
172                .with_error_code("E075")
173                .with_sub_diagnostics(diags)
174                .into()); // FIXME: these sub-diagnostics aren't printed to the console
175        }
176    }
177    Ok(generator.types_index)
178}

180impl<'ink> DataTypeGenerator<'ink, '_> {
181    /// generates the members of an opaque struct and associates its initial values
182    fn expand_opaque_types(&mut self, data_type: &DataType) -> Result<(), CodegenError> {
183        let information = data_type.get_type_information();
184        if let DataTypeInformation::Struct { source, members, .. } = information {
185            let members = members
186                .iter()
187                .filter(|it| !(it.is_temp() || it.is_return() || it.is_var_external()))
188                .map(|m| self.types_index.get_associated_type(m.get_type_name()))
189                .collect::<Result<Vec<BasicTypeEnum>, _>>()?;
190
191            let struct_type = match source {
192                StructSource::Pou(..) => self.types_index.get_associated_pou_type(data_type.get_name()),
193                StructSource::OriginalDeclaration | StructSource::Internal(_) => {
194                    self.types_index.get_associated_type(data_type.get_name())
195                }
196            }
197            .map(BasicTypeEnum::into_struct_type)?;
198
199            struct_type.set_body(members.as_slice(), false);
200        }
201        Ok(())
202    }

204    /// Creates an llvm type to be associated with the given data type.
205    /// Generates only an opaque type for structs.
206    /// Eagerly generates but does not associate nested array and referenced aliased types
207    fn create_type(&mut self, name: &str, data_type: &DataType) -> Result<AnyTypeEnum<'ink>, CodegenError> {
208        let information = data_type.get_type_information();
209        match information {
210            DataTypeInformation::Struct { source, .. } => match source {
211                StructSource::Pou(..) => self
212                    .types_index
213                    .get_associated_pou_type(data_type.get_name())
214                    .map(|res| res.as_any_type_enum()),
215                StructSource::OriginalDeclaration => self
216                    .types_index
217                    .get_associated_type(data_type.get_name())
218                    .map(|res| res.as_any_type_enum()),
219                StructSource::Internal(_) => self
220                    .types_index
221                    .get_associated_type(data_type.get_name())
222                    .map(|res| res.as_any_type_enum()),
223            },
224
225            // We distinguish between two types of arrays, normal and variable length ones.
226            // Variable Length Arrays are defined by having unknown sizes at compile time, thus we handle
227            // these two cases based on whether the {start,end}-offset equals `TypeSize::Undetermined`.
228            DataTypeInformation::Array { inner_type_name, dimensions, .. } => {
229                if dimensions.iter().any(|dimension| dimension.is_undetermined()) {
230                    self.create_type(inner_type_name, self.index.get_type(inner_type_name)?)
231                } else {
232                    self.index
233                        .get_effective_type_by_name(inner_type_name)
234                        .map_err(Into::into)
235                        .and_then(|inner_type| self.create_type(inner_type_name, inner_type))
236                        .and_then(|inner_type| self.create_nested_array_type(inner_type, dimensions))
237                        .map(|it| it.as_any_type_enum())
238                }
239            }
240            DataTypeInformation::Integer { size, .. } => {
241                Ok(get_llvm_int_type(self.llvm.context, *size, name).into())
242            }
243            DataTypeInformation::Enum { name, referenced_type, .. } => {
244                let effective_type = self.index.get_effective_type_or_void_by_name(referenced_type);
245                if let DataTypeInformation::Integer { .. } = effective_type.get_type_information() {
246                    self.create_type(name, effective_type)
247                } else {
248                    Err(Diagnostic::new(format!(
249                        "Invalid type nature for generic argument. {} is no `ANY_INT`.",
250                        effective_type.get_name()
251                    ))
252                    .with_error_code("E062")
253                    .into())
254                }
255            }
256            DataTypeInformation::Float { size, .. } => {
257                Ok(get_llvm_float_type(self.llvm.context, *size, name).into())
258            }
259            DataTypeInformation::String { size, encoding } => {
260                let base_type = if *encoding == StringEncoding::Utf8 {
261                    self.llvm.context.i8_type()
262                } else {
263                    self.llvm.context.i16_type()
264                };
265
266                let string_size = size
267                    .as_int_value(self.index)
268                    .map_err(|it| Diagnostic::codegen_error(it.as_str(), SourceLocation::undefined()))?
269                    as u32;
270                Ok(base_type.array_type(string_size).into())
271            }
272            DataTypeInformation::SubRange { referenced_type, .. }
273            | DataTypeInformation::Alias { referenced_type, .. } => self
274                .index
275                .get_effective_type_by_name(referenced_type)
276                .map_err(Into::into)
277                .and_then(|data_type| self.create_type(name, data_type)),
278            DataTypeInformation::Void => Ok(get_llvm_int_type(self.llvm.context, 32, "Void").into()),
279            DataTypeInformation::Pointer { inner_type_name, .. } => {
280                let inner_type = if information.is_function_pointer() {
281                    self.create_function_type(inner_type_name)?.as_any_type_enum()
282                } else {
283                    self.create_type(inner_type_name, self.index.get_type(inner_type_name)?)?
284                };
285
286                Ok(inner_type.create_ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)).into())
287            }
288            DataTypeInformation::Generic { .. } => {
289                unreachable!("Generic types should not be generated")
290            }
291        }
292    }

294    fn create_function_type(&mut self, method_name: &str) -> Result<FunctionType<'ink>, Diagnostic> {
295        let return_type = self
296            .types_index
297            .find_associated_type(self.index.get_return_type_or_void(method_name).get_name())
298            .map(|opt| opt.as_any_type_enum())
299            .unwrap_or(self.llvm.context.void_type().as_any_type_enum());
300
301        let mut parameter_types = Vec::new();
302
303        match self.index.find_pou(method_name) {
304            Some(PouIndexEntry::Method { parent_name, .. }) => {
305                let ty = self.types_index.get_associated_type(parent_name).expect("must exist");
306                let ty_ptr = ty.ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)).into();
307
308                // Methods are defined as functions in the LLVM IR, but carry the underlying POU type as their
309                // first parameter to operate on them, hence push the POU type to the very first position.
310                parameter_types.push(ty_ptr);
311
312                for parameter in self.index.get_declared_parameters(method_name) {
313                    // Instead of relying on the LLVM index, we create data-types on the fly here because some
314                    // types have not yet been visited and as a result may not be in the index. For example at
315                    // the time of writing this the index was not able to find a input parameter of type
316                    // `__auto_pointer_to_DINT`, consequently panicking
317                    let ty = self.create_type(
318                        parameter.get_name(),
319                        self.index.get_type(&parameter.data_type_name).expect("must exist"),
320                    )?;
321
322                    parameter_types.push(ty.try_into().unwrap());
323                }
324            }
325
326            Some(PouIndexEntry::FunctionBlock { name, .. }) => {
327                let ty = self.types_index.get_associated_type(name).expect("must exist");
328                let ty_ptr = ty.ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)).into();
329
330                // Function blocks are a bit "weird" in that they only expect an instance argument even if
331                // they define input, output and/or inout parameters. Effectively, while being methods per-se,
332                // their calling convention differs from regular methods.
333                parameter_types.push(ty_ptr);
334            }
335
336            _ => unreachable!("internal error, invalid method call"),
337        }
338
339        let fn_type = match return_type {
340            AnyTypeEnum::ArrayType(value) => value.fn_type(parameter_types.as_slice(), false),
341            AnyTypeEnum::FloatType(value) => value.fn_type(parameter_types.as_slice(), false),
342            AnyTypeEnum::IntType(value) => value.fn_type(parameter_types.as_slice(), false),
343            AnyTypeEnum::PointerType(value) => value.fn_type(parameter_types.as_slice(), false),
344            AnyTypeEnum::StructType(value) => value.fn_type(parameter_types.as_slice(), false),
345            AnyTypeEnum::VectorType(value) => value.fn_type(parameter_types.as_slice(), false),
346            AnyTypeEnum::VoidType(value) => value.fn_type(parameter_types.as_slice(), false),
347            AnyTypeEnum::ScalableVectorType(value) => value.fn_type(parameter_types.as_slice(), false),
348            AnyTypeEnum::FunctionType(_) => unreachable!(),
349        };
350
351        Ok(fn_type)
352    }

354    fn generate_initial_value(
355        &mut self,
356        data_type: &DataType,
357    ) -> Result<Option<BasicValueEnum<'ink>>, CodegenError> {
358        let information = data_type.get_type_information();
359        match information {
360            DataTypeInformation::Struct { source, members, .. } => {
361                let member_names_and_initializers = members
362                    .iter()
363                    .filter(|it| it.get_variable_type() != VariableType::Temp)
364                    .map(|it| {
365                        self.generate_initial_value_for_variable(it).and_then(|v| match v {
366                            Some(v) => Ok((it.get_qualified_name(), v)),
367                            None => self
368                                .types_index
369                                .get_associated_type(it.get_type_name())
370                                .map(get_default_for)
371                                .map(|v| (it.get_qualified_name(), v)),
372                        })
373                    })
374                    .collect::<Result<Vec<(&str, BasicValueEnum)>, _>>()?;
375
376                let mut member_values: Vec<BasicValueEnum> = Vec::new();
377                for (name, v) in &member_names_and_initializers {
378                    self.types_index.associate_initial_value(name, *v)?;
379                    member_values.push(*v);
380                }
381
382                let struct_type = match source {
383                    StructSource::Pou(..) => self.types_index.get_associated_pou_type(data_type.get_name()),
384                    StructSource::OriginalDeclaration => {
385                        self.types_index.get_associated_type(data_type.get_name())
386                    }
387                    StructSource::Internal(_) => self.types_index.get_associated_type(data_type.get_name()),
388                }?
389                .into_struct_type();
390
391                Ok(Some(struct_type.const_named_struct(&member_values).as_basic_value_enum()))
392            }
393            DataTypeInformation::Array { .. } => self.generate_array_initializer(
394                data_type,
395                |stmt| matches!(stmt.stmt, AstStatement::Literal(AstLiteral::Array { .. })),
396                "LiteralArray",
397            ),
398            DataTypeInformation::String { .. } => self.generate_array_initializer(
399                data_type,
400                |stmt| matches!(stmt.stmt, AstStatement::Literal(AstLiteral::String { .. })),
401                "LiteralString",
402            ),
403            DataTypeInformation::SubRange { referenced_type, .. } => {
404                self.generate_initial_value_for_type(data_type, referenced_type)
405            }
406            DataTypeInformation::Alias { referenced_type, .. } => {
407                self.generate_initial_value_for_type(data_type, referenced_type)
408            }
409            DataTypeInformation::Enum { referenced_type, .. } => {
410                self.generate_initial_value_for_type(data_type, referenced_type)
411            }
412            //all other types (scalars, pointer and void)
413            _ => Ok(None),
414        }
415    }

417    /// generates and returns an optional inital value at the given declared variable
418    /// if no initial value is defined, it returns the initial value of the variable's
419    /// datatype or Ok(None) if the type also has no declared default value
420    fn generate_initial_value_for_variable(
421        &mut self,
422        variable: &VariableIndexEntry,
423    ) -> Result<Option<BasicValueEnum<'ink>>, CodegenError> {
424        let initializer = variable
425            .initial_value
426            .and_then(|it| self.index.get_const_expressions().get_constant_statement(&it));
427        self.generate_initializer(variable.get_qualified_name(), initializer, variable.get_type_name())
428    }

430    /// generates and returns an optional inital value at the given dataType
431    /// if no initial value is defined, it returns  an optional initial value of
432    /// the aliased type (referenced_type)
433    fn generate_initial_value_for_type(
434        &mut self,
435        data_type: &DataType,
436        referenced_type: &str,
437    ) -> Result<Option<BasicValueEnum<'ink>>, CodegenError> {
438        self.generate_initializer(
439            data_type.get_name(),
440            self.index.get_const_expressions().maybe_get_constant_statement(&data_type.initial_value),
441            referenced_type,
442        )
443    }

445    /// generates the given initializer-statement if one is present
446    /// if no initializer is present, it returns an optional default value
447    /// of the given datatype.
448    /// Errors will be reported using the given qualified_name
449    fn generate_initializer(
450        &mut self,
451        qualified_name: &str,
452        initializer: Option<&AstNode>,
453        data_type_name: &str,
454    ) -> Result<Option<BasicValueEnum<'ink>>, CodegenError> {
455        if let Some(initializer) = initializer {
456            let generator = ExpressionCodeGenerator::new_context_free(
457                self.llvm,
458                self.index,
459                self.annotations,
460                &self.types_index,
461            );
462
463            let lhs_type = self.index.get_type_information_or_void(data_type_name);
464            if !initializer.is_literal() && (lhs_type.is_pointer() || lhs_type.is_alias()) {
465                return Ok(self
466                    .types_index
467                    .find_associated_type(lhs_type.get_name())
468                    .map(|it| it.const_zero().as_basic_value_enum()));
469            };
470
471            generator
472                .generate_expression(initializer)
473                .map(Some)
474                .map_err(|_| Diagnostic::cannot_generate_initializer(qualified_name, initializer).into())
475        } else {
476            // if there's no initializer defined for this alias, we go and check the aliased type for an initial value
477            self.index
478                .get_type(data_type_name)
479                .map_err(Into::into)
480                .and_then(|referenced_data_type| self.generate_initial_value(referenced_data_type))
481        }
482    }

484    /// generates and associates the given array-datatype (used for arrays and strings)
485    fn generate_array_initializer(
486        &self,
487        data_type: &DataType,
488        predicate: fn(&AstNode) -> bool,
489        expected_ast: &str,
490    ) -> Result<Option<BasicValueEnum<'ink>>, CodegenError> {
491        if let Some(initializer) =
492            self.index.get_const_expressions().maybe_get_constant_statement(&data_type.initial_value)
493        {
494            if predicate(initializer) {
495                let generator = ExpressionCodeGenerator::new_context_free(
496                    self.llvm,
497                    self.index,
498                    self.annotations,
499                    &self.types_index,
500                );
501                Ok(Some(generator.generate_literal(initializer)?.get_basic_value_enum()))
502            } else {
503                Err(Diagnostic::codegen_error(
504                    format!("Expected {expected_ast} but found {initializer:?}"),
505                    initializer,
506                )
507                .into())
508            }
509        } else {
510            Ok(None)
511        }
512    }

514    /// creates the llvm types for a multi-dimensional array
515    ///
516    /// an array with multiple dimensions will be flattened into a long
517    /// 1-dimensional array.
518    /// e.g. `arr: ARRAY[0..2, 0..2] OF INT` produces the same result like
519    /// `arr: ARRAY[0..3] OF INT`.
520    fn create_nested_array_type(
521        &self,
522        inner_type: AnyTypeEnum<'ink>,
523        dimensions: &[Dimension],
524    ) -> Result<AnyTypeEnum<'ink>, CodegenError> {
525        let len = dimensions
526            .iter()
527            .map(|dimension| {
528                dimension
529                    .get_length(self.index)
530                    .map_err(|it| Diagnostic::codegen_error(it.as_str(), SourceLocation::undefined()))
531            })
532            .collect::<Result<Vec<u32>, Diagnostic>>()?
533            .into_iter()
534            .reduce(|a, b| a * b)
535            .ok_or_else(|| {
536                Diagnostic::codegen_error("Invalid array dimensions", SourceLocation::undefined())
537            })?;
538
539        let result = match inner_type {
540            AnyTypeEnum::IntType(ty) => ty.array_type(len),
541            AnyTypeEnum::FloatType(ty) => ty.array_type(len),
542            AnyTypeEnum::StructType(ty) => ty.array_type(len),
543            AnyTypeEnum::ArrayType(ty) => ty.array_type(len),
544            AnyTypeEnum::PointerType(ty) => ty.array_type(len),
545            AnyTypeEnum::VectorType(ty) => ty.array_type(len),
546            AnyTypeEnum::ScalableVectorType(ty) => ty.array_type(len),
547
548            AnyTypeEnum::FunctionType(_) => unimplemented!("function types are not supported in arrays"),
549            AnyTypeEnum::VoidType(_) => unimplemented!("void types not supported in arrays"),
550        }
551        .as_any_type_enum();
552
553        Ok(result)
554    }

556    fn generate_debug_types(&mut self, types: &VecDeque<(&str, &DataType)>) -> Result<(), CodegenError> {
557        for (_, data_type) in types.iter().filter(|(_, data_type)| data_type.is_backed_by_struct()) {
558            self.debug.register_debug_type(data_type.get_name(), data_type, self.index, &self.types_index)?;
559        }
560
561        Ok(())
562    }

565pub fn get_default_for(basic_type: BasicTypeEnum) -> BasicValueEnum {
566    match basic_type {
567        BasicTypeEnum::ArrayType(t) => t.const_zero().into(),
568        BasicTypeEnum::FloatType(t) => t.const_zero().into(),
569        BasicTypeEnum::IntType(t) => t.const_zero().into(),
570        BasicTypeEnum::PointerType(t) => t.const_zero().into(),
571        BasicTypeEnum::StructType(t) => t.const_zero().into(),
572        BasicTypeEnum::VectorType(t) => t.const_zero().into(),
573        BasicTypeEnum::ScalableVectorType(t) => t.const_zero().into(),
574    }
