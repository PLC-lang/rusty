79#[derive(Debug)]
80pub enum ExpressionValue<'ink> {
81    /// A Locator-Value
82    /// An lvalue (locator value) represents an object that occupies some identifiable location in memory (i.e. has an address).
83    LValue(PointerValue<'ink>),
84    /// An expression that does not represent an object occupying some identifiable location in memory.

88impl<'ink> ExpressionValue<'ink> {
89    /// returns the value represented by this ExpressionValue
90    pub fn get_basic_value_enum(&self) -> BasicValueEnum<'ink> {
91        match self {
92            ExpressionValue::LValue(it) => it.as_basic_value_enum(),
93            ExpressionValue::RValue(it) => it.to_owned(),
94        }
95    }
96
97    /// returns the given expression value as an r-value which means that it will load
98    /// the pointer, if this is an l_value
99    pub fn as_r_value(&self, llvm: &Llvm<'ink>, load_name: Option<String>) -> BasicValueEnum<'ink> {
100        match self {
101            ExpressionValue::LValue(it) => llvm.load_pointer(it, load_name.as_deref().unwrap_or("")),
102            ExpressionValue::RValue(it) => it.to_owned(),
103        }
104    }

169    /// entry point into the expression generator.
170    /// generates the given expression and returns the resulting BasicValueEnum
171    pub fn generate_expression(&self, expression: &AstNode) -> Result<BasicValueEnum<'ink>, Diagnostic> {
172        // If the expression was replaced by the resolver, generate the replacement
173        if let Some(StatementAnnotation::ReplacementAst { statement }) = self.annotations.get(expression) {
174            // we trust that the validator only passed us valid parameters (so left & right should be same type)
175            return self.generate_expression(statement);
176        }
177
178        let v = self
179            .generate_expression_value(expression)?
180            .as_r_value(self.llvm, self.get_load_name(expression))
181            .as_basic_value_enum();
182
183        let Some(target_type) = self.annotations.get_type_hint(expression, self.index) else {
184            // no type-hint -> we can return the value as is
185            return Ok(v);
186        };
187        let actual_type = self.annotations.get_type_or_void(expression, self.index);
188        Ok(cast_if_needed!(self, target_type, actual_type, v, self.annotations.get(expression)))
189    }

199    pub fn generate_expression_value(
200        &self,
201        expression: &AstNode,
202    ) -> Result<ExpressionValue<'ink>, Diagnostic> {
203        //see if this is a constant - maybe we can short curcuit this codegen
204        if let Some(StatementAnnotation::Variable {
205            qualified_name, constant: true, resulting_type, ..
206        }) = self.annotations.get(expression)
207        {
208            if !self.index.get_type_information_or_void(resulting_type).is_aggregate() {
209                match self.generate_constant_expression(qualified_name, expression) {
210                    // We return here if constant propagation worked
211                    Ok(expr) => return Ok(expr),
212                    // ...and fall-back to generating the expression further down if it didn't
213                    Err(why) => log::info!("{why}"),
214                }
215            }
216        }
217
218        // generate the expression
219        match expression.get_stmt() {
220            AstStatement::This => {
221                self.function_context.ok_or_else(|| {
222                    Diagnostic::codegen_error("Cannot use 'this' without context", expression)
223                })?;
224                let Some(this_name) = self.annotations.get_call_name(expression) else {
225                    unreachable!("this should have a name");
226                };
227                let this_value =
228                    self.llvm_index.find_loaded_associated_variable_value(this_name).ok_or_else(|| {
229                        let message = format!("Cannot find '{}' in associated variable values", this_name);
230                        Diagnostic::codegen_error(message, expression)
231                    })?;
232                Ok(ExpressionValue::LValue(this_value))
233            }
234            AstStatement::ReferenceExpr(data) => {
235                let res =
236                    self.generate_reference_expression(&data.access, data.base.as_deref(), expression)?;
237                let val = match res {
238                    ExpressionValue::LValue(val) => {
239                        ExpressionValue::LValue(self.auto_deref_if_necessary(val, expression))
240                    }
241                    ExpressionValue::RValue(val) => {
242                        let val = if val.is_pointer_value() {
243                            self.auto_deref_if_necessary(val.into_pointer_value(), expression)
244                                .as_basic_value_enum()
245                        } else {
246                            val
247                        };
248                        ExpressionValue::RValue(val)
249                    }
250                };
251                Ok(val)
252            }
253            AstStatement::HardwareAccess(..) => self
254                .create_llvm_pointer_value_for_reference(None, "address", expression)
255                .map(ExpressionValue::LValue),
256            AstStatement::BinaryExpression(data) => self
257                .generate_binary_expression(&data.left, &data.right, &data.operator, expression)
258                .map(ExpressionValue::RValue),
259            AstStatement::CallStatement(data) => {
260                self.generate_call_statement(&data.operator, data.parameters.as_deref())
261            }
262            AstStatement::UnaryExpression(data) => {
263                self.generate_unary_expression(&data.operator, &data.value).map(ExpressionValue::RValue)
264            }
265            AstStatement::ParenExpression(expr) => self.generate_expression_value(expr),
266            //fallback
267            _ => self.generate_literal(expression),
268        }
269    }

271    /// Propagate the constant value of the constant reference to  `qualified_name`.
272    /// - `qualified _name` the qualified name of the referenced constant variable we want to propagate
273    /// - `expression` the original expression
274    fn generate_constant_expression(
275        &self,
276        qualified_name: &str,
277        expression: &AstNode,
278    ) -> Result<ExpressionValue<'ink>, Diagnostic> {
279        let const_expression = self
280            .index
281            // try to find a constant variable
282            .find_variable(None, &qualified_name.split('.').collect::<Vec<_>>())
283            // or else try to find an enum element
284            .or_else(|| self.index.find_enum_variant_by_qualified_name(qualified_name))
285            // if this is no constant we have a problem
286            .filter(|v| v.is_constant())
287            .and_then(|v| v.initial_value)
288            // fetch the constant's initial value fron the const-expressions arena
289            .and_then(|constant_variable| {
290                self.index.get_const_expressions().get_resolved_constant_statement(&constant_variable)
291            })
292            .ok_or_else(|| {
293                // We'll _probably_ land here because we're dealing with aggregate types, see also
294                // https://github.com/PLC-lang/rusty/issues/288
295                let message = format!("Cannot propagate constant value for '{qualified_name:}'");
296                Diagnostic::codegen_error(message, expression)
297            })?;
298
299        //  generate the resulting constant-expression (which should be a Value, no ptr-reference)
300        self.generate_expression_value(const_expression)
301    }

303    /// Generate an access to the appropriate GOT entry to achieve an access to the given base
304    /// lvalue.
305    pub fn generate_got_access(
306        &self,
307        context: &AstNode,
308        llvm_type: &BasicTypeEnum<'ink>,
309    ) -> Result<Option<PointerValue<'ink>>, Diagnostic> {
310        match self.annotations.get(context) {
311            Some(StatementAnnotation::Variable { qualified_name, .. }) => {
312                // We will generate a GEP, which has as its base address the magic constant which
313                // will eventually be replaced by the location of the GOT.
314                let base =
315                    self.llvm.context.i64_type().const_int(0xdeadbeef00000000, false).const_to_pointer(
316                        llvm_type.ptr_type(AddressSpace::default()).ptr_type(AddressSpace::default()),
317                    );
318
319                self.llvm_index
320                    .find_got_index(qualified_name)
321                    .map(|idx| {
322                        let ptr = self.llvm.load_array_element(
323                            base,
324                            &[self.llvm.context.i32_type().const_int(idx, false)],
325                            "",
326                        )?;
327                        Ok(self.llvm.load_pointer(&ptr, "").into_pointer_value())
328                    })
329                    .transpose()
330            }
331            _ => Ok(None),
332        }
333    }

335    /// Generate an access to the appropriate GOT entry to achieve a call to the given function.
336    pub fn generate_got_call(
337        &self,
338        qualified_name: &str,
339        function_type: &FunctionType<'ink>,
340        args: &[BasicMetadataValueEnum<'ink>],
341    ) -> Result<Option<CallSiteValue<'ink>>, Diagnostic> {
342        // We will generate a GEP, which has as its base address the magic constant which
343        // will eventually be replaced by the location of the GOT.
344        let base = self.llvm.context.i64_type().const_int(0xdeadbeef00000000, false).const_to_pointer(
345            function_type.ptr_type(AddressSpace::default()).ptr_type(AddressSpace::default()),
346        );
347
348        self.llvm_index
349            .find_got_index(qualified_name)
350            .map(|idx| {
351                let mut ptr = self.llvm.load_array_element(
352                    base,
353                    &[self.llvm.context.i32_type().const_int(idx, false)],
354                    "",
355                )?;
356                ptr = self.llvm.load_pointer(&ptr, "").into_pointer_value();
357                let callable = CallableValue::try_from(ptr)
358                    .map_err(|_| Diagnostic::new("Pointer was not a function pointer"))?;
359
360                Ok(self.llvm.builder.build_call(callable, args, "call"))
361            })
362            .transpose()
363    }

365    /// generates a binary expression (e.g. a + b, x AND y, etc.) and returns the resulting `BasicValueEnum`
366    /// - `left` the AstStatement left of the operator
367    /// - `right` the AstStatement right of the operator
368    /// - `operator` the binary expression's operator
369    /// - `expression` the whole expression for diagnostic reasons
370    fn generate_binary_expression(
371        &self,
372        left: &AstNode,
373        right: &AstNode,
374        operator: &Operator,
375        expression: &AstNode,
376    ) -> Result<BasicValueEnum<'ink>, Diagnostic> {
377        let l_type_hint = self.get_type_hint_for(left)?;
378        let ltype = self.index.get_intrinsic_type_by_name(l_type_hint.get_name()).get_type_information();
379        let r_type_hint = self.get_type_hint_for(right)?;
380        let rtype = self.index.get_intrinsic_type_by_name(r_type_hint.get_name()).get_type_information();
381        if ltype.is_bool() && rtype.is_bool() {
382            return self.generate_bool_binary_expression(operator, left, right);
383        }
384        if ltype.is_int() && rtype.is_int() {
385            Ok(self.create_llvm_int_binary_expression(
386                operator,
387                self.generate_expression(left)?,
388                self.generate_expression(right)?,
389            ))
390        } else if ltype.is_float() && rtype.is_float() {
391            Ok(self.create_llvm_float_binary_expression(
392                operator,
393                self.generate_expression(left)?,
394                self.generate_expression(right)?,
395            ))
396        } else if (ltype.is_pointer() && rtype.is_int())
397            || (ltype.is_int() && rtype.is_pointer())
398            || (ltype.is_pointer() && rtype.is_pointer())
399        {
400            self.create_llvm_binary_expression_for_pointer(operator, left, ltype, right, rtype, expression)
401        } else {
402            self.create_llvm_generic_binary_expression(left, right, expression)
403        }
404    }

406    pub fn generate_direct_access_index(
407        &self,
408        access: &DirectAccessType,
409        index: &AstNode,
410        access_type: &DataTypeInformation,
411        target_type: &DataType,
412    ) -> Result<IntValue<'ink>, Diagnostic> {
413        let reference = self.generate_expression(index)?;
414        //Load the reference
415        if reference.is_int_value() {
416            //This cast is needed to convert the index/reference to the type of original expression
417            //being accessed.
418            //The reason is that llvm expects a shift operation to happen on the same type, and
419            //this is what the direct access will eventually end up in.
420            let reference =
421                cast_if_needed!(self, target_type, self.get_type_hint_for(index)?, reference, None)
422                    .into_int_value();
423            //Multiply by the bitwitdh
424            if access.get_bit_width() > 1 {
425                let bitwidth =
426                    reference.get_type().const_int(access.get_bit_width(), access_type.is_signed_int());
427
428                Ok(self.llvm.builder.build_int_mul(reference, bitwidth, ""))
429            } else {
430                Ok(reference)
431            }
432        } else {
433            Err(Diagnostic::new(format!("Cannot cast from {} to Integer Type", access_type.get_name()))
434                .with_error_code("E051")
435                .with_location(index))
436        }
437    }

439    /// generates a Unary-Expression e.g. -<expr> or !<expr>
440    fn generate_unary_expression(
441        &self,
442        unary_operator: &Operator,
443        expression: &AstNode,
444    ) -> Result<BasicValueEnum<'ink>, Diagnostic> {
445        let value = match unary_operator {
446            Operator::Not => {
447                let operator = self.generate_expression(expression)?.into_int_value();
448                let operator = if self
449                    .get_type_hint_for(expression)
450                    .map(|it| it.get_type_information().is_bool())
451                    .unwrap_or_default()
452                {
453                    to_i1(operator, &self.llvm.builder)
454                } else {
455                    operator
456                };
457
458                Ok(self.llvm.builder.build_not(operator, "tmpVar").as_basic_value_enum())
459            }
460            Operator::Minus => {
461                let generated_exp = self.generate_expression(expression)?;
462                if generated_exp.is_float_value() {
463                    Ok(self
464                        .llvm
465                        .builder
466                        .build_float_neg(generated_exp.into_float_value(), "tmpVar")
467                        .as_basic_value_enum())
468                } else if generated_exp.is_int_value() {
469                    Ok(self
470                        .llvm
471                        .builder
472                        .build_int_neg(generated_exp.into_int_value(), "tmpVar")
473                        .as_basic_value_enum())
474                } else {
475                    Err(Diagnostic::codegen_error("Negated expression must be numeric", expression))
476                }
477            }
478            Operator::Plus => self.generate_expression(expression),
479            _ => unimplemented!(),
480        };
481        value
482    }

484    /// generates the given call-statement <operator>(<parameters>)
485    /// returns the call's result as a BasicValueEnum (may be a void-type for PROGRAMs)
486    ///
487    /// - `operator` - the expression that points to the callable instance (e.g. a PROGRAM, FUNCTION or FUNCTION_BLOCK instance)
488    /// - `parameters` - an optional StatementList of parameters
489    pub fn generate_call_statement(
490        &self,
491        operator: &AstNode,
492        parameters: Option<&AstNode>,
493    ) -> Result<ExpressionValue<'ink>, Diagnostic> {
494        if self.annotations.get(operator).is_some_and(StatementAnnotation::is_fnptr) {
495            return self.generate_fnptr_call(operator, parameters);
496        }
497
498        // find the pou we're calling
499        let pou = self.annotations.get_call_name(operator).zip(self.annotations.get_qualified_name(operator))
500            .and_then(|(call_name, qualified_name)| {
501                self.index.find_pou(call_name)
502                    // for some functions (builtins) the call name does not exist in the index, we try to call with the originally defined generic functions
503                    .or_else(|| self.index.find_pou(qualified_name))
504            })
505            .or_else(||
506                // some rare situations have a callstatement that's not properly annotated (e.g. checkRange-call of ranged datatypes)
507                if let Some(name) = operator.get_flat_reference_name() {
508                    self.index.find_pou(name)
509                } else {
510                    None
511                })
512            .ok_or_else(|| Diagnostic::cannot_generate_call_statement(operator))?;
513
514        // find corresponding implementation
515        let implementation = pou
516            .find_implementation(self.index)
517            .ok_or_else(|| Diagnostic::cannot_generate_call_statement(operator))?;
518
519        let parameters_list = parameters.map(flatten_expression_list).unwrap_or_default();
520        let implementation_name = &implementation.get_call_name();
521        // if the function is builtin, generate a basic value enum for it
522        if let Some(builtin) = self.index.get_builtin_function(implementation_name) {
523            // adr, ref, etc.
524            return builtin.codegen(self, parameters_list.as_slice(), operator.get_location());
525        }
526
527        let arguments_list = self.generate_pou_call_arguments_list(
528            pou,
529            parameters_list.as_slice(),
530            implementation,
531            operator,
532            self.get_function_context(operator)?,
533        )?;
534
535        let function = self
536            .llvm_index
537            .find_associated_implementation(implementation_name) // using the non error option to control the output error
538            .ok_or_else(|| {
539                Diagnostic::codegen_error(
540                    format!("No callable implementation associated to {implementation_name:?}"),
541                    operator,
542                )
543            })?;
544
545        // generate the debug statetment for a call
546        self.register_debug_location(operator);
547
548        // Check for the function within the GOT. If it's there, we need to generate an indirect
549        // call to its location within the GOT, which should contain a function pointer.
550        // First get the function type so our function pointer can have the correct type.
551        let qualified_name = self
552            .annotations
553            .get_qualified_name(operator)
554            .expect("Shouldn't have got this far without a name for the function");
555        let function_type = function.get_type();
556        let call = self
557            .generate_got_call(qualified_name, &function_type, &arguments_list)?
558            .unwrap_or_else(|| self.llvm.builder.build_call(function, &arguments_list, "call"));
559
560        // if the target is a function, declare the struct locally
561        // assign all parameters into the struct values
562
563        // so grab either:
564        // - the call's return value
565        // - or a null-ptr
566        let value = call
567            .try_as_basic_value()
568            .either(Ok, |_| {
569                // we return an uninitialized int pointer for void methods :-/
570                // dont deref it!!
571                Ok(get_llvm_int_type(self.llvm.context, INT_SIZE, INT_TYPE)
572                    .ptr_type(AddressSpace::from(ADDRESS_SPACE_CONST))
573                    .const_null()
574                    .as_basic_value_enum())
575            })
576            .map(ExpressionValue::RValue);
577
578        // after the call we need to copy the values for assigned outputs
579        // this is only necessary for outputs defined as `rusty::index::ArgumentType::ByVal` (PROGRAM, FUNCTION_BLOCK)
580        // FUNCTION outputs are defined as `rusty::index::ArgumentType::ByRef` // FIXME(mhasel): for standard-compliance functions also need to support VAR_OUTPUT
581        if !(pou.is_function() || pou.is_method()) {
582            let parameter_struct = match arguments_list.first() {
583                Some(v) => v.into_pointer_value(),
584                None => self.generate_lvalue(operator)?,
585            };
586
587            self.assign_output_values(parameter_struct, implementation_name, parameters_list)?
588        }
589
590        value
591    }

593    fn generate_fnptr_call(
594        &self,
595        operator: &AstNode,
596        arguments: Option<&AstNode>,
597    ) -> Result<ExpressionValue<'ink>, Diagnostic> {
598        let Some(ReferenceExpr { base: Some(ref base), .. }) = operator.get_deref_expr() else {
599            // XXX: This would fail for auto-deref pointers, but given (for now) function pointers are never
600            // auto-derefed this should be fine.
601            unreachable!("internal error, invalid method call")
602        };
603
604        let qualified_pou_name = self.annotations.get(operator).unwrap().qualified_name().unwrap();
605        let impl_entry = self.index.find_implementation_by_name(qualified_pou_name).unwrap();
606        debug_assert!(
607            impl_entry.is_method() | impl_entry.is_function_block(),
608            "internal error, invalid method call"
609        );
610
611        // Get the associated variable then load it, e.g. `%localFnPtrVariable = alloca void (%Fb*)*, align 8`
612        // followed by `%1 = load void (%Fb*)*, void (%Fb*)** %localFnPtrVariable, align 8``
613        let function_pointer_value = match self.generate_expression_value(base)? {
614            ExpressionValue::LValue(value) => self.llvm.load_pointer(&value, "").into_pointer_value(),
615            ExpressionValue::RValue(_) => unreachable!(),
616        };
617
618        // Generate the argument list; our assumption is function pointers are only supported for methods and
619        // direct function block calls, hence we explicitly fetch the instance argument from the list. In
620        // terms of lowered ST code you can imagine something alike `fnPtr^(instanceFb, arg1, ..., argN)`
621        let (instance, arguments_raw, arguments_llvm) = {
622            let arguments = arguments.map(flatten_expression_list).unwrap_or_default();
623            let (instance, arguments) = match arguments.len() {
624                0 => panic!("invalid lowered code, no instance argument found"),
625                1 => (self.generate_lvalue(arguments[0])?, Vec::new()),
626                _ => (self.generate_lvalue(arguments[0])?, arguments[1..].to_vec()),
627            };
628
629            let mut generated_arguments = match &impl_entry.implementation_type {
630                ImplementationType::Method => self.generate_function_arguments(
631                    self.index.find_pou(qualified_pou_name).unwrap(),
632                    &arguments,
633                    self.index.get_declared_parameters(qualified_pou_name),
634                )?,
635
636                // Function Block body calls have a slightly different calling convention compared to regular
637                // methods. Specifically the arguments aren't passed to the call itself but rather the
638                // instance struct is gep'ed and the arguments are stored into the gep'ed pointer value.
639                // Best to call `--ir` on a simple function block body call to see the generated IR.
640                ImplementationType::FunctionBlock => {
641                    self.generate_stateful_pou_arguments(qualified_pou_name, None, instance, &arguments)?;
642                    Vec::new()
643                }
644
645                _ => unreachable!("internal error, invalid method call"),
646            };
647
648            generated_arguments.insert(0, instance.clone().as_basic_value_enum().into());
649            (instance, arguments, generated_arguments)
650        };
651
652        // Finally generate the function pointer call
653        let callable = CallableValue::try_from(function_pointer_value).unwrap();
654        let call = self.llvm.builder.build_call(callable, &arguments_llvm, "fnptr_call");
655
656        let value = match call.try_as_basic_value() {
657            Either::Left(value) => value,
658            Either::Right(_) => get_llvm_int_type(self.llvm.context, INT_SIZE, INT_TYPE)
659                .ptr_type(AddressSpace::from(ADDRESS_SPACE_CONST))
660                .const_null()
661                .as_basic_value_enum(),
662        };
663
664        // Output variables are assigned after the function block call, effectively gep'ing the instance
665        // struct fetching the output values
666        if impl_entry.is_function_block() {
667            self.assign_output_values(instance, qualified_pou_name, arguments_raw)?
668        }
669
670        Ok(ExpressionValue::RValue(value))
671    }

674    /// - `parameter_struct` a pointer to a struct-instance that holds all function-parameters
675    /// - `function_name` the name of the callable
676    /// - `parameters` vec of passed parameters to the call

678        &self,
679        parameter_struct: PointerValue<'ink>,
680        function_name: &str,

696                        .and_then(StatementAnnotation::get_location_in_parent)
697                        .expect("arguments must have a type hint"),
698                    parameter_struct,
699                })?
700            }
701        }
702
703        Ok(())

718    pub fn generate_assignment_with_direct_access(
719        &self,
720        left_statement: &AstNode,
721        left_value: IntValue,
722        left_pointer: PointerValue,
723        right_type: &DataType,
724        right_expr: BasicValueEnum,
725    ) -> Result<(), Diagnostic> {
726        let Some((target, access_sequence)) = collect_base_and_direct_access_for_assignment(left_statement)
727        else {
728            unreachable!("Invalid direct-access expression: {left_statement:#?}")
729        };
730
731        let type_left = self.get_type_hint_for(target)?;
732        let type_right =
733            if let DataTypeInformation::Integer { semantic_size: Some(typesystem::U1_SIZE), .. } =
734                *right_type.get_type_information()
735            {
736                // we need to switch to a faked u1 type, when dealing with a single bit
737                self.index.get_type_or_panic(typesystem::U1_TYPE)
738            } else {
739                right_type
740            };
741
742        let Some((element, direct_access)) = access_sequence.split_first() else { unreachable!("") };
743
744        // Build index
745        let mut index = if let AstStatement::DirectAccess(data, ..) = element.get_stmt() {
746            self.generate_direct_access_index(
747                &data.access,
748                &data.index,
749                type_right.get_type_information(),
750                type_left,
751            )
752        } else {
753            // TODO: using the global context we could get a slice here; currently not possible because the
754            //       global context isn't passed into codegen
755            Err(Diagnostic::new(format!("{element:?} not a direct access"))
756                .with_error_code("E055")
757                .with_location(*element))
758        }?;
759        for element in direct_access {
760            let rhs_next = if let AstStatement::DirectAccess(data, ..) = element.get_stmt() {
761                self.generate_direct_access_index(
762                    &data.access,
763                    &data.index,
764                    type_right.get_type_information(),
765                    type_left,
766                )
767            } else {
768                // TODO: using the global context we could get a slice here; currently not possible because the
769                //       global context isn't passed into codegen
770                Err(Diagnostic::new(format!("{element:?} not a direct access"))
771                    .with_error_code("E055")
772                    .with_location(*element))
773            }?;
774            index = self.llvm.builder.build_int_add(index, rhs_next, "");
775        }
776
777        //Build mask for the index
778        //Get the target bit type as all ones
779        let rhs_type = self.llvm_index.get_associated_type(type_right.get_name())?.into_int_type();
780        let ones = rhs_type.const_all_ones();
781
782        //Extend the mask to the target type
783        let extended_mask = self.llvm.builder.build_int_z_extend(ones, left_value.get_type(), "ext");
784        //Position the ones in their correct locations
785        let shifted_mask = self.llvm.builder.build_left_shift(extended_mask, index, "shift");
786        //Invert the mask
787        let mask = self.llvm.builder.build_not(shifted_mask, "invert");
788        //And the result with the mask to erase the set bits at the target location
789        let and_value = self.llvm.builder.build_and(left_value, mask, "erase");
790
791        //Cast the right side to the left side type
792        let lhs = cast_if_needed!(self, type_left, type_right, right_expr, None).into_int_value();
793        //Shift left by the direct access
794        let value = self.llvm.builder.build_left_shift(lhs, index, "value");
795
796        //OR the result and store it in the left side
797        let or_value = self.llvm.builder.build_or(and_value, value, "or");
798        self.llvm.builder.build_store(left_pointer, or_value);
799
800        Ok(())
801    }

803    fn generate_output_assignment_with_direct_access(
804        &self,
805        left_statement: &AstNode,
806        left_pointer: PointerValue,
807        right_pointer: PointerValue,
808        right_type: &DataType,
809    ) -> Result<(), Diagnostic> {
810        let left_value = self.llvm.builder.build_load(left_pointer, "").into_int_value();
811
812        //Generate an expression for the right size
813        let right = self.llvm.builder.build_load(right_pointer, "");
814        self.generate_assignment_with_direct_access(
815            left_statement,
816            left_value,
817            left_pointer,
818            right_type,
819            right,
820        )?;
821
822        Ok(())
823    }

825    fn generate_output_assignment(&self, context: &CallParameterAssignment) -> Result<(), Diagnostic> {
826        let &CallParameterAssignment { assignment: expr, function_name, index, parameter_struct } = context;
827        let builder = &self.llvm.builder;
828
829        // We don't want to generate any code if the right side of an assignment is empty, e.g. `foo(out =>)`
830        if expr.is_empty_statement() {
831            return Ok(());
832        }
833
834        let parameter = self.index.get_declared_parameter(function_name, index).expect("must exist");
835
836        match expr.get_stmt() {
837            AstStatement::ReferenceExpr(_) if expr.has_direct_access() => {
838                let rhs_type = {
839                    let pou = self.index.find_pou(function_name).unwrap();
840                    let pou_struct = &pou.find_instance_struct_type(self.index).unwrap().information;
841                    let DataTypeInformation::Struct { members, .. } = pou_struct else { unreachable!() };
842
843                    self.index.find_effective_type_by_name(&members[index as usize].data_type_name).unwrap()
844                };
845
846                let AstStatement::ReferenceExpr(ReferenceExpr {
847                    access: ReferenceAccess::Member(member),
848                    base,
849                }) = &expr.get_stmt()
850                else {
851                    unreachable!("must be a bitaccess, will return early for all other cases")
852                };
853
854                if let AstStatement::DirectAccess(_) = member.as_ref().get_stmt() {
855                    // Given `foo.bar.baz.%W1.%B1.%X3`, we want to grab the base i.e. `foo.bar.baz`
856                    let (Some(base), _) = (base, ..) else { panic!() };
857                    let (base, _) = collect_base_and_direct_access_for_assignment(base).unwrap();
858
859                    let lhs = self.generate_expression_value(base)?.get_basic_value_enum();
860                    let rhs = self.llvm.builder.build_struct_gep(parameter_struct, index, "").unwrap();
861
862                    self.generate_output_assignment_with_direct_access(
863                        expr,
864                        lhs.into_pointer_value(),
865                        rhs,
866                        rhs_type,
867                    )?;
868                };
869            }
870
871            _ => {
872                let assigned_output = self.generate_lvalue(expr)?;
873                let assigned_output_type =
874                    self.annotations.get_type_or_void(expr, self.index).get_type_information();
875                let output = builder.build_struct_gep(parameter_struct, index, "").map_err(|_| {
876                    Diagnostic::codegen_error(
877                        format!("Cannot build generate parameter: {parameter:#?}"),
878                        &parameter.source_location,
879                    )
880                })?;
881
882                let output_value_type = self.index.get_type_information_or_void(parameter.get_type_name());
883
884                if assigned_output_type.is_aggregate() && output_value_type.is_aggregate() {
885                    self.build_memcpy(
886                        assigned_output,
887                        assigned_output_type,
888                        expr.get_location(),
889                        output,
890                        output_value_type,
891                        parameter.source_location.clone(),
892                    )?;
893                } else {
894                    let output_value = builder.build_load(output, "");
895                    builder.build_store(assigned_output, output_value);
896                }
897            }
898        };
899
900        Ok(())
901    }

929    /// generates the argument list for a call to a pou
930    /// a call to a function returns a Vec with all parameters for the function,
931    /// a call to a Program/Fb will return a Vec with a single struct carrying all parameters
932    fn generate_pou_call_arguments_list(
933        &self,
934        pou: &PouIndexEntry,
935        passed_parameters: &[&AstNode],
936        implementation: &ImplementationIndexEntry,
937        operator: &AstNode,
938        function_context: &'b FunctionContext<'ink, 'b>,
939    ) -> Result<Vec<BasicMetadataValueEnum<'ink>>, Diagnostic> {
940        match pou {
941            PouIndexEntry::Function { .. } => {
942                // we're calling a function
943                let declared_parameters = self.index.get_declared_parameters(implementation.get_type_name());
944                self.generate_function_arguments(pou, passed_parameters, declared_parameters)
945            }
946            PouIndexEntry::Method { .. } => {
947                let class_ptr = self.generate_lvalue(operator).or_else(|_| {
948                    // this might be a local method
949                    function_context
950                        .function
951                        .get_first_param()
952                        .map(|class_ptr| class_ptr.into_pointer_value())
953                        .ok_or_else(|| Diagnostic::cannot_generate_call_statement(operator))
954                })?;
955                let declared_parameters = self.index.get_declared_parameters(implementation.get_type_name());
956                let mut parameters =
957                    self.generate_function_arguments(pou, passed_parameters, declared_parameters)?;
958                parameters.insert(0, class_ptr.into());
959                Ok(parameters)
960            }
961            PouIndexEntry::Action { .. }
962                if try_from!(operator, ReferenceExpr).is_some_and(|it| it.base.is_none()) =>
963            {
964                // special handling for local actions, get the parameter from the function context
965                let call_ptr = function_context
966                    .function
967                    .get_first_param()
968                    .map(|call_ptr| call_ptr.into_pointer_value())
969                    .ok_or_else(|| Diagnostic::cannot_generate_call_statement(operator))?;
970
971                self.generate_stateful_pou_arguments(
972                    &implementation.get_call_name_for_ir(),
973                    None,
974                    call_ptr,
975                    passed_parameters,
976                )
977            }
978            _ => {
979                let call_ptr = self.generate_lvalue(operator)?;
980                self.generate_stateful_pou_arguments(
981                    &implementation.get_call_name_for_ir(),
982                    None,
983                    call_ptr,
984                    passed_parameters,
985                )
986            }
987        }
988    }

990    fn generate_function_arguments(
991        &self,
992        pou: &PouIndexEntry,
993        arguments: &[&AstNode],
994        declared_parameters: Vec<&VariableIndexEntry>,
995    ) -> Result<Vec<BasicMetadataValueEnum<'ink>>, Diagnostic> {
996        let mut result = Vec::new();
997        let mut variadic_parameters = Vec::new();
998        let mut passed_param_indices = Vec::new();
999        for (i, argument) in arguments.iter().enumerate() {
1000            let (i, argument, _) = get_implicit_call_parameter(argument, &declared_parameters, i)?;
1001
1002            // parameter_info includes the declaration type and type name
1003            let parameter_info = declared_parameters
1004                .get(i)
1005                .map(|it| {
1006                    let name = it.get_type_name();
1007                    if let Some(DataTypeInformation::Pointer {
1008                        inner_type_name, auto_deref: Some(_), ..
1009                    }) = self.index.find_effective_type_info(name)
1010                    {
1011                        // for auto_deref pointers (VAR_INPUT {ref}, VAR_IN_OUT) we call generate_argument_by_ref()
1012                        // we need the inner_type and not pointer to type otherwise we would generate a double pointer
1013                        Some((it.get_declaration_type(), inner_type_name.as_str()))
1014                    } else {
1015                        Some((it.get_declaration_type(), name))
1016                    }
1017                })
1018                // TODO : Is this idomatic, we need to wrap in ok because the next step does not necessarily fail
1019                .map(Ok)
1020                // None -> possibly variadic
1021                .unwrap_or_else(|| {
1022                    // if we are dealing with a variadic function, we can accept all extra parameters
1023                    if pou.is_variadic() {
1024                        variadic_parameters.push(argument);
1025                        Ok(None)
1026                    } else {
1027                        // we are not variadic, we have too many parameters here
1028                        Err(Diagnostic::codegen_error("Too many parameters", argument))
1029                    }
1030                })?;
1031
1032            if let Some((declaration_type, type_name)) = parameter_info {
1033                let argument: BasicValueEnum = if declaration_type.is_by_ref()
1034                    || (self.index.get_effective_type_or_void_by_name(type_name).is_aggregate_type()
1035                        && declaration_type.is_input())
1036                {
1037                    let declared_parameter = declared_parameters.get(i);
1038                    self.generate_argument_by_ref(argument, type_name, declared_parameter.copied())?
1039                } else {
1040                    // by val
1041                    if !argument.is_empty_statement() {
1042                        self.generate_expression(argument)?
1043                    } else if let Some(param) = declared_parameters.get(i) {
1044                        self.generate_empty_expression(param)?
1045                    } else {
1046                        unreachable!("Statement param must have an index entry at this point.");
1047                    }
1048                };
1049                result.push((i, argument));
1050            }
1051
1052            passed_param_indices.push(i);
1053        }
1054
1055        // handle missing parameters, generate empty expression
1056        if declared_parameters.len() > passed_param_indices.len() {
1057            for (i, param) in declared_parameters.into_iter().enumerate() {
1058                if !passed_param_indices.contains(&i) {
1059                    let generated_exp = self.generate_empty_expression(param)?;
1060                    result.push((i, generated_exp));
1061                }
1062            }
1063        }
1064
1065        // push variadic collection and optionally the variadic size
1066        if pou.is_variadic() {
1067            let last_location = result.len();
1068            for (i, parameter) in
1069                self.generate_variadic_arguments_list(pou, &variadic_parameters)?.into_iter().enumerate()
1070            {
1071                result.push((i + last_location, parameter));
1072            }
1073        }
1074
1075        result.sort_by(|(idx_a, _), (idx_b, _)| idx_a.cmp(idx_b));
1076        Ok(result.into_iter().map(|(_, v)| v.into()).collect::<Vec<BasicMetadataValueEnum>>())
1077    }

1079    /// generates a value that is passed by reference
1080    /// this generates and returns a PointerValue
1081    /// pointing to the given `argument`
1082    fn generate_argument_by_ref(
1083        &self,
1084        argument: &AstNode,
1085        type_name: &str,
1086        declared_parameter: Option<&VariableIndexEntry>,
1087    ) -> Result<BasicValueEnum<'ink>, Diagnostic> {
1088        if argument.is_empty_statement() {
1089            // Uninitialized var_output / var_in_out
1090            let v_type = self
1091                .llvm_index
1092                .find_associated_type(type_name)
1093                .ok_or_else(|| Diagnostic::unknown_type(type_name, argument))?;
1094
1095            let ptr_value = self.llvm.builder.build_alloca(v_type, "");
1096            if let Some(p) = declared_parameter {
1097                if let Some(initial_value) =
1098                    self.get_initial_value(&p.initial_value, &self.get_parameter_type(p))
1099                {
1100                    let value = self.generate_expression(initial_value)?;
1101                    self.llvm.builder.build_store(ptr_value, value);
1102                }
1103            }
1104
1105            return Ok(ptr_value.into());
1106        }
1107
1108        // Generate the element pointer, then...
1109        let value = {
1110            let value = self.generate_expression_value(argument)?;
1111            match value {
1112                ExpressionValue::LValue(v) => v,
1113                ExpressionValue::RValue(_v) => {
1114                    // Passed a literal to a byref parameter?
1115                    let value = self.generate_expression(argument)?;
1116                    let argument = self.llvm.builder.build_alloca(value.get_type(), "");
1117                    self.llvm.builder.build_store(argument, value);
1118                    argument
1119                }
1120            }
1121        };
1122
1123        // ...check if we can bitcast a reference to their hinted type
1124        if let Some(hint) = self.annotations.get_type_hint(argument, self.index) {
1125            let actual_type = self.annotations.get_type_or_void(argument, self.index);
1126            let actual_type_info = self.index.find_elementary_pointer_type(&actual_type.information);
1127            let target_type_info = self.index.find_elementary_pointer_type(&hint.information);
1128
1129            if target_type_info.is_vla() {
1130                // XXX: Calling `cast_if_needed` will result in an `alloca` call for EVERY function call.
1131                // LLVM might be able to optimize it away but ideally we find a solution for this at some
1132                // point? For a more in-depth description see the `pass` function in `vla_adr.rs`
1133                return Ok(cast_if_needed!(
1134                    self,
1135                    hint,
1136                    actual_type,
1137                    value.into(),
1138                    self.annotations.get(argument)
1139                ));
1140            };
1141
1142            // From https://llvm.org/docs/LangRef.html#bitcast-to-instruction: The ‘bitcast’ instruction takes
1143            // a value to cast, which must be a **non-aggregate** first class value [...]
1144            if !actual_type_info.is_aggregate() && actual_type_info != target_type_info {
1145                return Ok(self.llvm.builder.build_bitcast(
1146                    value,
1147                    self.llvm_index.get_associated_type(hint.get_name())?,
1148                    "",
1149                ));
1150            }
1151        }
1152
1153        // ...check if we can bitcast an array to a pointer, i.e. `[81 x i8]*` should be passed as a `i8*`
1154        if value.get_type().get_element_type().is_array_type() {
1155            let res = self.llvm.builder.build_bitcast(
1156                value,
1157                value
1158                    .get_type()
1159                    .get_element_type()
1160                    .into_array_type()
1161                    .get_element_type()
1162                    .ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)),
1163                "",
1164            );
1165
1166            return Ok(res.into_pointer_value().into());
1167        }
1168
1169        // ...otherwise no bitcasting was needed, thus return the generated element pointer as is
1170        Ok(value.into())
1171    }

1173    pub fn generate_variadic_arguments_list(
1174        &self,
1175        pou: &PouIndexEntry,
1176        variadic_params: &[&AstNode],
1177    ) -> Result<Vec<BasicValueEnum<'ink>>, Diagnostic> {
1178        // get the real varargs from the index
1179        if let Some((var_args, argument_type)) = self
1180            .index
1181            .get_variadic_member(pou.get_name())
1182            .and_then(|it| it.get_varargs().zip(Some(it.get_declaration_type())))
1183        {
1184            let generated_params = variadic_params
1185                .iter()
1186                .map(|param_statement| {
1187                    self.get_type_hint_for(param_statement).map(|it| it.get_name()).and_then(|type_name| {
1188                        // if the variadic is defined in a by_ref block, we need to pass the argument as reference
1189                        if argument_type.is_by_ref() {
1190                            self.generate_argument_by_ref(
1191                                param_statement,
1192                                type_name,
1193                                self.index.get_variadic_member(pou.get_name()),
1194                            )
1195                        } else {
1196                            self.generate_expression(param_statement)
1197                        }
1198                    })
1199                })
1200                .collect::<Result<Vec<_>, _>>()?;
1201
1202            // for sized variadics we create an array and store all the arguments in that array
1203            if let VarArgs::Sized(Some(type_name)) = var_args {
1204                let ty = self.llvm_index.get_associated_type(type_name).map(|it| {
1205                    if argument_type.is_by_ref() && it.is_array_type() {
1206                        it.into_array_type().get_element_type()
1207                    } else {
1208                        it
1209                    }
1210                })?;
1211                // if the variadic argument is ByRef, wrap it in a pointer.
1212                let ty = if argument_type.is_by_ref() {
1213                    ty.ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)).into()
1214                } else {
1215                    ty
1216                };
1217
1218                let size = generated_params.len();
1219                let size_param = self.llvm.i32_type().const_int(size as u64, true);
1220
1221                let arr = ty.array_type(size as u32);
1222                let arr_storage = self.llvm.builder.build_alloca(arr, "");
1223                for (i, ele) in generated_params.iter().enumerate() {
1224                    let ele_ptr = self.llvm.load_array_element(
1225                        arr_storage,
1226                        &[
1227                            self.llvm.context.i32_type().const_zero(),
1228                            self.llvm.context.i32_type().const_int(i as u64, true),
1229                        ],
1230                        "",
1231                    )?;
1232                    self.llvm.builder.build_store(ele_ptr, *ele);
1233                }
1234
1235                // bitcast the array to pointer so it matches the declared function signature
1236                let arr_storage = self.llvm.builder.build_bitcast(
1237                    arr_storage,
1238                    ty.ptr_type(AddressSpace::from(ADDRESS_SPACE_GENERIC)),
1239                    "",
1240                );
1241
1242                Ok(vec![size_param.into(), arr_storage])
1243            } else {
1244                Ok(generated_params)
1245            }
1246        } else {
1247            unreachable!("Function must be variadic")
1248        }
1249    }

1361    /// generates the appropriate value for the given expression where the expression
1362    /// is a call's implicit argument (hence: foo(3), not foo(in := 3))
1363    fn generate_nameless_parameter(
1364        &self,
1365        param_context: &CallParameterAssignment,
1366    ) -> Result<Option<BasicValueEnum<'ink>>, Diagnostic> {
1367        let builder = &self.llvm.builder;
1368        let function_name = param_context.function_name;
1369        let index = param_context.index;
1370        let parameter_struct = param_context.parameter_struct;
1371        let expression = param_context.assignment;
1372        if let Some(parameter) = self.index.get_declared_parameter(function_name, index) {
1373            // this happens before the pou call
1374            // before the call statement we may only consider inputs and inouts
1375            // after the call we need to copy the output values to the correct assigned variables
1376            if matches!(parameter.get_variable_type(), VariableType::Output) {
1377                return Ok(None);
1378            }
1379
1380            let pointer_to_param = builder.build_struct_gep(parameter_struct, index, "").map_err(|_| {
1381                Diagnostic::codegen_error(
1382                    format!("Cannot build generate parameter: {expression:#?}"),
1383                    expression,
1384                )
1385            })?;
1386
1387            let parameter = self
1388                .index
1389                .find_parameter(function_name, index)
1390                .and_then(|var| self.index.find_effective_type_by_name(var.get_type_name()))
1391                .map(|var| var.get_type_information())
1392                .unwrap_or_else(|| self.index.get_void_type().get_type_information());
1393
1394            if let DataTypeInformation::Pointer { auto_deref: Some(_), inner_type_name, .. } = parameter {
1395                //this is a VAR_IN_OUT assignment, so don't load the value, assign the pointer
1396                //expression may be empty -> generate a local variable for it
1397                let generated_exp = if expression.is_empty_statement() {
1398                    let temp_type = self
1399                        .llvm_index
1400                        .find_associated_type(inner_type_name)
1401                        .ok_or_else(|| Diagnostic::unknown_type(parameter.get_name(), expression))?;
1402                    builder.build_alloca(temp_type, "empty_varinout").as_basic_value_enum()
1403                } else {
1404                    self.generate_lvalue(expression)?.as_basic_value_enum()
1405                };
1406                builder.build_store(pointer_to_param, generated_exp);
1407            } else {
1408                self.generate_store(pointer_to_param, parameter, expression)?;
1409            };
1410
1411            Ok(None)
1412        } else {
1413            Ok(Some(self.generate_expression(expression)?))
1414        }
1415    }

1417    /// generates the appropriate value for the given `right` expression  that is passed in a call
1418    /// to the parameter represented by `left`
1419    /// it's a call's explicit argument (hence: foo(left := right), not foo(right))
1420    fn generate_formal_parameter(
1421        &self,
1422        param_context: &CallParameterAssignment,
1423        left: &AstNode,
1424        right: &AstNode,
1425    ) -> Result<(), Diagnostic> {
1426        let function_name = param_context.function_name;
1427        let parameter_struct = param_context.parameter_struct;
1428
1429        if let Some(StatementAnnotation::Variable { qualified_name, .. }) = self.annotations.get(left) {
1430            let parameter = self
1431                .index
1432                .find_fully_qualified_variable(qualified_name)
1433                .ok_or_else(|| Diagnostic::unresolved_reference(qualified_name, left))?;
1434            let index = parameter.get_location_in_parent();
1435
1436            // don't generate param assignments for empty statements, with the exception
1437            // of VAR_IN_OUT params - they need an address to point to
1438            let is_auto_deref = self
1439                .index
1440                .find_effective_type_by_name(parameter.get_type_name())

1442                .unwrap_or(self.index.get_void_type().get_type_information())
1443                .is_auto_deref();

1446                    assignment: right,
1447                    function_name,
1448                    index,
1449                    parameter_struct,

1608    /// generates a gep statement for a array-reference with an optional qualifier
1609    ///
1610    /// - `qualifier` an optional qualifier for a reference (e.g. myStruct.x[2] where myStruct is the qualifier for x)
1611    /// - `reference` the reference-statement pointing to the array
1612    /// - `access` the accessor expression (the expression between the brackets: reference[access])
1613    fn generate_element_pointer_for_array(
1614        &self,
1615        reference: &AstNode,
1616        access: &AstNode,
1617    ) -> Result<PointerValue<'ink>, Diagnostic> {
1618        //Load the reference
1619        self.generate_expression_value(reference)
1620            .map(|it| it.get_basic_value_enum().into_pointer_value())
1621            .and_then(|lvalue| {
1622                if let DataTypeInformation::Array { dimensions, .. } =
1623                    self.get_type_hint_info_for(reference)?
1624                {
1625                    // make sure dimensions match statement list
1626                    let statements = access.get_as_list();
1627                    if statements.is_empty() || statements.len() != dimensions.len() {
1628                        return Err(Diagnostic::codegen_error("Invalid array access", access));
1629                    }
1630
1631                    // e.g. an array like `ARRAY[0..3, 0..2, 0..1] OF ...` has the lengths [ 4 , 3 , 2 ]
1632                    let lengths = dimensions
1633                        .iter()
1634                        .map(|d| d.get_length(self.index))
1635                        .collect::<Result<Vec<_>, _>>()
1636                        .map_err(|msg| {
1637                            Diagnostic::codegen_error(
1638                                format!("Invalid array dimensions access: {msg}").as_str(),
1639                                access,
1640                            )
1641                        })?;
1642
1643                    // the portion indicates how many elements are represented by the corresponding dimension
1644                    // the first dimensions corresponds to the number of elements of the rest of the dimensions
1645                    //          - portion = 6 (size to the right = 3 x 2 = 6)
1646                    //         /        - portion = 2 (because the size to the right of this dimension = 2)
1647                    //        /        /         - the last dimension is directly translated into array-coordinates (skip-size = 1)
1648                    //                /         /
1649                    // [    4   ,    3    ,    2  ]
1650                    let dimension_portions = (0..lengths.len())
1651                        .map(|index| {
1652                            if index == lengths.len() - 1 {
1653                                1
1654                            } else {
1655                                lengths[index + 1..lengths.len()].iter().product()
1656                            }
1657                        })
1658                        .collect::<Vec<u32>>();
1659
1660                    let accessors_and_portions = statements
1661                        .iter()
1662                        .zip(dimensions)
1663                        .map(|(statement, dimension)|
1664                            // generate array-accessors
1665                            self.generate_access_for_dimension(dimension, statement))
1666                        .zip(dimension_portions);
1667
1668                    // accessing [ 1, 2, 2] means to access [ 1*6 + 2*2 + 2*1 ] = 12
1669                    let (index_access, _) = accessors_and_portions.fold(
1670                        (Ok(self.llvm.i32_type().const_zero().as_basic_value_enum()), 1),
1671                        |(accumulated_value, _), (current_v, current_portion)| {
1672                            let result = accumulated_value.and_then(|last_v| {
1673                                current_v.map(|v| {
1674                                    let current_portion_value = self
1675                                        .llvm
1676                                        .i32_type()
1677                                        .const_int(current_portion as u64, false)
1678                                        .as_basic_value_enum();
1679                                    // multiply the accessor with the dimension's portion
1680                                    let m_v = self.create_llvm_int_binary_expression(
1681                                        &Operator::Multiplication,
1682                                        current_portion_value,
1683                                        v,
1684                                    );
1685                                    // take the sum of the mulitlication and the previous accumulated_value
1686                                    // this now becomes the new accumulated value
1687                                    self.create_llvm_int_binary_expression(&Operator::Plus, m_v, last_v)
1688                                })
1689                            });
1690                            (result, 0 /* the 0 will be ignored */)
1691                        },
1692                    );
1693
1694                    // make sure we got an int-value
1695                    let index_access: IntValue = index_access.and_then(|it| {
1696                        it.try_into()
1697                            .map_err(|_| Diagnostic::codegen_error("non-numeric index-access", access))
1698                    })?;
1699
1700                    let accessor_sequence = if lvalue.get_type().get_element_type().is_array_type() {
1701                        // e.g.: [81 x i32]*
1702                        // the first index (0) will point to the array -> [81 x i32]
1703                        // the second index (index_access) will point to the element in the array
1704                        vec![self.llvm.i32_type().const_zero(), index_access]
1705                    } else {
1706                        // lvalue is a pointer to type -> e.g.: i32*
1707                        // only one index (index_access) is needed to access the element
1708
1709                        // IGNORE the additional first index (0)
1710                        // it would point to -> i32
1711                        // we can't access any element of i32
1712                        vec![index_access]
1713                    };
1714
1715                    // load the access from that array
1716                    let pointer = self.llvm.load_array_element(lvalue, &accessor_sequence, "tmpVar")?;
1717
1718                    return Ok(pointer);
1719                }
1720                Err(Diagnostic::codegen_error("Invalid array access", access))
1721            })
1722    }

1724    /// generates the result of an pointer binary-expression
1725    ///
1726    /// - `operator` the binary operator
1727    /// - `left` the left side of the binary expression, needs to be an pointer/int-value
1728    /// - `left_type` DataTypeInformation of the left side
1729    /// - `right` the right side of the binary expression, needs to be an pointer/int-value
1730    /// - `right_type` DataTypeInformation of the right side
1731    /// - `expression` the binary expression
1732    pub fn create_llvm_binary_expression_for_pointer(
1733        &self,
1734        operator: &Operator,
1735        left: &AstNode,
1736        left_type: &DataTypeInformation,
1737        right: &AstNode,
1738        right_type: &DataTypeInformation,
1739        expression: &AstNode,
1740    ) -> Result<BasicValueEnum<'ink>, Diagnostic> {
1741        let left_expr = self.generate_expression(left)?;
1742        let right_expr = self.generate_expression(right)?;
1743
1744        let result = match operator {
1745            Operator::Plus | Operator::Minus => {
1746                let (ptr, index, name) = if left_type.is_pointer() && right_type.is_int() {
1747                    let ptr = left_expr.into_pointer_value();
1748                    let index = right_expr.into_int_value();
1749                    let name = format!("access_{}", left_type.get_name());
1750                    (Some(ptr), Some(index), Some(name))
1751                } else if left_type.is_int() && right_type.is_pointer() {
1752                    let ptr = right_expr.into_pointer_value();
1753                    let index = left_expr.into_int_value();
1754                    let name = format!("access_{}", right_type.get_name());
1755                    (Some(ptr), Some(index), Some(name))
1756                } else {
1757                    // if left and right are both pointers we can not perform plus/minus
1758                    (None, None, None)
1759                };

1761                if let (Some(ptr), Some(mut index), Some(name)) = (ptr, index, name) {
1762                    // if operator is minus we need to negate the index
1763                    if let Operator::Minus = operator {
1764                        index = self.int_neg(index);
1765                    }

1784                .as_basic_value_enum()),
1785            Operator::NotEqual => Ok(self
1786                .llvm
1787                .builder
1788                .build_int_compare(
1789                    IntPredicate::NE,
1790                    self.convert_to_int_value_if_pointer(left_expr),
1791                    self.convert_to_int_value_if_pointer(right_expr),
1792                    "tmpVar",
1793                )
1794                .as_basic_value_enum()),
1795            Operator::Less => Ok(self
1796                .llvm
1797                .builder
1798                .build_int_compare(
1799                    IntPredicate::SLT,
1800                    self.convert_to_int_value_if_pointer(left_expr),
1801                    self.convert_to_int_value_if_pointer(right_expr),
1802                    "tmpVar",
1803                )
1804                .as_basic_value_enum()),
1805            Operator::Greater => Ok(self
1806                .llvm
1807                .builder
1808                .build_int_compare(
1809                    IntPredicate::SGT,
1810                    self.convert_to_int_value_if_pointer(left_expr),
1811                    self.convert_to_int_value_if_pointer(right_expr),

2383    /// generates a phi-expression (&& or || expression) with respect to short-circuit evaluation
2384    ///
2385    /// - `operator` AND / OR
2386    /// - `left` the left side of the expression as an i1 value
2387    /// - `right` the right side of an expression as an i1 value
2388    pub fn generate_bool_short_circuit_expression(
2389        &self,
2390        operator: &Operator,
2391        left: &AstNode,
2392        right: &AstNode,
2393    ) -> Result<BasicValueEnum<'ink>, Diagnostic> {
2394        let builder = &self.llvm.builder;
2395        let lhs = to_i1(self.generate_expression(left)?.into_int_value(), builder);
2396        let function = self.get_function_context(left)?.function;
2397
2398        let right_branch = self.llvm.context.append_basic_block(function, "");
2399        let continue_branch = self.llvm.context.append_basic_block(function, "");
2400
2401        let final_left_block = builder.get_insert_block().expect(INTERNAL_LLVM_ERROR);
2402        //Compare left to 0
2403
2404        match operator {
2405            Operator::Or => builder.build_conditional_branch(lhs, continue_branch, right_branch),
2406            Operator::And => builder.build_conditional_branch(lhs, right_branch, continue_branch),
2407            _ => {
2408                return Err(Diagnostic::codegen_error(
2409                    format!("Cannot generate phi-expression for operator {operator:}"),
2410                    left,
2411                ));
2412            }
2413        };
2414
2415        builder.position_at_end(right_branch);
2416        let rhs = to_i1(self.generate_expression(right)?.into_int_value(), builder);
2417        let final_right_block = builder.get_insert_block().expect(INTERNAL_LLVM_ERROR);
2418        builder.build_unconditional_branch(continue_branch);
2419
2420        builder.position_at_end(continue_branch);
2421        //Generate phi
2422        let phi_value = builder.build_phi(lhs.get_type(), "");
2423        //assert
2424        phi_value.add_incoming(&[(&lhs, final_left_block), (&rhs, final_right_block)]);
2425
2426        Ok(phi_value.as_basic_value())
2427    }

2495    fn build_memcpy(
2496        &self,
2497        left: inkwell::values::PointerValue<'ink>,
2498        left_type: &DataTypeInformation,
2499        left_location: SourceLocation,
2500        right: inkwell::values::PointerValue<'ink>,
2501        right_type: &DataTypeInformation,
2502        right_location: SourceLocation,
2503    ) -> Result<PointerValue<'ink>, Diagnostic> {
2504        let (size, alignment) = match (left_type, right_type) {
2505            (
2506                DataTypeInformation::String { size: lsize, .. },
2507                DataTypeInformation::String { size: rsize, .. },
2508            ) => {
2509                let target_size = lsize
2510                    .as_int_value(self.index)
2511                    .map_err(|err| Diagnostic::codegen_error(err.as_str(), &left_location))?;
2512                let value_size = rsize
2513                    .as_int_value(self.index)
2514                    .map_err(|err| Diagnostic::codegen_error(err.as_str(), right_location))?;
2515                let size = std::cmp::min(target_size - 1, value_size);
2516                //FIXME: use the target_layout for this operation
2517                let alignment = left_type.get_string_character_width(self.index).value();
2518                //Multiply by the string alignment to copy enough for widestrings
2519                //This is done at compile time to avoid generating an extra mul
2520                let size = self.llvm.context.i32_type().const_int((size * alignment as i64) as u64, true);
2521                (size, alignment)
2522            }
2523            (DataTypeInformation::Array { .. }, DataTypeInformation::Array { .. })
2524            | (DataTypeInformation::Struct { .. }, DataTypeInformation::Struct { .. }) => {
2525                let size = self.llvm_index.get_associated_type(right_type.get_name())?.size_of().ok_or_else(
2526                    || {
2527                        Diagnostic::codegen_error(
2528                            format!("Unknown size of type {}.", right_type.get_name()).as_str(),
2529                            right_location,
2530                        )
2531                    },
2532                )?;
2533                (size, 1)
2534            }
2535            _ => unreachable!("memcpy is not used for non-aggregate types"),
2536        };
2537
2538        self.llvm
2539            .builder
2540            .build_memcpy(left, alignment, right, alignment, size)
2541            .map_err(|err| Diagnostic::codegen_error(err, left_location))
2542    }

2560    /// Generate a GEP instruction, accessing the array pointed to within the VLA struct at runtime
2561    fn generate_element_pointer_for_vla(
2562        &self,
2563        reference: ExpressionValue<'ink>,
2564        reference_annotation: &StatementAnnotation,
2565        access: &AstNode,
2566    ) -> Result<PointerValue<'ink>, Diagnostic> {
2567        let builder = &self.llvm.builder;
2568
2569        // array access is either directly on a reference or on another array access (ARRAY OF ARRAY)
2570
2571        let StatementAnnotation::Variable { resulting_type: reference_type, .. } = reference_annotation
2572        else {
2573            unreachable!();
2574        };
2575
2576        let struct_ptr = reference.get_basic_value_enum().into_pointer_value();
2577        // GEPs into the VLA struct, getting an LValue for the array pointer and the dimension array and
2578        // dereferences the former
2579        let arr_ptr_gep = self.llvm.builder.build_struct_gep(struct_ptr, 0, "vla_arr_gep").map_err(|_| {
2580            Diagnostic::codegen_error("Cannot access VLA array pointer", access.get_location())
2581        })?;
2582        let vla_arr_ptr = builder.build_load(arr_ptr_gep, "vla_arr_ptr").into_pointer_value();
2583        // get pointer to array containing dimension information
2584        let dim_arr_gep = builder.build_struct_gep(struct_ptr, 1, "dim_arr").unwrap();
2585
2586        // get lengths of dimensions
2587        let type_ = self.index.get_type_information_or_void(reference_type);
2588        let Some(ndims) = type_.get_type_information().get_dimension_count() else { unreachable!() };
2589
2590        // get the start/end offsets for each dimension ( ARRAY[4..10, -4..4] ...)
2591        let index_offsets = get_indices(self.llvm, ndims, dim_arr_gep);
2592
2593        // calculate the required offset from the array pointer for the given accessor statements. this is
2594        // relatively straightforward for single-dimensional arrays, but is quite costly (O(n²)) for multi-dimensional arrays
2595        let access_statements = access.get_as_list();
2596        let accessor = if access_statements.len() == 1 {
2597            let Some(stmt) = access_statements.first() else {
2598                unreachable!("Must have exactly 1 access statement")
2599            };
2600            //turn it into i32 immediately
2601            // this is the  same logic we use for normal array accessors
2602            let access_value = cast_if_needed!(
2603                self,
2604                self.index.get_type(DINT_TYPE)?,
2605                self.get_type_hint_for(stmt)?,
2606                self.generate_expression(stmt)?,
2607                None
2608            );
2609
2610            // if start offset is not 0, adjust the access value accordingly
2611            let Some(start_offset) = index_offsets.first().map(|(start, _)| *start) else {
2612                unreachable!("VLA must have information about dimension offsets")
2613            };
2614            self.create_llvm_int_binary_expression(&Operator::Minus, access_value, start_offset.into())
2615                .into_int_value()
2616        } else {
2617            // see https://plc-lang.github.io/rusty/arch/codegen.html#multi-dimensional-arrays
2618            // for more details on multi-dimensional array accessor calculation
2619            let accessors = access_statements
2620                .iter()
2621                .map(|it| {
2622                    let value = self.generate_expression(it).expect("Uncaught invalid accessor statement");
2623                    //turn it into i32 immediately
2624                    // this is the  same logic we use for normal array accessors
2625                    Ok(cast_if_needed!(
2626                        self,
2627                        self.index.get_type(DINT_TYPE)?,
2628                        self.get_type_hint_for(it)?,
2629                        value,
2630                        None
2631                    )
2632                    .into_int_value())
2633                })
2634                .collect::<Result<Vec<_>, Diagnostic>>()?;
2635
2636            if access_statements.len() != index_offsets.len() {
2637                unreachable!("Amount of access statements and dimensions does not match.")
2638            }
2639
2640            // length of a dimension is 'end - start + 1'
2641            let lengths = get_dimension_lengths(self.llvm, &index_offsets);
2642
2643            // calculate the accessor multiplicators for each dimension.
2644            let dimension_offsets = get_vla_accessor_factors(self.llvm, &lengths);
2645
2646            // adjust accessors for 0-indexing
2647            let adjusted_accessors = normalize_offsets(self.llvm, &accessors, &index_offsets);
2648
2649            // calculate the resulting accessor for the given accessor statements and dimension offsets
2650            int_value_multiply_accumulate(
2651                self.llvm,
2652                &adjusted_accessors.iter().zip(&dimension_offsets).collect::<Vec<_>>(),
2653            )
2654        };
2655
2656        Ok(unsafe { builder.build_in_bounds_gep(vla_arr_ptr, &[accessor], "arr_val") })
2657    }

2659    /// generates a reference expression (member, index, deref, etc.)
2660    ///
2661    /// - `access` the ReferenceAccess of the reference to generate
2662    /// - `base` the "previous" segment of an optional qualified reference-access
2663    /// - `original_expression` the original ast-statement used to report Diagnostics
2664    pub(crate) fn generate_reference_expression(
2665        &self,
2666        access: &ReferenceAccess,
2667        base: Option<&AstNode>,
2668        original_expression: &AstNode,
2669    ) -> Result<ExpressionValue<'ink>, Diagnostic> {
2670        match (access, base) {
2671            // `.foo`
2672            (ReferenceAccess::Global(node), _) => {
2673                let name = node.get_flat_reference_name().unwrap_or("unknown");
2674
2675                self.create_llvm_pointer_value_for_reference(
2676                    None,
2677                    self.get_load_name(node).as_deref().unwrap_or(name),
2678                    node,
2679                ).map(ExpressionValue::LValue)
2680            }
2681
2682            // `base.member` or just `member`
2683            (ReferenceAccess::Member(member), base) => {
2684                let base_value = base.map(|it| self.generate_expression_value(it)).transpose()?;
2685
2686                if let AstStatement::DirectAccess(data) = member.as_ref().get_stmt() {
2687                    let (Some(base), Some(base_value)) = (base, base_value) else {
2688                        return Err(Diagnostic::codegen_error("Cannot generate DirectAccess without base value.", original_expression));
2689                    };
2690                    self.generate_direct_access_expression(base, &base_value, member, &data.access, &data.index)
2691                } else {
2692                    let member_name = member.get_flat_reference_name().unwrap_or("unknown");
2693
2694                    self.create_llvm_pointer_value_for_reference(
2695                        base_value.map(|it| it.get_basic_value_enum().into_pointer_value()).as_ref(),
2696                        self.get_load_name(member).as_deref().unwrap_or(member_name),
2697                        original_expression,
2698                    ).map(ExpressionValue::LValue)
2699                }
2700            }
2701
2702            // `base[idx]`
2703            (ReferenceAccess::Index(array_idx), Some(base)) => {
2704                if self.annotations.get_type_or_void(base, self.index).is_vla() {
2705                    // vla array needs special handling
2706                    self.generate_element_pointer_for_vla(
2707                        self.generate_expression_value(base)?,
2708                        self.annotations.get(base).expect(""),
2709                        array_idx.as_ref(),
2710                    )
2711                        .map_err(|_| unreachable!("invalid access statement"))
2712                        .map(ExpressionValue::LValue)
2713                } else {
2714                    // normal array expression
2715                    self.generate_element_pointer_for_array(base, array_idx).map(ExpressionValue::LValue)
2716                }
2717            }
2718
2719            // `INT#target` (INT = base)
2720            (ReferenceAccess::Cast(target), Some(base)) => {
2721                if target.as_ref().is_identifier() {
2722                    let mr =
2723                        AstFactory::create_member_reference(target.as_ref().clone(), None, target.get_id());
2724                    self.generate_expression_value(&mr)
2725                } else if target.as_ref().is_literal(){
2726                    self.generate_expression_value(target.as_ref())
2727                } else {
2728                    // Otherwise just bitcast the target to the given type
2729                    let base_type = self.annotations.get_type_or_void(base, self.index);
2730                    let base_type_name = base_type.get_name();
2731
2732                    // Generate the value we're casting
2733                    let target_value = self.generate_expression_value(target.as_ref())?;
2734
2735                    // Get the LLVM type for the cast target
2736                    let target_llvm_type = self.llvm_index.get_associated_type(base_type_name)
2737                        .map(|t| t.ptr_type(AddressSpace::from(0)))
2738                        .unwrap_or_else(|_| self.llvm.context.i8_type().ptr_type(AddressSpace::from(0)));
2739
2740                    // Perform the bitcast
2741                    let basic_value = target_value.get_basic_value_enum();
2742                    let cast_ptr = self.llvm.builder.build_bitcast(basic_value, target_llvm_type, "cast");
2743                    let cast_value = ExpressionValue::RValue(cast_ptr);
2744
2745                    Ok(cast_value)
2746                }
2747            }
2748
2749            // `base^`
2750            (ReferenceAccess::Deref, Some(base)) => {
2751                let ptr = self.generate_expression_value(base)?;
2752                Ok(ExpressionValue::LValue(
2753                    self.llvm
2754                        .load_pointer(&ptr.get_basic_value_enum().into_pointer_value(), "deref")
2755                        .into_pointer_value(),
2756                ))
2757            }
2758
2759            // `&base`
2760            (ReferenceAccess::Address, Some(base)) => {
2761                let lvalue = self.generate_expression_value(base)?;
2762                Ok(ExpressionValue::RValue(lvalue.get_basic_value_enum()))
2763            }
2764
2765            (ReferenceAccess::Index(_), None)   // [idx];
2766            | (ReferenceAccess::Cast(_), None)  // INT#;
2767            | (ReferenceAccess::Deref, None)    // ^;
2768            | (ReferenceAccess::Address, None)  // &;
2769            => Err(Diagnostic::codegen_error(
2770                "Expected a base expression, but found none",
2771                original_expression,
2772            ))
2773        }
2774    }

2776    /// generates a direct-access expression like `x.%B4`
2777    /// - `qualifier` the qualifier statement (see `x` above)
2778    /// - `qualifier_value` the generated value of the qualifier
2779    /// - `member` the member AstStatement (see `%B4`above)
2780    /// - `access` the type of access (see `B` above)
2781    fn generate_direct_access_expression(
2782        &self,
2783        qualifier: &AstNode,
2784        qualifier_value: &ExpressionValue<'ink>,
2785        member: &AstNode,
2786        access: &DirectAccessType,
2787        index: &AstNode,
2788    ) -> Result<ExpressionValue<'ink>, Diagnostic> {
2789        let loaded_base_value = qualifier_value.as_r_value(self.llvm, self.get_load_name(qualifier));
2790        let datatype = self.get_type_hint_info_for(member)?;
2791        let base_type = self.get_type_hint_for(qualifier)?;
2792
2793        //Generate and load the index value
2794        let rhs = self.generate_direct_access_index(access, index, datatype, base_type)?;
2795        //Shift the qualifer value right by the index value
2796        let shift = self.llvm.builder.build_right_shift(
2797            loaded_base_value.into_int_value(),
2798            rhs,
2799            base_type.get_type_information().is_signed_int(),
2800            "shift",
2801        );
2802        //Trunc the result to the get only the target size
2803        let value = self.llvm.builder.build_int_truncate_or_bit_cast(
2804            shift,
2805            self.llvm_index.get_associated_type(datatype.get_name())?.into_int_type(),
2806            "",
2807        );
2808
2809        let result = if datatype.get_type_information().is_bool() {
2810            // since booleans are i1 internally, but i8 in llvm, we need to bitwise-AND the value with 1 to make sure we end up with the expected value
2811            self.llvm.builder.build_and(value, self.llvm.context.i8_type().const_int(1, false), "")
2812        } else {
2813            value
2814        };
2815
2816        Ok(ExpressionValue::RValue(result.as_basic_value_enum()))
2817    }

2826pub fn get_implicit_call_parameter<'a>(

2828    parameters: &[&VariableIndexEntry],

2833        AstStatement::Assignment(data) | AstStatement::OutputAssignment(data) => {

2835                return Err(
2836                    //TODO: use global context to get an expression slice
2837                    Diagnostic::new("Expression is not assignable")
2838                        .with_error_code("E050")
2839                        .with_location(argument),
2840                );
2841            };
2842
2843            let loc = parameters
2844                .iter()
2845                .position(|p| p.get_name().eq_ignore_ascii_case(left_name))
2846                .ok_or_else(|| Diagnostic::unresolved_reference(left_name, data.left.as_ref()))?;
2847
2848            (loc, data.right.as_ref(), false)
2849        }
2850
2851        // Implicit
2852        _ => (idx, argument, true),
2853    };

2946/// Computes the product of all elements in a collection of IntValues
2947///
2948/// a <- a * b
2949fn int_value_product<'ink>(llvm: &Llvm<'ink>, values: &[IntValue<'ink>]) -> IntValue<'ink> {
2950    // initialize the accumulator with 1
2951    let accum_ptr = llvm.builder.build_alloca(llvm.i32_type(), "accum");
2952    llvm.builder.build_store(accum_ptr, llvm.i32_type().const_int(1, false));
2953    for val in values {
2954        // load previous value from accumulator and multiply with current value
2955        let product = llvm.builder.build_int_mul(
2956            llvm.builder.build_load(accum_ptr, "load_accum").into_int_value(),
2957            *val,
2958            "product",
2959        );
2960        // store new value into accumulator
2961        llvm.builder.build_store(accum_ptr, product);
2962    }
2963
2964    llvm.builder.build_load(accum_ptr, "accessor_factor").into_int_value()
2965}

2967/// Iterates over a collection of tuples, computes the product of the two numbers
2968/// and adds that product to an accumulator.
2969///
2970/// a <- a + (b * c)
2971fn int_value_multiply_accumulate<'ink>(
2972    llvm: &Llvm<'ink>,
2973    values: &[(&IntValue<'ink>, &IntValue<'ink>)],
2974) -> IntValue<'ink> {
2975    // initialize the accumulator with 0
2976    let accum = llvm.builder.build_alloca(llvm.i32_type(), "accum");
2977    llvm.builder.build_store(accum, llvm.i32_type().const_zero());
2978    for (left, right) in values {
2979        // multiply accessor with dimension factor
2980        let product = llvm.builder.build_int_mul(**left, **right, "multiply");
2981        // load previous value from accum and add product
2982        let curr = llvm.builder.build_int_add(
2983            llvm.builder.build_load(accum, "load_accum").into_int_value(),
2984            product,
2985            "accumulate",
2986        );
2987        // store new value into accumulator
2988        llvm.builder.build_store(accum, curr);
2989    }
2990    llvm.builder.build_load(accum, "accessor").into_int_value()
2991}
