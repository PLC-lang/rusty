35/// the full context when generating statements inside a POU
36#[derive(Debug)]
37pub struct FunctionContext<'ink, 'b> {
38    /// the current pou's name. This means that a variable x may refer to "`linking_context`.x"
39    pub linking_context: &'b ImplementationIndexEntry,
40    /// the llvm function to generate statements into
41    pub function: FunctionValue<'ink>,
42    /// The blocks/labels this function can use
43    pub blocks: FxHashMap<String, BasicBlock<'ink>>,
44}

46/// the StatementCodeGenerator is used to generate statements (For, If, etc.) or expressions (references, literals, etc.)
47pub struct StatementCodeGenerator<'a, 'b> {
48    llvm: &'b Llvm<'a>,
49    index: &'b Index,
50    annotations: &'b AstAnnotations,
51    llvm_index: &'b LlvmTypedIndex<'a>,
52    function_context: &'b FunctionContext<'a, 'b>,
53
54    pub load_prefix: String,
55    pub load_suffix: String,
56
57    /// the block to jump to when you want to exit the loop
58    pub current_loop_exit: Option<BasicBlock<'a>>,
59    /// the block to jump to when you want to continue the loop
60    pub current_loop_continue: Option<BasicBlock<'a>>,
61
62    pub debug: &'b DebugBuilderEnum<'a>,
63}

104    /// generates a list of statements
105    pub fn generate_body(&self, statements: &[AstNode]) -> Result<(), CodegenError> {
106        let mut child_index = LlvmTypedIndex::create_child(self.llvm_index);
107        for s in statements {
108            child_index = self.generate_statement(child_index, s)?;
109        }
110        Ok(())
111    }

113    /// some versions of llvm will crash on two consecutive return or
114    /// unconditional jump statements. the solution is to insert another
115    /// building block before the second one, so the don't directly
116    /// follow each other. this is what we call a buffer block.
117    fn generate_buffer_block(&self) {
118        let (builder, _, context) = self.get_llvm_deps();
119        let buffer_block = context
120            .insert_basic_block_after(builder.get_insert_block().expect(INTERNAL_LLVM_ERROR), "buffer_block");
121        builder.position_at_end(buffer_block);
122    }

124    /// genertes a single statement
125    ///
126    /// - `statement` the statement to be generated
127    pub fn generate_statement(
128        &self,
129        mut llvm_index: LlvmTypedIndex<'b>,
130        statement: &AstNode,
131    ) -> Result<LlvmTypedIndex<'b>, CodegenError> {
132        match statement.get_stmt() {
133            AstStatement::EmptyStatement(..) => {
134                //nothing to generate
135            }
136            AstStatement::Assignment(data, ..) => {
137                self.generate_assignment_statement(&llvm_index, &data.left, &data.right)?;
138            }
139            AstStatement::RefAssignment(data, ..) => {
140                self.generate_ref_assignment(&llvm_index, &data.left, &data.right)?;
141            }
142            AstStatement::ControlStatement(ctl_statement, ..) => {
143                self.generate_control_statement(&llvm_index, ctl_statement)?
144            }
145            AstStatement::ReturnStatement(ReturnStatement { condition }) => match condition {
146                Some(condition) => {
147                    self.generate_conditional_return(&llvm_index, statement, condition)?;
148                }
149                None => {
150                    self.register_debug_location(statement);
151                    self.generate_return_statement()?;
152                    self.generate_buffer_block(); // XXX(volsa): This is not needed on x86 but if removed segfaults on ARM
153                }
154            },
155            AstStatement::LabelStatement(LabelStatement { name }) => {
156                if let Some(block) = self.function_context.blocks.get(name) {
157                    //unconditionally jump to the label
158                    self.register_debug_location(statement);
159                    self.llvm.builder.build_unconditional_branch(*block)?;
160                    //Place the current instert block at the label statement
161                    self.llvm.builder.position_at_end(*block);
162                }
163            }
164            AstStatement::JumpStatement(JumpStatement { condition, .. }) => {
165                //Find the label to jump to
166                let Some(then_block) = self.annotations.get(statement).and_then(|label| {
167                    if let StatementAnnotation::Label { name } = label {
168                        self.function_context.blocks.get(name)
169                    } else {
170                        None
171                    }
172                }) else {
173                    return Err(Diagnostic::codegen_error(
174                        "Could not find label for {statement:?}",
175                        statement,
176                    )
177                    .into());
178                };
179                //Set current location as else block
180                let current_block = self.llvm.builder.get_insert_block().expect(INTERNAL_LLVM_ERROR);
181                let else_block = self.llvm.context.insert_basic_block_after(current_block, "else_block");
182
183                self.register_debug_location(condition);
184                let expression_generator = self.create_expr_generator(&llvm_index);
185                let condition = expression_generator.generate_expression(condition)?;
186
187                self.register_debug_location(statement);
188                self.llvm.builder.build_conditional_branch(
189                    condition.into_int_value(),
190                    *then_block,
191                    else_block,
192                )?;
193                // Make sure further code is at the else block
194                self.llvm.builder.position_at_end(else_block);
195            }
196            AstStatement::ExitStatement(_) => {
197                if let Some(exit_block) = &self.current_loop_exit {
198                    self.register_debug_location(statement);
199                    self.llvm.builder.build_unconditional_branch(*exit_block)?;
200                    self.generate_buffer_block();
201                } else {
202                    return Err(Diagnostic::codegen_error(
203                        "Cannot break out of loop when not inside a loop",
204                        statement,
205                    )
206                    .into());
207                }
208            }
209            AstStatement::ContinueStatement(_) => {
210                if let Some(cont_block) = &self.current_loop_continue {
211                    self.llvm.builder.build_unconditional_branch(*cont_block)?;
212                    self.generate_buffer_block();
213                } else {
214                    return Err(Diagnostic::codegen_error(
215                        "Cannot continue loop when not inside a loop",
216                        statement,
217                    )
218                    .into());
219                }
220            }
221            AstStatement::ExpressionList(statements) => {
222                let mut llvm_index = LlvmTypedIndex::create_child(&llvm_index);
223                for stmt in statements {
224                    llvm_index = self.generate_statement(llvm_index, stmt)?;
225                }
226            }
227            AstStatement::AllocationStatement(Allocation { name, reference_type }) => {
228                let ty =
229                    llvm_index.find_associated_type(reference_type).expect("Type must exist at this point");
230                let value = self.llvm.builder.build_alloca(ty, name)?;
231                self.llvm.generate_variable_initializer(
232                    &llvm_index,
233                    self.index,
234                    (name, reference_type, &statement.location),
235                    value,
236                    None,
237                    &self.create_expr_generator(&llvm_index),
238                )?;
239                llvm_index.associate_loaded_local_variable(
240                    self.function_context.linking_context.get_type_name(),
241                    name,
242                    value,
243                )?;
244            }
245            _ => {
246                self.create_expr_generator(&llvm_index).generate_expression(statement)?;
247            }
248        }
249        Ok(llvm_index)
250    }

252    /// genertes a single statement
253    ///
254    /// - `statement` the control statement to be generated
255    pub fn generate_control_statement(
256        &self,
257        llvm_index: &'a LlvmTypedIndex<'b>,
258        statement: &AstControlStatement,
259    ) -> Result<(), CodegenError> {
260        match statement {
261            AstControlStatement::If(ifstmt) => self.generate_if_statement(llvm_index, ifstmt),
262            AstControlStatement::ForLoop(for_stmt) => self.generate_for_statement(llvm_index, for_stmt),
263            AstControlStatement::WhileLoop(stmt) | AstControlStatement::RepeatLoop(stmt) => {
264                self.generate_loop_statement(llvm_index, stmt)
265            }
266            AstControlStatement::Case(stmt) => self.generate_case_statement(llvm_index, stmt),
267        }
268    }

270    /// Generates IR for a `REF=` assignment, which is syntactic sugar for an assignment where the
271    /// right-hand side is wrapped in a `REF(...)` call. Specifically `foo REF= bar` and
272    /// `foo := REF(bar)` are the same.
273    ///
274    /// Note: Although somewhat similar to the [`generate_assignment_statement`] function, we can't
275    /// apply the code here because the left side of a `REF=` assignment is flagged as auto-deref.
276    /// For `REF=` assignments we don't want (and can't) deref without generating incorrect IR.
277    pub fn generate_ref_assignment(
278        &self,
279        llvm_index: &'a LlvmTypedIndex<'b>,
280        left: &AstNode,
281        right: &AstNode,
282    ) -> Result<(), CodegenError> {
283        let exp = self.create_expr_generator(llvm_index);
284        let ref_builtin = self.index.get_builtin_function("REF").expect("REF must exist");
285
286        let AstStatement::ReferenceExpr(data) = &left.stmt else {
287            unreachable!("should be covered by a validation")
288        };
289
290        let left_ptr_val = {
291            let expr = exp.generate_reference_expression(&data.access, data.base.as_deref(), left)?;
292            expr.get_basic_value_enum().into_pointer_value()
293        };
294        let right_expr_val = if right.is_zero() {
295            exp.generate_literal(right)?
296        } else {
297            ref_builtin.codegen(&exp, &[right], right.get_location())?
298        };
299
300        self.llvm.builder.build_store(left_ptr_val, right_expr_val.get_basic_value_enum())?;
301        Ok(())
302    }

304    /// generates an assignment statement _left_ := _right_
305    ///
306    /// `left_statement` the left side of the assignment
307    /// `right_statement` the right side of the assignment
308    pub fn generate_assignment_statement(
309        &self,
310        llvm_index: &'a LlvmTypedIndex<'b>,
311        left_statement: &AstNode,
312        right_statement: &AstNode,
313    ) -> Result<(), CodegenError> {
314        //Register any debug info for the store
315        self.register_debug_location(left_statement);
316        //TODO: Looks hacky, the strings will be similar so we should look into making the assignment a bit nicer.
317        if left_statement.has_direct_access() {
318            return self.generate_assignment_statement_direct_access(
319                llvm_index,
320                left_statement,
321                right_statement,
322            );
323        }
324
325        if self.annotations.get(left_statement).is_some_and(|it| {
326            // TODO(mhasel): ideally the resolver decides which assignment statement to call when lowering the init functions,
327            // but that requires refactoring of how `aliases` and `reference to` LHS/RHS nodes are annotated. this is a workaround.
328            self.function_context.linking_context.is_init() && (it.is_alias() || it.is_reference_to())
329        }) {
330            return self.generate_ref_assignment(llvm_index, left_statement, right_statement);
331        };
332
333        let exp_gen = self.create_expr_generator(llvm_index);
334        let left: PointerValue = exp_gen.generate_expression_value(left_statement).and_then(|it| {
335            it.get_basic_value_enum()
336                .try_into()
337                .map_err(|err| CodegenError::new(format!("{err:?}").as_str(), left_statement))
338        })?;
339
340        let left_type = exp_gen.get_type_hint_info_for(left_statement)?;
341        // if the lhs-type is a subrange type we may need to generate a check-call
342        // e.g. x := y,  ==> x := CheckSignedInt(y);
343        let range_checked_right_side = if let DataTypeInformation::SubRange { .. } = left_type {
344            // there is a sub-range defined, so we need to wrap the right side into the check function if it exists
345            self.annotations.get_hidden_function_call(right_statement)
346        } else {
347            None
348        };
349
350        let right_statement = range_checked_right_side.unwrap_or(right_statement);
351
352        exp_gen.generate_store(left, left_type, right_statement)?;
353        Ok(())
354    }

356    fn register_debug_location(&self, statement: &AstNode) {
357        let line = statement.get_location().get_line_plus_one();
358        let column = statement.get_location().get_column();
359        self.debug.set_debug_location(self.llvm, self.function_context, line, column);
360    }

362    fn generate_assignment_statement_direct_access(
363        &self,
364        llvm_index: &'a LlvmTypedIndex<'b>,
365        left_statement: &AstNode,
366        right_statement: &AstNode,
367    ) -> Result<(), CodegenError> {
368        let exp_gen = self.create_expr_generator(llvm_index);
369
370        // Left pointer
371        let Some((base, _)) = collect_base_and_direct_access_for_assignment(left_statement) else {
372            unreachable!("Invalid direct-access expression: {left_statement:#?}")
373        };
374        let left_expr_value = exp_gen.generate_expression_value(base)?;
375        let left_value = left_expr_value.as_r_value(self.llvm, None)?.into_int_value();
376        let left_pointer = left_expr_value.get_basic_value_enum().into_pointer_value();
377
378        // Generate an expression for the right size
379        let right_type = exp_gen.get_type_hint_for(right_statement)?;
380        let right_expr = exp_gen.generate_expression(right_statement)?;
381
382        exp_gen.generate_assignment_with_direct_access(
383            left_statement,
384            left_value,
385            left_pointer,
386            right_type,
387            right_expr,
388        )
389    }

391    /// generates a for-loop statement
392    ///
393    /// FOR `counter` := `start` TO `end` BY `by_step` DO
394    ///
395    /// - `counter` the counter variable
396    /// - `start` the value indicating the start of the for loop
397    /// - `end` the value indicating the end of the for loop
398    /// - `by_step` the step of the loop
399    /// - `body` the statements inside the for-loop
400    fn generate_for_statement(
401        &self,
402        llvm_index: &'a LlvmTypedIndex<'b>,
403        stmt: &ForLoopStatement,
404    ) -> Result<(), CodegenError> {
405        let (builder, current_function, context) = self.get_llvm_deps();
406        let exp_gen = self.create_expr_generator(llvm_index);
407
408        let end_ty = self.annotations.get_type_or_void(&stmt.end, self.index);
409        let counter_ty = self.annotations.get_type_or_void(&stmt.counter, self.index);
410        let cast_target_ty = get_bigger_type(self.index.get_type_or_panic(DINT_TYPE), counter_ty, self.index);
411        let cast_target_llty = llvm_index.find_associated_type(cast_target_ty.get_name()).unwrap();
412
413        let step_ty = stmt.by_step.as_ref().map(|it| {
414            self.register_debug_location(it);
415            self.annotations.get_type_or_void(it, self.index)
416        });
417
418        let eval_step = || {
419            step_ty.map_or_else(
420                || self.llvm.create_const_numeric(&cast_target_llty, "1", SourceLocation::internal()),
421                |step_ty| {
422                    let step = exp_gen.generate_expression(stmt.by_step.as_ref().unwrap())?;
423                    cast_if_needed!(exp_gen, cast_target_ty, step_ty, step, None)
424                },
425            )
426        };
427
428        let predicate_incrementing = context.append_basic_block(current_function, "predicate_sle");
429        let predicate_decrementing = context.append_basic_block(current_function, "predicate_sge");
430        let loop_body = context.append_basic_block(current_function, "loop");
431        let increment = context.append_basic_block(current_function, "increment");
432        let afterloop = context.append_basic_block(current_function, "continue");
433
434        self.generate_assignment_statement(llvm_index, &stmt.counter, &stmt.start)?;
435        let counter = exp_gen.generate_lvalue(&stmt.counter)?;
436        let counter_pointee = {
437            let datatype = self.annotations.get_type(&stmt.counter, self.index).unwrap();
438            self.llvm_index.get_associated_type(&datatype.name).unwrap()
439        };
440
441        // generate loop predicate selector. since `STEP` can be a reference, this needs to be a runtime eval
442        // XXX(mhasel): IR could possibly be improved by generating phi instructions.
443        //              Candidate for frontend optimization for builds without optimization when `STEP`
444        //              is a compile-time constant
445        let is_incrementing = builder.build_int_compare(
446            inkwell::IntPredicate::SGT,
447            eval_step()?.into_int_value(),
448            self.llvm
449                .create_const_numeric(&cast_target_llty, "0", SourceLocation::internal())?
450                .into_int_value(),
451            "is_incrementing",
452        )?;
453        builder.build_conditional_branch(is_incrementing, predicate_incrementing, predicate_decrementing)?;
454        // generate predicates for incrementing and decrementing counters
455        let generate_predicate = |predicate| {
456            builder.position_at_end(match predicate {
457                inkwell::IntPredicate::SLE => predicate_incrementing,
458                inkwell::IntPredicate::SGE => predicate_decrementing,
459                _ => unreachable!(),
460            });
461
462            let end = exp_gen.generate_expression_value(&stmt.end).unwrap();
463            let end_value = match end {
464                ExpressionValue::LValue(value, pointee) => builder.build_load(pointee, value, "")?,
465                ExpressionValue::RValue(val) => val,
466            };
467
468            let counter_value = builder.build_load(counter_pointee, counter, "")?;
469            let cmp = builder.build_int_compare(
470                predicate,
471                cast_if_needed!(exp_gen, cast_target_ty, counter_ty, counter_value, None)?.into_int_value(),
472                cast_if_needed!(exp_gen, cast_target_ty, end_ty, end_value, None)?.into_int_value(),
473                "condition",
474            )?;
475            builder.build_conditional_branch(cmp, loop_body, afterloop)?;
476            Ok::<(), CodegenError>(())
477        };
478        generate_predicate(inkwell::IntPredicate::SLE)?;
479        generate_predicate(inkwell::IntPredicate::SGE)?;
480
481        // generate loop body
482        builder.position_at_end(loop_body);
483        let body_builder = StatementCodeGenerator {
484            current_loop_continue: Some(increment),
485            current_loop_exit: Some(afterloop),
486            load_prefix: self.load_prefix.clone(),
487            load_suffix: self.load_suffix.clone(),
488            ..*self
489        };
490        body_builder.generate_body(&stmt.body)?;
491        // Place debug location to end of loop
492        self.debug.set_debug_location(
493            self.llvm,
494            self.function_context,
495            stmt.end_location.get_line_plus_one(),
496            stmt.end_location.get_column(),
497        );
498
499        // increment counter
500        builder.build_unconditional_branch(increment)?;
501        builder.position_at_end(increment);
502        let counter_value = builder.build_load(counter_pointee, counter, "")?;
503        let inc = inkwell::values::BasicValue::as_basic_value_enum(&builder.build_int_add(
504            eval_step()?.into_int_value(),
505            cast_if_needed!(exp_gen, cast_target_ty, counter_ty, counter_value, None)?.into_int_value(),
506            "next",
507        )?);
508        builder.build_store(
509            counter,
510            cast_if_needed!(exp_gen, counter_ty, cast_target_ty, inc, None)?.into_int_value(),
511        )?;
512
513        // check condition
514        builder.build_conditional_branch(is_incrementing, predicate_incrementing, predicate_decrementing)?;
515        // continue
516        builder.position_at_end(afterloop);
517        Ok(())
518    }

520    /// genertes a case statement
521    ///
522    /// CASE selector OF
523    /// conditional_block#1:
524    /// conditional_block#2:
525    /// END_CASE;
526    ///
527    /// - `selector` the case's selector expression
528    /// - `conditional_blocks` all case-blocks including the condition and the body
529    /// - `else_body` the statements in the else-block
530    fn generate_case_statement(
531        &self,
532        llvm_index: &'a LlvmTypedIndex<'b>,
533        stmt: &CaseStatement,
534    ) -> Result<(), CodegenError> {
535        let (builder, current_function, context) = self.get_llvm_deps();
536        //Continue
537        let continue_block = context.append_basic_block(current_function, "continue");
538
539        let basic_block = builder.get_insert_block().expect(INTERNAL_LLVM_ERROR);
540        let exp_gen = self.create_expr_generator(llvm_index);
541        let selector_statement = exp_gen.generate_expression(&stmt.selector)?;
542
543        let mut cases = Vec::new();
544        let else_block = context.append_basic_block(current_function, "else");
545        let mut current_else_block = else_block;
546
547        for conditional_block in &stmt.case_blocks {
548            //craete a block for the case's body
549            let case_block = context.prepend_basic_block(else_block, "case");
550
551            //flatten the expression list into a vector of expressions
552            let expressions = flatten_expression_list(&conditional_block.condition);
553            for s in expressions {
554                if let AstStatement::RangeStatement(data, ..) = s.get_stmt() {
555                    //if this is a range statement, we generate an if (x >= start && x <= end) then the else-section
556                    builder.position_at_end(current_else_block);
557                    // since the if's generate additional blocks, we use the last one as the else-section
558                    current_else_block = self.generate_case_range_condition(
559                        llvm_index,
560                        &stmt.selector,
561                        data.start.as_ref(),
562                        data.end.as_ref(),
563                        case_block,
564                    )?;
565                } else {
566                    // this should be a a literal or a reference to a constant
567                    builder.position_at_end(basic_block);
568                    let condition = exp_gen.generate_expression(s)?; //TODO : Is a type conversion needed here?
569                                                                     // collect all literal case blocks to pass to the llvm switch-statement
570                    cases.push((condition.into_int_value(), case_block));
571                }
572            }
573            //generate the case's body
574            builder.position_at_end(case_block);
575            self.generate_body(&conditional_block.body)?;
576            //Add debug location to the end of the case
577            self.debug.set_debug_location(
578                self.llvm,
579                self.function_context,
580                stmt.end_location.get_line_plus_one(),
581                stmt.end_location.get_column(),
582            );
583            // skip all other case-bodies
584            builder.build_unconditional_branch(continue_block)?;
585        }
586        // current-else is the last else-block generated by the range-expressions
587        builder.position_at_end(current_else_block);
588        self.generate_body(&stmt.else_block)?;
589        //Add debug location to the end of the case
590        self.debug.set_debug_location(
591            self.llvm,
592            self.function_context,
593            stmt.end_location.get_line_plus_one(),
594            stmt.end_location.get_column(),
595        );
596        builder.build_unconditional_branch(continue_block)?;
597        continue_block.move_after(current_else_block).expect(INTERNAL_LLVM_ERROR);
598
599        // now that we collected all cases, go back to the initial block and generate the switch-statement
600        builder.position_at_end(basic_block);
601
602        self.register_debug_location(&stmt.selector);
603        builder.build_switch(selector_statement.into_int_value(), else_block, &cases)?;
604
605        builder.position_at_end(continue_block);
606        Ok(())
607    }

609    /// returns the new block to use as else
610    ///
611    ///
612    fn generate_case_range_condition(
613        &self,
614        llvm_index: &'a LlvmTypedIndex<'b>,
615        selector: &AstNode,
616        start: &AstNode,
617        end: &AstNode,
618        match_block: BasicBlock,
619    ) -> Result<BasicBlock<'_>, CodegenError> {
620        let (builder, _, context) = self.get_llvm_deps();
621
622        let range_then = context
623            .insert_basic_block_after(builder.get_insert_block().expect(INTERNAL_LLVM_ERROR), "range_then");
624        let range_else = context.insert_basic_block_after(range_then, "range_else");
625        let exp_gen = self.create_expr_generator(llvm_index);
626        let lower_bound = {
627            self.register_debug_location(start);
628            let start_val = exp_gen.generate_expression(start)?;
629            self.register_debug_location(selector);
630            let selector_val = exp_gen.generate_expression(selector)?;
631            exp_gen.create_llvm_int_binary_expression(&Operator::GreaterOrEqual, selector_val, start_val)?
632        };
633
634        //jmp to continue if the value is smaller than start
635        builder.build_conditional_branch(
636            to_i1(lower_bound.into_int_value(), builder)?,
637            range_then,
638            range_else,
639        )?;
640        builder.position_at_end(range_then);
641        let upper_bound = {
642            self.register_debug_location(end);
643            let end_val = exp_gen.generate_expression(end)?;
644            self.register_debug_location(selector);
645            let selector_val = exp_gen.generate_expression(selector)?;
646            exp_gen.create_llvm_int_binary_expression(&Operator::LessOrEqual, selector_val, end_val)?
647        };
648        builder.build_conditional_branch(
649            to_i1(upper_bound.into_int_value(), builder)?,
650            match_block,
651            range_else,
652        )?;
653        Ok(range_else)
654    }

656    /// generates a while statement
657    ///
658    /// WHILE condition DO
659    ///     body
660    /// END_WHILE
661    ///
662    /// - `condition` the while's condition
663    /// - `body` the while's body statements
664    fn generate_loop_statement(
665        &self,
666        llvm_index: &'a LlvmTypedIndex<'b>,
667        stmt: &LoopStatement,
668    ) -> Result<(), CodegenError> {
669        let builder = &self.llvm.builder;
670        let basic_block = builder.get_insert_block().expect(INTERNAL_LLVM_ERROR);
671        let (condition_block, _) =
672            self.generate_base_while_statement(llvm_index, &stmt.condition, &stmt.body, &stmt.end_location)?;
673
674        let continue_block = builder.get_insert_block().expect(INTERNAL_LLVM_ERROR);
675
676        builder.position_at_end(basic_block);
677        builder.build_unconditional_branch(condition_block)?;
678
679        builder.position_at_end(continue_block);
680        Ok(())
681    }

683    /// utility method for while and repeat loops
684    fn generate_base_while_statement(
685        &self,
686        llvm_index: &'a LlvmTypedIndex<'b>,
687        condition: &AstNode,
688        body: &[AstNode],
689        end_location: &SourceLocation,
690    ) -> Result<(BasicBlock<'_>, BasicBlock<'_>), CodegenError> {
691        let (builder, current_function, context) = self.get_llvm_deps();
692        let condition_check = context.append_basic_block(current_function, "condition_check");
693        let while_body = context.append_basic_block(current_function, "while_body");
694        let continue_block = context.append_basic_block(current_function, "continue");
695
696        //Check loop condition
697        builder.position_at_end(condition_check);
698        self.register_debug_location(condition);
699        let condition_value = self.create_expr_generator(llvm_index).generate_expression(condition)?;
700        builder.build_conditional_branch(
701            to_i1(condition_value.into_int_value(), builder)?,
702            while_body,
703            continue_block,
704        )?;
705
706        //Enter the for loop
707        builder.position_at_end(while_body);
708        let body_generator = StatementCodeGenerator {
709            current_loop_exit: Some(continue_block),
710            current_loop_continue: Some(condition_check),
711            load_prefix: self.load_prefix.clone(),
712            load_suffix: self.load_suffix.clone(),
713            ..*self
714        };
715        body_generator.generate_body(body)?;
716        //Set the debug location to the end of the loop
717        self.debug.set_debug_location(
718            self.llvm,
719            self.function_context,
720            end_location.get_line_plus_one(),
721            end_location.get_column(),
722        );
723        //Loop back
724        builder.build_unconditional_branch(condition_check)?;
725
726        //Continue
727        builder.position_at_end(continue_block);
728        Ok((condition_check, while_body))
729    }

731    /// generates an IF-Statement
732    ///
733    /// - `conditional_blocks` a list of conditions + bodies for every if  (respectivle else-if)
734    /// - `else_body` the list of statements in the else-block
735    fn generate_if_statement(
736        &self,
737        llvm_index: &'a LlvmTypedIndex<'b>,
738        stmt: &IfStatement,
739    ) -> Result<(), CodegenError> {
740        let (builder, current_function, context) = self.get_llvm_deps();
741        let mut blocks = vec![builder.get_insert_block().expect(INTERNAL_LLVM_ERROR)];
742        for _ in 1..stmt.blocks.len() {
743            blocks.push(context.append_basic_block(current_function, "branch"));
744        }
745
746        let else_block = if !stmt.else_block.is_empty() {
747            let result = context.append_basic_block(current_function, "else");
748            blocks.push(result);
749            Some(result)
750        } else {
751            None
752        };
753        //Continue
754        let continue_block = context.append_basic_block(current_function, "continue");
755        blocks.push(continue_block);
756
757        for (i, block) in stmt.blocks.iter().enumerate() {
758            let then_block = blocks[i];
759            let else_block = blocks[i + 1];
760
761            builder.position_at_end(then_block);
762
763            self.register_debug_location(&block.condition);
764            let condition = self.create_expr_generator(llvm_index).generate_expression(&block.condition)?;
765            let conditional_block = context.prepend_basic_block(else_block, "condition_body");
766
767            //Generate if statement condition
768            builder.build_conditional_branch(
769                to_i1(condition.into_int_value(), builder)?,
770                conditional_block,
771                else_block,
772            )?;
773
774            //Generate if statement content
775
776            builder.position_at_end(conditional_block);
777            self.generate_body(&block.body)?;
778            // Place debug location to end of if
779            self.debug.set_debug_location(
780                self.llvm,
781                self.function_context,
782                stmt.end_location.get_line_plus_one(),
783                stmt.end_location.get_column(),
784            );
785            builder.build_unconditional_branch(continue_block)?;
786        }
787        //Else
788
789        if let Some(else_block) = else_block {
790            builder.position_at_end(else_block);
791            self.generate_body(&stmt.else_block)?;
792            // Place debug location to end of if
793            self.debug.set_debug_location(
794                self.llvm,
795                self.function_context,
796                stmt.end_location.get_line_plus_one(),
797                stmt.end_location.get_column(),
798            );
799            builder.build_unconditional_branch(continue_block)?;
800        }
801        //Continue
802        builder.position_at_end(continue_block);
803        Ok(())
804    }

806    /// generates the function's return statement only if the given pou_type is a `PouType::Function`
807    ///
808    /// a function returns the value of the local variable that has the function's name
809    pub fn generate_return_statement(&self) -> Result<(), CodegenError> {
810        if let Some(ret_v) =
811            self.index.find_return_variable(self.function_context.linking_context.get_type_name())
812        {
813            if self
814                .index
815                .find_effective_type_by_name(ret_v.get_type_name())
816                .map(|it| it.is_aggregate_type())
817                .unwrap_or(false)
818            {
819                //generate return void
820                self.llvm.builder.build_return(None)?;
821            } else {
822                // generate return statement
823                let call_name = &self.function_context.linking_context.get_call_name_for_ir();
824                let var_name = format!("{call_name}_ret"); // TODO: Naming convention (see plc_util/src/convention.rs)
825                let ret_name = ret_v.get_qualified_name();
826                let value_ptr =
827                    self.llvm_index.find_loaded_associated_variable_value(ret_name).ok_or_else(|| {
828                        CodegenError::new(
829                            format!("Cannot generate return variable for {call_name:}"),
830                            SourceLocation::undefined(),
831                        )
832                    })?;
833
834                let pointee = self.llvm_index.get_associated_type(ret_v.get_type_name()).unwrap();
835                let loaded_value = self.llvm.load_pointer(pointee, &value_ptr, var_name.as_str())?;
836                self.llvm.builder.build_return(Some(&loaded_value))?;
837            }
838        } else {
839            self.llvm.builder.build_return(None)?;
840        }
841        Ok(())
842    }

844    /// Generates LLVM IR for conditional returns, which return if a given condition evaluates to true and
845    /// does nothing otherwise.
846    pub fn generate_conditional_return(
847        &'a self,
848        llvm_index: &'a LlvmTypedIndex<'b>,
849        statement: &AstNode,
850        condition: &AstNode,
851    ) -> Result<(), CodegenError> {
852        let expression_generator = self.create_expr_generator(llvm_index);
853
854        self.register_debug_location(condition);
855        let condition = expression_generator.generate_expression(condition)?;
856
857        let then_block = self.llvm.context.append_basic_block(self.function_context.function, "then_block");
858        let else_block = self.llvm.context.append_basic_block(self.function_context.function, "else_block");
859
860        self.llvm.builder.build_conditional_branch(
861            to_i1(condition.into_int_value(), &self.llvm.builder)?,
862            then_block,
863            else_block,
864        )?;
865
866        self.llvm.builder.position_at_end(then_block);
867        self.register_debug_location(statement);
868        self.generate_return_statement()?;
869        self.llvm.builder.position_at_end(else_block);
870
871        Ok(())
872    }

879/// Deconstructs assignments such as `a.b.c.%W3.%X2 := 2` into a base statement and its direct-access sequences.
880/// For the given example this function would return `(Node(a.b.c), vec![Node(%W3), Node(%X2)])`
881fn collect_base_and_direct_access_for_assignment(
882    left_statement: &AstNode,
883) -> Option<(&AstNode, Vec<&AstNode>)> {
884    let mut current = Some(left_statement);
885    let mut access_sequence = Vec::new();
886    while let Some(AstStatement::ReferenceExpr(ReferenceExpr { access: ReferenceAccess::Member(m), base })) =
887        current.map(|it| it.get_stmt())
888    {
889        if matches!(m.get_stmt(), AstStatement::DirectAccess { .. }) {
890            access_sequence.insert(0, m.as_ref());
891            current = base.as_deref();
892        } else {
893            break;
894        }
895    }
896    current.zip(Some(access_sequence))
897}
