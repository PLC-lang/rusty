35/// the full context when generating statements inside a POU
36#[derive(Debug)]
37pub struct FunctionContext<'ink, 'b> {
38    /// the current pou's name. This means that a variable x may refer to "`linking_context`.x"
39    pub linking_context: &'b ImplementationIndexEntry,
40    /// the llvm function to generate statements into
41    pub function: FunctionValue<'ink>,
42    /// The blocks/labels this function can use
43    pub blocks: FxHashMap<String, BasicBlock<'ink>>,
44}

46/// the StatementCodeGenerator is used to generate statements (For, If, etc.) or expressions (references, literals, etc.)
47pub struct StatementCodeGenerator<'a, 'b> {
48    llvm: &'b Llvm<'a>,
49    index: &'b Index,
50    annotations: &'b AstAnnotations,
51    llvm_index: &'b LlvmTypedIndex<'a>,
52    function_context: &'b FunctionContext<'a, 'b>,
53
54    pub load_prefix: String,
55    pub load_suffix: String,
56
57    /// the block to jump to when you want to exit the loop
58    pub current_loop_exit: Option<BasicBlock<'a>>,
59    /// the block to jump to when you want to continue the loop
60    pub current_loop_continue: Option<BasicBlock<'a>>,
61
62    pub debug: &'b DebugBuilderEnum<'a>,
63}

104    /// generates a list of statements
105    pub fn generate_body(&self, statements: &[AstNode]) -> Result<(), Diagnostic> {
106        let mut child_index = LlvmTypedIndex::create_child(self.llvm_index);
107        for s in statements {
108            child_index = self.generate_statement(child_index, s)?;
109        }
110        Ok(())
111    }

113    /// some versions of llvm will crash on two consecutive return or
114    /// unconditional jump statements. the solution is to insert another
115    /// building block before the second one, so the don't directly
116    /// follow each other. this is what we call a buffer block.
117    fn generate_buffer_block(&self) {
118        let (builder, _, context) = self.get_llvm_deps();
119        let buffer_block = context
120            .insert_basic_block_after(builder.get_insert_block().expect(INTERNAL_LLVM_ERROR), "buffer_block");
121        builder.position_at_end(buffer_block);
122    }

124    /// genertes a single statement
125    ///
126    /// - `statement` the statement to be generated
127    pub fn generate_statement(
128        &self,
129        mut llvm_index: LlvmTypedIndex<'b>,
130        statement: &AstNode,
131    ) -> Result<LlvmTypedIndex<'b>, Diagnostic> {
132        match statement.get_stmt() {
133            AstStatement::EmptyStatement(..) => {
134                //nothing to generate
135            }
136            AstStatement::Assignment(data, ..) => {
137                self.generate_assignment_statement(&llvm_index, &data.left, &data.right)?;
138            }
139            AstStatement::RefAssignment(data, ..) => {
140                self.generate_ref_assignment(&llvm_index, &data.left, &data.right)?;
141            }
142            AstStatement::ControlStatement(ctl_statement, ..) => {
143                self.generate_control_statement(&llvm_index, ctl_statement)?
144            }
145            AstStatement::ReturnStatement(ReturnStatement { condition }) => match condition {
146                Some(condition) => {
147                    self.generate_conditional_return(&llvm_index, statement, condition)?;
148                }
149                None => {
150                    self.register_debug_location(statement);
151                    self.generate_return_statement()?;
152                    self.generate_buffer_block(); // XXX(volsa): This is not needed on x86 but if removed segfaults on ARM
153                }
154            },
155            AstStatement::LabelStatement(LabelStatement { name }) => {
156                if let Some(block) = self.function_context.blocks.get(name) {
157                    //unconditionally jump to the label
158                    self.register_debug_location(statement);
159                    self.llvm.builder.build_unconditional_branch(*block);
160                    //Place the current instert block at the label statement
161                    self.llvm.builder.position_at_end(*block);
162                }
163            }
164            AstStatement::JumpStatement(JumpStatement { condition, .. }) => {
165                //Find the label to jump to
166                let Some(then_block) = self.annotations.get(statement).and_then(|label| {
167                    if let StatementAnnotation::Label { name } = label {
168                        self.function_context.blocks.get(name)
169                    } else {
170                        None
171                    }
172                }) else {
173                    return Err(Diagnostic::codegen_error(
174                        "Could not find label for {statement:?}",
175                        statement,
176                    ));
177                };
178                //Set current location as else block
179                let current_block = self.llvm.builder.get_insert_block().expect(INTERNAL_LLVM_ERROR);
180                let else_block = self.llvm.context.insert_basic_block_after(current_block, "else_block");
181
182                self.register_debug_location(condition);
183                let expression_generator = self.create_expr_generator(&llvm_index);
184                let condition = expression_generator.generate_expression(condition)?;
185
186                self.register_debug_location(statement);
187                self.llvm.builder.build_conditional_branch(
188                    condition.into_int_value(),
189                    *then_block,
190                    else_block,
191                );
192                // Make sure further code is at the else block
193                self.llvm.builder.position_at_end(else_block);
194            }
195            AstStatement::ExitStatement(_) => {
196                if let Some(exit_block) = &self.current_loop_exit {
197                    self.register_debug_location(statement);
198                    self.llvm.builder.build_unconditional_branch(*exit_block);
199                    self.generate_buffer_block();
200                } else {
201                    return Err(Diagnostic::codegen_error(
202                        "Cannot break out of loop when not inside a loop",
203                        statement,
204                    ));
205                }
206            }
207            AstStatement::ContinueStatement(_) => {
208                if let Some(cont_block) = &self.current_loop_continue {
209                    self.llvm.builder.build_unconditional_branch(*cont_block);
210                    self.generate_buffer_block();
211                } else {
212                    return Err(Diagnostic::codegen_error(
213                        "Cannot continue loop when not inside a loop",
214                        statement,
215                    ));
216                }
217            }
218            AstStatement::ExpressionList(statements) => {
219                let mut llvm_index = LlvmTypedIndex::create_child(&llvm_index);
220                for stmt in statements {
221                    llvm_index = self.generate_statement(llvm_index, stmt)?;
222                }
223            }
224            AstStatement::AllocationStatement(Allocation { name, reference_type }) => {
225                let ty =
226                    llvm_index.find_associated_type(reference_type).expect("Type must exist at this point");
227                let value = self.llvm.builder.build_alloca(ty, name);
228                self.llvm.generate_variable_initializer(
229                    &llvm_index,
230                    self.index,
231                    (name, reference_type, &statement.location),
232                    value,
233                    None,
234                    &self.create_expr_generator(&llvm_index),
235                )?;
236                llvm_index.associate_loaded_local_variable(
237                    self.function_context.linking_context.get_type_name(),
238                    name,
239                    value,
240                )?;
241            }
242            _ => {
243                self.create_expr_generator(&llvm_index).generate_expression(statement)?;
244            }
245        }
246        Ok(llvm_index)
247    }

249    /// genertes a single statement
250    ///
251    /// - `statement` the control statement to be generated
252    pub fn generate_control_statement(
253        &self,
254        llvm_index: &'a LlvmTypedIndex<'b>,
255        statement: &AstControlStatement,
256    ) -> Result<(), Diagnostic> {
257        match statement {
258            AstControlStatement::If(ifstmt) => self.generate_if_statement(llvm_index, ifstmt),
259            AstControlStatement::ForLoop(for_stmt) => self.generate_for_statement(llvm_index, for_stmt),
260            AstControlStatement::WhileLoop(stmt) | AstControlStatement::RepeatLoop(stmt) => {
261                self.generate_loop_statement(llvm_index, stmt)
262            }
263            AstControlStatement::Case(stmt) => self.generate_case_statement(llvm_index, stmt),
264        }
265    }

267    /// Generates IR for a `REF=` assignment, which is syntactic sugar for an assignment where the
268    /// right-hand side is wrapped in a `REF(...)` call. Specifically `foo REF= bar` and
269    /// `foo := REF(bar)` are the same.
270    ///
271    /// Note: Although somewhat similar to the [`generate_assignment_statement`] function, we can't
272    /// apply the code here because the left side of a `REF=` assignment is flagged as auto-deref.
273    /// For `REF=` assignments we don't want (and can't) deref without generating incorrect IR.
274    pub fn generate_ref_assignment(
275        &self,
276        llvm_index: &'a LlvmTypedIndex<'b>,
277        left: &AstNode,
278        right: &AstNode,
279    ) -> Result<(), Diagnostic> {
280        let exp = self.create_expr_generator(llvm_index);
281        let ref_builtin = self.index.get_builtin_function("REF").expect("REF must exist");
282
283        let AstStatement::ReferenceExpr(data) = &left.stmt else {
284            unreachable!("should be covered by a validation")
285        };
286
287        let left_ptr_val = {
288            let expr = exp.generate_reference_expression(&data.access, data.base.as_deref(), left)?;
289            expr.get_basic_value_enum().into_pointer_value()
290        };
291        let right_expr_val = if right.is_zero() {
292            exp.generate_literal(right)?
293        } else {
294            ref_builtin.codegen(&exp, &[right], right.get_location())?
295        };
296
297        self.llvm.builder.build_store(left_ptr_val, right_expr_val.get_basic_value_enum());
298        Ok(())
299    }

301    /// generates an assignment statement _left_ := _right_
302    ///
303    /// `left_statement` the left side of the assignment
304    /// `right_statement` the right side of the assignment
305    pub fn generate_assignment_statement(
306        &self,
307        llvm_index: &'a LlvmTypedIndex<'b>,
308        left_statement: &AstNode,
309        right_statement: &AstNode,
310    ) -> Result<(), Diagnostic> {
311        //Register any debug info for the store
312        self.register_debug_location(left_statement);
313        //TODO: Looks hacky, the strings will be similar so we should look into making the assignment a bit nicer.
314        if left_statement.has_direct_access() {
315            return self.generate_assignment_statement_direct_access(
316                llvm_index,
317                left_statement,
318                right_statement,
319            );
320        }
321
322        if self.annotations.get(left_statement).is_some_and(|it| {
323            // TODO(mhasel): ideally the resolver decides which assignment statement to call when lowering the init functions,
324            // but that requires refactoring of how `aliases` and `reference to` LHS/RHS nodes are annotated. this is a workaround.
325            self.function_context.linking_context.is_init() && (it.is_alias() || it.is_reference_to())
326        }) {
327            return self.generate_ref_assignment(llvm_index, left_statement, right_statement);
328        };
329
330        let exp_gen = self.create_expr_generator(llvm_index);
331        let left: PointerValue = exp_gen.generate_expression_value(left_statement).and_then(|it| {
332            it.get_basic_value_enum()
333                .try_into()
334                .map_err(|err| Diagnostic::codegen_error(format!("{err:?}").as_str(), left_statement))
335        })?;
336
337        let left_type = exp_gen.get_type_hint_info_for(left_statement)?;
338        // if the lhs-type is a subrange type we may need to generate a check-call
339        // e.g. x := y,  ==> x := CheckSignedInt(y);
340        let range_checked_right_side = if let DataTypeInformation::SubRange { .. } = left_type {
341            // there is a sub-range defined, so we need to wrap the right side into the check function if it exists
342            self.annotations.get_hidden_function_call(right_statement)
343        } else {
344            None
345        };
346
347        let right_statement = range_checked_right_side.unwrap_or(right_statement);
348
349        exp_gen.generate_store(left, left_type, right_statement)?;
350        Ok(())
351    }

355        let column = statement.get_location().get_column();

359    fn generate_assignment_statement_direct_access(
360        &self,
361        llvm_index: &'a LlvmTypedIndex<'b>,
362        left_statement: &AstNode,
363        right_statement: &AstNode,
364    ) -> Result<(), Diagnostic> {
365        let exp_gen = self.create_expr_generator(llvm_index);
366
367        // Left pointer
368        let Some((base, _)) = collect_base_and_direct_access_for_assignment(left_statement) else {
369            unreachable!("Invalid direct-access expression: {left_statement:#?}")
370        };
371        let left_expr_value = exp_gen.generate_expression_value(base)?;
372        let left_value = left_expr_value.as_r_value(self.llvm, None).into_int_value();
373        let left_pointer = left_expr_value.get_basic_value_enum().into_pointer_value();
374
375        // Generate an expression for the right size
376        let right_type = exp_gen.get_type_hint_for(right_statement)?;
377        let right_expr = exp_gen.generate_expression(right_statement)?;
378
379        exp_gen.generate_assignment_with_direct_access(
380            left_statement,
381            left_value,
382            left_pointer,
383            right_type,
384            right_expr,
385        )
386    }

388    /// generates a for-loop statement
389    ///
390    /// FOR `counter` := `start` TO `end` BY `by_step` DO
391    ///
392    /// - `counter` the counter variable
393    /// - `start` the value indicating the start of the for loop
394    /// - `end` the value indicating the end of the for loop
395    /// - `by_step` the step of the loop
396    /// - `body` the statements inside the for-loop
397    fn generate_for_statement(
398        &self,
399        llvm_index: &'a LlvmTypedIndex<'b>,
400        stmt: &ForLoopStatement,
401    ) -> Result<(), Diagnostic> {
402        let (builder, current_function, context) = self.get_llvm_deps();
403        let exp_gen = self.create_expr_generator(llvm_index);
404
405        let end_ty = self.annotations.get_type_or_void(&stmt.end, self.index);
406        let counter_ty = self.annotations.get_type_or_void(&stmt.counter, self.index);
407        let cast_target_ty = get_bigger_type(self.index.get_type_or_panic(DINT_TYPE), counter_ty, self.index);
408        let cast_target_llty = llvm_index.find_associated_type(cast_target_ty.get_name()).unwrap();
409
410        let step_ty = stmt.by_step.as_ref().map(|it| {
411            self.register_debug_location(it);
412            self.annotations.get_type_or_void(it, self.index)
413        });
414
415        let eval_step = || {
416            step_ty.map_or_else(
417                || self.llvm.create_const_numeric(&cast_target_llty, "1", SourceLocation::internal()),
418                |step_ty| {
419                    let step = exp_gen.generate_expression(stmt.by_step.as_ref().unwrap())?;
420                    Ok(cast_if_needed!(exp_gen, cast_target_ty, step_ty, step, None))
421                },
422            )
423        };
424
425        let predicate_incrementing = context.append_basic_block(current_function, "predicate_sle");
426        let predicate_decrementing = context.append_basic_block(current_function, "predicate_sge");
427        let loop_body = context.append_basic_block(current_function, "loop");
428        let increment = context.append_basic_block(current_function, "increment");
429        let afterloop = context.append_basic_block(current_function, "continue");
430
431        self.generate_assignment_statement(llvm_index, &stmt.counter, &stmt.start)?;
432        let counter = exp_gen.generate_lvalue(&stmt.counter)?;
433
434        // generate loop predicate selector. since `STEP` can be a reference, this needs to be a runtime eval
435        // XXX(mhasel): IR could possibly be improved by generating phi instructions.
436        //              Candidate for frontend optimization for builds without optimization when `STEP`
437        //              is a compile-time constant
438        let is_incrementing = builder.build_int_compare(
439            inkwell::IntPredicate::SGT,
440            eval_step()?.into_int_value(),
441            self.llvm
442                .create_const_numeric(&cast_target_llty, "0", SourceLocation::internal())?
443                .into_int_value(),
444            "is_incrementing",
445        );
446        builder.build_conditional_branch(is_incrementing, predicate_incrementing, predicate_decrementing);
447        // generate predicates for incrementing and decrementing counters
448        let generate_predicate = |predicate| {
449            builder.position_at_end(match predicate {
450                inkwell::IntPredicate::SLE => predicate_incrementing,
451                inkwell::IntPredicate::SGE => predicate_decrementing,
452                _ => unreachable!(),
453            });
454
455            let end = exp_gen.generate_expression_value(&stmt.end).unwrap();
456            let end_value = match end {
457                ExpressionValue::LValue(ptr) => builder.build_load(ptr, ""),
458                ExpressionValue::RValue(val) => val,
459            };
460            let counter_value = builder.build_load(counter, "");
461            let cmp = builder.build_int_compare(
462                predicate,
463                cast_if_needed!(exp_gen, cast_target_ty, counter_ty, counter_value, None).into_int_value(),
464                cast_if_needed!(exp_gen, cast_target_ty, end_ty, end_value, None).into_int_value(),
465                "condition",
466            );
467            builder.build_conditional_branch(cmp, loop_body, afterloop);
468        };
469        generate_predicate(inkwell::IntPredicate::SLE);
470        generate_predicate(inkwell::IntPredicate::SGE);
471
472        // generate loop body
473        builder.position_at_end(loop_body);
474        let body_builder = StatementCodeGenerator {
475            current_loop_continue: Some(increment),
476            current_loop_exit: Some(afterloop),
477            load_prefix: self.load_prefix.clone(),
478            load_suffix: self.load_suffix.clone(),
479            ..*self
480        };
481        body_builder.generate_body(&stmt.body)?;
482        // Place debug location to end of loop
483        self.debug.set_debug_location(
484            self.llvm,
485            self.function_context,
486            stmt.end_location.get_line_plus_one(),
487            stmt.end_location.get_column(),
488        );
489
490        // increment counter
491        builder.build_unconditional_branch(increment);
492        builder.position_at_end(increment);
493        let counter_value = builder.build_load(counter, "");
494        let inc = inkwell::values::BasicValue::as_basic_value_enum(&builder.build_int_add(
495            eval_step()?.into_int_value(),
496            cast_if_needed!(exp_gen, cast_target_ty, counter_ty, counter_value, None).into_int_value(),
497            "next",
498        ));
499        builder.build_store(
500            counter,
501            cast_if_needed!(exp_gen, counter_ty, cast_target_ty, inc, None).into_int_value(),
502        );
503
504        // check condition
505        builder.build_conditional_branch(is_incrementing, predicate_incrementing, predicate_decrementing);
506        // continue
507        builder.position_at_end(afterloop);
508        Ok(())
509    }

511    /// genertes a case statement
512    ///
513    /// CASE selector OF
514    /// conditional_block#1:
515    /// conditional_block#2:
516    /// END_CASE;
517    ///
518    /// - `selector` the case's selector expression
519    /// - `conditional_blocks` all case-blocks including the condition and the body
520    /// - `else_body` the statements in the else-block
521    fn generate_case_statement(
522        &self,
523        llvm_index: &'a LlvmTypedIndex<'b>,
524        stmt: &CaseStatement,
525    ) -> Result<(), Diagnostic> {
526        let (builder, current_function, context) = self.get_llvm_deps();
527        //Continue
528        let continue_block = context.append_basic_block(current_function, "continue");
529
530        let basic_block = builder.get_insert_block().expect(INTERNAL_LLVM_ERROR);
531        let exp_gen = self.create_expr_generator(llvm_index);
532        let selector_statement = exp_gen.generate_expression(&stmt.selector)?;
533
534        let mut cases = Vec::new();
535        let else_block = context.append_basic_block(current_function, "else");
536        let mut current_else_block = else_block;
537
538        for conditional_block in &stmt.case_blocks {
539            //craete a block for the case's body
540            let case_block = context.prepend_basic_block(else_block, "case");
541
542            //flatten the expression list into a vector of expressions
543            let expressions = flatten_expression_list(&conditional_block.condition);
544            for s in expressions {
545                if let AstStatement::RangeStatement(data, ..) = s.get_stmt() {
546                    //if this is a range statement, we generate an if (x >= start && x <= end) then the else-section
547                    builder.position_at_end(current_else_block);
548                    // since the if's generate additional blocks, we use the last one as the else-section
549                    current_else_block = self.generate_case_range_condition(
550                        llvm_index,
551                        &stmt.selector,
552                        data.start.as_ref(),
553                        data.end.as_ref(),
554                        case_block,
555                    )?;
556                } else {
557                    // this should be a a literal or a reference to a constant
558                    builder.position_at_end(basic_block);
559                    let condition = exp_gen.generate_expression(s)?; //TODO : Is a type conversion needed here?
560                                                                     // collect all literal case blocks to pass to the llvm switch-statement
561                    cases.push((condition.into_int_value(), case_block));
562                }
563            }
564            //generate the case's body
565            builder.position_at_end(case_block);
566            self.generate_body(&conditional_block.body)?;
567            //Add debug location to the end of the case
568            self.debug.set_debug_location(
569                self.llvm,
570                self.function_context,
571                stmt.end_location.get_line_plus_one(),
572                stmt.end_location.get_column(),
573            );
574            // skip all other case-bodies
575            builder.build_unconditional_branch(continue_block);
576        }
577        // current-else is the last else-block generated by the range-expressions
578        builder.position_at_end(current_else_block);
579        self.generate_body(&stmt.else_block)?;
580        //Add debug location to the end of the case
581        self.debug.set_debug_location(
582            self.llvm,
583            self.function_context,
584            stmt.end_location.get_line_plus_one(),
585            stmt.end_location.get_column(),
586        );
587        builder.build_unconditional_branch(continue_block);
588        continue_block.move_after(current_else_block).expect(INTERNAL_LLVM_ERROR);
589
590        // now that we collected all cases, go back to the initial block and generate the switch-statement
591        builder.position_at_end(basic_block);
592
593        self.register_debug_location(&stmt.selector);
594        builder.build_switch(selector_statement.into_int_value(), else_block, &cases);
595
596        builder.position_at_end(continue_block);
597        Ok(())
598    }

600    /// returns the new block to use as else
601    ///
602    ///
603    fn generate_case_range_condition(
604        &self,
605        llvm_index: &'a LlvmTypedIndex<'b>,
606        selector: &AstNode,
607        start: &AstNode,
608        end: &AstNode,
609        match_block: BasicBlock,
610    ) -> Result<BasicBlock<'_>, Diagnostic> {
611        let (builder, _, context) = self.get_llvm_deps();
612
613        let range_then = context
614            .insert_basic_block_after(builder.get_insert_block().expect(INTERNAL_LLVM_ERROR), "range_then");
615        let range_else = context.insert_basic_block_after(range_then, "range_else");
616        let exp_gen = self.create_expr_generator(llvm_index);
617        let lower_bound = {
618            self.register_debug_location(start);
619            let start_val = exp_gen.generate_expression(start)?;
620            self.register_debug_location(selector);
621            let selector_val = exp_gen.generate_expression(selector)?;
622            exp_gen.create_llvm_int_binary_expression(&Operator::GreaterOrEqual, selector_val, start_val)
623        };
624
625        //jmp to continue if the value is smaller than start
626        builder.build_conditional_branch(
627            to_i1(lower_bound.into_int_value(), builder),
628            range_then,
629            range_else,
630        );
631        builder.position_at_end(range_then);
632        let upper_bound = {
633            self.register_debug_location(end);
634            let end_val = exp_gen.generate_expression(end)?;
635            self.register_debug_location(selector);
636            let selector_val = exp_gen.generate_expression(selector)?;
637            exp_gen.create_llvm_int_binary_expression(&Operator::LessOrEqual, selector_val, end_val)
638        };
639        builder.build_conditional_branch(
640            to_i1(upper_bound.into_int_value(), builder),
641            match_block,
642            range_else,
643        );
644        Ok(range_else)
645    }

647    /// generates a while statement
648    ///
649    /// WHILE condition DO
650    ///     body
651    /// END_WHILE
652    ///
653    /// - `condition` the while's condition
654    /// - `body` the while's body statements
655    fn generate_loop_statement(
656        &self,
657        llvm_index: &'a LlvmTypedIndex<'b>,
658        stmt: &LoopStatement,
659    ) -> Result<(), Diagnostic> {
660        let builder = &self.llvm.builder;
661        let basic_block = builder.get_insert_block().expect(INTERNAL_LLVM_ERROR);
662        let (condition_block, _) =
663            self.generate_base_while_statement(llvm_index, &stmt.condition, &stmt.body, &stmt.end_location)?;
664
665        let continue_block = builder.get_insert_block().expect(INTERNAL_LLVM_ERROR);
666
667        builder.position_at_end(basic_block);
668        builder.build_unconditional_branch(condition_block);
669
670        builder.position_at_end(continue_block);
671        Ok(())
672    }

674    /// utility method for while and repeat loops
675    fn generate_base_while_statement(
676        &self,
677        llvm_index: &'a LlvmTypedIndex<'b>,
678        condition: &AstNode,
679        body: &[AstNode],
680        end_location: &SourceLocation,
681    ) -> Result<(BasicBlock<'_>, BasicBlock<'_>), Diagnostic> {
682        let (builder, current_function, context) = self.get_llvm_deps();
683        let condition_check = context.append_basic_block(current_function, "condition_check");
684        let while_body = context.append_basic_block(current_function, "while_body");
685        let continue_block = context.append_basic_block(current_function, "continue");
686
687        //Check loop condition
688        builder.position_at_end(condition_check);
689        self.register_debug_location(condition);
690        let condition_value = self.create_expr_generator(llvm_index).generate_expression(condition)?;
691        builder.build_conditional_branch(
692            to_i1(condition_value.into_int_value(), builder),
693            while_body,
694            continue_block,
695        );
696
697        //Enter the for loop
698        builder.position_at_end(while_body);
699        let body_generator = StatementCodeGenerator {
700            current_loop_exit: Some(continue_block),
701            current_loop_continue: Some(condition_check),
702            load_prefix: self.load_prefix.clone(),
703            load_suffix: self.load_suffix.clone(),
704            ..*self
705        };
706        body_generator.generate_body(body)?;
707        //Set the debug location to the end of the loop
708        self.debug.set_debug_location(
709            self.llvm,
710            self.function_context,
711            end_location.get_line_plus_one(),
712            end_location.get_column(),
713        );
714        //Loop back
715        builder.build_unconditional_branch(condition_check);
716
717        //Continue
718        builder.position_at_end(continue_block);
719        Ok((condition_check, while_body))
720    }

722    /// generates an IF-Statement
723    ///
724    /// - `conditional_blocks` a list of conditions + bodies for every if  (respectivle else-if)
725    /// - `else_body` the list of statements in the else-block
726    fn generate_if_statement(
727        &self,
728        llvm_index: &'a LlvmTypedIndex<'b>,
729        stmt: &IfStatement,
730    ) -> Result<(), Diagnostic> {
731        let (builder, current_function, context) = self.get_llvm_deps();
732        let mut blocks = vec![builder.get_insert_block().expect(INTERNAL_LLVM_ERROR)];
733        for _ in 1..stmt.blocks.len() {
734            blocks.push(context.append_basic_block(current_function, "branch"));
735        }
736
737        let else_block = if !stmt.else_block.is_empty() {
738            let result = context.append_basic_block(current_function, "else");
739            blocks.push(result);
740            Some(result)
741        } else {
742            None
743        };
744        //Continue
745        let continue_block = context.append_basic_block(current_function, "continue");
746        blocks.push(continue_block);
747
748        for (i, block) in stmt.blocks.iter().enumerate() {
749            let then_block = blocks[i];
750            let else_block = blocks[i + 1];
751
752            builder.position_at_end(then_block);
753
754            self.register_debug_location(&block.condition);
755            let condition = self.create_expr_generator(llvm_index).generate_expression(&block.condition)?;
756            let conditional_block = context.prepend_basic_block(else_block, "condition_body");
757
758            //Generate if statement condition
759            builder.build_conditional_branch(
760                to_i1(condition.into_int_value(), builder),
761                conditional_block,
762                else_block,
763            );
764
765            //Generate if statement content
766
767            builder.position_at_end(conditional_block);
768            self.generate_body(&block.body)?;
769            // Place debug location to end of if
770            self.debug.set_debug_location(
771                self.llvm,
772                self.function_context,
773                stmt.end_location.get_line_plus_one(),
774                stmt.end_location.get_column(),
775            );
776            builder.build_unconditional_branch(continue_block);
777        }
778        //Else
779
780        if let Some(else_block) = else_block {
781            builder.position_at_end(else_block);
782            self.generate_body(&stmt.else_block)?;
783            // Place debug location to end of if
784            self.debug.set_debug_location(
785                self.llvm,
786                self.function_context,
787                stmt.end_location.get_line_plus_one(),
788                stmt.end_location.get_column(),
789            );
790            builder.build_unconditional_branch(continue_block);
791        }
792        //Continue
793        builder.position_at_end(continue_block);
794        Ok(())
795    }

797    /// generates the function's return statement only if the given pou_type is a `PouType::Function`
798    ///
799    /// a function returns the value of the local variable that has the function's name
800    pub fn generate_return_statement(&self) -> Result<(), Diagnostic> {
801        if let Some(ret_v) =
802            self.index.find_return_variable(self.function_context.linking_context.get_type_name())
803        {
804            if self
805                .index
806                .find_effective_type_by_name(ret_v.get_type_name())
807                .map(|it| it.is_aggregate_type())
808                .unwrap_or(false)
809            {
810                //generate return void
811                self.llvm.builder.build_return(None);
812            } else {
813                // generate return statement
814                let call_name = &self.function_context.linking_context.get_call_name_for_ir();
815                let var_name = format!("{call_name}_ret"); // TODO: Naming convention (see plc_util/src/convention.rs)
816                let ret_name = ret_v.get_qualified_name();
817                let value_ptr =
818                    self.llvm_index.find_loaded_associated_variable_value(ret_name).ok_or_else(|| {
819                        Diagnostic::codegen_error(
820                            format!("Cannot generate return variable for {call_name:}"),
821                            SourceLocation::undefined(),
822                        )
823                    })?;
824                let loaded_value = self.llvm.load_pointer(&value_ptr, var_name.as_str());
825                self.llvm.builder.build_return(Some(&loaded_value));
826            }
827        } else {
828            self.llvm.builder.build_return(None);
829        }
830        Ok(())
831    }

833    /// Generates LLVM IR for conditional returns, which return if a given condition evaluates to true and
834    /// does nothing otherwise.
835    pub fn generate_conditional_return(
836        &'a self,
837        llvm_index: &'a LlvmTypedIndex<'b>,
838        statement: &AstNode,
839        condition: &AstNode,
840    ) -> Result<(), Diagnostic> {
841        let expression_generator = self.create_expr_generator(llvm_index);
842
843        self.register_debug_location(condition);
844        let condition = expression_generator.generate_expression(condition)?;
845
846        let then_block = self.llvm.context.append_basic_block(self.function_context.function, "then_block");
847        let else_block = self.llvm.context.append_basic_block(self.function_context.function, "else_block");
848
849        self.llvm.builder.build_conditional_branch(
850            to_i1(condition.into_int_value(), &self.llvm.builder),
851            then_block,
852            else_block,
853        );
854
855        self.llvm.builder.position_at_end(then_block);
856        self.register_debug_location(statement);
857        self.generate_return_statement()?;
858        self.llvm.builder.position_at_end(else_block);
859
860        Ok(())
861    }

868/// Deconstructs assignments such as `a.b.c.%W3.%X2 := 2` into a base statement and its direct-access sequences.
869/// For the given example this function would return `(Node(a.b.c), vec![Node(%W3), Node(%X2)])`
870fn collect_base_and_direct_access_for_assignment(
871    left_statement: &AstNode,
872) -> Option<(&AstNode, Vec<&AstNode>)> {
873    let mut current = Some(left_statement);
874    let mut access_sequence = Vec::new();
875    while let Some(AstStatement::ReferenceExpr(ReferenceExpr { access: ReferenceAccess::Member(m), base })) =
876        current.map(|it| it.get_stmt())
877    {
878        if matches!(m.get_stmt(), AstStatement::DirectAccess { .. }) {
879            access_sequence.insert(0, m.as_ref());
880            current = base.as_deref();
881        } else {
882            break;
883        }
884    }
885    current.zip(Some(access_sequence))
886}
