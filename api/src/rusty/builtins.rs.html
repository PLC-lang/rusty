610fn validate_types(
611    validator: &mut Validator,
612    parameters: &Option<&AstNode>,
613    annotations: &dyn AnnotationMap,
614    index: &Index,
615) {
616    let Some(params) = parameters else { return };
617
618    let types: Vec<_> =
619        flatten_expression_list(params).into_iter().map(|it| extract_actual_parameter(it).clone()).collect();
620    let mut types = types.iter().peekable();
621
622    while let Some(left) = types.next() {
623        if let Some(right) = types.peek() {
624            validate_type_compatibility(validator, annotations, index, left, right);
625        }
626    }
627}

629fn validate_builtin_symbol_parameter_count(
630    validator: &mut Validator,
631    operator: &AstNode,
632    parameters: Option<&AstNode>,
633    operation: Operator,
634) {
635    let Some(params) = parameters else {
636        validator.push_diagnostic(Diagnostic::invalid_argument_count(2, 0, operator));
637        return;
638    };
639
640    let count = flatten_expression_list(params).len();
641    match operation {
642        // non-extensible operators
643        Operator::Minus | Operator::Division | Operator::NotEqual => {
644            if count != 2 {
645                validator.push_diagnostic(Diagnostic::invalid_argument_count(2, count, operator));
646            }
647        }
648        _ => {
649            if count < 2 {
650                validator.push_diagnostic(Diagnostic::invalid_argument_count(2, count, operator));
651            }
652        }
653    }
654}

656// creates nested BinaryExpressions for each parameter, such that
657// GT(a, b, c, d) ends up as (a > b) & (b > c) & (c > d)
658fn annotate_comparison_function(
659    annotator: &mut TypeAnnotator,
660    statement: &AstNode,
661    operator: &AstNode,
662    parameters: &AstNode,
663    ctx: VisitorContext,
664    operation: Operator,
665) {
666    let mut ctx = ctx;
667    let params_flattened = flatten_expression_list(parameters);
668    if params_flattened.iter().any(|it| {
669        !annotator
670            .annotation_map
671            .get_type_or_void(it, annotator.index)
672            .has_nature(TypeNature::Elementary, annotator.index)
673    }) {
674        // we are trying to call this function with a non-elementary type, so we redirect back to the resolver
675        annotator.annotate_call_statement(operator, Some(parameters), &ctx);
676        return;
677    }
678
679    let comparisons = params_flattened
680        .windows(2)
681        .map(|window| {
682            AstFactory::create_binary_expression(
683                window[0].clone(),
684                operation,
685                window[1].clone(),
686                ctx.id_provider.next_id(),
687            )
688        })
689        .collect::<Vec<_>>();
690    let Some(new_statement) = comparisons.first() else {
691        // no windows => less than 2 parameters, caught during validation
692        return;
693    };
694    let mut new_statement = new_statement.clone();
695    comparisons.into_iter().skip(1).for_each(|right| {
696        new_statement = AstFactory::create_binary_expression(
697            new_statement.clone(),
698            Operator::And,
699            right,
700            ctx.id_provider.next_id(),
701        )
702    });
703
704    annotator.visit_statement(&ctx, &new_statement);
705    annotator.update_expected_types(annotator.index.get_type_or_panic(typesystem::BOOL_TYPE), &new_statement);
706    annotator.annotate(statement, StatementAnnotation::ReplacementAst { statement: new_statement });
707    annotator.update_expected_types(annotator.index.get_type_or_panic(typesystem::BOOL_TYPE), statement);
708}

710fn annotate_arithmetic_function(
711    annotator: &mut TypeAnnotator,
712    statement: &AstNode,
713    operator: &AstNode,
714    parameters: &AstNode,
715    ctx: VisitorContext,
716    operation: Operator,
717) {
718    let params = flatten_expression_list(parameters);
719    let params_extracted: Vec<_> =
720        params.iter().map(|param| extract_actual_parameter(param).clone()).collect();
721
722    // Add type hints (only named arguments)
723    params
724        .iter()
725        .zip(&params_extracted)
726        .filter(|(it, _)| matches!(it.get_stmt(), AstStatement::Assignment(_)))
727        .for_each(|(_, extracted)| {
728            let param_type = annotator
729                .annotation_map
730                .get_type_or_void(extracted, annotator.index)
731                .get_type_information()
732                .get_name()
733                .to_owned();
734            annotator.annotation_map.annotate_type_hint(extracted, StatementAnnotation::value(param_type));
735        });
736
737    if params_extracted.iter().any(|param| {
738        !annotator
739            .annotation_map
740            .get_type_or_void(param, annotator.index)
741            .has_nature(TypeNature::Num, annotator.index)
742    }) {
743        // we are trying to call this function with a non-numerical type, so we redirect back to the resolver
744        annotator.annotate_call_statement(operator, Some(parameters), &ctx);
745        return;
746    }
747
748    let mut ctx = ctx;
749    // find biggest type to later annotate it as type hint. this is done in a closure to avoid a borrow-checker tantrum later on due to
750    // mutable and immutable borrow of TypeAnnotator
751    let find_biggest_param_type_name = |annotator: &TypeAnnotator| {
752        let mut bigger = annotator
753            .annotation_map
754            .get_type_or_void(params_extracted.first().expect("must have this parameter"), annotator.index);
755
756        for param in params_extracted.iter().skip(1) {
757            let right_type = annotator.annotation_map.get_type_or_void(param, annotator.index);
758            bigger = get_bigger_type(bigger, right_type, annotator.index);
759        }
760
761        bigger.get_name().to_owned()
762    };
763
764    let bigger_type = find_biggest_param_type_name(annotator);
765
766    // create nested AstStatement::BinaryExpression for each parameter, such that
767    // ADD(a, b, c, d) ends up as (((a + b) + c) + d)
768    let left = (*params_extracted.first().expect("Must exist")).clone();
769    let new_statement = params_extracted.into_iter().skip(1).fold(left, |left, right| {
770        AstFactory::create_binary_expression(left, operation, right.clone(), ctx.id_provider.next_id())
771    });
772
773    annotator.visit_statement(&ctx, &new_statement);
774    annotator.update_expected_types(annotator.index.get_type_or_panic(&bigger_type), &new_statement);
775    annotator.annotate(statement, StatementAnnotation::ReplacementAst { statement: new_statement });
776    annotator.update_expected_types(annotator.index.get_type_or_panic(&bigger_type), statement);
777}

779fn annotate_variable_length_array_bound_function(
780    annotator: &mut TypeAnnotator,
781    parameters: Option<&AstNode>,
782) {
783    let Some(parameters) = parameters else {
784        return;
785    };
786    let params = ast::flatten_expression_list(parameters);
787    let vla = params.first().expect("must exist; covered by validation");
788    let vla_param = extract_actual_parameter(vla);
789    // if the VLA parameter is a VLA struct, annotate it as such
790    let vla_type = annotator.annotation_map.get_type_or_void(vla_param, annotator.index);
791    let vla_type_name = if vla_type.get_nature() == TypeNature::__VLA {
792        vla_type.get_name()
793    } else {
794        // otherwise annotate it with an internal, reserved VLA type
795        typesystem::__VLA_TYPE
796    };
797    annotator.annotation_map.annotate_type_hint(vla_param, StatementAnnotation::value(vla_type_name));
798    if let Some(dim) = params.get(1) {
799        let dim_param = extract_actual_parameter(dim);
800        let dim_type = annotator.annotation_map.get_type_or_void(dim_param, annotator.index);
801        if !dim_type.is_void() {
802            // Use the actual type of the dimension parameter
803            annotator
804                .annotation_map
805                .annotate_type_hint(dim_param, StatementAnnotation::value(dim_type.get_name()));
806        } else {
807            // Fallback to a default integer type if no type is available
808            annotator.annotation_map.annotate_type_hint(dim_param, StatementAnnotation::value("DINT"));
809        }
810    }
811}

813fn validate_variable_length_array_bound_function(
814    validator: &mut Validator,
815    operator: &AstNode,
816    parameters: Option<&AstNode>,
817    annotations: &dyn AnnotationMap,
818    index: &Index,
819) {
820    let Some(parameters) = parameters else {
821        validator.push_diagnostic(Diagnostic::invalid_argument_count(2, 0, operator));
822
823        // no params, nothing to validate
824        return;
825    };
826
827    let params = ast::flatten_expression_list(parameters);
828
829    if let &[vla, dim] = params.as_slice() {
830        let [actual_vla, actual_idx] = [vla, dim].map(extract_actual_parameter);
831
832        let idx_type = annotations.get_type_or_void(actual_idx, index);
833
834        if !idx_type.has_nature(TypeNature::Int, index) {
835            validator.push_diagnostic(
836                Diagnostic::new(format!(
837                    "Invalid type nature for generic argument. {} is no {}",
838                    idx_type.get_name(),
839                    TypeNature::Int
840                ))
841                .with_error_code("E062")
842                .with_location(actual_idx),
843            )
844        }
845
846        // TODO: consider adding validation for consts and enums once https://github.com/PLC-lang/rusty/issues/847 has been implemented
847        if let AstStatement::Literal(AstLiteral::Integer(dimension_idx)) = actual_idx.get_stmt() {
848            let dimension_idx = *dimension_idx as usize;
849
850            let Some(n_dimensions) =
851                annotations.get_type_or_void(actual_vla, index).get_type_information().get_dimension_count()
852            else {
853                // not a vla, validated via type nature
854                return;
855            };
856
857            if dimension_idx < 1 || dimension_idx > n_dimensions {
858                validator.push_diagnostic(
859                    Diagnostic::new("Index out of bound").with_error_code("E046").with_location(operator),
860                )
861            }
862        };
863    } else {
864        validator.push_diagnostic(Diagnostic::invalid_argument_count(2, params.len(), operator));
865    }
866}
867
868fn validate_argument_count(
869    validator: &mut Validator,
870    operator: &AstNode,
871    parameters: &Option<&AstNode>,
872    expected: usize,
873) {
874    let Some(params) = parameters else {
875        validator.push_diagnostic(Diagnostic::invalid_argument_count(expected, 0, operator));
876        return;
877    };
878
879    let params = flatten_expression_list(params);
880
881    if params.len() != expected {
882        validator.push_diagnostic(Diagnostic::invalid_argument_count(expected, params.len(), operator));
883    }
884}

886/// Helper function to extract the actual parameter from Assignment nodes when dealing with named arguments
887/// For named arguments like `func(param := value)`, the AST contains an Assignment node where we need
888/// to extract the right-hand side (the actual value). For positional arguments, use the parameter directly.
889fn extract_actual_parameter(param: &AstNode) -> &AstNode {
890    if let AstStatement::Assignment(assignment) = param.get_stmt() {
891        // Named argument: extract the actual value from the right side of the assignment
892        assignment.right.as_ref()
893    } else {
894        // Positional argument: use the parameter directly
895        param
896    }
897}

899/// Generates the code for the LOWER- AND UPPER_BOUND built-in functions, returning an error if the function
900/// arguments are incorrect.
901fn generate_variable_length_array_bound_function<'ink>(
902    generator: &ExpressionCodeGenerator<'ink, '_>,
903    params: &[&AstNode],
904    is_lower: bool,
905    location: SourceLocation,
906) -> Result<ExpressionValue<'ink>, CodegenError> {
907    let llvm = generator.llvm;
908    let builder = &generator.llvm.builder;
909
910    if let &[vla, dim] = params {
911        let [actual_vla, actual_dim] = [vla, dim].map(extract_actual_parameter);
912
913        let data_type_information =
914            generator.annotations.get_type_or_void(actual_vla, generator.index).get_type_information();
915
916        // TODO: most of the codegen errors should already be caught during validation.
917        // once we abort codegen on critical errors, revisit and change to unreachable where possible
918        if !data_type_information.is_vla() {
919            return Err(CodegenError::GenericError(
920                format!("Expected VLA type, received {}", data_type_information.get_name()),
921                location,
922            ));
923        };
924
925        let vla = generator.generate_lvalue(actual_vla).unwrap();
926        let dim = builder.build_struct_gep(vla, 1, "dim").unwrap();
927
928        let accessor = match actual_dim.get_stmt() {
929            // e.g. LOWER_BOUND(arr, 1)
930            AstStatement::Literal(kind) => {
931                let AstLiteral::Integer(value) = kind else {
932                    let Some(type_name) = get_literal_actual_signed_type_name(kind, false) else {
933                        unreachable!("type cannot be VOID")
934                    };
935                    return Err(CodegenError::GenericError(
936                        format!("Invalid literal type. Expected INT type, received {type_name} type"),
937                        location,
938                    ));
939                };
940                // array offset start- and end-values are adjacent values in a flattened array -> 2 values per dimension, so in order
941                // to read the correct values, the given index needs to be doubled. Additionally, the value is adjusted for 0-indexing.
942                let offset = if is_lower { (value - 1) as u64 * 2 } else { (value - 1) as u64 * 2 + 1 };
943                llvm.i32_type().const_int(offset, false)
944            }
945            // e.g. LOWER_BOUND(arr, idx + 3)
946            _ => {
947                let expression_value = generator.generate_expression(actual_dim)?;
948                if !expression_value.is_int_value() {
949                    todo!()
950                };
951                // this operation mirrors the offset calculation of literal ints, but at runtime
952                let offset = builder.build_int_mul(
953                    llvm.i32_type().const_int(2, false),
954                    builder.build_int_sub(
955                        expression_value.into_int_value(),
956                        llvm.i32_type().const_int(1, false),
957                        "",
958                    )?,
959                    "",
960                )?;
961                if !is_lower {
962                    builder.build_int_add(offset, llvm.i32_type().const_int(1, false), "")?
963                } else {
964                    offset
965                }
966            }
967        };
968        let gep_bound =
969            unsafe { llvm.builder.build_in_bounds_gep(dim, &[llvm.i32_type().const_zero(), accessor], "") }?;
970        let bound = llvm.builder.build_load(gep_bound, "")?;
971
972        Ok(ExpressionValue::RValue(bound))
973    } else {
974        Err(CodegenError::GenericError("Invalid signature for LOWER_BOUND/UPPER_BOUND".to_string(), location))
975    }
976}
