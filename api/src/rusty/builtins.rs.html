603fn validate_types(
604    validator: &mut Validator,
605    parameters: &Option<&AstNode>,
606    annotations: &dyn AnnotationMap,
607    index: &Index,
608) {
609    let Some(params) = parameters else { return };
610
611    let types = flatten_expression_list(params);
612    let mut types = types.iter().peekable();
613
614    while let Some(left) = types.next() {
615        if let Some(right) = types.peek() {
616            validate_type_compatibility(validator, annotations, index, left, right);
617        }
618    }
619}

621fn validate_builtin_symbol_parameter_count(
622    validator: &mut Validator,
623    operator: &AstNode,
624    parameters: Option<&AstNode>,
625    operation: Operator,
626) {
627    let Some(params) = parameters else {
628        validator.push_diagnostic(Diagnostic::invalid_argument_count(2, 0, operator));
629        return;
630    };
631
632    let count = flatten_expression_list(params).len();
633    match operation {
634        // non-extensible operators
635        Operator::Minus | Operator::Division | Operator::NotEqual => {
636            if count != 2 {
637                validator.push_diagnostic(Diagnostic::invalid_argument_count(2, count, operator));
638            }
639        }
640        _ => {
641            if count < 2 {
642                validator.push_diagnostic(Diagnostic::invalid_argument_count(2, count, operator));
643            }
644        }
645    }
646}

648// creates nested BinaryExpressions for each parameter, such that
649// GT(a, b, c, d) ends up as (a > b) & (b > c) & (c > d)
650fn annotate_comparison_function(
651    annotator: &mut TypeAnnotator,
652    statement: &AstNode,
653    operator: &AstNode,
654    parameters: &AstNode,
655    ctx: VisitorContext,
656    operation: Operator,
657) {
658    let mut ctx = ctx;
659    let params_flattened = flatten_expression_list(parameters);
660    if params_flattened.iter().any(|it| {
661        !annotator
662            .annotation_map
663            .get_type_or_void(it, annotator.index)
664            .has_nature(TypeNature::Elementary, annotator.index)
665    }) {
666        // we are trying to call this function with a non-elementary type, so we redirect back to the resolver
667        annotator.annotate_call_statement(operator, Some(parameters), &ctx);
668        return;
669    }
670
671    let comparisons = params_flattened
672        .windows(2)
673        .map(|window| {
674            AstFactory::create_binary_expression(
675                window[0].clone(),
676                operation,
677                window[1].clone(),
678                ctx.id_provider.next_id(),
679            )
680        })
681        .collect::<Vec<_>>();
682    let Some(new_statement) = comparisons.first() else {
683        // no windows => less than 2 parameters, caught during validation
684        return;
685    };
686    let mut new_statement = new_statement.clone();
687    comparisons.into_iter().skip(1).for_each(|right| {
688        new_statement = AstFactory::create_binary_expression(
689            new_statement.clone(),
690            Operator::And,
691            right,
692            ctx.id_provider.next_id(),
693        )
694    });
695
696    annotator.visit_statement(&ctx, &new_statement);
697    annotator.update_expected_types(annotator.index.get_type_or_panic(typesystem::BOOL_TYPE), &new_statement);
698    annotator.annotate(statement, StatementAnnotation::ReplacementAst { statement: new_statement });
699    annotator.update_expected_types(annotator.index.get_type_or_panic(typesystem::BOOL_TYPE), statement);
700}

702fn annotate_arithmetic_function(
703    annotator: &mut TypeAnnotator,
704    statement: &AstNode,
705    operator: &AstNode,
706    parameters: &AstNode,
707    ctx: VisitorContext,
708    operation: Operator,
709) {
710    let params_flattened = flatten_expression_list(parameters);
711    if params_flattened.iter().any(|it| {
712        !annotator
713            .annotation_map
714            .get_type_or_void(it, annotator.index)
715            .has_nature(TypeNature::Num, annotator.index)
716    }) {
717        // we are trying to call this function with a non-numerical type, so we redirect back to the resolver
718        annotator.annotate_call_statement(operator, Some(parameters), &ctx);
719        return;
720    }
721
722    let mut ctx = ctx;
723    // find biggest type to later annotate it as type hint. this is done in a closure to avoid a borrow-checker tantrum later on due to
724    // mutable and immutable borrow of TypeAnnotator
725    let find_biggest_param_type_name = |annotator: &TypeAnnotator| {
726        let mut bigger = annotator
727            .annotation_map
728            .get_type_or_void(params_flattened.first().expect("must have this parameter"), annotator.index);
729
730        for param in params_flattened.iter().skip(1) {
731            let right_type = annotator.annotation_map.get_type_or_void(param, annotator.index);
732            bigger = get_bigger_type(bigger, right_type, annotator.index);
733        }
734
735        bigger.get_name().to_owned()
736    };
737
738    let bigger_type = find_biggest_param_type_name(annotator);
739
740    // create nested AstStatement::BinaryExpression for each parameter, such that
741    // ADD(a, b, c, d) ends up as (((a + b) + c) + d)
742    let left = (*params_flattened.first().expect("Must exist")).clone();
743    let new_statement = params_flattened.into_iter().skip(1).fold(left, |left, right| {
744        AstFactory::create_binary_expression(left, operation, right.clone(), ctx.id_provider.next_id())
745    });
746
747    annotator.visit_statement(&ctx, &new_statement);
748    annotator.update_expected_types(annotator.index.get_type_or_panic(&bigger_type), &new_statement);
749    annotator.annotate(statement, StatementAnnotation::ReplacementAst { statement: new_statement });
750    annotator.update_expected_types(annotator.index.get_type_or_panic(&bigger_type), statement);
751}

753fn annotate_variable_length_array_bound_function(
754    annotator: &mut TypeAnnotator,
755    parameters: Option<&AstNode>,
756) {
757    let Some(parameters) = parameters else {
758        // caught during validation
759        return;
760    };
761    let params = ast::flatten_expression_list(parameters);
762    let Some(vla) = params.first() else {
763        // caught during validation
764        return;
765    };
766
767    // if the VLA parameter is a VLA struct, annotate it as such
768    let vla_type = annotator.annotation_map.get_type_or_void(vla, annotator.index);
769    let vla_type_name = if vla_type.get_nature() == TypeNature::__VLA {
770        vla_type.get_name()
771    } else {
772        // otherwise annotate it with an internal, reserved VLA type
773        typesystem::__VLA_TYPE
774    };
775
776    annotator.annotation_map.annotate_type_hint(vla, StatementAnnotation::value(vla_type_name));
777}

779fn validate_variable_length_array_bound_function(
780    validator: &mut Validator,
781    operator: &AstNode,
782    parameters: Option<&AstNode>,
783    annotations: &dyn AnnotationMap,
784    index: &Index,
785) {
786    let Some(parameters) = parameters else {
787        validator.push_diagnostic(Diagnostic::invalid_argument_count(2, 0, operator));
788
789        // no params, nothing to validate
790        return;
791    };
792
793    let params = ast::flatten_expression_list(parameters);
794
795    if params.len() > 2 {
796        validator.push_diagnostic(Diagnostic::invalid_argument_count(2, params.len(), operator));
797    }
798
799    match (params.first(), params.get(1)) {
800        (Some(vla), Some(idx)) => {
801            let idx_type = annotations.get_type_or_void(idx, index);
802
803            if !idx_type.has_nature(TypeNature::Int, index) {
804                validator.push_diagnostic(
805                    Diagnostic::new(format!(
806                        "Invalid type nature for generic argument. {} is no {}",
807                        idx_type.get_name(),
808                        TypeNature::Int
809                    ))
810                    .with_error_code("E062")
811                    .with_location(*idx),
812                )
813            }
814
815            // TODO: consider adding validation for consts and enums once https://github.com/PLC-lang/rusty/issues/847 has been implemented
816            if let AstStatement::Literal(AstLiteral::Integer(dimension_idx)) = idx.get_stmt() {
817                let dimension_idx = *dimension_idx as usize;
818
819                let Some(n_dimensions) =
820                    annotations.get_type_or_void(vla, index).get_type_information().get_dimension_count()
821                else {
822                    // not a vla, validated via type nature
823                    return;
824                };
825
826                if dimension_idx < 1 || dimension_idx > n_dimensions {
827                    validator.push_diagnostic(
828                        Diagnostic::new("Index out of bound").with_error_code("E046").with_location(operator),
829                    )
830                }
831            };
832        }
833        (Some(_), None) => validator.push_diagnostic(Diagnostic::invalid_argument_count(2, 1, operator)),
834        _ => unreachable!(),
835    }
836}

838fn validate_argument_count(
839    validator: &mut Validator,
840    operator: &AstNode,
841    parameters: &Option<&AstNode>,
842    expected: usize,
843) {
844    let Some(params) = parameters else {
845        validator.push_diagnostic(Diagnostic::invalid_argument_count(expected, 0, operator));
846        return;
847    };
848
849    let params = flatten_expression_list(params);
850
851    if params.len() != expected {
852        validator.push_diagnostic(Diagnostic::invalid_argument_count(expected, params.len(), operator));
853    }
854}

856/// Generates the code for the LOWER- AND UPPER_BOUND built-in functions, returning an error if the function
857/// arguments are incorrect.
858fn generate_variable_length_array_bound_function<'ink>(
859    generator: &ExpressionCodeGenerator<'ink, '_>,
860    params: &[&AstNode],
861    is_lower: bool,
862    location: SourceLocation,
863) -> Result<ExpressionValue<'ink>, Diagnostic> {
864    let llvm = generator.llvm;
865    let builder = &generator.llvm.builder;
866    let data_type_information =
867        generator.annotations.get_type_or_void(params[0], generator.index).get_type_information();
868
869    // TODO: most of the codegen errors should already be caught during validation.
870    // once we abort codegen on critical errors, revisit and change to unreachable where possible
871    if !data_type_information.is_vla() {
872        return Err(Diagnostic::codegen_error(
873            format!("Expected VLA type, received {}", data_type_information.get_name()),
874            location,
875        ));
876    };
877
878    let vla = generator.generate_lvalue(params[0]).unwrap();
879    let dim = builder.build_struct_gep(vla, 1, "dim").unwrap();
880
881    let accessor = match params[1].get_stmt() {
882        // e.g. LOWER_BOUND(arr, 1)
883        AstStatement::Literal(kind) => {
884            let AstLiteral::Integer(value) = kind else {
885                let Some(type_name) = get_literal_actual_signed_type_name(kind, false) else {
886                    unreachable!("type cannot be VOID")
887                };
888                return Err(Diagnostic::codegen_error(
889                    format!("Invalid literal type. Expected INT type, received {type_name} type"),
890                    location,
891                ));
892            };
893            // array offset start- and end-values are adjacent values in a flattened array -> 2 values per dimension, so in order
894            // to read the correct values, the given index needs to be doubled. Additionally, the value is adjusted for 0-indexing.
895            let offset = if is_lower { (value - 1) as u64 * 2 } else { (value - 1) as u64 * 2 + 1 };
896            llvm.i32_type().const_int(offset, false)
897        }
898        // e.g. LOWER_BOUND(arr, idx + 3)
899        _ => {
900            let expression_value = generator.generate_expression(params[1])?;
901            if !expression_value.is_int_value() {
902                todo!()
903            };
904            // this operation mirrors the offset calculation of literal ints, but at runtime
905            let offset = builder.build_int_mul(
906                llvm.i32_type().const_int(2, false),
907                builder.build_int_sub(
908                    expression_value.into_int_value(),
909                    llvm.i32_type().const_int(1, false),
910                    "",
911                ),
912                "",
913            );
914            if !is_lower {
915                builder.build_int_add(offset, llvm.i32_type().const_int(1, false), "")
916            } else {
917                offset
918            }
919        }
920    };
921
922    let gep_bound =
923        unsafe { llvm.builder.build_in_bounds_gep(dim, &[llvm.i32_type().const_zero(), accessor], "") };
924    let bound = llvm.builder.build_load(gep_bound, "");
925
926    Ok(ExpressionValue::RValue(bound))
927}

968pub fn parse_built_ins(id_provider: IdProvider) -> CompilationUnit {
969    let src = BUILTIN.values().map(|it| it.decl).collect::<Vec<&str>>().join(" ");
970    let mut unit = parser::parse(
971        lexer::lex_with_ids(&src, id_provider.clone(), SourceLocationFactory::internal(&src)),
972        LinkageType::BuiltIn,
973        "<builtin>",
974    )
975    .0;
976
977    pre_process(&mut unit, id_provider);
978    unit
979}
