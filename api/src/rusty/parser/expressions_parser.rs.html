21macro_rules! parse_left_associative_expression {
22    ($lexer: expr, $action : expr,
23        $( $pattern:pat_param )|+,
24    ) => {
25        {
26            let mut left = $action($lexer);
27            while matches!($lexer.token, $( $pattern )|+)  {
28                let operator = match to_operator(&$lexer.token) {
29                    Some(operator) => operator,
30                    None => break,
31                };
32                $lexer.advance();
33                let right = $action($lexer);
34                left = AstFactory::create_binary_expression(left, operator, right, $lexer.next_id());
35            }
36            left
37        }
38    };
39}

41/// parse_expression(): returns expression as Statement. if a parse error
42/// is encountered, the erroneous part of the AST will consist of an
43/// EmptyStatement and a diagnostic will be logged. That case is different from
44/// only an EmptyStatement returned, which does not denote an error condition.
45pub fn parse_expression(lexer: &mut ParseSession) -> AstNode {
46    if lexer.token == KeywordSemicolon {
47        AstFactory::create_empty_statement(lexer.location(), lexer.next_id())
48    } else {
49        parse_expression_list(lexer)
50    }
51}
52
53pub fn parse_expression_list(lexer: &mut ParseSession) -> AstNode {
54    let start = lexer.location();
55    let left = parse_range_statement(lexer);
56    if lexer.token == KeywordComma {
57        let mut expressions = vec![];
58        // this starts an expression list
59        while lexer.token == KeywordComma {
60            lexer.advance();
61            if !lexer.closes_open_region(&lexer.token) {
62                expressions.push(parse_range_statement(lexer));
63            }
64        }
65        // we may have parsed no additional expression because of trailing comma
66        if !expressions.is_empty() {
67            expressions.insert(0, left);
68            return AstFactory::create_expression_list(
69                expressions,
70                start.span(&lexer.last_location()),
71                lexer.next_id(),
72            );
73        }
74    }
75    left
76}

78pub(crate) fn parse_range_statement(lexer: &mut ParseSession) -> AstNode {
79    let start = parse_or_expression(lexer);
80
81    if lexer.token == KeywordDotDot {
82        lexer.advance();
83        let end = parse_or_expression(lexer);
84        return AstFactory::create_range_statement(start, end, lexer.next_id());
85    }
86    start
87}

132// Expoent **
133fn parse_exponent_expression(lexer: &mut ParseSession) -> AstNode {
134    //This is always parsed as a function call to the EXPT function
135    //Parse left
136    let mut left = parse_unary_expression(lexer);
137    while matches!(lexer.token, OperatorExponent) {
138        lexer.advance();
139        let right = parse_unary_expression(lexer);
140        let span = left.get_location().span(&right.get_location());
141        left =
142            AstFactory::create_call_to_with_ids("EXPT", vec![left, right], &span, lexer.id_provider.clone());
143    }
144    left
145}

147// UNARY -x, NOT x
148fn parse_unary_expression(lexer: &mut ParseSession) -> AstNode {
149    // collect all consecutive operators
150    let start_location = lexer.location();
151    let mut operators = vec![];
152    while let Some(operator) = match lexer.token {
153        OperatorNot => Some(Operator::Not),
154        OperatorPlus => Some(Operator::Plus),
155        OperatorMinus => Some(Operator::Minus),
156        _ => None,
157    } {
158        operators.push(operator);
159        lexer.advance();
160    }
161    // created nested statements if necessary (e.g. &&)
162    let init = parse_leaf_expression(lexer);
163    operators.iter().rev().fold(init, |expression, operator| {
164        let expression_location = expression.get_location();
165        let location = start_location.span(&expression_location);
166
167        match (&operator, &expression.get_stmt()) {
168            (Operator::Minus, AstStatement::Literal(AstLiteral::Integer(value))) => {
169                AstNode::new_literal(AstLiteral::new_integer(-value), lexer.next_id(), location)
170            }
171
172            (Operator::Plus, AstStatement::Literal(AstLiteral::Integer(value))) => {
173                AstNode::new_literal(AstLiteral::new_integer(*value), lexer.next_id(), location)
174            }
175
176            // Return the reference itself instead of wrapping it inside a `AstStatement::UnaryExpression`
177            (Operator::Plus, AstStatement::Identifier(name)) => {
178                AstFactory::create_identifier(name, &location, lexer.next_id())
179            }
180
181            _ => AstFactory::create_unary_expression(*operator, expression, location, lexer.next_id()),
182        }
183    })
184}

186fn to_operator(token: &Token) -> Option<Operator> {
187    match token {
188        OperatorPlus => Some(Operator::Plus),
189        OperatorMinus => Some(Operator::Minus),
190        OperatorMultiplication => Some(Operator::Multiplication),
191        OperatorExponent => Some(Operator::Exponentiation),
192        OperatorDivision => Some(Operator::Division),
193        OperatorEqual => Some(Operator::Equal),
194        OperatorNotEqual => Some(Operator::NotEqual),
195        OperatorLess => Some(Operator::Less),
196        OperatorGreater => Some(Operator::Greater),
197        OperatorLessOrEqual => Some(Operator::LessOrEqual),
198        OperatorGreaterOrEqual => Some(Operator::GreaterOrEqual),
199        OperatorModulo => Some(Operator::Modulo),
200        OperatorAnd | OperatorAmp => Some(Operator::And),
201        OperatorOr => Some(Operator::Or),
202        OperatorXor => Some(Operator::Xor),
203        OperatorNot => Some(Operator::Not),
204        _ => None,
205    }
206}

208// Literals, Identifiers, etc.
209fn parse_leaf_expression(lexer: &mut ParseSession) -> AstNode {
210    let literal_parse_result = match lexer.token {
211        OperatorMultiplication => parse_vla_range(lexer),
212        _ => parse_call_statement(lexer),
213    };
214
215    match literal_parse_result {
216        Some(statement) => match lexer.token {
217            KeywordAssignment => {
218                lexer.advance();
219                AstFactory::create_assignment(statement, parse_range_statement(lexer), lexer.next_id())
220            }
221            KeywordOutputAssignment => {
222                lexer.advance();
223                AstFactory::create_output_assignment(statement, parse_range_statement(lexer), lexer.next_id())
224            }
225            KeywordReferenceAssignment => {
226                lexer.advance();
227                AstFactory::create_ref_assignment(statement, parse_range_statement(lexer), lexer.next_id())
228            }
229            _ => statement,
230        },
231        None => {
232            let statement = AstFactory::create_empty_statement(
233                lexer.diagnostics.last().map_or(SourceLocation::undefined(), |d| d.get_location()),
234                lexer.next_id(),
235            );
236            statement
237        }
238    }
239}

241/// parse an expression at the bottom of the parse-tree.
242/// leaf-expressions are literals, identifier, direct-access and parenthesized expressions
243/// (since the parentheses change the parse-priority)
244fn parse_atomic_leaf_expression(lexer: &mut ParseSession<'_>) -> Option<AstNode> {
245    // Check if we're dealing with a number that has an explicit '+' or '-' sign...
246
247    match lexer.token {
248        OperatorPlus | OperatorMinus => {
249            let is_negative = lexer.token == OperatorMinus;
250            lexer.advance();
251
252            match lexer.token {
253                LiteralInteger => parse_literal_number(lexer, is_negative),
254                LiteralIntegerBin => parse_literal_number_with_modifier(lexer, 2, is_negative),
255                LiteralIntegerOct => parse_literal_number_with_modifier(lexer, 8, is_negative),
256                LiteralIntegerHex => parse_literal_number_with_modifier(lexer, 16, is_negative),
257                _ => {
258                    lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
259                        "Numeric Literal",
260                        lexer.slice(),
261                        lexer.location(),
262                    ));
263                    None
264                }
265            }
266        }
267        KeywordParensOpen => {
268            parse_any_in_region(lexer, vec![KeywordParensClose], |lexer| {
269                lexer.advance(); // eat KeywordParensOpen
270
271                let start = lexer.last_location();
272                let expr = parse_expression(lexer);
273
274                Some(AstFactory::create_paren_expression(
275                    expr,
276                    start.span(&lexer.location()),
277                    lexer.next_id(),
278                ))
279            })
280        }
281        Identifier => Some(parse_identifier(lexer)),
282        KeywordSuper => {
283            lexer.advance();
284            Some(AstFactory::create_super_reference(
285                lexer.last_location(),
286                lexer.try_consume(OperatorDeref).then_some(()),
287                lexer.next_id(),
288            ))
289        }
290        KeywordThis => {
291            lexer.advance();
292            Some(AstFactory::create_this_reference(lexer.last_location(), lexer.next_id()))
293        }
294        HardwareAccess((hw_type, access_type)) => parse_hardware_access(lexer, hw_type, access_type),
295        LiteralInteger => parse_literal_number(lexer, false),
296        LiteralIntegerBin => parse_literal_number_with_modifier(lexer, 2, false),
297        LiteralIntegerOct => parse_literal_number_with_modifier(lexer, 8, false),
298        LiteralIntegerHex => parse_literal_number_with_modifier(lexer, 16, false),
299        LiteralDate => parse_literal_date(lexer),
300        LiteralTimeOfDay => parse_literal_time_of_day(lexer),
301        LiteralTime => parse_literal_time(lexer),
302        LiteralDateAndTime => parse_literal_date_and_time(lexer),
303        LiteralString => parse_literal_string(lexer, false),
304        LiteralWideString => parse_literal_string(lexer, true),
305        LiteralTrue => parse_bool_literal(lexer, true),
306        LiteralFalse => parse_bool_literal(lexer, false),
307        LiteralNull => parse_null_literal(lexer),
308        KeywordSquareParensOpen => parse_array_literal(lexer),
309        DirectAccess(access) => parse_direct_access(lexer, access),
310        _ => {
311            if lexer.closing_keywords.contains(&vec![KeywordParensClose])
312                && matches!(lexer.last_token, KeywordOutputAssignment | KeywordAssignment)
313            {
314                // due to closing keyword ')' and last_token '=>' / ':='
315                // we are probably in a call statement missing a parameter assignment 'foo(param := );
316                // optional parameter assignments are allowed, validation should handle any unwanted cases
317                Some(AstFactory::create_empty_statement(lexer.location(), lexer.next_id()))
318            } else {
319                lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
320                    "Literal",
321                    lexer.slice(),
322                    lexer.location(),
323                ));
324                None
325            }
326        }
327    }
328}

334fn parse_vla_range(lexer: &mut ParseSession) -> Option<AstNode> {
335    lexer.advance();
336    Some(AstFactory::create_vla_range_statement(lexer.last_location(), lexer.next_id()))

339fn parse_array_literal(lexer: &mut ParseSession) -> Option<AstNode> {
340    let start = lexer.range().start;
341    expect_token!(lexer, KeywordSquareParensOpen, None);
342    lexer.advance();
343    let elements = Some(Box::new(parse_expression(lexer)));
344    let end = lexer.range().end;
345    expect_token!(lexer, KeywordSquareParensClose, None);
346    lexer.advance();
347
348    Some(AstNode::new_literal(
349        AstLiteral::new_array(elements),
350        lexer.next_id(),
351        lexer.source_range_factory.create_range(start..end),
352    ))
353}

372pub fn parse_call_statement(lexer: &mut ParseSession) -> Option<AstNode> {
373    let reference = parse_qualified_reference(lexer)?;
374    let reference_loc = reference.get_location();
375
376    // We're not dealing with a call statement here
377    if !lexer.try_consume(KeywordParensOpen) {
378        return Some(reference);
379    }
380
381    let call = if lexer.try_consume(KeywordParensClose) {
382        AstFactory::create_call_statement(
383            reference,
384            None,
385            lexer.next_id(),
386            reference_loc.span(&lexer.location()),
387        )
388    } else {
389        parse_any_in_region(lexer, vec![KeywordParensClose], |lexer| {
390            AstFactory::create_call_statement(
391                reference,
392                Some(parse_expression_list(lexer)),
393                lexer.next_id(),
394                reference_loc.span(&lexer.location()),
395            )
396        })
397    };
398
399    // Are we dealing with an array-index access directly after the call, e.g. `foo()[...]`?
400    if lexer.try_consume(KeywordSquareParensOpen) {
401        let index = parse_any_in_region(lexer, vec![KeywordSquareParensClose], parse_expression);
402        let statement = AstFactory::create_index_reference(
403            index,
404            Some(call),
405            lexer.next_id(),
406            SourceLocation::undefined(),
407        );
408
409        return Some(statement);
410    }
411
412    Some(call)
413}

415pub fn parse_qualified_reference(lexer: &mut ParseSession) -> Option<AstNode> {
416    let mut current = None;
417    let mut pos = lexer.parse_progress - 1; // force an initial loop
418
419    // as long as we parse something we keep eating stuff eagerly
420    while lexer.parse_progress > pos {
421        pos = lexer.parse_progress;
422        match (
423            current,
424            // only test for the tokens without eating it
425            [KeywordDot, KeywordSquareParensOpen, OperatorDeref, TypeCastPrefix]
426                .into_iter()
427                .find(|it| lexer.token == *it),
428        ) {
429            // No base, No token -> Beginning of a qualified reference
430            (None, None) => {
431                let exp = parse_atomic_leaf_expression(lexer)?;
432                // pack if this is something to be resolved
433                current = if exp.is_identifier() {
434                    Some(AstFactory::create_member_reference(exp, None, lexer.next_id()))
435                } else {
436                    Some(exp)
437                };
438            }
439            // Global Namespace Operator, e.g. `.foo`
440            (None, Some(KeywordDot)) => {
441                let location_dot = lexer.location();
442                lexer.advance();
443
444                let expr = parse_atomic_leaf_expression(lexer)?;
445                let location = location_dot.span(&expr.location);
446
447                current = Some(AstFactory::create_global_reference(lexer.next_id(), expr, location));
448            }
449            // base._ -> a segment of a qualified reference, we stand right after the dot
450            (Some(base), Some(KeywordDot)) => {
451                lexer.advance();
452                let member = if lexer.token == LiteralInteger {
453                    let index = parse_strict_literal_integer(lexer)?;
454                    let location = index.get_location();
455                    AstFactory::create_direct_access(DirectAccessType::Bit, index, lexer.next_id(), location)
456                } else {
457                    parse_atomic_leaf_expression(lexer)?
458                };
459                current = Some(AstFactory::create_member_reference(member, Some(base), lexer.next_id()));
460            }
461            // CAST-Statement: INT#a.b.c
462            // this means INT#(a.b.c) rather than (INT#a).b.c
463            (_, Some(TypeCastPrefix)) => {
464                let location_start = lexer.range().start;
465                let location = lexer.location();
466                let mut type_name = lexer.slice_and_advance();
467                type_name.pop(); // get rid of the "#" at the end
468                let stmt = parse_atomic_leaf_expression(lexer)?;
469                let end = stmt.get_location();
470                let type_range = lexer
471                    .source_range_factory
472                    .create_range(location_start..(location_start + type_name.len()));
473                current = Some(AstFactory::create_cast_statement(
474                    AstFactory::create_member_reference(
475                        AstFactory::create_identifier(type_name.as_str(), &type_range, lexer.next_id()),
476                        None,
477                        lexer.next_id(),
478                    ),
479                    stmt,
480                    &location.span(&end),
481                    lexer.next_id(),
482                ));
483            }
484            (Some(base), Some(KeywordSquareParensOpen)) => {
485                lexer.advance();
486                let index_reference =
487                    parse_any_in_region(lexer, vec![KeywordSquareParensClose], parse_expression);
488                let new_location = base.get_location().span(&lexer.last_location());
489                current = Some({
490                    AstFactory::create_index_reference(
491                        index_reference,
492                        Some(base),
493                        lexer.next_id(),
494                        new_location,
495                    )
496                })
497            }
498            (Some(base), Some(OperatorDeref)) => {
499                lexer.advance();
500                let new_location = base.get_location().span(&lexer.last_location());
501                current = Some(AstFactory::create_deref_reference(base, lexer.next_id(), new_location))
502            }
503            (last_current, _) => {
504                current = last_current; // exit the loop
505            }
506        }
507    }
508    match current {
509        Some(current) => Some(current),
510        None => parse_atomic_leaf_expression(lexer),
511    }
512}

514fn parse_direct_access(lexer: &mut ParseSession, access: DirectAccessType) -> Option<AstNode> {
515    //Consume the direct access
516    let location = lexer.location();
517    lexer.advance();
518    //The next token can either be an integer or an identifier
519    let index = match lexer.token {
520        LiteralInteger => parse_strict_literal_integer(lexer),
521        Identifier => {
522            let location = lexer.location();
523            Some(AstFactory::create_member_reference(
524                AstFactory::create_identifier(lexer.slice_and_advance().as_str(), location, lexer.next_id()),
525                None,
526                lexer.next_id(),
527            ))
528        }
529        _ => {
530            lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
531                "Integer or Reference",
532                lexer.slice(),
533                lexer.location(),
534            ));
535            None
536        }
537    }?;
538    let location = location.span(&lexer.last_location());
539    Some(AstFactory::create_direct_access(access, index, lexer.next_id(), location))
540}

542fn parse_literal_number_with_modifier(
543    lexer: &mut ParseSession,
544    radix: u32,
545    is_negative: bool,
546) -> Option<AstNode> {
547    // we can safely unwrap the number string, since the token has
548    // been matched using regular expressions
549    let location = lexer.location();
550    let token = lexer.slice_and_advance();
551    let number_str = token.split('#').next_back().expect("token with '#'");
552    let number_str = number_str.replace('_', "");
553
554    // again, the parsed number can be safely unwrapped.
555    let value = i128::from_str_radix(number_str.as_str(), radix).expect("valid i128");
556    let value = if is_negative { -value } else { value };
557    Some(AstNode::new_literal(AstLiteral::new_integer(value), lexer.next_id(), location))
558}

560fn parse_literal_number(lexer: &mut ParseSession, is_negative: bool) -> Option<AstNode> {
561    let location = if is_negative {
562        //correct the location if we just parsed a minus before
563        lexer.last_range.start..lexer.range().end
564    } else {
565        lexer.range()
566    };
567    let result = lexer.slice_and_advance();
568    if result.to_lowercase().contains('e') {
569        let value = result.replace('_', "");
570        //Treat exponents as reals
571        return Some(AstNode::new_literal(
572            AstLiteral::new_real(value),
573            lexer.next_id(),
574            lexer.source_range_factory.create_range(location),
575        ));
576    } else if lexer.try_consume(KeywordDot) {
577        return parse_literal_real(lexer, result, location, is_negative);
578    } else if lexer.try_consume(KeywordParensOpen) {
579        let start = location.start;
580        let multiplier = match result.parse::<u32>() {
581            Ok(v) => Some(v),
582            Err(e) => {
583                lexer.accept_diagnostic(
584                    Diagnostic::new(format!("Failed to parse number {result}"))
585                        .with_error_code("E011")
586                        .with_location(lexer.source_range_factory.create_range(location))
587                        .with_internal_error(e.into()),
588                );
589                None
590            }
591        }?;
592        let element = parse_expression(lexer);
593        expect_token!(lexer, KeywordParensClose, None);
594        let end = lexer.range().end;
595        lexer.advance();
596        return Some(AstFactory::create_multiplied_statement(
597            multiplier,
598            element,
599            lexer.source_range_factory.create_range(start..end),
600            lexer.next_id(),
601        ));
602    }
603
604    // parsed number value can be safely unwrapped
605    let result = result.replace('_', "");
606
607    let value = result.parse::<i128>().expect("valid i128");
608    let value = if is_negative { -value } else { value };
609
610    Some(AstNode::new_literal(
611        AstLiteral::new_integer(value),
612        lexer.next_id(),
613        lexer.source_range_factory.create_range(location),
614    ))
615}

617/// Parses a literal integer without considering Signs or the Possibility of a Floating Point/ Exponent
618pub fn parse_strict_literal_integer(lexer: &mut ParseSession) -> Option<AstNode> {
619    //correct the location if we just parsed a minus before
620    let location = lexer.location();
621    let result = lexer.slice_and_advance();
622    // parsed number value can be safely unwrapped
623    let result = result.replace('_', "");
624    if result.to_lowercase().contains('e') {
625        lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
626            "Integer",
627            &format!("Exponent value: {result}"),
628            location,
629        ));
630        None
631    } else {
632        let value = result.parse::<i128>().expect("valid i128");
633        Some(AstNode::new_literal(AstLiteral::new_integer(value), lexer.next_id(), location))
634    }
635}

637fn parse_number<F: FromStr>(lexer: &mut ParseSession, text: &str, location: &SourceLocation) -> Option<F> {
638    match text.parse::<F>() {
639        Ok(v) => Some(v),
640        Err(_) => {
641            lexer.accept_diagnostic(
642                Diagnostic::new(format!("Failed to parse number {text}"))
643                    .with_error_code("E011")
644                    .with_location(location),
645            );
646            None
647        }
648    }
649}

651fn parse_date_from_string(
652    lexer: &mut ParseSession,
653    text: &str,
654    location: SourceLocation,
655    id: AstId,
656) -> Option<AstNode> {
657    let mut segments = text.split('-');
658
659    //we can safely expect 3 numbers
660    let year = segments
661        .next()
662        .map(|s| parse_number::<i32>(lexer, s, &location))
663        .expect("year-segment - tokenizer broken?")?;
664    let month = segments
665        .next()
666        .map(|s| parse_number::<u32>(lexer, s, &location))
667        .expect("month-segment - tokenizer broken?")?;
668    let day = segments
669        .next()
670        .map(|s| parse_number::<u32>(lexer, s, &location))
671        .expect("day-segment - tokenizer broken?")?;
672
673    Some(AstNode::new_literal(AstLiteral::new_date(year, month, day), id, location))
674}

676fn parse_literal_date_and_time(lexer: &mut ParseSession) -> Option<AstNode> {
677    let location = lexer.location();
678    //get rid of D# or DATE#
679    let slice = lexer.slice_and_advance();
680    let hash_location = slice.find('#').unwrap_or_default();
681    let last_minus_location = slice.rfind('-').expect("unexpected date-and-time syntax");
682
683    let (_, date_and_time) = slice.split_at(hash_location + 1); //get rid of the prefix
684    let (date, time) = date_and_time.split_at(last_minus_location - hash_location);
685
686    //we can safely expect 3 numbers
687    let mut segments = date.split('-');
688    let msg = "unexpected date-and-time syntax";
689    let year = parse_number::<i32>(lexer, segments.next().expect(msg), &location)?;
690    let month = parse_number::<u32>(lexer, segments.next().expect(msg), &location)?;
691    let day = parse_number::<u32>(lexer, segments.next().expect(msg), &location)?;
692
693    //we can safely expect 3 numbers
694    let mut segments = time.split(':');
695    let (hour, min, sec, nano) = parse_time_of_day(lexer, &mut segments, &location)?;
696
697    Some(AstNode::new_literal(
698        AstLiteral::new_date_and_time(year, month, day, hour, min, sec, nano),
699        lexer.next_id(),
700        location,
701    ))
702}

728fn parse_time_of_day(
729    lexer: &mut ParseSession,
730    time: &mut Split<char>,
731    location: &SourceLocation,
732) -> Option<(u32, u32, u32, u32)> {
733    let hour = parse_number::<u32>(lexer, time.next().expect("expected hours"), location)?;
734    let min = parse_number::<u32>(lexer, time.next().expect("expected minutes"), location)?;
735
736    // doesn't necessarily have to have seconds, e.g [12:00] is also valid
737    let sec = match time.next() {
738        Some(v) => parse_number::<f64>(lexer, v, location)?,
739        None => 0.0,
740    };
741
742    let nano = (sec.fract() * 1e+9_f64).round() as u32;
743
744    Some((hour, min, sec.floor() as u32, nano))
745}

747fn parse_literal_time(lexer: &mut ParseSession) -> Option<AstNode> {
748    const POS_D: usize = 0;
749    const POS_H: usize = 1;
750    const POS_M: usize = 2;
751    const POS_S: usize = 3;
752    const POS_MS: usize = 4;
753    const POS_US: usize = 5;
754    const POS_NS: usize = 6;
755    let location = lexer.location();
756    //get rid of T# or TIME#
757    let slice = lexer.slice_and_advance();
758    let (_, slice) = slice.split_at(slice.find('#').unwrap_or_default() + 1); //get rid of the prefix
759
760    let mut chars = slice.char_indices();
761    let mut char = chars.next();
762
763    let is_negative = char.map(|(_, c)| c == '-').unwrap_or(false);
764    if is_negative {
765        char = chars.next();
766    }
767
768    let mut values: [Option<f64>; 7] = [None, None, None, None, None, None, None];
769
770    let mut prev_pos = POS_D;
771    while char.is_some() {
772        //expect a number
773        let number = {
774            let start = char.expect("char").0;
775            //just eat all the digits
776            char = chars.find(|(_, ch)| !ch.is_ascii_digit() && !ch.eq(&'.'));
777            match char {
778                None => {
779                    lexer.accept_diagnostic(
780                        Diagnostic::new("Invalid TIME Literal: Cannot parse segment.")
781                            .with_error_code("E010")
782                            .with_location(location),
783                    );
784                    return None;
785                }
786                Some((index, _)) => parse_number::<f64>(lexer, &slice[start..index], &location)?,
787            }
788        };
789
790        //expect a unit
791        let unit = {
792            let start = match char {
793                Some((index, _)) => index,
794                None => {
795                    lexer.accept_diagnostic(
796                        Diagnostic::new("Invalid TIME Literal: Missing unit (d|h|m|s|ms|us|ns)")
797                            .with_error_code("E010")
798                            .with_location(location),
799                    );
800                    return None;
801                }
802            };
803
804            //just eat all the characters
805            char = chars.find(|(_, ch)| !ch.is_ascii_alphabetic());
806            &slice[start..char.unwrap_or((slice.len(), ' ')).0]
807        }
808        .to_lowercase();
809
810        //now assign the number to the according segment of the value's array
811        let position = match unit.as_str() {
812            "d" => Some(POS_D),
813            "h" => Some(POS_H),
814            "m" => Some(POS_M),
815            "s" => Some(POS_S),
816            "ms" => Some(POS_MS),
817            "us" => Some(POS_US),
818            "ns" => Some(POS_NS),
819            _ => None,
820        };
821        if let Some(position) = position {
822            //check if we assign out of order - every assignment before must have been a smaller position
823            if prev_pos > position {
824                lexer.accept_diagnostic(
825                    Diagnostic::new("Invalid TIME Literal: segments out of order, use d-h-m-s-ms")
826                        .with_error_code("E010")
827                        .with_location(location),
828                );
829                return None;
830            }
831            prev_pos = position; //remember that we wrote this position
832
833            if values[position].is_some() {
834                lexer.accept_diagnostic(
835                    Diagnostic::new("Invalid TIME Literal: segments must be unique")
836                        .with_error_code("E010")
837                        .with_location(location),
838                );
839                return None;
840            }
841            values[position] = Some(number); //store the number
842        } else {
843            lexer.accept_diagnostic(
844                Diagnostic::new(format!("Invalid TIME Literal: illegal unit '{unit}'"))
845                    .with_error_code("E010")
846                    .with_location(location),
847            );
848            return None;
849        }
850    }
851
852    Some(AstNode::new_literal(
853        AstLiteral::Time(Time {
854            day: values[POS_D].unwrap_or_default(),
855            hour: values[POS_H].unwrap_or_default(),
856            min: values[POS_M].unwrap_or_default(),
857            sec: values[POS_S].unwrap_or_default(),
858            milli: values[POS_MS].unwrap_or_default(),
859            micro: values[POS_US].unwrap_or_default(),
860            nano: values[POS_NS].map(|it| it as u32).unwrap_or(0u32),
861            negative: is_negative,
862        }),
863        lexer.next_id(),
864        location,
865    ))
866}

872fn handle_special_chars(string: &str, is_wide: bool) -> String {
873    let (re, re_hex) = if is_wide {
874        (
875            Regex::new(r#"(\$([lLnNpPrRtT$"]))"#).expect("valid regex"), //Cannot fail
876            Regex::new(r"(\$([[:xdigit:]]{2}){2})+").expect("valid regex"), //Cannot fail
877        )
878    } else {
879        (
880            Regex::new(r"(\$([lLnNpPrRtT$']))").expect("valid regex"), //Cannot fail
881            Regex::new(r"(\$([[:xdigit:]]{2}))+").expect("valid regex"), //Cannot fail
882        )
883    };
884
885    // separated re and re_hex to minimize copying
886    let res = re.replace_all(string, |caps: &Captures| {
887        let cap_str = &caps[1];
888        match cap_str {
889            "$l" | "$L" => "\n",
890            "$n" | "$N" => "\n",
891            "$p" | "$P" => "\x0C",
892            "$r" | "$R" => "\r",
893            "$t" | "$T" => "\t",
894            "$$" => "$",
895            "$'" => "\'",
896            "$\"" => "\"",
897            _ => unreachable!(),
898        }
899    });
900
901    re_hex
902        .replace_all(&res, |caps: &Captures| {
903            let hex = &caps[0];
904            let hex_vals: Vec<&str> = hex.split('$').filter(|it| !it.is_empty()).collect();
905            let res = if is_wide {
906                let hex_vals: Vec<u16> =
907                    hex_vals.iter().map(|it| u16::from_str_radix(it, 16).unwrap_or_default()).collect();
908                String::from_utf16_lossy(&hex_vals)
909            } else {
910                let hex_vals: Vec<u8> =
911                    hex_vals.iter().map(|it| u8::from_str_radix(it, 16).unwrap_or_default()).collect();
912                String::from_utf8_lossy(&hex_vals).to_string()
913            };
914            res
915        })
916        .into()
917}

919fn parse_literal_string(lexer: &mut ParseSession, is_wide: bool) -> Option<AstNode> {
920    let result = lexer.slice();
921    let location = lexer.location();
922
923    let string_literal = Some(AstNode::new_literal(
924        AstLiteral::new_string(handle_special_chars(&trim_quotes(result), is_wide), is_wide),
925        lexer.next_id(),
926        location,
927    ));
928    lexer.advance();
929    string_literal
930}

932fn parse_literal_real(
933    lexer: &mut ParseSession,
934    integer: String,
935    integer_range: Range<usize>,
936    is_negative: bool,
937) -> Option<AstNode> {
938    if lexer.token == LiteralInteger {
939        let start = integer_range.start;
940        let end = lexer.range().end;
941        let fractional = lexer.slice_and_advance();
942        let value = format!("{}{}.{}", if is_negative { "-" } else { "" }, integer, fractional);
943        let new_location = lexer.source_range_factory.create_range(start..end);
944
945        Some(AstNode::new_literal(AstLiteral::new_real(value), lexer.next_id(), new_location))
946    } else {
947        lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
948            "LiteralInteger or LiteralExponent",
949            lexer.slice(),
950            lexer.location(),
951        ));
952        None
953    }
954}
