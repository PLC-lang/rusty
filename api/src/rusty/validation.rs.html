40pub struct ValidationContext<'s, T: AnnotationMap> {
41    annotations: &'s T,
42    index: &'s Index,
43    /// the type_name of the context for a reference (e.g. `a.b` where `a`'s type is the context of `b`)
44    qualifier: Option<&'s str>,
45    is_call: bool,
46    is_cast: bool,
47}
48
49impl<'s, T: AnnotationMap> ValidationContext<'s, T> {
50    fn with_qualifier(&self, qualifier: &'s str) -> Self {
51        ValidationContext { qualifier: Some(qualifier), ..self.to_owned() }
52    }
53
54    fn with_optional_qualifier(&self, qualifier: Option<&'s str>) -> Self {
55        ValidationContext { qualifier, ..self.to_owned() }
56    }
57
58    fn find_pou(&self, stmt: &AstNode) -> Option<&PouIndexEntry> {
59        self.annotations.get_call_name(stmt).and_then(|pou_name| {
60            self.index
61                // check if this is an instance of a function block and get the type name
62                .find_callable_instance_variable(self.qualifier, &[pou_name])
63                .map(|it| it.get_type_name())
64                // if it is not an instance, check if we are dealing with an action and get the base POU name
65                .or_else(|| self.index.find_implementation_by_name(pou_name).map(|it| it.get_type_name()))
66                // we didn't encounter an instance or action call, keep initial name
67                .or(Some(pou_name))
68                .and_then(|name| self.index.find_pou(name))
69        })
70    }
71
72    fn set_is_call(&self) -> Self {
73        ValidationContext { is_call: true, ..self.to_owned() }
74    }
75
76    fn is_call(&self) -> bool {
77        self.is_call
78    }
79
80    fn set_cast(&self) -> Self {
81        ValidationContext { is_cast: true, ..self.to_owned() }
82    }
83}

85impl<T: AnnotationMap> Clone for ValidationContext<'_, T> {
86    fn clone(&self) -> Self {
87        ValidationContext {

89            index: self.index,
90            qualifier: self.qualifier,
91            is_call: self.is_call,
92            is_cast: self.is_cast,
93        }
94    }
95}

97/// This trait should be implemented by any validator used by `validation::Validator`
98pub trait Validators {
99    fn push_diagnostic(&mut self, diagnostic: Diagnostic);
100
101    fn take_diagnostics(&mut self) -> Vec<Diagnostic>;
102}

104pub struct Validator<'a> {
105    context: &'a GlobalContext,
106    diagnostics: Vec<Diagnostic>,
107    global_validator: GlobalValidator,
108    recursive_validator: RecursiveValidator,
109}
110
111impl Validators for Validator<'_> {
112    fn push_diagnostic(&mut self, diagnostic: Diagnostic) {
113        self.diagnostics.push(diagnostic);
114    }
115    fn take_diagnostics(&mut self) -> Vec<Diagnostic> {
116        std::mem::take(&mut self.diagnostics)
117    }
118}
119
120impl<'a> Validator<'a> {
121    pub fn new(context: &'a GlobalContext) -> Validator<'a> {
122        Validator {
123            context,
124            diagnostics: Vec::new(),
125            global_validator: GlobalValidator::new(),
126            recursive_validator: RecursiveValidator::new(),
127        }
128    }
129
130    pub fn get_type_name_or_slice(&self, dt: &DataType) -> String {
131        if dt.is_internal() {
132            return dt.get_type_information().get_inner_name().to_string();
133        }
134
135        self.context.slice(&dt.location)
136    }
137
138    pub fn diagnostics(&mut self) -> Vec<Diagnostic> {
139        let mut all_diagnostics = Vec::new();
140        all_diagnostics.append(&mut self.take_diagnostics());
141        all_diagnostics.append(&mut self.global_validator.take_diagnostics());
142        all_diagnostics.append(&mut self.recursive_validator.take_diagnostics());
143        all_diagnostics
144    }
145
146    pub fn perform_global_validation(&mut self, index: &Index) {
147        self.global_validator.validate(index);
148        self.recursive_validator.validate(index);
149        self.validate_configured_templates(index);
150
151        // XXX: To avoid bloating up this function any further, maybe package logic into seperate module or
152        //      function if another global check is introduced (including the overflow checks)?
153        // Find and report const-expressions that would overflow
154        for it in index.get_const_expressions().into_iter() {
155            let Some(expr) = index.get_const_expressions().find_const_expression(&it.0) else { continue };
156            if let ConstExpression::Unresolvable { reason, .. } = expr {
157                if let UnresolvableKind::Overflow(reason, location) = reason.as_ref() {
158                    self.push_diagnostic(
159                        Diagnostic::new(reason).with_error_code("E039").with_location(location),
160                    );
161                }
162            };
163        }
164    }

166    pub fn visit_unit<T: AnnotationMap>(&mut self, annotations: &T, index: &Index, unit: &CompilationUnit) {
167        let context =
168            ValidationContext { annotations, index, qualifier: None, is_call: false, is_cast: false };
169        // Validate POU and declared Variables
170        for pou in &unit.pous {
171            let context = context.with_qualifier(pou.name.as_str());
172
173            visit_pou(self, pou, &context);
174            visit_property(self, &context);
175        }
176
177        // Validate user declared types
178        for t in &unit.user_types {
179            visit_user_type_declaration(self, t, &context);
180        }
181
182        // Validate config variables (VAR_CONFIG)
183        for variable in &unit.var_config {
184            visit_config_variable(self, variable, &context);
185        }
186
187        // Validate global variables
188        for gv in &unit.global_vars {
189            visit_variable_block(self, None, gv, &context);
190        }
191
192        // Validate implementations
193        for implementation in &unit.implementations {
194            visit_implementation(self, implementation, &context);
195        }
196
197        for interface in &unit.interfaces {
198            visit_interface(self, interface, &context);
199        }
200    }

202    pub fn validate_configured_templates(&mut self, index: &Index) {
203        // get config variables as string, along with their locations
204        let config_variables =
205            index.get_config_variables().iter().fold(FxHashMap::default(), |mut acc, var| {
206                match ExpressionPath::try_from(var) {
207                    Ok(p) => p.expand(index).into_iter().for_each(|p| {
208                        acc.entry(p)
209                            .and_modify(|entry: &mut Vec<SourceLocation>| entry.push(var.location.clone()))
210                            .or_insert(vec![var.location.clone()]);
211                    }),
212                    Err(e) => {
213                        self.diagnostics.extend(e);
214                    }
215                }
216                acc
217            });
218        // check for ambiguously configured template-variables
219        config_variables.values().for_each(|loc| {
220            if loc.len() > 1 {
221                self.diagnostics.push(
222                    Diagnostic::new("Template variable configured multiple times")
223                        .with_error_code("E108")
224                        .with_location(&loc[0])
225                        .with_secondary_locations(loc.iter().skip(1).cloned().collect()),
226                )
227            }
228        });
229
230        // collect all template-instances
231        let instances = index
232            .filter_instances(|entry, _| !entry.is_constant())
233            .filter(|(_, entry)| {
234                entry.get_hardware_binding().is_some_and(|opt| opt.access == DirectAccessType::Template)
235            })
236            .fold(vec![], |mut acc, (path, idxentry)| {
237                let path = path.expand(index);
238                let is_array = path.len() > 1;
239                path.into_iter().for_each(|p| {
240                    acc.push((p, (idxentry, is_array)));
241                });
242                acc
243            });
244
245        let mut incomplete_array_configurations = FxIndexSet::default();
246        // validate if all template-instances are configured in VAR_CONFIG blocks
247        for (segments, (val, is_array)) in instances {
248            if !config_variables.contains_key(&segments) {
249                if is_array {
250                    incomplete_array_configurations.insert(&val.source_location);
251                } else {
252                    self.diagnostics.push(
253                        Diagnostic::new("Template-variable must have a configuration")
254                            .with_error_code("E107")
255                            .with_location(&val.source_location),
256                    );
257                }
258            }
259        }
260
261        // report arrays which have elements that are missing configuration
262        incomplete_array_configurations.iter().for_each(|array_location| {
263            self.diagnostics.push(
264                Diagnostic::new("One or more template-elements in array have not been configured")
265                    .with_location(*array_location)
266                    .with_error_code("E107"),
267            );
268        });
269    }
