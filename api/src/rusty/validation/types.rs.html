15pub fn visit_data_type_declaration<T: AnnotationMap>(
16    validator: &mut Validator,
17    declaration: &DataTypeDeclaration,
18    context: &ValidationContext<T>,
19) {
20    if declaration.get_location().is_internal() {
21        return;
22    }
23
24    match declaration {
25        DataTypeDeclaration::Reference { referenced_type, location } => {
26            if context.index.find_effective_type_by_name(referenced_type).is_none() {
27                validator.push_diagnostic(Diagnostic::unknown_type(referenced_type, location));
28            };
29        }
30        DataTypeDeclaration::Definition { data_type, location, .. } => {
31            visit_data_type(validator, data_type, location, context)
32        }
33        DataTypeDeclaration::Aggregate { .. } => {}
34    }
35}

37pub fn visit_data_type<T: AnnotationMap>(
38    validator: &mut Validator,
39    data_type: &DataType,
40    location: &SourceLocation,
41    context: &ValidationContext<T>,
42) {
43    validate_data_type(validator, data_type, location);
44
45    let context = &context.with_optional_qualifier(data_type.get_name());
46    match data_type {
47        DataType::StructType { variables, .. } => {
48            variables.iter().for_each(|v| visit_variable(validator, v, context))
49        }
50        DataType::ArrayType { referenced_type, .. } => {
51            visit_data_type_declaration(validator, referenced_type, context)
52        }
53        DataType::VarArgs { referenced_type: Some(referenced_type), .. } => {
54            visit_data_type_declaration(validator, referenced_type.as_ref(), context);
55        }
56        DataType::PointerType { referenced_type, .. } => {
57            visit_data_type_declaration(validator, referenced_type.as_ref(), context);
58        }
59        DataType::EnumType { numeric_type, .. } => {
60            if let Some(resolved_type) = context.index.find_effective_type_by_name(numeric_type) {
61                let type_info = resolved_type.get_type_information();
62                if !type_info.is_int() || type_info.is_date_or_time_type() {
63                    validator.push_diagnostic(
64                        Diagnostic::new(format!(
65                            "Invalid type '{}' for enum. Only integer types are allowed",
66                            numeric_type
67                        ))
68                        .with_error_code("E122")
69                        .with_location(location),
70                    );
71                }
72            } else {
73                validator.push_diagnostic(Diagnostic::unknown_type(numeric_type, location));
74            }
75        }
76        _ => {}
77    }
78}

80fn validate_data_type(validator: &mut Validator, data_type: &DataType, location: &SourceLocation) {
81    if location.is_internal() {
82        return;
83    }
84
85    match data_type {
86        DataType::StructType { variables, .. } if variables.is_empty() => {
87            validator.push_diagnostic(
88                Diagnostic::new("Variable block is empty").with_error_code("E028").with_location(location),
89            );
90        }
91        DataType::EnumType {
92            elements: AstNode { stmt: AstStatement::ExpressionList(expressions), .. },
93            ..
94        } if expressions.is_empty() => {
95            validator.push_diagnostic(
96                Diagnostic::new("Variable block is empty").with_error_code("E028").with_location(location),
97            );
98        }
99        DataType::VarArgs { referenced_type: None, sized: true } => validator.push_diagnostic(
100            Diagnostic::new("Missing datatype: Sized Variadics require a known datatype.")
101                .with_error_code("E038")
102                .with_location(location),
103        ),
104        _ => {}
105    }
106}

108pub fn visit_user_type_declaration<T: AnnotationMap>(
109    validator: &mut Validator,
110    user_type: &UserTypeDeclaration,
111    context: &ValidationContext<T>,
112) {
113    visit_data_type(validator, &user_type.data_type, &user_type.location, context);
114    validate_array_assignment(validator, context, user_type);
115}

117pub fn data_type_is_fb_or_class_instance(type_name: &str, index: &Index) -> bool {
118    let data_type_info = index.find_effective_type_by_name(type_name).map_or_else(
119        || index.get_void_type().get_type_information(),
120        crate::typesystem::DataType::get_type_information,
121    );
122
123    if let DataTypeInformation::Struct {
124        source: StructSource::Pou(PouType::FunctionBlock) | StructSource::Pou(PouType::Class),
125        ..
126    } = data_type_info
127    {
128        return true;
129    }
130
131    match data_type_info {
132        DataTypeInformation::Struct { members, .. } =>
133        //see if any member is fb or class intance
134        {
135            members.iter().any(|member| data_type_is_fb_or_class_instance(member.get_type_name(), index))
136        }
137        DataTypeInformation::Array { inner_type_name, .. } => {
138            data_type_is_fb_or_class_instance(inner_type_name.as_str(), index)
139        }
140        DataTypeInformation::Pointer { inner_type_name, .. } => {
141            data_type_is_fb_or_class_instance(inner_type_name.as_str(), index)
142        }
143        DataTypeInformation::Alias { referenced_type, .. } => {
144            data_type_is_fb_or_class_instance(referenced_type.as_str(), index)
145        }
146        _ => false,
147    }
