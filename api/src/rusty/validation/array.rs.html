1//! This module is responsible for validating array assignments both in their syntax and semantics.
2//!
3//! Specifically this module checks if the array assignments start with a leading `[` symbol and the fed
4//! elements are less-equal to the arrays size. As an example, `foo : ARRAY[1..2] OF DINT := (1, 2, 3)`
5//! violates both the syntax and semantic of array assignments.
6//!
7//! Design note: Because we distinguish between variables inside VAR blocks [`plc_ast::ast::Variable`]
8//! and POU bodies [`plc_ast::ast::AstStatementKind`] and how we interact with them (e.g. infering types of
9//! [`plc_ast::ast::Variable`] from the AstAnnotation being impossible right now) a wrapper enum was
10//! introduced to make the validation code as generic as possible.
11

23/// Indicates whether an array was assigned in a VAR block or a POU body
24#[derive(Debug, Clone, Copy)]
25pub(super) enum StatementWrapper<'a> {
26    Statement(&'a AstNode),
27    Variable(&'a Variable),
28}
29
30impl<'a> From<&'a AstNode> for StatementWrapper<'a> {
31    fn from(value: &'a AstNode) -> Self {
32        Self::Statement(value)
33    }
34}
35
36impl<'a> From<&'a Variable> for StatementWrapper<'a> {
37    fn from(value: &'a Variable) -> Self {
38        Self::Variable(value)
39    }
40}

42pub(super) fn validate_array_assignment<'a, T, S>(
43    validator: &mut Validator,
44    context: &ValidationContext<T>,
45    statement: S,
46) where
47    T: AnnotationMap,
48    S: Into<StatementWrapper<'a>> + Copy,
49{
50    let statement = statement.into();
51
52    let Some(rhs_stmt) = statement.rhs_statement() else { return };
53    let Some(lhs_info) = statement.lhs_info(context) else { return };
54
55    if !lhs_info.is_array() {
56        return;
57    }
58
59    validate_array(validator, context, &statement, lhs_info, rhs_stmt);
60    validate_array_of_structs(validator, context, lhs_info, rhs_stmt);
61}
62
63fn validate_array<T: AnnotationMap>(
64    validator: &mut Validator,
65    context: &ValidationContext<T>,
66    statement: &StatementWrapper,
67    lhs_type: &DataTypeInformation,
68    rhs_stmt: &AstNode,
69) {
70    let stmt_rhs = peel(rhs_stmt);
71    if !(stmt_rhs.is_literal_array() || stmt_rhs.is_reference() || stmt_rhs.is_call()) {
72        validator.push_diagnostic(
73            Diagnostic::new("Array assignments must be surrounded with `[]`")
74                .with_error_code("E043")
75                .with_location(stmt_rhs),
76        );
77        return; // Return here, because array size validation is error-prone with incorrect assignments
78    }
79
80    let len_lhs = lhs_type.get_array_length(context.index).unwrap_or(0);
81    let len_rhs = statement_to_array_length(context, stmt_rhs);
82
83    if len_lhs == 0 {
84        return;
85    }
86
87    if len_lhs < len_rhs {
88        let name = statement.lhs_name(validator.context);
89        let location = stmt_rhs.get_location();
90        validator.push_diagnostic(
91            Diagnostic::new(format!(
92                "Array `{name}` has a size of {len_lhs}, but {len_rhs} elements were provided"
93            ))
94            .with_error_code("E043")
95            .with_location(location),
96        );
97    }
98}

100fn validate_array_of_structs<T: AnnotationMap>(
101    validator: &mut Validator,
102    context: &ValidationContext<T>,
103    lhs_type: &DataTypeInformation,
104    rhs_stmt: &AstNode,
105) {
106    let Some(array_type_name) = lhs_type.get_inner_array_type_name() else { return };
107    let Some(dti) = context.index.find_effective_type_by_name(array_type_name) else { return };
108    if !dti.is_struct() {
109        return;
110    }
111
112    let AstStatement::Literal(AstLiteral::Array(array)) = rhs_stmt.get_stmt() else { return };
113    let Some(elements) = array.elements().map(AstNode::get_stmt) else { return };
114
115    match elements {
116        AstStatement::ExpressionList(expressions) => {
117            for invalid in expressions.iter().filter(|it| !it.is_paren()) {
118                validator.push_diagnostic(
119                    Diagnostic::new("Struct initializers within arrays have to be wrapped by `()`")
120                        .with_error_code("E043")
121                        .with_location(invalid),
122                );
123            }
124        }
125
126        // arr := [foo := 0]
127        AstStatement::Assignment(..) => {
128            validator.push_diagnostic(
129                Diagnostic::new("Struct initializers within arrays have to be wrapped by `()`")
130                    .with_error_code("E043")
131                    .with_location(rhs_stmt),
132            );
133        }
134
135        _ => (),
136    }
137}

139/// Takes an [`AstStatementKind`] and returns its length as if it was an array. For example calling this function
140/// on an expression-list such as `[(...), (...)]` would return 2.
141fn statement_to_array_length<T: AnnotationMap>(context: &ValidationContext<T>, statement: &AstNode) -> usize {
142    match statement.get_stmt() {
143        AstStatement::Literal(AstLiteral::Array(arr)) => match arr.elements() {
144            Some(AstNode { stmt: AstStatement::ExpressionList(expressions), .. }) => {
145                expressions.iter().map(|it| statement_to_array_length(context, it)).sum::<usize>()
146            }
147
148            Some(any) => statement_to_array_length(context, any),
149            None => 0,
150        },
151
152        AstStatement::CallStatement(_) => context
153            .annotations
154            .get_type(statement, context.index)
155            .and_then(|it| it.information.get_array_length(context.index))
156            .unwrap_or(0),
157
158        AstStatement::MultipliedStatement(data) => data.multiplier as usize,
159        AstStatement::ExpressionList { .. } | AstStatement::ParenExpression(_) => 1,
160
161        // Any literal other than an array can be counted as 1
162        AstStatement::Literal { .. } => 1,
163
164        _any => {
165            // XXX: Not sure what else could be in here
166            log::debug!("Array size-counting for {statement:?} not covered; validation _might_ be wrong");
167            0
168        }
169    }
170}

172impl<'a> StatementWrapper<'a> {
173    fn lhs_name(&self, context: &GlobalContext) -> String {
174        match self {
175            StatementWrapper::Variable(variable) => variable.name.clone(),
176            StatementWrapper::Statement(statement) => {
177                let AstStatement::Assignment(data) = &statement.stmt else { return "".to_string() };
178                context.slice(&data.left.location)
179            }
180        }
181    }
182
183    fn rhs_statement(&self) -> Option<&'a AstNode> {
184        match self {
185            StatementWrapper::Variable(variable) => variable.initializer.as_ref(),
186            StatementWrapper::Statement(statement) => {
187                let AstStatement::Assignment(data) = &statement.stmt else { return None };
188                Some(&data.right)
189            }
190        }
191    }
192
193    fn lhs_info<T>(&self, context: &'a ValidationContext<T>) -> Option<&'a DataTypeInformation>
194    where
195        T: AnnotationMap,
196    {
197        let ty = match self {
198            StatementWrapper::Statement(statement) => {
199                let AstNode { stmt: AstStatement::Assignment(data), .. } = statement else { return None };
200                context.annotations.get_type(&data.left, context.index).map(|it| it.get_type_information())
201            }
202
203            StatementWrapper::Variable(variable) => variable
204                .data_type_declaration
205                .get_referenced_type()
206                .and_then(|it| context.index.find_effective_type_info(&it)),
207        }?;
208
209        match ty {
210            DataTypeInformation::Pointer { .. } => Some(context.index.find_elementary_pointer_type(ty)),
211            _ => Some(ty),
212        }
213    }
214}
215
216fn peel(node: &AstNode) -> &AstNode {
217    match &node.stmt {
218        AstStatement::ParenExpression(expr) => peel(expr),
219        _ => node,
220    }
