14/// Validator that does not check a dedicated file but rather
15/// uses the index to validate the project as a whole.
16/// It performs validations including:
17///  - naming-conflicts
18///  - <tbc>
19#[derive(Default, Validators)]
20pub struct GlobalValidator {
21    diagnostics: Vec<Diagnostic>,
22}
23
24impl GlobalValidator {
25    pub fn new() -> GlobalValidator {
26        GlobalValidator { diagnostics: Vec::new() }
27    }

29    /// reports a name-conflict for the given name. the locations indicate the
30    /// locations of the declared symbols that make up the conflict. this method will
31    /// create a diagnostic per location where it attaches the other locations as additional information.
32    fn report_name_conflict(
33        &mut self,
34        name: &str,
35        locations: &[&SourceLocation],
36        additional_text: Option<&str>,
37    ) {
38        for v in locations.iter() {
39            let others = locations.iter().filter(|it| *it != v).map(|it| (*it).clone()).collect::<Vec<_>>();
40
41            // If the SourceRange of `v` is undefined, we can assume the user choose a name which clashes
42            // with an (internal) built-in datatype, hence the undefined location.
43            if v.is_builtin_internal() || v.is_undefined() {
44                for other in others {
45                    self.diagnostics.push(
46                        Diagnostic::new(format!(
47                            "{name} can not be used as a name because it is a built-in datatype"
48                        ))
49                        .with_location(other)
50                        .with_error_code("E004"),
51                    );
52                }
53            } else {
54                let additional_text = additional_text.unwrap_or("Duplicate symbol.");
55                self.push_diagnostic(
56                    Diagnostic::new(format!("{name}: {additional_text}"))
57                        .with_error_code("E004")
58                        .with_location(*v)
59                        .with_secondary_locations(others),
60                );
61            }
62        }
63    }

65    /// checks all symbols of the given index for naming conflicts.
66    /// all problems will be reported to self.diagnostics
67    pub fn validate(&mut self, index: &Index) {
68        // everything callable (funks, global FB-instances, programs)
69        self.validate_unique_callables(index);
70
71        // everything that can be a type (DTs, FBs)
72        self.validate_unique_datatypes(index);
73
74        // globals + PRGs
75        self.validate_unique_variables(index);
76
77        // all POUs
78        self.validate_unique_pous(index);
79
80        // all interfaces
81        self.validate_unique_interfaces(index);
82    }

84    /// validates following uniqueness-clusters:
85    /// - globals + programs
86    /// - member-variables
87    /// - enums
88    fn validate_unique_variables(&mut self, index: &Index) {
89        let globals = index.get_globals().values().map(|g| (g.get_name(), &g.source_location));
90        let prgs = index
91            .get_pous()
92            .values()
93            .filter(|pou| matches!(pou, PouIndexEntry::Program { .. }))
94            .map(|p| (p.get_name(), p.get_location()));
95
96        self.check_uniqueness_of_cluster(globals.chain(prgs), Some("Ambiguous global variable."));
97
98        // Report name conflicts between any member variables in the VAR block
99        for ty in index.get_types().values().chain(index.get_pou_types().values()) {
100            let mut groups: FxHashMap<&str, Vec<(&str, &SourceLocation)>> = FxHashMap::default();
101            let properties = index.get_properties_in_pou(ty.get_name());
102            let properties = properties.iter().map(|it| (it.name.as_str(), &it.location));
103
104            for (variable_name, location) in
105                ty.get_members().iter().map(|it| (it.get_name(), &it.source_location)).chain(properties)
106            {
107                let group = groups.entry(variable_name).or_default();
108                group.push((variable_name, location));
109            }
110
111            for duplicates in groups.values().filter(|it| it.len() > 1) {
112                let locations = duplicates.iter().map(|(_, location)| *location).collect::<Vec<_>>();
113                self.report_name_conflict(duplicates[0].0, &locations, None);
114            }
115        }
116
117        // Report name conflicts between enum variants and any other member variable in the VAR block
118        for pou in index.get_pou_types().values() {
119            let mut groups: FxHashMap<&str, Vec<&VariableIndexEntry>> = FxHashMap::default();
120            let variants = index.get_enum_variants_in_pou(pou.get_name());
121
122            for variant in variants {
123                let group = groups.entry(variant.get_name()).or_default();
124                group.push(variant);
125            }
126
127            for member in helper::get_non_enum_pou_members(index, pou.get_name()) {
128                if let Some(variant) = groups.get(member.get_name()) {
129                    let locations = vec![&member.source_location, &variant[0].source_location];
130                    self.report_name_conflict(member.get_name(), &locations, None);
131                }
132            }
133        }
134    }

136    ///validates uniqueness of datatypes (types + functionblocks + classes)
137    fn validate_unique_datatypes(&mut self, index: &Index) {
138        let all_declared_types = index.get_types().values().filter_map(|dt| {
139            let name = dt.get_name();
140            if !index.get_type_information_or_void(name).is_generic(index) {
141                Some((name, &dt.location))
142            } else {
143                None
144            }
145        });
146
147        let all_function_blocks = index
148            .get_pous()
149            .values()
150            .filter(|p| p.is_function_block() || p.is_class())
151            .map(|p| (p.get_name(), p.get_location()));
152        let vec = all_declared_types.collect::<Vec<_>>();
153        let all_declared_types = vec.into_iter();
154        self.check_uniqueness_of_cluster(
155            all_declared_types.chain(all_function_blocks),
156            Some("Ambiguous datatype."),
157        );
158    }

160    /// validates the uniqueness of everything callable (global fb-instances + programs + functions )
161    fn validate_unique_callables(&mut self, index: &Index) {
162        let all_fb_instances = index
163            .get_globals()
164            .values()
165            .filter(|g| {
166                index
167                    .find_effective_type_by_name(g.get_type_name())
168                    .map(|it| {
169                        matches!(
170                            it.information,
171                            DataTypeInformation::Struct {
172                                source: StructSource::Pou(PouType::FunctionBlock),
173                                ..
174                            }
175                        )
176                    })
177                    .unwrap_or(false)
178            })
179            .map(|it| (it.get_name(), &it.source_location));
180        let all_prgs = index
181            .get_pous()
182            .values()
183            .filter(|p| {
184                matches!(
185                    p,
186                    PouIndexEntry::Program { .. }
187                        | PouIndexEntry::Action { .. }
188                        // Properties are already covered somewhere else
189                        | PouIndexEntry::Method { property: None, .. }
190                )
191            })
192            .map(|it| (it.get_name(), it.get_location()));
193
194        let all_funcs = index
195            .get_pous()
196            .values()
197            .filter(|p| p.is_function() && !p.is_generic())
198            .map(|it| (it.get_name(), it.get_location()));
199
200        self.check_uniqueness_of_cluster(
201            all_fb_instances.chain(all_prgs).chain(all_funcs),
202            Some("Ambiguous callable symbol."),
203        );
204    }

206    ///validate the uniqueness of POUs (programs, functions, function_blocks, classes)
207    fn validate_unique_pous(&mut self, index: &Index) {
208        //inner filter
209        fn only_toplevel_pous(pou: &&PouIndexEntry) -> bool {
210            !pou.is_action() && !pou.is_method() && !pou.is_generic()
211        }
212
213        let pou_clusters = index
214            .get_pous()
215            .entries()
216            .filter(|(_, entries_per_name)| entries_per_name.iter().filter(only_toplevel_pous).count() > 1)
217            .map(|(name, pous)| {
218                (
219                    name.as_str(),
220                    pous.iter()
221                        .filter(|p| only_toplevel_pous(p) && !p.is_generic())
222                        .map(|p| p.get_location()),
223                )
224            });
225
226        for (name, cluster) in pou_clusters {
227            self.report_name_conflict(name, &cluster.collect::<Vec<_>>(), None);
228        }
229    }
230
231    fn validate_unique_interfaces(&mut self, index: &Index) {
232        let interfaces = index
233            .get_interfaces()
234            .values()
235            .map(|interface| (interface.get_name(), interface.get_name_location()));
236
237        self.check_uniqueness_of_cluster(interfaces, Some("Ambiguous interface"));
238    }
239
240    fn check_uniqueness_of_cluster<'a, T>(&mut self, cluster: T, additional_text: Option<&str>)
241    where
242        T: Iterator<Item = (&'a str, &'a SourceLocation)>,
243    {
244        let mut cluster_map: SymbolMap<&str, &SourceLocation> = SymbolMap::default();
245        for (name, loc) in cluster {
246            cluster_map.insert(name, loc);
247        }
248        for (name, locations) in cluster_map.entries().filter(|(_, v)| v.len() > 1) {
249            self.report_name_conflict(name, locations, additional_text);
250        }
251    }
