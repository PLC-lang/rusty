13/// Validator that does not check a dedicated file but rather
14/// uses the index to validate the project as a whole.
15/// It performs validations including:
16///  - naming-conflicts
17///  - <tbc>
18#[derive(Default, Validators)]
19pub struct GlobalValidator {
20    diagnostics: Vec<Diagnostic>,
21}
22
23impl GlobalValidator {
24    pub fn new() -> GlobalValidator {
25        GlobalValidator { diagnostics: Vec::new() }
26    }

28    /// reports a name-conflict for the given name. the locations indicate the
29    /// locations of the declared symbols that make up the conflict. this method will
30    /// create a diagnostic per location where it attaches the other locations as additional information.
31    fn report_name_conflict(
32        &mut self,
33        name: &str,
34        locations: &[&SourceLocation],
35        additional_text: Option<&str>,
36    ) {
37        for v in locations.iter() {
38            let others = locations.iter().filter(|it| *it != v).map(|it| (*it).clone()).collect::<Vec<_>>();
39
40            // If the SourceRange of `v` is undefined, we can assume the user choose a name which clashes
41            // with an (internal) built-in datatype, hence the undefined location.
42            if v.is_builtin_internal() || v.is_undefined() {
43                for other in others {
44                    self.diagnostics.push(
45                        Diagnostic::new(format!(
46                            "{name} can not be used as a name because it is a built-in datatype"
47                        ))
48                        .with_location(other)
49                        .with_error_code("E004"),
50                    );
51                }
52            } else {
53                let additional_text = additional_text.unwrap_or("Duplicate symbol.");
54                self.push_diagnostic(
55                    Diagnostic::new(format!("{name}: {additional_text}"))
56                        .with_error_code("E004")
57                        .with_location(*v)
58                        .with_secondary_locations(others),
59                );
60            }
61        }
62    }

64    /// checks all symbols of the given index for naming conflicts.
65    /// all problems will be reported to self.diagnostics
66    pub fn validate(&mut self, index: &Index) {
67        // everything callable (funks, global FB-instances, programs)
68        self.validate_unique_callables(index);
69
70        // everything that can be a type (DTs, FBs)
71        self.validate_unique_datatypes(index);
72
73        // globals + PRGs
74        self.validate_unique_variables(index);
75
76        // all POUs
77        self.validate_unique_pous(index);
78
79        // all interfaces
80        self.validate_unique_interfaces(index);
81    }

83    /// validates following uniqueness-clusters:
84    /// - globals + programs
85    /// - member-variables
86    /// - enums
87    fn validate_unique_variables(&mut self, index: &Index) {
88        let globals = index.get_globals().values().map(|g| (g.get_name(), &g.source_location));
89        let prgs = index
90            .get_pous()
91            .values()
92            .filter(|pou| matches!(pou, PouIndexEntry::Program { .. }))
93            .map(|p| (p.get_name(), p.get_location()));
94
95        self.check_uniqueness_of_cluster(globals.chain(prgs), Some("Ambiguous global variable."));
96
97        // Report name conflicts between any member variables in the VAR block
98        for ty in index.get_types().values().chain(index.get_pou_types().values()) {
99            let mut groups: FxHashMap<&str, Vec<(&str, &SourceLocation)>> = FxHashMap::default();
100            let properties = index.get_properties_in_pou(ty.get_name());
101            let properties = properties.iter().map(|it| (it.name.as_str(), &it.location));
102
103            for (variable_name, location) in
104                ty.get_members().iter().map(|it| (it.get_name(), &it.source_location)).chain(properties)
105            {
106                let group = groups.entry(variable_name).or_default();
107                group.push((variable_name, location));
108            }
109
110            for duplicates in groups.values().filter(|it| it.len() > 1) {
111                let locations = duplicates.iter().map(|(_, location)| *location).collect::<Vec<_>>();
112                self.report_name_conflict(duplicates[0].0, &locations, None);
113            }
114        }
115    }

117    ///validates uniqueness of datatypes (types + functionblocks + classes)
118    fn validate_unique_datatypes(&mut self, index: &Index) {
119        let all_declared_types = index.get_types().values().filter_map(|dt| {
120            let name = dt.get_name();
121            if !index.get_type_information_or_void(name).is_generic(index) {
122                Some((name, &dt.location))
123            } else {
124                None
125            }
126        });
127
128        let all_function_blocks = index
129            .get_pous()
130            .values()
131            .filter(|p| p.is_function_block() || p.is_class())
132            .map(|p| (p.get_name(), p.get_location()));
133        let vec = all_declared_types.collect::<Vec<_>>();
134        let all_declared_types = vec.into_iter();
135        self.check_uniqueness_of_cluster(
136            all_declared_types.chain(all_function_blocks),
137            Some("Ambiguous datatype."),
138        );
139    }

141    /// validates the uniqueness of everything callable (global fb-instances + programs + functions )
142    fn validate_unique_callables(&mut self, index: &Index) {
143        let all_fb_instances = index
144            .get_globals()
145            .values()
146            .filter(|g| {
147                index
148                    .find_effective_type_by_name(g.get_type_name())
149                    .map(|it| {
150                        matches!(
151                            it.information,
152                            DataTypeInformation::Struct {
153                                source: StructSource::Pou(PouType::FunctionBlock),
154                                ..
155                            }
156                        )
157                    })
158                    .unwrap_or(false)
159            })
160            .map(|it| (it.get_name(), &it.source_location));
161        let all_prgs = index
162            .get_pous()
163            .values()
164            .filter(|p| {
165                matches!(
166                    p,
167                    PouIndexEntry::Program { .. }
168                        | PouIndexEntry::Action { .. }
169                        // Properties are already covered somewhere else
170                        | PouIndexEntry::Method { property: None, .. }
171                )
172            })
173            .map(|it| (it.get_name(), it.get_location()));
174
175        let all_funcs = index
176            .get_pous()
177            .values()
178            .filter(|p| p.is_function() && !p.is_generic())
179            .map(|it| (it.get_name(), it.get_location()));
180
181        self.check_uniqueness_of_cluster(
182            all_fb_instances.chain(all_prgs).chain(all_funcs),
183            Some("Ambiguous callable symbol."),
184        );
185    }

187    ///validate the uniqueness of POUs (programs, functions, function_blocks, classes)
188    fn validate_unique_pous(&mut self, index: &Index) {
189        //inner filter
190        fn only_toplevel_pous(pou: &&PouIndexEntry) -> bool {
191            !pou.is_action() && !pou.is_method() && !pou.is_generic()
192        }
193
194        let pou_clusters = index
195            .get_pous()
196            .entries()
197            .filter(|(_, entries_per_name)| entries_per_name.iter().filter(only_toplevel_pous).count() > 1)
198            .map(|(name, pous)| {
199                (
200                    name.as_str(),
201                    pous.iter()
202                        .filter(|p| only_toplevel_pous(p) && !p.is_generic())
203                        .map(|p| p.get_location()),
204                )
205            });
206
207        for (name, cluster) in pou_clusters {
208            self.report_name_conflict(name, &cluster.collect::<Vec<_>>(), None);
209        }
210    }
211
212    fn validate_unique_interfaces(&mut self, index: &Index) {
213        let interfaces = index
214            .get_interfaces()
215            .values()
216            .map(|interface| (interface.get_name(), interface.get_name_location()));
217
218        self.check_uniqueness_of_cluster(interfaces, Some("Ambiguous interface"));
219    }
220
221    fn check_uniqueness_of_cluster<'a, T>(&mut self, cluster: T, additional_text: Option<&str>)
222    where
223        T: Iterator<Item = (&'a str, &'a SourceLocation)>,
224    {
225        let mut cluster_map: SymbolMap<&str, &SourceLocation> = SymbolMap::default();
226        for (name, loc) in cluster {
227            cluster_map.insert(name, loc);
228        }
229        for (name, locations) in cluster_map.entries().filter(|(_, v)| v.len() > 1) {
230            self.report_name_conflict(name, locations, additional_text);
231        }
232    }
