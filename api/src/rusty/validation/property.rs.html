1//! Module validating properties
2//!
3//! Properties are lowered into methods, so most validation logic is already handled by existing code.
4//! However, some of these validations produce generic error messages. This module provides more specific
5//! validations to improve the error reporting experience for users with regards to properties.
6

23pub fn visit_property<T>(validator: &mut Validator, context: &ValidationContext<T>)
24where
25    T: AnnotationMap,
26{
27    let Some(pou) = context.index.find_pou(context.qualifier.unwrap_or_default()) else {
28        return;
29    };
30
31    validate_definition(validator, context, pou);
32    validate_name_clashes(validator, context, pou);
33    validate_overridden_signatures(validator, context, pou);
34}

36fn property_accessor_datatype<'idx>(
37    index: &'idx Index,
38    property: &PropertyBlock,
39    kind: PropertyKind,
40) -> Option<&'idx DataType> {
41    property.implementations.iter().find(|implementation| implementation.kind == kind).map(|implementation| {
42        let name = implementation.datatype.get_name().unwrap_or_default();
43        index.get_effective_type_or_void_by_name(name)
44    })
45}
46
47fn property_datatype<'idx>(index: &'idx Index, property: &PropertyBlock) -> Option<&'idx DataType> {
48    property.implementations.first().map(|implementation| {
49        let name = implementation.datatype.get_name().unwrap_or_default();
50        index.get_effective_type_or_void_by_name(name)
51    })
52}
53
54fn property_types_match(index: &Index, left: Option<&DataType>, right: Option<&DataType>) -> bool {
55    matches!((left, right), (Some(left), Some(right)) if types::are_equal_types(index, left, right))
56}
57
58fn local_property_types_conflict(index: &Index, property: &PropertyBlock) -> bool {
59    let get = property_accessor_datatype(index, property, PropertyKind::Get);
60    let set = property_accessor_datatype(index, property, PropertyKind::Set);
61
62    matches!((get, set), (Some(get), Some(set)) if !types::are_equal_types(index, get, set))
63}

65fn validate_property_definition(validator: &mut Validator, index: &Index, property: &PropertyBlock) {
66    let mut count_get = 0;
67    let mut count_set = 0;
68
69    for implementation in &property.implementations {
70        for variable in &implementation.variable_blocks {
71            if variable.location.is_internal() {
72                continue;
73            }
74
75            if !matches!(variable.kind, VariableBlockType::Local | VariableBlockType::Temp) {
76                validator.push_diagnostic(
77                    Diagnostic::new("Properties only allow variable blocks of type VAR")
78                        .with_location(&property.ident.location)
79                        .with_secondary_location(&variable.location)
80                        .with_error_code("E116"),
81                );
82            }
83        }
84
85        match implementation.kind {
86            PropertyKind::Get => count_get += 1,
87            PropertyKind::Set => count_set += 1,
88        }
89    }
90
91    if local_property_types_conflict(index, property) {
92        let getter = property
93            .implementations
94            .iter()
95            .find(|implementation| implementation.kind == PropertyKind::Get)
96            .expect("getter must exist when local property types conflict");
97        let setter = property
98            .implementations
99            .iter()
100            .find(|implementation| implementation.kind == PropertyKind::Set)
101            .expect("setter must exist when local property types conflict");
102
103        validator.push_diagnostic(
104            Diagnostic::new(format!(
105                "Property `{}` has conflicting datatypes across PROPERTY_GET / PROPERTY_SET",
106                property.ident.name
107            ))
108            .with_location(&property.ident.location)
109            .with_secondary_locations(vec![getter.datatype.get_location(), setter.datatype.get_location()])
110            .with_error_code("E112"),
111        );
112    }
113
114    if count_get > 1 {
115        validator.push_diagnostic(
116            Diagnostic::new("Property has multiple PROPERTY_GET blocks")
117                .with_location(&property.ident.location)
118                .with_error_code("E004"),
119        );
120    }
121
122    if count_set > 1 {
123        validator.push_diagnostic(
124            Diagnostic::new("Property has multiple PROPERTY_SET blocks")
125                .with_location(&property.ident.location)
126                .with_error_code("E004"),
127        );
128    }
129}

131fn validate_definition<T>(validator: &mut Validator, context: &ValidationContext<T>, pou: &PouIndexEntry)
132where
133    T: AnnotationMap,
134{
135    for property in pou.get_properties_vec() {
136        validate_property_definition(validator, context.index, property);
137    }
138}

140fn validate_name_clashes<T>(validator: &mut Validator, context: &ValidationContext<T>, pou: &PouIndexEntry)
141where
142    T: AnnotationMap,
143{
144    let mut seen = HashSet::new();
145    let mut super_class = pou.get_super_class();
146    while let Some(parent_str) = super_class {
147        if !seen.insert(parent_str) {
148            break;
149        }
150
151        let Some(pou_parent) = context.index.find_pou(parent_str) else {
152            break;
153        };
154
155        // Check if any property in the current POU clashes with a variable in the parent POU
156        for property in pou.get_properties_vec() {
157            if let Some(variable) = context.index.find_local_member(parent_str, &property.ident.name) {
158                validator.push_diagnostic(
159                    Diagnostic::new(format!(
160                        "Name conflict between property and variable `{}` defined in POU `{}` and `{}`",
161                        property.ident.name,
162                        pou_parent.get_name(),
163                        pou.get_name()
164                    ))
165                    .with_error_code("E021")
166                    .with_location(&property.ident.location)
167                    .with_secondary_location(&variable.source_location),
168                );
169            }
170        }
171
172        // Check if any variable in the current POU clashes with a property in the parent POU
173        for member in context.index.get_pou_members(context.qualifier.unwrap_or_default()) {
174            if let Some(property) = pou_parent.get_property(member.get_name()) {
175                validator.push_diagnostic(
176                    Diagnostic::new(format!(
177                        "Name conflict between property and variable `{}` defined in POU `{}` and `{}`",
178                        property.name,
179                        pou_parent.get_name(),
180                        pou.get_name(),
181                    ))
182                    .with_error_code("E021")
183                    .with_location(&property.location)
184                    .with_secondary_location(&member.source_location),
185                );
186            }
187        }
188
189        super_class = pou_parent.get_super_class();
190    }
191}

193/// Validates if a derived property is redefined with a conflicting signature
194fn validate_overridden_signatures<T>(
195    validator: &mut Validator,
196    context: &ValidationContext<T>,
197    pou: &PouIndexEntry,
198) where
199    T: AnnotationMap,
200{
201    if pou.get_properties().is_none() {
202        return;
203    }
204
205    let mut seen = HashSet::new();
206    let mut super_class = pou.get_super_class();
207    while let Some(parent_str) = super_class {
208        if !seen.insert(parent_str) {
209            break;
210        }
211
212        let Some(pou_parent) = context.index.find_pou(parent_str) else {
213            break;
214        };
215
216        // No conflicts if one of the two has no properties
217        let Some(properties_child) = pou.get_properties() else {
218            super_class = pou_parent.get_super_class();
219            continue;
220        };
221
222        let Some(properties_parent) = pou_parent.get_properties() else {
223            super_class = pou_parent.get_super_class();
224            continue;
225        };
226
227        for (name, property_child) in properties_child {
228            if let Some(property_parent) = properties_parent.get(name) {
229                let child_type = property_datatype(context.index, property_child);
230                let parent_type = property_datatype(context.index, property_parent);
231
232                if !property_types_match(context.index, child_type, parent_type) {
233                    validator.push_diagnostic(
234                        Diagnostic::new(format!(
235                            "Overridden property `{}` has different signatures in POU `{}` and `{}`",
236                            property_child.ident.name,
237                            pou.get_name(),
238                            pou_parent.get_name()
239                        ))
240                        .with_error_code("E112")
241                        .with_location(&property_child.ident.location)
242                        .with_secondary_location(&property_parent.ident.location),
243                    );
244                }
245            }
246        }
247
248        super_class = pou_parent.get_super_class();
249    }
250}

252pub(crate) fn validate_properties_in_interfaces<T>(
253    validator: &mut Validator,
254    context: &ValidationContext<'_, T>,
255    interface: &Interface,
256) where
257    T: AnnotationMap,
258{
259    let interface = context.index.find_interface(&interface.ident.name).expect("must exist");
260
261    for property in &interface.properties {
262        validate_property_definition(validator, context.index, property);
263    }
264
265    // Retrieve all properties an interface inherits directly or indirectly and map them into tuples of
266    // (<interface name>, <property defined in that interface>)
267    let derived_properties = interface
268        .get_interface_hierarchy(context.index)
269        .iter()
270        .map(|it| (&it.ident, &it.properties))
271        .flat_map(|(name, properties)| properties.iter().map(|property| (name.clone(), property)))
272        .collect_vec();
273
274    // Group all these properties by their name
275    let mut clusters: FxHashMap<String, Vec<(Identifier, &PropertyBlock)>> = FxHashMap::default();
276    for (intf_ident, property) in derived_properties {
277        clusters.entry(property.ident.name.clone()).or_default().push((intf_ident, property));
278    }
279
280    // Check if properties in these clusters have the same type, otherwise we can't implement them in e.g. a FB
281    for ((left_intf_ident, left_property), (right_intf_ident, right_property)) in
282        clusters.values().filter(|properties| properties.len() > 1).flatten().tuple_windows()
283    {
284        let left_type = property_datatype(context.index, left_property);
285        let right_type = property_datatype(context.index, right_property);
286
287        if !property_types_match(context.index, left_type, right_type) {
288            validator.push_diagnostic(
289                Diagnostic::new(format!(
290                    "Property `{}` defined in interface `{}` and `{}` have different datatypes",
291                    left_property.ident.name, left_intf_ident.name, right_intf_ident.name
292                ))
293                .with_error_code("E112")
294                .with_location(&interface.ident.location)
295                .with_secondary_locations(vec![
296                    left_property
297                        .implementations
298                        .first()
299                        .map(|it| it.datatype.get_location())
300                        .unwrap_or_else(|| left_property.ident.location.clone()),
301                    right_property
302                        .implementations
303                        .first()
304                        .map(|it| it.datatype.get_location())
305                        .unwrap_or_else(|| right_property.ident.location.clone()),
306                ]),
307            );
308        }
309    }
