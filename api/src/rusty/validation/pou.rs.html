15pub fn visit_pou<T: AnnotationMap>(validator: &mut Validator, pou: &Pou, context: &ValidationContext<'_, T>) {
16    if pou.linkage != LinkageType::External {
17        validate_pou(validator, pou);
18        validate_interface_impl(validator, context, pou);
19        validate_base_class(validator, context, pou);
20        validate_methods_overrides(validator, context, pou.id, &pou.name, &pou.name_location);
21        if let PouType::Method { .. } = pou.kind {
22            validate_method(validator, pou, context);
23        }
24
25        for block in &pou.variable_blocks {
26            visit_variable_block(validator, Some(pou), block, context);
27        }
28    }
29}

31fn validate_methods_overrides<T: AnnotationMap>(
32    validator: &mut Validator,
33    context: &ValidationContext<'_, T>,
34    ast_id: AstId,
35    container_name: &str,
36    primary_location: &SourceLocation,
37) {
38    let Some(StatementAnnotation::MethodDeclarations { declarations }) =
39        context.annotations.get_with_id(ast_id)
40    else {
41        return;
42    };
43    declarations.iter().for_each(|(_, decl)| {
44        // validate that abstract signatures all match
45        // Concrete to abstract methods are checked at a different stage
46        let methods = decl
47            .iter()
48            .filter(|it| it.is_abstract())
49            .flat_map(|it| context.index.find_pou(it.get_qualified_name()))
50            // We already have a specialized validation for properties and would like to avoid duplicates here
51            .filter(|it| !it.is_property());
52        // XXX(ghha) should this not be combinations instead of tuple_windows?
53        for (method1, method2) in methods.tuple_windows() {
54            let diagnostics = validate_method_signature(context.index, method1, method2, primary_location);
55            if !diagnostics.is_empty() {
56                validator.push_diagnostic(
57                    Diagnostic::new(format!(
58                        "{} in `{}` is declared with conflicting signatures in `{}` and `{}`",
59                        method1.get_method_name().expect("must be a method"),
60                        container_name,
61                        method1.get_parent_pou_name().unwrap(),
62                        method2.get_parent_pou_name().unwrap()
63                    ))
64                    .with_error_code("E111")
65                    .with_location(primary_location)
66                    .with_secondary_location(method1.get_location())
67                    .with_secondary_location(method2.get_location())
68                    .with_sub_diagnostics(diagnostics),
69                );
70                // We want to early return here otherwise we could spam the user with lots of (valid) but identical
71                // diagnostics reported earlier.
72                return;
73            }
74        }
75
76        // we only want to validate existing concrete implementations for POUs, not interfaces
77        if context.index.find_pou(container_name).is_some() {
78            // validate that each abstract method has at least one concrete implementation
79            let abstracts = decl
80                .iter()
81                .filter(|it| it.is_abstract())
82                .map(|it| (it.get_qualifier(), context.index.find_pou(it.get_qualified_name()).unwrap()));
83
84            // Expecting only one concrete implementation
85            let concrete = decl
86                .iter()
87                .filter(|it| it.is_concrete())
88                .map(|it| context.index.find_pou(it.get_qualified_name()).unwrap())
89                .next();
90            // Validate that each concrete method which has an abstract counterpart has the same signature
91            if let Some(method_impl) = concrete {
92                abstracts.for_each(|(_, method_ref)| {
93                    let diagnostics =
94                        validate_method_signature(context.index, method_ref, method_impl, primary_location);
95                    for diagnostic in diagnostics {
96                        validator.push_diagnostic(diagnostic);
97                    }
98                });
99            } else {
100                abstracts.for_each(|(name, intf)| {
101                    validator.push_diagnostic(
102                        Diagnostic::new(format!(
103                            "{} defined in interface `{}` is missing in POU `{}`",
104                            intf.get_method_name().expect("must be a method"),
105                            name,
106                            container_name
107                        ))
108                        .with_error_code("E112")
109                        .with_location(primary_location)
110                        .with_secondary_location(intf.get_location()),
111                    );
112                })
113            };
114        }
115    });
116}

118fn validate_method<T: AnnotationMap>(
119    validator: &mut Validator<'_>,
120    pou: &Pou,
121    context: &ValidationContext<'_, T>,
122) {
123    let Some(StatementAnnotation::Override { definitions }) = context.annotations.get_with_id(pou.id) else {
124        // No override
125        return;
126    };
127    let Some(method_impl) = context.index.find_pou(&pou.name) else {
128        // Method does not exist
129        return;
130    };
131
132    //Only validate methods that are not abstract, abstract methods are validated in validate_implemented_methods
133    let overridden_concrete_methods = definitions
134        .iter()
135        .filter(|it| it.is_concrete())
136        .flat_map(|it| context.index.find_pou(it.get_qualified_name()))
137        .collect::<Vec<_>>();
138    overridden_concrete_methods.iter().for_each(|method_ref| {
139        let diagnostics =
140            validate_method_signature(context.index, method_ref, method_impl, &pou.name_location);
141        for diagnostic in diagnostics {
142            validator.push_diagnostic(diagnostic);
143        }
144    })
145}

147fn validate_base_class<T: AnnotationMap>(
148    validator: &mut Validator<'_>,
149    context: &ValidationContext<'_, T>,
150    pou: &Pou,
151) {
152    if let Some(Identifier { name, location }) = &pou.super_class {
153        // Check if the interfaces are implemented on the correct POU types
154        if !matches!(pou.kind, PouType::FunctionBlock | PouType::Class) {
155            validator.push_diagnostic(
156                Diagnostic::new("Subclassing is only allowed in `CLASS` and `FUNCTION_BLOCK`")
157                    .with_error_code("E110")
158                    .with_location(&pou.name_location),
159            );
160        }
161
162        if context.index.find_pou(name).is_none() {
163            validator.push_diagnostic(
164                Diagnostic::new(format!("Base `{}` does not exist", name))
165                    .with_error_code("E048")
166                    .with_location(location),
167            );
168        }
169    };
170}

172fn validate_interface_impl<T>(validator: &mut Validator, ctxt: &ValidationContext<'_, T>, pou: &Pou)
173where
174    T: AnnotationMap,
175{
176    // No interfaces declared to implement
177    if pou.interfaces.is_empty() {
178        return;
179    }
180
181    // Check if the interfaces are implemented on the correct POU types
182    if !matches!(pou.kind, PouType::FunctionBlock | PouType::Class) {
183        let location = {
184            let location_first = pou.interfaces.first().unwrap();
185            let location_last = pou.interfaces.last().unwrap();
186
187            location_first.location.span(&location_last.location)
188        };
189
190        validator.push_diagnostic(
191            Diagnostic::new("Interfaces can only be implemented by `CLASS` or `FUNCTION_BLOCK`")
192                .with_error_code("E110")
193                .with_location(location),
194        );
195    }
196
197    // Check if the declared interfaces exist, i.e. the comma seperated identifiers after `[...] IMPLEMENTS`
198    for declaration in &pou.interfaces {
199        if ctxt.index.find_interface(&declaration.name).is_none() {
200            validator.push_diagnostic(
201                Diagnostic::new(format!("Interface `{}` does not exist", declaration.name))
202                    .with_error_code("E048")
203                    .with_location(&declaration.location),
204            );
205        }
206    }
207}

209pub fn visit_implementation<T: AnnotationMap>(
210    validator: &mut Validator,
211    implementation: &Implementation,
212    context: &ValidationContext<'_, T>,
213) {
214    if implementation.pou_type == PouType::Class && !implementation.statements.is_empty() {
215        validator.push_diagnostic(
216            Diagnostic::new("A class cannot have an implementation")
217                .with_error_code("E017")
218                .with_location(&implementation.location),
219        );
220    }
221    if implementation.linkage != LinkageType::External {
222        validate_action_container(validator, implementation);
223        // Validate the label uniqueness
224
225        if let Some(labels) = context.index.get_labels(&implementation.name) {
226            for (_, labels) in labels.entries() {
227                let mut label_iter = labels.iter();
228                if let Some(first) = label_iter.next() {
229                    if let Some(second) = label_iter.next() {
230                        //Collect remaining
231                        let mut locations: Vec<_> = label_iter.map(|it| it.location.clone()).collect();
232                        locations.push(first.location.clone());
233                        locations.push(second.location.clone());
234                        validator.push_diagnostic(
235                            Diagnostic::new(format!("{}: Duplicate label.", &first.name))
236                                .with_error_code("E018")
237                                .with_location(&first.location)
238                                .with_secondary_locations(locations),
239                        );
240                    }
241                }
242            }
243        }
244        implementation.statements.iter().for_each(|s| {
245            visit_statement(validator, s, &context.with_qualifier(implementation.name.as_str()))
246        });
247    }
248}

250fn validate_pou(validator: &mut Validator, pou: &Pou) {
251    if pou.kind == PouType::Class {
252        validate_class(validator, pou);
253    };
254
255    // If the POU is not a function or method, it cannot have a return type
256    if !matches!(pou.kind, PouType::Function | PouType::Method { .. }) {
257        if let Some(start_return_type) = &pou.return_type {
258            validator.push_diagnostic(
259                Diagnostic::new(format!("POU Type {:?} does not support a return type", pou.kind))
260                    .with_error_code("E026")
261                    .with_location(start_return_type.get_location()),
262            )
263        }
264    }
265}

267fn validate_class(validator: &mut Validator, pou: &Pou) {
268    // var in/out/inout blocks are not allowed inside of class declaration
269    // TODO: This should be on each block
270    if pou.variable_blocks.iter().any(|it| {
271        matches!(it.kind, VariableBlockType::InOut | VariableBlockType::Input(_) | VariableBlockType::Output)
272    }) {
273        validator.push_diagnostic(
274            Diagnostic::new("A class cannot contain `VAR_INPUT`, `VAR_IN_OUT`, or `VAR_OUTPUT` blocks")
275                .with_error_code("E019")
276                .with_location(&pou.name_location),
277        );
278    }
279}

283        validator.push_diagnostic(

291pub fn visit_interface<T: AnnotationMap>(
292    validator: &mut Validator,
293    interface: &Interface,
294    context: &ValidationContext<'_, T>,
295) {
296    let Identifier { name, location } = &interface.ident;
297    let entry = context.index.find_interface(name).expect("must exist");
298    entry.get_extensions().iter().for_each(|declaration| {
299        if context.index.find_interface(&declaration.name).is_none() {
300            validator.push_diagnostic(
301                Diagnostic::new(format!("Interface `{}` does not exist", declaration.name))
302                    .with_error_code("E048")
303                    .with_location(&declaration.location),
304            );
305        }
306    });
307    validate_methods_overrides(validator, context, interface.id, name, location);
308    property::validate_properties_in_interfaces(validator, context, interface);
309}

321    pub fn validate_method_signature(
322        index: &Index,
323        method_ref: &PouIndexEntry,
324        method_impl: &PouIndexEntry,
325        primary_location: &SourceLocation,
326    ) -> Vec<Diagnostic> {
327        let ctxt = Context::new(index, method_ref, method_impl, primary_location);
328        let mut validator = SignatureValidator::new(&ctxt);
329        validator.validate();
330        validator.diagnostics
331    }
332
333    struct Context<'idx> {
334        index: &'idx Index,
335        method_ref: &'idx PouIndexEntry,
336        method_impl: &'idx PouIndexEntry,
337        primary_location: &'idx SourceLocation,
338    }
339
340    impl<'idx> Context<'idx> {
341        fn new(
342            index: &'idx Index,
343            method_ref: &'idx PouIndexEntry,
344            method_impl: &'idx PouIndexEntry,
345            primary_location: &'idx SourceLocation,
346        ) -> Self {
347            Self { index, method_ref, method_impl, primary_location }
348        }
349
350        /// Returns a tuple of the return [DataType]s of the method reference and the method implementation
351        fn get_return_types(&self) -> (&DataType, &DataType) {
352            let method_ref_return_type_name = self.method_ref.get_return_type().unwrap_or_default();
353            let method_impl_return_type_name = self.method_impl.get_return_type().unwrap_or_default();
354            let return_type_ref = self.index.get_effective_type_or_void_by_name(method_ref_return_type_name);
355            let return_type_impl =
356                self.index.get_effective_type_or_void_by_name(method_impl_return_type_name);
357
358            (return_type_ref, return_type_impl)
359        }
360    }
361
362    struct SignatureValidator<'b, 'idx> {
363        context: &'b Context<'idx>,
364        diagnostics: Vec<Diagnostic>,
365    }
366
367    impl<'b, 'idx> SignatureValidator<'b, 'idx> {
368        fn new(context: &'b Context<'idx>) -> Self {
369            Self { context, diagnostics: Vec::new() }
370        }
371
372        fn validate(&mut self) {
373            // Check if the return types match
374            self.validate_return_types();
375
376            // Check if the parameters match; note that the order of the parameters is important due to implicit calls.
377            self.validate_parameters();
378        }
379
380        fn validate_return_types(&mut self) {
381            let (return_type_ref, return_type_impl) = self.context.get_return_types();
382            if let Some(sub_diagnostics) = self.validate_types(return_type_impl, return_type_ref) {
383                self.diagnostics.push(
384                    Diagnostic::new(
385                        "Derived methods with conflicting signatures, return types do not match:",
386                    )
387                    .with_location(self.context.primary_location)
388                    .with_error_code("E112")
389                    .with_sub_diagnostics(sub_diagnostics),
390                );
391            }
392        }
393
394        fn validate_parameters(&mut self) {
395            let context = self.context;
396            let method_impl = context.method_impl;
397            let method_ref = context.method_ref;
398            let method_name = context.method_ref.get_flat_reference_name();
399            let parameters_ref = context.index.get_declared_parameters(method_ref.get_name());
400            let parameters_impl = context.index.get_declared_parameters(method_impl.get_name());
401            let mut diagnostics = vec![];
402
403            // Conditionally skip the first parameter if the return type is aggregate.
404            // Return types have already been validated and we don't want to show errors
405            // for internally modified code.
406            let (return_type_ref, return_type_impl) = context.get_return_types();
407            parameters_ref
408                .iter()
409                .skip(return_type_ref.get_type_information().is_aggregate() as usize)
410                .zip_longest(parameters_impl.iter().skip(return_type_impl.get_type_information().is_aggregate() as usize))
411                .for_each(|pair|
412            {
413                match pair {
414                    itertools::EitherOrBoth::Both(parameter_ref, parameter_impl) => {
415                        // Name
416                        if parameter_impl.get_name() != parameter_ref.get_name() {
417                            diagnostics.push(
418                                Diagnostic::new(format!(
419                                    "Expected parameter `{}` but got `{}`",
420                                    parameter_ref.get_name(),
421                                    parameter_impl.get_name()
422                                ))
423                                .with_error_code("E118")
424                                .with_secondary_location(&parameter_ref.source_location)
425                                .with_secondary_location(&parameter_impl.source_location),
426                            );
427                        }
428
429                        // Type
430                        let impl_ty = context
431                            .index
432                            .get_effective_type_or_void_by_name(parameter_impl.get_type_name());
433                        let ref_ty = context
434                            .index
435                            .get_effective_type_or_void_by_name(parameter_ref.get_type_name());
436
437                        if let Some(sub_diagnostics) = self.validate_types(impl_ty, ref_ty) {
438                            diagnostics.push(
439                                Diagnostic::new(format!(
440                                    "Parameter `{}` has conflicting type declarations:",
441                                    parameter_ref.get_name(),
442                                ))
443                                .with_error_code("E118")
444                                .with_sub_diagnostics(sub_diagnostics)
445                                .with_secondary_location(method_impl)
446                                .with_secondary_location(&parameter_ref.source_location)
447                            );
448                        }
449
450                        // Declaration Type (VAR_INPUT, VAR_OUTPUT, VAR_IN_OUT)
451                        if parameter_impl.get_declaration_type() != parameter_ref.get_declaration_type() {
452                            diagnostics.push(
453                                Diagnostic::new(format!(
454                                    "Expected parameter `{}` to have `{}` as its declaration type but got `{}`",
455                                    parameter_impl.get_name(),
456                                    parameter_ref.get_declaration_type().get_inner(),
457                                    parameter_impl.get_declaration_type().get_inner(),
458                                ))
459                                .with_error_code("E118")
460                                .with_secondary_location(method_impl)
461                                .with_secondary_location(&parameter_ref.source_location),
462                            );
463                        }
464                    }
465
466                    itertools::EitherOrBoth::Left(parameter_ref) => {
467                        diagnostics.push(
468                            Diagnostic::new(format!(
469                                "Parameter `{} : {}` missing in method `{}`",
470                                parameter_ref.get_name(),
471                                parameter_ref.get_type_name(),
472                                method_name,
473                            ))
474                            .with_error_code("E118")
475                            .with_secondary_location(method_impl)
476                            .with_secondary_location(&parameter_ref.source_location),
477                        );
478                    }
479
480                    // Exceeding parameters in the POU, which we did not catch in the for loop above because we were only
481                    // iterating over the interface parameters; anyhow any exceeding parameter is considered an error because
482                    // the function signature no longer holds
483                    itertools::EitherOrBoth::Right(parameter_impl) => {
484                        diagnostics.push(
485                            Diagnostic::new(format!(
486                                "`{}` has more parameters than the method defined in `{}`",
487                                method_name,
488                                method_ref.get_parent_pou_name().unwrap(),
489                            ))
490                            .with_error_code("E118")
491                            .with_secondary_location(&parameter_impl.source_location)
492                            .with_secondary_location(method_ref),
493                        );
494                    }
495                }
496            });
497
498            if !diagnostics.is_empty() {
499                self.diagnostics.push(
500                    Diagnostic::new("Derived methods with conflicting signatures, parameters do not match:")
501                        .with_error_code("E112")
502                        .with_location(self.context.primary_location)
503                        .with_sub_diagnostics(diagnostics),
504                );
505            }
506        }

508        fn validate_types(&self, left: &DataType, right: &DataType) -> Option<Vec<Diagnostic>> {
509            let l_type_info = left.get_type_information();
510            let r_type_info = right.get_type_information();
511            if l_type_info == r_type_info {
512                return None;
513            }
514            let context = self.context;
515            match (l_type_info, r_type_info) {
516                (DataTypeInformation::Array { .. }, DataTypeInformation::Array { .. }) => {
517                    self.validate_array_types(left, right)
518                }
519                (
520                    DataTypeInformation::Pointer { inner_type_name: l_inner, auto_deref: l_auto, .. },
521                    DataTypeInformation::Pointer { inner_type_name: r_inner, auto_deref: r_auto, .. },
522                ) => {
523                    if l_auto.is_some() || r_auto.is_some() {
524                        let left = l_auto
525                            .map(|_| context.index.get_effective_type_or_void_by_name(l_inner))
526                            .unwrap_or(left);
527                        let right = r_auto
528                            .map(|_| context.index.get_effective_type_or_void_by_name(r_inner))
529                            .unwrap_or(right);
530                        return self.validate_types(left, right);
531                    }
532                    self.validate_types(
533                        context.index.get_effective_type_or_void_by_name(l_inner),
534                        context.index.get_effective_type_or_void_by_name(r_inner),
535                    )
536                }
537                (DataTypeInformation::Pointer { inner_type_name, auto_deref: Some(_), .. }, _) => {
538                    let inner_type = context.index.get_effective_type_or_void_by_name(inner_type_name);
539                    self.validate_types(inner_type, right)
540                }
541                (_, DataTypeInformation::Pointer { inner_type_name, auto_deref: Some(_), .. }) => {
542                    let inner_type = context.index.get_effective_type_or_void_by_name(inner_type_name);
543                    self.validate_types(left, inner_type)
544                }
545                (DataTypeInformation::String { .. }, DataTypeInformation::String { .. }) => {
546                    self.validate_string_types(l_type_info, r_type_info)
547                }
548                (
549                    DataTypeInformation::SubRange { sub_range: _l_range, .. },
550                    DataTypeInformation::SubRange { sub_range: _r_range, .. },
551                ) => {
552                    self.validate_types(
553                        context.index.get_intrinsic_type(left),
554                        context.index.get_intrinsic_type(right),
555                    )
556
557                    // FIXME: properly validate the ranges (folded constants are problematic)
558                }
559                _ => Some(vec![self.create_diagnostic(l_type_info.get_name(), r_type_info.get_name())]),
560            }
561        }
562
563        fn create_diagnostic(&self, left: &str, right: &str) -> Diagnostic {
564            Diagnostic::new(format!(
565                "Type `{right}` declared in `{}` but `{}` declared type `{left}`",
566                self.context.method_ref.get_name(),
567                self.context.method_impl.get_name()
568            ))
569            .with_error_code("E118")
570            .with_secondary_location(self.context.method_impl)
571            .with_secondary_location(self.context.method_ref)
572        }

574        fn validate_array_types(&self, left: &DataType, right: &DataType) -> Option<Vec<Diagnostic>> {
575            let context = self.context;
576            let method_impl = context.method_impl;
577            let method_ref = context.method_ref;
578
579            let mut sub_diagnostics = vec![];
580            let mut left_type = left;
581            let mut right_type = right;
582
583            while let (Some(l_inner), Some(r_inner)) = (
584                left_type.get_type_information().get_inner_array_type_name(),
585                right_type.get_type_information().get_inner_array_type_name(),
586            ) {
587                // we found two inner types, so the previous iteration was array types. check their dimensions
588                sub_diagnostics.extend(self.validate_array_dimensions(left_type, right));
589                left_type = context.index.get_effective_type_or_void_by_name(l_inner);
590                right_type = context.index.get_effective_type_or_void_by_name(r_inner);
591            }
592
593            let left_name = left_type.get_name();
594            let right_name = right_type.get_name();
595            if left_name != right_name {
596                sub_diagnostics.push(
597                    Diagnostic::new(format!(
598                        "Expected array of type `{}` but got `{}`",
599                        right_name, left_name
600                    ))
601                    .with_error_code("E118")
602                    .with_secondary_location(method_impl)
603                    .with_secondary_location(method_ref),
604                )
605            };
606
607            (!sub_diagnostics.is_empty()).then_some(sub_diagnostics)
608        }
609
610        fn validate_array_dimensions(&self, left: &DataType, right: &DataType) -> Vec<Diagnostic> {
611            let context = self.context;
612            let left_type = left.get_type_information();
613            let right_type = right.get_type_information();
614            let l_dims = left_type.get_dimensions().expect("Array type without dimensions");
615            let r_dims = right_type.get_dimensions().expect("Array type without dimensions");
616
617            l_dims
618                .iter()
619                .zip_longest(r_dims.iter())
620                .filter_map(|pair| {
621                    match pair {
622                        itertools::EitherOrBoth::Both(l, r) => {
623                            match (l.get_range(context.index), r.get_range(context.index)) {
624                                (Ok(l_range), Ok(r_range)) => (l_range != r_range).then_some(
625                                    Diagnostic::new(format!(
626                                        "Array range declared as `[{}..{}]` but implemented as `[{}..{}]`",
627                                        r_range.start, r_range.end, l_range.start, l_range.end
628                                    ))
629                                    .with_error_code("E118")
630                                    .with_secondary_location(left.location.clone())
631                                    .with_secondary_location(right.location.clone()),
632                                ),
633                                _ => {
634                                    // Expression in array dimension could not be evaluated. this should already have raised an error elsewhere
635                                    None
636                                }
637                            }
638                        }
639                        itertools::EitherOrBoth::Left(_) | itertools::EitherOrBoth::Right(_) => Some(
640                            Diagnostic::new(format!(
641                                "Array declared with `{}` dimension{} but implemented with `{}`",
642                                l_dims.len(),
643                                if l_dims.len() == 1 { "" } else { "s" },
644                                r_dims.len()
645                            ))
646                            .with_error_code("E118")
647                            .with_secondary_location(right.location.clone())
648                            .with_secondary_location(context.method_impl)
649                            .with_secondary_location(left.location.clone())
650                            .with_secondary_location(context.method_ref),
651                        ),
652                    }
653                })
654                .collect()
655        }
656
657        fn validate_string_types(
658            &self,
659            left: &DataTypeInformation,
660            right: &DataTypeInformation,
661        ) -> Option<Vec<Diagnostic>> {
662            let ctxt = self.context;
663            let method_impl = ctxt.method_impl;
664            let method_ref = ctxt.method_ref;
665            //FIXME: this is not accurate, we should not do our own data layout calculations
666            let l_encoding = left.get_string_character_width(ctxt.index);
667            let r_encoding = right.get_string_character_width(ctxt.index);
668            let left_length = left.get_size_in_bits(ctxt.index).unwrap() / l_encoding.bits();
669            let right_length = right.get_size_in_bits(ctxt.index).unwrap() / r_encoding.bits();
670            let mut sub_diagnostics = vec![];
671            (left_length != right_length).then(|| {
672                sub_diagnostics.push(
673                    Diagnostic::new(format!(
674                        "Expected string of length `{}` but got string of length `{}`",
675                        right_length, left_length
676                    ))
677                    .with_error_code("E118")
678                    .with_secondary_location(method_impl)
679                    .with_secondary_location(method_ref),
680                );
681            });
682            (l_encoding != r_encoding).then(|| {
683                sub_diagnostics.push(self.create_diagnostic(left.get_name(), right.get_name()));
684            });
685
686            (!sub_diagnostics.is_empty()).then_some(sub_diagnostics)
687        }
688    }
