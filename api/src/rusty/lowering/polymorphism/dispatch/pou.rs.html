1//! Lowering of method calls into indirect calls through virtual tables
2//!
3//! This module is responsible for lowering / transforming method calls into method calls through the
4//! virtual table (for information regarding virtual tables refer to [`crate::lowering::polymorphism::table`]). In a
5//! nutshell it will transform a method call such as `ptr^.foo()` into `__vtable_Fb#(ptr^.__vtable^).foo^(ptr^)`.
6//!
7//! However, not all method calls must be lowered but rather the following cases:
8//!
9//! # 1. Method calls within methods (and function block bodies)
10//! The reason we want to lower all method calls within (other) methods is for the fact that a non
11//! overridden method may make use of an overridden method. For example consider
12//! ```text
13//! FUNCTION_BLOCK A
14//!     METHOD getName: STRING
15//!         getName := 'A';
16//!     END_METHOD
17//!
18//!     METHOD printName
19//!         printf('name = %s$N', ADR(getName()));
20//!     END_METHOD
21//! END_FUNCTION_BLOCK
22//!
23//! FUNCTION_BLOCK B EXTENDS A
24//!     METHOD getName: STRING
25//!         getName := 'B';
26//!     END_METHOD
27//!
28//!     METHOD persistName
29//!         // Persist name to some file
30//!     END_METHOD
31//! END_FUNCTION_BLOCK
32//!
33//! FUNCTION main
34//!     VAR
35//!         instanceA: A;
36//!         instanceB: B;
37//!
38//!         refInstanceA: POINTER TO A;
39//!     END_VAR
40//!
41//!     refInstanceA := ADR(instanceA);
42//!     refInstanceA^.printName(); // Calls `A::printName` which calls `A::getName` yielding "name = A"
43//!
44//!     refInstanceA := ADR(instanceB);
45//!     refInstanceA^.printName(); // Calls `A::printName` which calls `B::getName` yielding "name = B"
46//! END_FUNCTION
47//! ```
48//!
49//! As described in the main function, the calls to `printName` must happen at runtime. Were that not the case
50//! then `printName` in A would resolve to `A::getName` at compile time, yielding an incorrect result for the
51//! second `refInstanceA^.printName()` call. Lowering the call to `printName` would result in
52//! `printf('name = %s$N', ADR(__vtable_A#(THIS^.__vtable^).getName^(THIS^))`.
53//!
54//! # 2. Method calls through a pointer variable pointing to a class or function block
55//! Essentially what is illustrated in 1. within the main function, consider:
56//!
57//! ```text
58//! FUNCTION main
59//!     VAR
60//!         name: STRING;
61//!         instanceA: A;
62//!         instanceB: B;
63//!
64//!         refInstanceA: POINTER TO A;
65//!     END_VAR
66//!
67//!     refInstanceA := ADR(instanceA);
68//!     name := refInstanceA^.getName(); // Calls `A::getName` yielding "name = A"
69//!
70//!     refInstanceA := ADR(instanceB);
71//!     name := refInstanceA^.getName(); // Calls `B::getName` yielding "name = B"
72//! END_FUNCTION
73//! ```
74//!
75//! While this is a simple example, and in theory compilers would be able to derive the correct method calls
76//! at compile time with some statical analysis, our compiler today is not able to do that. Specifically it
77//! does not know that the second `refInstanceA` variable is pointing at `B` and the pointer call could be
78//! simplified into a direct call to `B::getName()`. Instead, it relies on dynamic dispatch for a correct code
79//! execution. Again, this is done by accessing the virtual table and calling the function pointer behind it.
80//! In terms of ST code we would transform the calls into `__vtable_A#(refInstanceA^.__vtable^).getName^(refInstanceA^)`.
81//!
82//!
83//! One final thing to note, while the casting of the virtual tables into concrete types is not really
84//! interesting per-se, the upcasting from a child to its parent virtual table should at least be mentioned.
85//! That is, as long as the virtual table definitions are compatible, upcasting can be performed without any
86//! issues. Compatible here refers to the fact that the order of the member fields must be constant. More
87//! specifically, the methods must be defined in "ancestral hierarchical order". To illustrate with the
88//! previous examples, assume we have
89//! ```text
90//! TYPE __vtable_A:
91//!     STRUCT
92//!         getName: __FPOINTER TO A.getName := ADR(A.getName);
93//!         printName: __FPOINTER TO A.printName := ADR(A.printName);
94//!     END_STRUCT
95//! END_TYPE
96//!
97//! TYPE __vtable_B:
98//!     STRUCT
99//!         getName: __FPOINTER TO B.getName := ADR(B.getName);             // Overridden
100//!         printName: __FPOINTER TO A.printName := ADR(A.printName);       // Inherited
101//!         persistName: __FPOINTER TO B.persistName := ADR(B.persistName); // New
102//!     END_STRUCT
103//! ```
104//!
105//! We can safely cast from B to A's virtual table because the layout is compatible and it would result in
106//! `persistName` to be cut off in the cast. Were that not the case, e.g. if `getName` were to be swapped with
107//! `persistName` then the call to `getName` would silently result in calling `persistName`.

126pub struct PolymorphicCallLowerer<'a> {
127    pub ids: IdProvider,
128    pub index: &'a Index,
129    pub annotations: &'a AnnotationMapImpl,
130
131    pub in_method_or_function_block: Option<String>,
132}

134impl<'a> AstVisitorMut for PolymorphicCallLowerer<'a> {
135    fn visit_implementation(&mut self, implementation: &mut plc_ast::ast::Implementation) {
136        if implementation.location.is_internal() {
137            return;
138        }
139
140        self.in_method_or_function_block = match &implementation.pou_type {
141            PouType::FunctionBlock => Some(implementation.name.clone()),
142            PouType::Method { parent, .. } => Some(parent.clone()),
143            _ => None,
144        };
145
146        for statement in &mut implementation.statements {
147            statement.walk(self);
148        }
149
150        self.in_method_or_function_block = None;
151    }
152
153    fn visit_call_statement(&mut self, node: &mut plc_ast::ast::AstNode) {
154        // When dealing with a function call such as `ref^.foo()` we have to perform several steps to lower
155        // it into a form that can be executed by the codegen without any intervention from our side, namely:
156        // 1. We must add the expression (excluding the method name) as the first argument to the call
157        //    -> ref^.foo(ref^)
158        // 2. We must access the virtual table of the instance, a VOID pointer
159        //    -> ref^.__vtable^.foo(ref^)
160        // 3. We must cast the virtual table access into a concrete type
161        //    -> __vtable_<POU_NAME>#(ref^.__vtable^).foo(ref^)
162        // 4. We must dereference the method call, which is a function pointer
163        //    -> __vtable_<POU_NAME>#(ref^.__vtable^).foo^(ref^)
164        //
165        // The final result transforms ref^.foo() into __vtable_<POU_NAME>#(ref^.__vtable^).foo^(ref^)
166        let prev = node.as_string();
167        let AstStatement::CallStatement(CallStatement { operator, parameters }) = &mut node.stmt else {
168            unreachable!();
169        };
170
171        // We need to walk the parameters before deciding to potentially stop here, because parameters may
172        // also contain polymorphic calls that need to be lowered, e.g. `functionCall(methodCall())`
173        if let Some(ref mut parameters) = parameters {
174            parameters.walk(self);
175        }
176
177        if !self.is_polymorphic_call_candidate(operator) {
178            return;
179        };
180
181        let unit_name = match self.in_method_or_function_block {
182            Some(ref name) => name.clone(),
183            None => match operator.get_base_ref_expr() {
184                Some(base) => {
185                    // When dealing with e.g. __main_myVariable
186                    let ty = self.annotations.get_type(base, self.index).unwrap();
187                    self.index.find_elementary_pointer_type(ty.get_type_information()).get_name().to_string()
188                }
189
190                None => {
191                    let ty = self.annotations.get_type(operator, self.index).unwrap();
192                    self.index.find_elementary_pointer_type(ty.get_type_information()).get_name().to_string()
193                }
194            },
195        };
196
197        log::trace!("lowering {}", operator.as_string());
198
199        // Pre-steps, add `__body` call when dealing with a direct function block and...
200        self.maybe_patch_body_operator(operator);
201        log::trace!("-1: {}", operator.as_string());
202
203        // ...add a `THIS^` base when dealing with method calls without a base, e.g. `foo()`
204        self.maybe_patch_this_base(operator);
205        log::trace!("0: {}", operator.as_string());
206
207        // Step 1: Add the expression (excluding the method name) as the first argument to the call
208        self.patch_instance_argument(operator, parameters);
209        log::trace!("1: {}", operator.as_string());
210
211        // Step 2: Patch a dereferenced virtual table access into the operator
212        self.patch_vtable_access(operator);
213        log::trace!("2: {}", operator.as_string());
214
215        // Step 3: Patch the virtual table cast into the operator
216        self.patch_vtable_cast(operator, &unit_name);
217        log::trace!("3: {}", operator.as_string());
218
219        // Step 4: Patch the method call to a dereferenced call
220        self.patch_method_call_deref(operator);
221        log::trace!("4: {}", operator.as_string());
222
223        log::debug!("converted `{}` into `{}`", prev, node.as_string());
224    }
225}

227impl<'a> PolymorphicCallLowerer<'a> {
228    pub fn new(ids: IdProvider, index: &'a Index, annotations: &'a AnnotationMapImpl) -> Self {
229        PolymorphicCallLowerer { ids, index, annotations, in_method_or_function_block: None }
230    }
231
232    pub fn lower(&mut self, units: &mut [CompilationUnit]) {
233        for unit in units {
234            self.visit_compilation_unit(unit);
235        }
236    }

238    /// Returns true if the given AST node is a candidate that needs to be lowered into a polymorphic call.
239    fn is_polymorphic_call_candidate(&self, operator: &AstNode) -> bool {
240        // We do not want to lower THIS and SUPER access, e.g. {THIS,SUPER}^() or {THIS,SUPER}^.foo()
241        if operator.is_super_or_super_deref()
242            || operator.get_base_ref_expr().is_some_and(AstNode::is_super_or_super_deref)
243            || operator.is_this()
244            || operator.is_this_deref()
245        {
246            return false;
247        }
248
249        if self.annotations.get(operator).is_some_and(|opt| opt.is_fnptr()) {
250            return false;
251        }
252
253        // Case 1 (Method call within methods or function block bodies)
254        if self.in_method_or_function_block.is_some()
255            && self.annotations.get_type(operator, self.index).is_some_and(DataType::is_method)
256            // Only lower something alike `THIS^.foo()` or `foo()` as opposed to `SUPER^.foo()` or `instanceFb.foo()`
257            && (operator.get_base_ref_expr().is_none() || operator.get_base_ref_expr().is_some_and(|opt| opt.is_this()))
258        {
259            return true;
260        }
261
262        // Case 2 (Method invocation via a pointer to a class or function block instance)
263        let AstStatement::ReferenceExpr(ReferenceExpr { access, base }) = &operator.stmt else {
264            return false;
265        };
266
267        match (access, base) {
268            // Dealing with `MyFbRef^.foo()`
269            (ReferenceAccess::Member(_), Some(base)) => self.is_polymorphic_call_candidate(base),
270
271            // Dealing with `MyFbRef^()`
272            (ReferenceAccess::Deref, Some(base)) => {
273                if self.annotations.get(operator).is_some_and(StatementAnnotation::is_fnptr) {
274                    return false;
275                };
276
277                let info = self.annotations.get_type_or_void(base, self.index).get_type_information();
278                let DataTypeInformation::Pointer { inner_type_name, .. } = info else {
279                    return false;
280                };
281
282                let inner_pointer_type = self.index.get_type_information_or_void(inner_type_name);
283                inner_pointer_type.is_class() || inner_pointer_type.is_function_block()
284            }
285
286            // Auto-deref, e.g. `refInstance: REFERENCE TO ...`
287            (ReferenceAccess::Member(member), None) => {
288                self.annotations.get(member).is_some_and(StatementAnnotation::is_reference_to)
289            }
290
291            _ => false,
292        }
293    }
294
295    fn maybe_patch_body_operator(&mut self, operator: &mut AstNode) {
296        if !self
297            .annotations
298            .get_type(operator, self.index)
299            .is_some_and(|opt| opt.information.is_function_block())
300        {
301            return;
302        }
303
304        let old_base = std::mem::take(operator);
305        let mut new_base = AstFactory::create_member_reference(
306            AstFactory::create_identifier("__body", SourceLocation::internal(), self.ids.next_id()),
307            Some(old_base),
308            self.ids.next_id(),
309        );
310
311        std::mem::swap(operator, &mut new_base);
312    }
313
314    fn maybe_patch_this_base(&mut self, operator: &mut AstNode) {
315        if !(self.in_method_or_function_block.is_some() && operator.get_base_ref_expr().is_none()) {
316            return;
317        }
318
319        let this_node = Box::new(AstFactory::create_deref_reference(
320            AstFactory::create_this_reference(SourceLocation::internal(), self.ids.next_id()),
321            self.ids.next_id(),
322            SourceLocation::internal(),
323        ));
324
325        operator.get_ref_expr_mut().unwrap().base.replace(this_node);
326    }
327
328    fn patch_instance_argument(&mut self, operator: &mut AstNode, parameters: &mut Option<Box<AstNode>>) {
329        // foo.bar()
330        // ^^^ base
331        let base = self.maybe_cast_instance(operator);
332
333        match parameters {
334            None => {
335                parameters.replace(Box::new(base));
336            }
337
338            Some(ref mut expr) => match &mut expr.stmt {
339                AstStatement::ExpressionList(expressions) => {
340                    expressions.insert(0, base);
341                }
342
343                _ => {
344                    let mut expressions = Box::new(AstFactory::create_expression_list(
345                        vec![base, std::mem::take(expr)],
346                        SourceLocation::internal(),
347                        self.ids.next_id(),
348                    ));
349
350                    std::mem::swap(expr, &mut expressions);
351                }
352            },
353        }
354    }
355
356    /// Patches a `__vtable` member access into the given node, e.g. `ref^.foo()` becomes `ref^.__vtable^.foo()`
357    fn patch_vtable_access(&mut self, node: &mut AstNode) {
358        let old_base = node.get_base_ref_expr_mut().unwrap(); // `ref^` in `ref^.foo()`
359
360        let mut new_base = AstFactory::create_deref_reference(
361            AstFactory::create_member_reference(
362                AstFactory::create_identifier("__vtable", SourceLocation::internal(), self.ids.next_id()),
363                Some(std::mem::take(old_base)),
364                self.ids.next_id(),
365            ),
366            self.ids.next_id(),
367            SourceLocation::internal(),
368        );
369
370        std::mem::swap(old_base, &mut new_base);
371    }
372
373    /// ref^.__vtable^.foo()` -> `__vtable_{POU_NAME}#(ref^.__vtable^).foo()
374    fn patch_vtable_cast(&mut self, node: &mut AstNode, pou_type_name: &str) {
375        let base_old = node.get_base_ref_expr_mut().unwrap(); // `ref^.__vtable^` in `ref^.__vtable^.foo()`
376        let base_old_paren = AstFactory::create_paren_expression(
377            std::mem::take(base_old),
378            SourceLocation::internal(),
379            self.ids.next_id(),
380        );
381
382        let mut base_new = AstFactory::create_cast_statement(
383            AstFactory::create_member_reference(
384                AstFactory::create_identifier(
385                    format!("__vtable_{pou_type_name}"),
386                    SourceLocation::internal(),
387                    self.ids.next_id(),
388                ),
389                None,
390                self.ids.next_id(),
391            ),
392            base_old_paren,
393            &SourceLocation::internal(),
394            self.ids.next_id(),
395        );
396
397        std::mem::swap(base_old, &mut base_new);
398    }
399
400    // `__vtable_{POU_NAME}#(ref^.__vtable^).foo()` -> `__vtable_{POU_NAME}#(ref^.__vtable^).foo^()`
401    fn patch_method_call_deref(&mut self, node: &mut AstNode) {
402        let mut base_new = AstFactory::create_deref_reference(
403            std::mem::take(node),
404            self.ids.next_id(),
405            SourceLocation::internal(),
406        );
407
408        std::mem::swap(node, &mut base_new);
409    }

411    /// Casts the instance argument to its concrete type, as defined by the method. While not strictly
412    /// necessary, because the code would compile without this step, it is required when generating IR using
413    /// the `--ir` flag. This is because some type-validation happens (afaik).
414    ///
415    /// For example `alpha(1, 2)` will be transformed into` `<...>.alpha^(A#(THIS^), 1, 2)` instead of
416    /// `<...>.alpha^(THIS^, 1, 2)`. The same would be true for `refInstance.alpha(1, 2)` which would become
417    /// `<...>.alpha^(A#(refInstance), 1, 2)`.
418    /// ```text
419    /// FUNCTION_BLOCK A
420    ///     METHOD alpha
421    ///         VAR_INPUT
422    ///             a, b: DINT;
423    ///         END_VAR
424    ///     END_METHOD
425    /// END_FUNCTION_BLOCK
426    ///
427    /// FUNCTION_BLOCK B EXTENDS A
428    ///     METHOD bravo
429    ///         alpha(1, 2);
430    ///     END_METHOD
431    /// END_FUNCTION_BLOCK
432    /// ```
433    fn maybe_cast_instance(&mut self, operator: &AstNode) -> AstNode {
434        let Some(method_owner) = self
435            .annotations
436            .get_type(operator, self.index)
437            .map(DataType::get_type_information)
438            .and_then(DataTypeInformation::get_method_owner)
439        else {
440            return operator.get_base_ref_expr().unwrap().clone();
441        };
442
443        let base = operator.get_base_ref_expr().unwrap();
444        AstFactory::create_cast_statement(
445            AstFactory::create_member_reference(
446                AstFactory::create_identifier(method_owner, SourceLocation::internal(), self.ids.next_id()),
447                None,
448                self.ids.next_id(),
449            ),
450            AstFactory::create_paren_expression(base.clone(), SourceLocation::internal(), self.ids.next_id()),
451            &SourceLocation::internal(),
452            self.ids.next_id(),
453        )
454    }

462    fn lower_statements(source: impl Into<SourceCode>, pous: &[&str]) -> Vec<String> {
463        let (_, project) = parse_and_annotate("unit-test", vec![source.into()]).unwrap();
464        let unit = project.units[0].get_unit();
465
466        let mut result = Vec::new();
467        for pou in pous {
468            result.push(format!("// Statements in {pou}"));
469            let statements = &unit.implementations.iter().find(|it| &it.name == pou).unwrap().statements;
470            let statements_str = statements.iter().map(|statement| statement.as_string()).collect::<Vec<_>>();
471
472            result.extend(statements_str);
473        }
474
475        result
476    }
