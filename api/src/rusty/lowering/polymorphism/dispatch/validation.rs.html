1//! Validation functions for interface polymorphism, called during lowering.
2//!
3//! Each function returns `Some(Diagnostic)` when the code is invalid, or `None` when valid.
4//!
5//! Note:
6//! These checks run inside [`super::interface::InterfaceDispatchLowerer`] rather than in the
7//! normal validation pass because the lowerer rewrites interface types to `__FATPOINTER` before
8//! the validator sees them. Running these checks here ensures error messages reference the
9//! user-written interface names, not internal types.
10//!
11//! Technically a `pre_generate` exists, which would mean polymorphic lowering happens after validation but
12//! other lowerers would then also need to change hooking from `post_annotate` to `pre_generate` and also
13//! break existing validations (some SUPER tests failed when I tried this). As a result, doing validations
14//! during lowering is the most pragmatic option right now, even if it's a bit inelegant.
15
16use plc_ast::ast::AstNode;
17use plc_diagnostics::diagnostics::Diagnostic;
18use plc_source::source_location::SourceLocation;
19
20use crate::{
21    index::Index, lowering::polymorphism::table::interface::helper as itable_helper, resolver::AnnotationMap,
22};

24/// Validates that a POU implements the target interface. Returns `Some(Diagnostic)` if not.
25///
26/// ```st
27/// refIA := instanceFbX;  // error: FbX doesn't implement IA
28/// refIA := instanceFbA;  // ok:    FbA IMPLEMENTS IA
29/// refIA := instanceFbB;  // ok:    FbB EXTENDS FbA, FbA IMPLEMENTS IA
30/// ```
31pub fn validate_pou_implements_interface(
32    index: &Index,
33    pou_name: &str,
34    interface_name: &str,
35    location: &SourceLocation,
36) -> Option<Diagnostic> {
37    let pou = index.find_pou(pou_name)?;
38    let all_interfaces = itable_helper::collect_interfaces_for_pou(index, pou);
39
40    if all_interfaces.contains(interface_name) {
41        return None;
42    }
43
44    Some(Diagnostic::invalid_interface_pou_assignment(pou_name, interface_name, location.clone()))
45}

47/// Validates that an interface-to-interface assignment is a valid upcast. Returns `Some(Diagnostic)` if not.
48///
49/// ```st
50/// refIA := refIB;  // ok:    IB EXTENDS IA
51/// refIB := refIA;  // error: downcast
52/// refIA := refIC;  // error: IC unrelated to IA
53/// ```
54pub fn validate_interface_assignment(
55    index: &Index,
56    source_iface: &str,
57    target_iface: &str,
58    location: &SourceLocation,
59) -> Option<Diagnostic> {
60    // Same type is always valid (handled before we get here, but be defensive).
61    if source_iface.eq_ignore_ascii_case(target_iface) {
62        return None;
63    }
64
65    let source = index.find_interface(source_iface)?;
66    let ancestors = source.get_interface_hierarchy(index);
67
68    // Check if target_iface is among the source's ancestors (which includes self).
69    if ancestors.iter().any(|a| a.get_name().eq_ignore_ascii_case(target_iface)) {
70        return None;
71    }
72
73    Some(Diagnostic::invalid_interface_assignment(source_iface, target_iface, location.clone()))
74}

76/// Validates that an interface reference is not called directly.
77///
78/// ```st
79/// refIA();     // error: a method name must be specified
80/// refIA.foo(); // ok:    call a declared interface method instead
81/// ```
82pub fn validate_direct_interface_call<T: AnnotationMap>(
83    annotations: &T,
84    index: &Index,
85    operator: &AstNode,
86) -> Option<Diagnostic> {
87    let operator_type = annotations.get_type_or_void(operator, index);
88    operator_type.is_interface().then(|| Diagnostic::invalid_direct_interface_call(operator.location.clone()))
89}
