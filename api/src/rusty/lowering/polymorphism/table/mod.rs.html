13/// Generates all polymorphism-related tables (virtual tables for POUs and, in the future,
14/// interface tables) by delegating to specialized generators.
15pub struct TableGenerator;
16
17impl TableGenerator {
18    pub fn generate(
19        ids: IdProvider,
20        generate_external_constructors: bool,
21        index: &Index,
22        units: &mut Vec<CompilationUnit>,
23    ) {
24        let mut vtable_gen = VirtualTableGenerator::new(ids.clone(), generate_external_constructors);
25        vtable_gen.generate(index, units);
26
27        let mut itable_gen = InterfaceTableGenerator::new(ids, generate_external_constructors);
28        itable_gen.generate(index, units);
29    }
30}

32/// Returns whether a POU with the given linkage should have its table instances
33/// defined in this compilation unit. External/Include POUs rely on the library
34/// to provide the instances, unless `--generate-external-constructors` is set
35/// for External POUs.
36pub fn is_internal_instance(linkage: LinkageType, generate_external_constructors: bool) -> bool {
37    match linkage {
38        LinkageType::External if generate_external_constructors => true,
39        LinkageType::External | LinkageType::Include => false,
40        _ => true,
41    }
42}
