1//! Lowering of method calls into indirect calls through virtual tables
2//!
3//! This module is responsible for lowering / transforming method calls into method calls through the
4//! virtual table (for information regarding virtual tables refer to [`crate::lowering::vtable`]). In a
5//! nutshell it will transform a method call such as `ptr^.foo()` into `__vtable_Fb#(ptr^.__vtable^).foo^(ptr^)`.
6//!
7//! However, not all method calls must be lowered but rather the following cases:
8//!
9//! # 1. Method calls within methods (and function block bodies)
10//! The reason we want to lower all method calls within (other) methods is for the fact that a non
11//! overridden method may make use of an overridden method. For example consider
12//! ```text
13//! FUNCTION_BLOCK A
14//!     METHOD getName: STRING
15//!         getName := 'A';
16//!     END_METHOD
17//!
18//!     METHOD printName
19//!         printf('name = %s$N', ADR(getName()));
20//!     END_METHOD
21//! END_FUNCTION_BLOCK
22//!
23//! FUNCTION_BLOCK B EXTENDS A
24//!     METHOD getName: STRING
25//!         getName := 'B';
26//!     END_METHOD
27//!
28//!     METHOD persistName
29//!         // Persist name to some file
30//!     END_METHOD
31//! END_FUNCTION_BLOCK
32//!
33//! FUNCTION main
34//!     VAR
35//!         instanceA: A;
36//!         instanceB: B;
37//!
38//!         refInstanceA: POINTER TO A;
39//!     END_VAR
40//!
41//!     refInstanceA := ADR(instanceA);
42//!     refInstanceA^.printName(); // Calls `A::printName` which calls `A::getName` yielding "name = A"
43//!
44//!     refInstanceA := ADR(instanceB);
45//!     refInstanceA^.printName(); // Calls `A::printName` which calls `B::getName` yielding "name = B"
46//! END_FUNCTION
47//! ```
48//!
49//! As described in the main function, the calls to `printName` must happen at runtime. Were that not the case
50//! then `printName` in A would resolve to `A::getName` at compile time, yielding an incorrect result for the
51//! second `refInstanceA^.printName()` call. Lowering the call to `printName` would result in
52//! `printf('name = %s$N', ADR(__vtable_A#(THIS^.__vtable^).getName^(THIS^))`.
53//!
54//! # 2. Method calls through a pointer variable pointing to a class or function block
55//! Essentially what is illustrated in 1. within the main function, consider:
56//!
57//! ```text
58//! FUNCTION main
59//!     VAR
60//!         name: STRING;
61//!         instanceA: A;
62//!         instanceB: B;
63//!
64//!         refInstanceA: POINTER TO A;
65//!     END_VAR
66//!
67//!     refInstanceA := ADR(instanceA);
68//!     name := refInstanceA^.getName(); // Calls `A::getName` yielding "name = A"
69//!
70//!     refInstanceA := ADR(instanceB);
71//!     name := refInstanceA^.getName(); // Calls `B::getName` yielding "name = B"
72//! END_FUNCTION
73//! ```
74//!
75//! While this is a simple example, and in theory compilers would be able to derive the correct method calls
76//! at compile time with some statical analysis, our compiler today is not able to do that. Specifically it
77//! does not know that the second `refInstanceA` variable is pointing at `B` and the pointer call could be
78//! simplified into a direct call to `B::getName()`. Instead, it relies on dynamic dispatch for a correct code
79//! execution. Again, this is done by accessing the virtual table and calling the function pointer behind it.
80//! In terms of ST code we would transform the calls into `__vtable_A#(refInstanceA^.__vtable^).getName^(refInstanceA^)`.
81//!
82//!
83//! One final thing to note, while the casting of the virtual tables into concrete types is not really
84//! interesting per-se, the upcasting from a child to its parent virtual table should at least be mentioned.
85//! That is, as long as the virtual table definitions are compatible, upcasting can be performed without any
86//! issues. Compatible here refers to the fact that the order of the member fields must be constant. More
87//! specifically, the methods must be defined in "ancestral hierarchical order". To illustrate with the
88//! previous examples, assume we have
89//! ```text
90//! TYPE __vtable_A:
91//!     STRUCT
92//!         getName: __FPOINTER TO A.getName := ADR(A.getName);
93//!         printName: __FPOINTER TO A.printName := ADR(A.printName);
94//!     END_STRUCT
95//! END_TYPE
96//!
97//! TYPE __vtable_B:
98//!     STRUCT
99//!         getName: __FPOINTER TO B.getName := ADR(B.getName);             // Overridden
100//!         printName: __FPOINTER TO A.printName := ADR(A.printName);       // Inherited
101//!         persistName: __FPOINTER TO B.persistName := ADR(B.persistName); // New
102//!     END_STRUCT
103//! ```
104//!
105//! We can safely cast from B to A's virtual table because the layout is compatible and it would result in
106//! `persistName` to be cut off in the cast. Were that not the case, e.g. if `getName` were to be swapped with
107//! `persistName` then the call to `getName` would silently result in calling `persistName`.
108

125pub struct PolymorphicCallLowerer {
126    pub ids: IdProvider,
127    pub index: Option<Index>,
128    pub annotations: Option<AnnotationMapImpl>,

133impl AstVisitorMut for PolymorphicCallLowerer {
134    fn visit_implementation(&mut self, implementation: &mut plc_ast::ast::Implementation) {
135        if implementation.location.is_internal() {
136            return;
137        }
138
139        self.in_method_or_function_block = match &implementation.pou_type {
140            PouType::FunctionBlock => Some(implementation.name.clone()),
141            PouType::Method { parent, .. } => Some(parent.clone()),
142            _ => None,
143        };
144
145        for statement in &mut implementation.statements {
146            statement.walk(self);
147        }
148
149        self.in_method_or_function_block = None;
150    }
151
152    fn visit_call_statement(&mut self, node: &mut plc_ast::ast::AstNode) {
153        // When dealing with a function call such as `ref^.foo()` we have to perform several steps to lower
154        // it into a form that can be executed by the codegen without any intervention from our side, namely:
155        // 1. We must add the expression (excluding the method name) as the first argument to the call
156        //    -> ref^.foo(ref^)
157        // 2. We must access the virtual table of the instance, a VOID pointer
158        //    -> ref^.__vtable^.foo(ref^)
159        // 3. We must cast the virtual table access into a concrete type
160        //    -> __vtable_<POU_NAME>#(ref^.__vtable^).foo(ref^)
161        // 4. We must dereference the method call, which is a function pointer
162        //    -> __vtable_<POU_NAME>#(ref^.__vtable^).foo^(ref^)
163        //
164        // The final result transforms ref^.foo() into __vtable_<POU_NAME>#(ref^.__vtable^).foo^(ref^)
165        let prev = node.as_string();
166        let AstStatement::CallStatement(CallStatement { operator, parameters }) = &mut node.stmt else {
167            unreachable!();
168        };
169
170        // We need to walk the parameters before deciding to potentially stop here, because parameters may
171        // also contain polymorphic calls that need to be lowered, e.g. `functionCall(methodCall())`
172        if let Some(ref mut parameters) = parameters {
173            parameters.walk(self);
174        }
175
176        if !self.is_polymorphic_call_candidate(operator) {
177            return;
178        };
179
180        let index = self.index.as_ref().unwrap();
181        let annotations = self.annotations.as_ref().unwrap();
182
183        let unit_name = match self.in_method_or_function_block {
184            Some(ref name) => name.clone(),
185            None => match operator.get_base_ref_expr() {
186                Some(base) => {
187                    // When dealing with e.g. __main_myVariable
188                    let ty = annotations.get_type(base, index).unwrap();
189                    index.find_elementary_pointer_type(ty.get_type_information()).get_name().to_string()
190                }
191
192                None => {
193                    let ty = annotations.get_type(operator, index).unwrap();
194                    index.find_elementary_pointer_type(ty.get_type_information()).get_name().to_string()
195                }
196            },
197        };
198
199        log::trace!("lowering {}", operator.as_string());
200
201        // Pre-steps, add `__body` call when dealing with a direct function block and...
202        self.maybe_patch_body_operator(operator);
203        log::trace!("-1: {}", operator.as_string());
204
205        // ...add a `THIS^` base when dealing with method calls without a base, e.g. `foo()`
206        self.maybe_patch_this_base(operator);
207        log::trace!("0: {}", operator.as_string());
208
209        // Step 1: Add the expression (excluding the method name) as the first argument to the call
210        self.patch_instance_argument(operator, parameters);
211        log::trace!("1: {}", operator.as_string());
212
213        // Step 2: Patch a dereferenced virtual table access into the operator
214        self.patch_vtable_access(operator);
215        log::trace!("2: {}", operator.as_string());
216
217        // Step 3: Patch the virtual table cast into the operator
218        self.patch_vtable_cast(operator, &unit_name);
219        log::trace!("3: {}", operator.as_string());
220
221        // Step 4: Patch the method call to a dereferenced call
222        self.patch_method_call_deref(operator);
223        log::trace!("4: {}", operator.as_string());
224
225        log::debug!("converted `{}` into `{}`", prev, node.as_string());
226    }
227}

229impl PolymorphicCallLowerer {
230    pub fn new(ids: IdProvider) -> PolymorphicCallLowerer {
231        PolymorphicCallLowerer { ids, index: None, annotations: None, in_method_or_function_block: None }
232    }
233
234    pub fn lower_unit(&mut self, unit: &mut CompilationUnit) {
235        self.visit_compilation_unit(unit);
236    }

238    /// Returns true if the given AST node is a candidate that needs to be lowered into a polymorphic call.
239    fn is_polymorphic_call_candidate(&self, operator: &AstNode) -> bool {
240        let index = self.index.as_ref().expect("index must be set before lowering");
241        let annotations = self.annotations.as_ref().expect("annotations must be set before lowering");
242
243        // We do not want to lower THIS and SUPER access, e.g. {THIS,SUPER}^() or {THIS,SUPER}^.foo()
244        if operator.is_super_or_super_deref()
245            || operator.get_base_ref_expr().is_some_and(AstNode::is_super_or_super_deref)
246            || operator.is_this()
247            || operator.is_this_deref()
248        {
249            return false;
250        }
251
252        if annotations.get(operator).is_some_and(|opt| opt.is_fnptr()) {
253            return false;
254        }
255
256        // Case 1 (Method call within methods or function block bodies)
257        if self.in_method_or_function_block.is_some()
258            && annotations.get_type(operator, index).is_some_and(DataType::is_method)
259            // Only lower something alike `THIS^.foo()` or `foo()` as opposed to `SUPER^.foo()` or `instanceFb.foo()`
260            && (operator.get_base_ref_expr().is_none() || operator.get_base_ref_expr().is_some_and(|opt| opt.is_this()))
261        {
262            return true;
263        }
264
265        // Case 2 (Method invocation via a pointer to a class or function block instance)
266        let AstStatement::ReferenceExpr(ReferenceExpr { access, base }) = &operator.stmt else {
267            return false;
268        };
269
270        match (access, base) {
271            // Dealing with `MyFbRef^.foo()`
272            (ReferenceAccess::Member(_), Some(base)) => self.is_polymorphic_call_candidate(base),
273
274            // Dealing with `MyFbRef^()`
275            (ReferenceAccess::Deref, Some(base)) => {
276                if annotations.get(operator).is_some_and(StatementAnnotation::is_fnptr) {
277                    return false;
278                };
279
280                let info = annotations.get_type_or_void(base, index).get_type_information();
281                let DataTypeInformation::Pointer { inner_type_name, .. } = info else {
282                    return false;
283                };
284
285                let inner_pointer_type = index.get_type_information_or_void(inner_type_name);
286                inner_pointer_type.is_class() || inner_pointer_type.is_function_block()
287            }
288
289            // Auto-deref, e.g. `refInstance: REFERENCE TO ...`
290            (ReferenceAccess::Member(member), None) => {
291                annotations.get(member).is_some_and(StatementAnnotation::is_reference_to)
292            }
293
294            _ => false,
295        }
296    }
297
298    fn maybe_patch_body_operator(&mut self, operator: &mut AstNode) {
299        let index = self.index.as_ref().unwrap();
300        let annotations = self.annotations.as_ref().unwrap();
301
302        if !annotations.get_type(operator, index).is_some_and(|opt| opt.information.is_function_block()) {
303            return;
304        }
305
306        let old_base = std::mem::take(operator);
307        let mut new_base = AstFactory::create_member_reference(
308            AstFactory::create_identifier("__body", SourceLocation::internal(), self.ids.next_id()),
309            Some(old_base),
310            self.ids.next_id(),
311        );
312
313        std::mem::swap(operator, &mut new_base);
314    }
315
316    fn maybe_patch_this_base(&mut self, operator: &mut AstNode) {
317        if !(self.in_method_or_function_block.is_some() && operator.get_base_ref_expr().is_none()) {
318            return;
319        }
320
321        let this_node = Box::new(AstFactory::create_deref_reference(
322            AstFactory::create_this_reference(SourceLocation::internal(), self.ids.next_id()),
323            self.ids.next_id(),
324            SourceLocation::internal(),
325        ));
326
327        operator.get_ref_expr_mut().unwrap().base.replace(this_node);
328    }
329
330    fn patch_instance_argument(&mut self, operator: &mut AstNode, parameters: &mut Option<Box<AstNode>>) {
331        // foo.bar()
332        // ^^^ base
333        let base = self.maybe_cast_instance(operator);
334
335        match parameters {
336            None => {
337                parameters.replace(Box::new(base));
338            }
339
340            Some(ref mut expr) => match &mut expr.stmt {
341                AstStatement::ExpressionList(expressions) => {
342                    expressions.insert(0, base);
343                }
344
345                _ => {
346                    let mut expressions = Box::new(AstFactory::create_expression_list(
347                        vec![base, std::mem::take(expr)],
348                        SourceLocation::internal(),
349                        self.ids.next_id(),
350                    ));
351
352                    std::mem::swap(expr, &mut expressions);
353                }
354            },
355        }
356    }
357
358    /// Patches a `__vtable` member access into the given node, e.g. `ref^.foo()` becomes `ref^.__vtable^.foo()`
359    fn patch_vtable_access(&mut self, node: &mut AstNode) {
360        let old_base = node.get_base_ref_expr_mut().unwrap(); // `ref^` in `ref^.foo()`
361
362        let mut new_base = AstFactory::create_deref_reference(
363            AstFactory::create_member_reference(
364                AstFactory::create_identifier("__vtable", SourceLocation::internal(), self.ids.next_id()),
365                Some(std::mem::take(old_base)),
366                self.ids.next_id(),
367            ),
368            self.ids.next_id(),
369            SourceLocation::internal(),
370        );
371
372        std::mem::swap(old_base, &mut new_base);
373    }
374
375    /// ref^.__vtable^.foo()` -> `__vtable_{POU_NAME}#(ref^.__vtable^).foo()
376    fn patch_vtable_cast(&mut self, node: &mut AstNode, pou_type_name: &str) {
377        let base_old = node.get_base_ref_expr_mut().unwrap(); // `ref^.__vtable^` in `ref^.__vtable^.foo()`
378        let base_old_paren = AstFactory::create_paren_expression(
379            std::mem::take(base_old),
380            SourceLocation::internal(),
381            self.ids.next_id(),
382        );
383
384        let mut base_new = AstFactory::create_cast_statement(
385            AstFactory::create_member_reference(
386                AstFactory::create_identifier(
387                    format!("__vtable_{pou_type_name}"),
388                    SourceLocation::internal(),
389                    self.ids.next_id(),
390                ),
391                None,
392                self.ids.next_id(),
393            ),
394            base_old_paren,
395            &SourceLocation::internal(),
396            self.ids.next_id(),
397        );
398
399        std::mem::swap(base_old, &mut base_new);
400    }
401
402    // `__vtable_{POU_NAME}#(ref^.__vtable^).foo()` -> `__vtable_{POU_NAME}#(ref^.__vtable^).foo^()`
403    fn patch_method_call_deref(&mut self, node: &mut AstNode) {
404        let mut base_new = AstFactory::create_deref_reference(
405            std::mem::take(node),
406            self.ids.next_id(),
407            SourceLocation::internal(),
408        );
409
410        std::mem::swap(node, &mut base_new);
411    }

413    /// Casts the instance argument to its concrete type, as defined by the method. While not strictly
414    /// necessary, because the code would compile without this step, it is required when generating IR using
415    /// the `--ir` flag. This is because some type-validation happens (afaik).
416    ///
417    /// For example `alpha(1, 2)` will be transformed into` `<...>.alpha^(A#(THIS^), 1, 2)` instead of
418    /// `<...>.alpha^(THIS^, 1, 2)`. The same would be true for `refInstance.alpha(1, 2)` which would become
419    /// `<...>.alpha^(A#(refInstance), 1, 2)`.
420    /// ```text
421    /// FUNCTION_BLOCK A
422    ///     METHOD alpha
423    ///         VAR_INPUT
424    ///             a, b: DINT;
425    ///         END_VAR
426    ///     END_METHOD
427    /// END_FUNCTION_BLOCK
428    ///
429    /// FUNCTION_BLOCK B EXTENDS A
430    ///     METHOD bravo
431    ///         alpha(1, 2);
432    ///     END_METHOD
433    /// END_FUNCTION_BLOCK
434    /// ```
435    fn maybe_cast_instance(&mut self, operator: &AstNode) -> AstNode {
436        let index = self.index.as_ref().unwrap();
437        let annotations = self.annotations.as_ref().unwrap();
438
439        let Some(method_owner) = annotations
440            .get_type(operator, index)
441            .map(DataType::get_type_information)
442            .and_then(DataTypeInformation::get_method_owner)
443        else {
444            return operator.get_base_ref_expr().unwrap().clone();
445        };
446
447        let base = operator.get_base_ref_expr().unwrap();
448        AstFactory::create_cast_statement(
449            AstFactory::create_member_reference(
450                AstFactory::create_identifier(method_owner, SourceLocation::internal(), self.ids.next_id()),
451                None,
452                self.ids.next_id(),
453            ),
454            AstFactory::create_paren_expression(base.clone(), SourceLocation::internal(), self.ids.next_id()),
455            &SourceLocation::internal(),
456            self.ids.next_id(),
457        )
458    }

466    fn lower_statements(source: impl Into<SourceCode>, pous: &[&str]) -> Vec<String> {
467        let (_, project) = parse_and_annotate("unit-test", vec![source.into()]).unwrap();
468        let unit = project.units[0].get_unit();
469
470        let mut result = Vec::new();
471        for pou in pous {
472            result.push(format!("// Statements in {pou}"));
473            let statements = &unit.implementations.iter().find(|it| &it.name == pou).unwrap().statements;
474            let statements_str = statements.iter().map(|statement| statement.as_string()).collect::<Vec<_>>();
475
476            result.extend(statements_str);
477        }
478
479        result
480    }
