44/// helper macro that calls visit_statement for all given statements
45/// use like `visit_all_statements!(self, ctx, stmt1, stmt2, stmt3, ...)`
46macro_rules! visit_all_statements {
47     ($self:expr, $ctx:expr, $last:expr ) => {
48         $self.visit_statement($ctx, $last);
49     };
50
51     ($self:expr, $ctx:expr, $head:expr, $($tail:expr), +) => {
52       $self.visit_statement($ctx, $head);
53       visit_all_statements!($self, $ctx, $($tail),+)
54     };
55   }

57/// Context object passed by the visitor
58/// Denotes the current context of expressions (e.g. the current pou, a defined context, etc.)
59///
60/// Helper methods `qualifier`, `current_pou` and `lhs_pou` copy the current context and
61/// change one field.
62#[derive(Clone, Default)]
63pub struct VisitorContext<'s> {
64    pub id_provider: IdProvider,
65
66    /// the type_name of the context for a reference (e.g. `a.b` where `a`'s type is the context of `b`)
67    qualifier: Option<&'s str>,
68
69    /// optional context for references (e.g. `x` may mean `POU.x` if used inside `POU`'s body)
70    pou: Option<&'s str>,
71
72    /// special context of the left-hand-side of an assignment in call statements
73    /// Inside the left hand side of an assignment is in the context of the call's POU
74    /// `foo(a := a)` actually means: `foo(foo.a := POU.a)`
75    lhs: Option<&'s str>,
76
77    /// Set to true when dealing with a member reference at the left-hand side of an assignment. This is
78    /// needed to correctly annotate the type of the property we're dealing with, i.e. a get- or set-accessor.
79    property_set: bool,
80
81    /// true if the expression passed a constant-variable on the way
82    /// e.g. true for `x` if x is declared in a constant block
83    /// e.g. true for `a.b.c` if either a,b or c is declared in a constant block
84    constant: bool,
85
86    /// true if the visitor entered a body (so no declarations)
87    in_body: bool,
88
89    /// true if the visitor entered a control statement
90    in_control: bool,
91
92    // what's the current strategy for resolving
93    resolve_strategy: Vec<ResolvingStrategy>,
94}

178impl TypeAnnotator<'_> {
179    pub fn annotate_with_id(&mut self, id: AstId, annotation: StatementAnnotation) {
180        match &annotation {
181            StatementAnnotation::Function { return_type, qualified_name, call_name, .. } => {
182                let name = call_name.as_ref().unwrap_or(qualified_name);
183                self.dependencies.insert(Dependency::Call(name.to_string()));
184                self.dependencies.extend(self.get_datatype_dependencies(name, FxIndexSet::default()));
185                self.dependencies.extend(self.get_datatype_dependencies(return_type, FxIndexSet::default()));
186            }
187            StatementAnnotation::Program { qualified_name } => {
188                self.dependencies.insert(Dependency::Call(qualified_name.to_string()));
189                self.dependencies
190                    .extend(self.get_datatype_dependencies(qualified_name, FxIndexSet::default()));
191            }
192            StatementAnnotation::Variable {
193                resulting_type,
194                qualified_name,
195                argument_type,
196                auto_deref,
197                ..
198            } => {
199                if matches!(argument_type.get_inner(), VariableType::Global) {
200                    match auto_deref {
201                        Some(AutoDerefType::Alias(inner)) | Some(AutoDerefType::Reference(inner)) => {
202                            self.dependencies.insert(Dependency::Datatype(inner.to_owned()));
203                        }
204                        _ => (),
205                    };
206                    self.dependencies
207                        .extend(self.get_datatype_dependencies(resulting_type, FxIndexSet::default()));
208                    self.dependencies.insert(Dependency::Variable(qualified_name.to_string()));
209                }
210            }
211            StatementAnnotation::Value { resulting_type } => {
212                self.dependencies.insert(Dependency::Datatype(resulting_type.to_string()));
213            }
214            _ => (),
215        };
216        self.annotation_map.annotate_with_id(id, annotation);
217    }
218
219    pub fn annotate(&mut self, s: &AstNode, annotation: StatementAnnotation) {
220        self.annotate_with_id(s.get_id(), annotation);
221    }

223    fn visit_compare_statement(&mut self, ctx: &VisitorContext, statement: &AstNode) {
224        let AstStatement::BinaryExpression(BinaryExpression { operator, left, right }) = statement.get_stmt()
225        else {
226            return;
227        };
228        let mut ctx = ctx.clone();
229        let call_statement = match operator {
230            // a <> b expression is handled as Not(Equal(a,b))
231            Operator::NotEqual => AstFactory::create_not_expression(
232                self.create_typed_compare_call_statement(&mut ctx, &Operator::Equal, left, right, statement),
233                statement.get_location(),
234                ctx.id_provider.next_id(),
235            ),
236            // a <= b expression is handled as a = b OR a < b
237            Operator::LessOrEqual => AstFactory::create_or_expression(
238                self.create_typed_compare_call_statement(&mut ctx, &Operator::Equal, left, right, statement),
239                self.create_typed_compare_call_statement(&mut ctx, &Operator::Less, left, right, statement),
240            ),
241            // a >= b expression is handled as a = b OR a > b
242            Operator::GreaterOrEqual => AstFactory::create_or_expression(
243                self.create_typed_compare_call_statement(&mut ctx, &Operator::Equal, left, right, statement),
244                self.create_typed_compare_call_statement(
245                    &mut ctx,
246                    &Operator::Greater,
247                    left,
248                    right,
249                    statement,
250                ),
251            ),
252            _ => self.create_typed_compare_call_statement(&mut ctx, operator, left, right, statement),
253        };
254        self.visit_statement(&ctx, &call_statement);
255        self.update_expected_types(self.index.get_type_or_panic(typesystem::BOOL_TYPE), &call_statement);
256        self.annotate(statement, StatementAnnotation::ReplacementAst { statement: call_statement });
257        self.update_expected_types(self.index.get_type_or_panic(typesystem::BOOL_TYPE), statement);
258    }
259
260    /// tries to call one of the EQUAL_XXX, LESS_XXX, GREATER_XXX functions for the
261    /// given type (of left). The given operator has to be a comparison-operator
262    fn create_typed_compare_call_statement(
263        &self,
264        ctx: &mut VisitorContext,
265        operator: &Operator,
266        left: &AstNode,
267        right: &AstNode,
268        statement: &AstNode,
269    ) -> AstNode {
270        let left_type = self
271            .annotation_map
272            .get_type_hint(left, self.index)
273            .unwrap_or_else(|| self.annotation_map.get_type_or_void(left, self.index));
274        let cmp_function_name = crate::typesystem::get_equals_function_name_for(
275            left_type.get_type_information().get_name(),
276            operator,
277        );
278
279        cmp_function_name
280            .map(|name| {
281                AstFactory::create_call_to(
282                    name,
283                    vec![left.clone(), right.clone()],
284                    ctx.id_provider.next_id(),
285                    left.get_id(),
286                    &statement.get_location(),
287                )
288            })
289            .unwrap_or(AstFactory::create_empty_statement(
290                statement.get_location(),
291                ctx.id_provider.next_id(),
292            ))
293    }

295    pub fn annotate_call_statement(
296        &mut self,
297        operator: &AstNode,
298        parameters_stmt: Option<&AstNode>,
299        ctx: &VisitorContext,
300    ) {
301        let parameters = if let Some(parameters) = parameters_stmt {
302            self.visit_statement(ctx, parameters);
303            flatten_expression_list(parameters)
304        } else {
305            vec![]
306        };
307
308        let mut generics_candidates: FxHashMap<String, Vec<String>> = FxHashMap::default();
309        let mut params = vec![];
310        let mut parameters = if self.annotation_map.get(operator).is_some_and(|opt| opt.is_fnptr()) {
311            // When dealing with a function pointer (which are only supported in the context of methods and
312            // direct function block calls), the first argument will be a instance of the POU, e.g.
313            // `fnPtrToMyFbEcho^(instanceFb)`, hence we must skip the first argument as otherwise the
314            // remaining arguments will receive an incorrect type hint. Again, for example assume we have
315            // `fnPtrToMyFbEcho^(instanceFb, 'stringValue', 5)` and we do not skip the first argument. Then,
316            // `instanceFB` will have a type-hint of "STRING" and `stringValue` will have a type-hint on
317            // `DINT`. This then results in an error in the codegen. Somewhat "ugly" I have to admit and a
318            // better approach would be to lower method calls from `fbInstance.echo('stringValue', 5)` to
319            // `fbInstance.echo(fbInstance, 'stringValue', 5)` but this has to do for now
320            parameters[1..].iter()
321        } else {
322            parameters.iter()
323        };
324
325        // If we are dealing with an action call statement, we need to get the declared parameters from the parent POU in order
326        // to annotate them with the correct type hint.
327        let operator_qualifier = &self.get_call_name(operator);
328        let implementation = self.index.find_implementation_by_name(operator_qualifier);
329        let operator_qualifier = implementation.map(|it| it.get_type_name()).unwrap_or(operator_qualifier);
330        for m in self.index.get_declared_parameters(operator_qualifier).into_iter() {
331            if let Some(p) = parameters.next() {
332                let type_name = m.get_type_name();
333                if let Some((key, candidate)) =
334                    TypeAnnotator::get_generic_candidate(self.index, &self.annotation_map, type_name, p)
335                {
336                    generics_candidates.entry(key.to_string()).or_default().push(candidate.to_string())
337                } else {
338                    params.push((p, type_name.to_string(), m.get_location_in_parent()))
339                }
340            }
341        }
342
343        //We possibly did not consume all parameters, see if the variadic arguments are derivable
344        match self.index.find_pou(operator_qualifier) {
345            Some(pou) if pou.is_variadic() => {
346                //get variadic argument type, if it is generic, update the generic candidates
347                if let Some(variadic) = self.index.get_variadic_member(pou.get_name()) {
348                    let type_name = variadic.get_type_name();
349                    for parameter in parameters {
350                        if let Some((key, candidate)) = TypeAnnotator::get_generic_candidate(
351                            self.index,
352                            &self.annotation_map,
353                            type_name,
354                            parameter,
355                        ) {
356                            generics_candidates
357                                .entry(key.to_string())
358                                .or_default()
359                                .push(candidate.to_string())
360                        } else {
361                            // intrinsic type promotion for variadics in order to be compatible with the C standard.
362                            // see ISO/IEC 9899:1999, 6.5.2.2 Function calls (https://www.open-std.org/jtc1/sc22/wg14/www/docs/n1256.pdf)
363                            // or https://en.cppreference.com/w/cpp/language/implicit_conversion#Integral_promotion
364                            // for more about default argument promotion.
365
366                            // varargs without a type declaration will be annotated "VOID", so in order to check if a
367                            // promotion is necessary, we need to first check the type of each parameter. in the case of numerical
368                            // types, we promote if the type is smaller than double/i32 (except for booleans).
369                            let type_name = if let Some(data_type) =
370                                self.annotation_map.get_type(parameter, self.index)
371                            {
372                                match &data_type.information {
373                                    DataTypeInformation::Float { .. } => get_bigger_type(
374                                        data_type,
375                                        self.index.get_type_or_panic(LREAL_TYPE),
376                                        self.index,
377                                    )
378                                    .get_name(),
379                                    DataTypeInformation::Integer { .. }
380                                        if !data_type.information.is_bool()
381                                            && !data_type.information.is_character() =>
382                                    {
383                                        get_bigger_type(
384                                            data_type,
385                                            self.index.get_type_or_panic(DINT_TYPE),
386                                            self.index,
387                                        )
388                                        .get_name()
389                                    }
390                                    _ => type_name,
391                                }
392                            } else {
393                                // default to original type in case no type could be found
394                                // and let the validator handle situations that might lead here
395                                type_name
396                            };
397
398                            params.push((
399                                parameter,
400                                type_name.to_string(),
401                                variadic.get_location_in_parent(),
402                            ));
403                        }
404                    }
405                }
406            }
407            _ => {}
408        }
409
410        for (p, name, position) in params {
411            self.annotate_parameters(p, &name, position as usize);
412        }
413
414        // Attempt to resolve the generic signature here
415        self.update_generic_call_statement(
416            generics_candidates,
417            operator_qualifier,
418            operator,
419            parameters_stmt,
420            ctx.to_owned(),
421        );
422    }
423}

453pub enum StatementAnnotation {

483        /// The defined qualified name of the function

487        /// The call name of the function iff it differs from the qualified name (generics)

491    /// with a virtual table.

493        /// The return type name of the function pointer

495

498        //      __think__ it might be neccessary because of named arguments? Obviously this limits the use
499        //      case of function pointers because you'd always reference a concrete function rather than some
500        //      generic function such as `DINT(STRING, INT)`. If we ever decide to support function pointers
501        //      for real, we probably want to include another syntax similar to C with the limitation of not
502        //      supporting named arguments.
503        /// The name of the referenced function, e.g. `MyFb.myMethod` in `POINTER TO MyFb.MyMethod := ADR(...)`
504        qualified_name: String,
505    },
506    /// a reference to a type (e.g. `INT`)
507    Type {
508        type_name: String,
509    },
510    /// a reference to a program call or reference (e.g. `PLC_PRG`)
511    Program {
512        qualified_name: String,
513    },
514    ReplacementAst {
515        statement: AstNode,
516    },
517    /// a reference to a label in a POU
518    Label {
519        name: String,
520    },
521    /// A method override
522    Override {
523        // The qualified name of all definitions of this method in interfaces or base classes
524        definitions: Vec<MethodDeclarationType>,
525    },
526    Super {
527        // Name of the super class (EXTENDS)
528        name: String,
529        // name of all interfaces implemented by the class
530        interfaces: Vec<String>,
531    },
532    MethodDeclarations {
533        declarations: FxHashMap<String, Vec<MethodDeclarationType>>,
534    },
535    Property {
536        // TODO: Is this neccesary?
537        name: String,
538    },
539    #[default]
540    None,
541}

546pub enum MethodDeclarationType {
547    Abstract(QualifiedName),
548    Concrete(QualifiedName),
549}
550
551impl MethodDeclarationType {
552    pub fn abstract_method(name: &str) -> Self {
553        MethodDeclarationType::Abstract(name.into())
554    }
555
556    pub fn concrete_method(name: &str) -> Self {
557        MethodDeclarationType::Concrete(name.into())
558    }
559
560    pub fn is_abstract(&self) -> bool {
561        matches!(self, Self::Abstract(_))
562    }
563
564    pub fn is_concrete(&self) -> bool {
565        matches!(self, Self::Concrete(_))
566    }
567
568    pub fn get_qualified_name(&self) -> &str {
569        match self {
570            MethodDeclarationType::Abstract(name) | MethodDeclarationType::Concrete(name) => name,
571        }
572    }
573
574    pub fn get_qualifier(&self) -> &str {
575        match self {
576            MethodDeclarationType::Abstract(name) | MethodDeclarationType::Concrete(name) => {
577                name.rsplit_once('.').map(|(qualifier, _)| qualifier).unwrap_or(name)
578            }
579        }
580    }
581
582    pub fn get_flat_name(&self) -> &str {
583        match self {
584            MethodDeclarationType::Abstract(name) | MethodDeclarationType::Concrete(name) => {
585                name.rsplit_once('.').map(|(_, flat_name)| flat_name).unwrap_or(name)
586            }
587        }
588    }

591trait InheritanceAnnotationConverter {
592    /// Returns a [`StatementAnnotation::MethodDeclarations`] for all method declarations of this entry,
593    /// including all inherited methods.
594    fn get_method_declarations_annotation(&self, index: &Index) -> Option<StatementAnnotation> {
595        let mut declarations = FxHashMap::<String, Vec<MethodDeclarationType>>::default();
596        self.get_method_declaration_types(index).into_iter().for_each(|method| {
597            declarations.entry(method.get_flat_name().to_string()).or_default().push(method)
598        });
599        (!declarations.is_empty()).then_some(StatementAnnotation::MethodDeclarations { declarations })
600    }
601
602    /// Returns a [`StatementAnnotation::Override`] for all method overrides of this entry,
603    /// including all inherited, overridden methods.
604    fn get_method_overrides_annotation(&self, index: &Index) -> Option<StatementAnnotation> {
605        // TODO: lazy inheritance iterator
606        let definitions = self.get_method_overrides(index);
607        (!definitions.is_empty()).then_some(StatementAnnotation::create_override(definitions))
608    }
609
610    /// Returns all method declaration-types (abstract/concrete) of this entry, including all inherited methods.
611    fn get_method_declaration_types(&self, index: &Index) -> Vec<MethodDeclarationType>;
612
613    /// Returns all overridden method declaration-types (abstract/concrete) of this entry, including all inherited methods.
614    fn get_method_overrides(&self, index: &Index) -> Vec<MethodDeclarationType>;
615}

621            .map(|method| MethodDeclarationType::abstract_method(method.get_name()))
622            .collect()
623    }
624
625    fn get_method_overrides(&self, index: &Index) -> Vec<MethodDeclarationType> {
626        let derived_methods = self.get_derived_methods(index);
627        self.methods
628            .iter()
629            .filter(|method_name| {
630                derived_methods.iter().any(|derived_method| {
631                    derived_method.get_call_name()
632                        == method_name.rsplit_once(".").map(|(_, it)| it).unwrap_or_default()
633                })
634            })
635            .map(|method_name| MethodDeclarationType::abstract_method(method_name))
636            .collect()
637    }

640impl InheritanceAnnotationConverter for PouIndexEntry {
641    fn get_method_declaration_types(&self, index: &Index) -> Vec<MethodDeclarationType> {
642        match self {
643            PouIndexEntry::Program { name, .. }
644            | PouIndexEntry::FunctionBlock { name, .. }
645            | PouIndexEntry::Class { name, .. } => {
646                //Find all declared methods
647                index
648                    .get_methods(name)
649                    .into_iter()
650                    .map(|it| MethodDeclarationType::concrete_method(it.get_name()))
651                    .chain(self.get_interfaces().iter().flat_map(|ident| {
652                        index
653                            .find_interface(ident)
654                            .into_iter()
655                            .flat_map(|interface| interface.get_method_declaration_types(index))
656                    }))
657                    .collect()
658            }
659            _ => vec![],
660        }
661    }
662
663    fn get_method_overrides(&self, index: &Index) -> Vec<MethodDeclarationType> {
664        let PouIndexEntry::Method { ref parent_name, declaration_kind: kind, .. } = self else {
665            return vec![];
666        };
667
668        let mut overrides = vec![];
669        let method_name = self.get_call_name();
670
671        // Annotate as concrete override if a super-class also defines this method
672        if let Some(inherited_method) = index
673            .find_pou(parent_name)
674            .and_then(|pou| pou.get_super_class())
675            .and_then(|super_name| index.find_method(super_name, method_name))
676        {
677            overrides.push(MethodDeclarationType::concrete_method(inherited_method.get_name()));
678        };
679
680        let interfaces = match kind {
681            ast::DeclarationKind::Abstract => index
682                .find_interface(parent_name)
683                .map(|it| it.get_extensions().iter().map(|it| it.name.as_str()).collect()),
684            ast::DeclarationKind::Concrete => index.find_pou(parent_name).map(|it| it.get_interfaces()),
685        }
686        .unwrap_or_default();
687        // Annotate all implemented methods which were inherited from interfaces
688        interfaces.iter().for_each(|interface| {
689            if let Some(interface) = index.find_interface(interface) {
690                interface
691                    .get_methods(index)
692                    .iter()
693                    .filter(|it| it.get_call_name() == method_name)
694                    .for_each(|it| overrides.push(MethodDeclarationType::abstract_method(it.get_name())));
695            }
696        });
697
698        overrides
699    }

1111impl<'i> TypeAnnotator<'i> {
1112    /// constructs a new TypeAnnotater that works with the given index for type-lookups
1113    pub fn new(index: &'i Index) -> TypeAnnotator<'i> {
1114        TypeAnnotator {
1115            annotation_map: AnnotationMapImpl::new(),
1116            index,
1117            dependencies: FxIndexSet::default(),
1118            string_literals: StringLiterals { utf08: FxHashSet::default(), utf16: FxHashSet::default() },
1119            jumps_to_annotate: FxHashMap::default(),
1120            scopes: Scopes::global(),
1121        }
1122    }
1123
1124    /// annotates the given AST elements with the type-name resulting for the statements/expressions.
1125    /// Returns an AnnotationMap with the resulting types for all visited Statements. See `AnnotationMap`
1126    pub fn visit_unit(
1127        index: &Index,
1128        unit: &'i CompilationUnit,
1129        id_provider: IdProvider,
1130    ) -> (AnnotationMapImpl, FxIndexSet<Dependency>, StringLiterals) {
1131        let mut visitor = TypeAnnotator::new(index);
1132        let ctx = &VisitorContext {
1133            id_provider,
1134            resolve_strategy: ResolvingStrategy::default_scopes(),
1135            ..Default::default()
1136        };
1137
1138        for global_variable in unit.global_vars.iter().flat_map(|it| it.variables.iter()) {
1139            visitor.dependencies.insert(Dependency::Variable(global_variable.name.to_string()));
1140            visitor.visit_variable(ctx, global_variable);
1141        }
1142
1143        for pou in &unit.pous {
1144            visitor.visit_pou(ctx, pou);
1145        }
1146
1147        for interface in &unit.interfaces {
1148            visitor.visit_interface(interface);
1149        }
1150
1151        for t in &unit.user_types {
1152            visitor.visit_user_type_declaration(t, ctx);
1153        }
1154
1155        let body_ctx = ctx.enter_body();
1156        for i in &unit.implementations {
1157            visitor.dependencies.extend(visitor.get_datatype_dependencies(&i.name, FxIndexSet::default()));
1158            i.statements.iter().for_each(|s| visitor.visit_statement(&body_ctx.with_pou(i.name.as_str()), s));
1159        }
1160
1161        for config_variable in &unit.var_config {
1162            visitor.visit_statement(ctx, &config_variable.reference);
1163        }
1164
1165        // enum initializers may have been introduced by the visitor (indexer)
1166        // so we should try to resolve and type-annotate them here as well
1167        for enum_element in index
1168            .get_all_enum_variants()
1169            .iter()
1170            .filter(|it| it.is_in_unit(unit.file.get_name().unwrap_or_default()))
1171        {
1172            //Add to dependency map
1173            visitor.dependencies.insert(Dependency::Variable(enum_element.get_qualified_name().to_string()));
1174            if let Some((Some(statement), scope)) =
1175                enum_element.initial_value.map(|i| index.get_const_expressions().find_expression(&i))
1176            {
1177                if let Some(scope) = scope {
1178                    visitor.visit_statement(&ctx.with_pou(scope), statement);
1179                } else {
1180                    visitor.visit_statement(ctx, statement);
1181                }
1182            }
1183        }
1184
1185        //Labels have been added to the index, annotate jumps with their appropriate labels
1186        for (pou, jumps) in visitor.jumps_to_annotate {
1187            for (label, nodes) in jumps {
1188                for node in nodes {
1189                    if let Some(label) = visitor.annotation_map.new_index.get_label(&pou, &label) {
1190                        visitor
1191                            .annotation_map
1192                            .type_map
1193                            .insert(node, StatementAnnotation::Label { name: label.name.clone() });
1194                    }
1195                }
1196            }
1197        }
1198
1199        (visitor.annotation_map, visitor.dependencies, visitor.string_literals)
1200    }
1201
1202    fn visit_user_type_declaration(&mut self, user_data_type: &UserTypeDeclaration, ctx: &VisitorContext) {
1203        self.visit_data_type(ctx, &user_data_type.data_type);
1204        if let Some(name) = user_data_type.data_type.get_name() {
1205            let ctx = &ctx.with_pou(name);
1206            if let Some((initializer, name)) =
1207                user_data_type.initializer.as_ref().zip(user_data_type.data_type.get_name())
1208            {
1209                self.visit_statement(ctx, initializer);
1210
1211                // Update the type-hint for the initializer
1212                if let Some(right_type) = self.index.find_effective_type_by_name(name) {
1213                    self.update_expected_types(right_type, initializer);
1214                }
1215            }
1216        } else {
1217            unreachable!("datatype without a name");
1218        }
1219    }
1220
1221    fn visit_pou(&mut self, ctx: &VisitorContext, pou: &'i Pou) {
1222        self.dependencies.insert(Dependency::Datatype(pou.name.clone()));
1223        if let Some(Identifier { name, .. }) = &pou.super_class {
1224            self.dependencies.insert(Dependency::Datatype(name.to_string()));
1225        }
1226        self.annotate_pou(pou);
1227        let pou_ctx = ctx.with_pou(pou.name.as_str());

1280    /// updates the expected types of statement on the right side of an assignment
1281    /// e.g. x : ARRAY [0..1] OF BYTE := [2,3];
1282    /// note that the left side needs to be annotated before this call
1283    fn update_right_hand_side_expected_type(
1284        &mut self,
1285        ctx: &VisitorContext,
1286        annotated_left_side: &AstNode,
1287        right_side: &AstNode,
1288    ) {
1289        if let AstStatement::ParenExpression(expr) = &right_side.stmt {
1290            self.update_right_hand_side_expected_type(ctx, annotated_left_side, expr);
1291            self.inherit_annotations(right_side, expr);
1292        }
1293
1294        if let Some(expected_type) = self.annotation_map.get_type(annotated_left_side, self.index).cloned() {
1295            // for assignments on SubRanges check if there are range type check functions
1296            if let DataTypeInformation::SubRange { sub_range, .. } = expected_type.get_type_information() {
1297                if let Some(statement) = self
1298                    .index
1299                    .find_range_check_implementation_for(expected_type.get_type_information())
1300                    .map(|f| {
1301                        AstFactory::create_call_to_check_function_ast(
1302                            f.get_call_name(),
1303                            right_side.clone(),
1304                            *sub_range.clone(),
1305                            &annotated_left_side.get_location(),
1306                            ctx.id_provider.clone(),
1307                        )
1308                    })
1309                {
1310                    self.visit_call_statement(&statement, ctx);
1311                    self.update_right_hand_side(&expected_type, &statement);
1312                    self.annotation_map.annotate_hidden_function_call(right_side, statement);
1313                } else {
1314                    self.update_right_hand_side(&expected_type, right_side);
1315                }
1316            } else {
1317                self.update_right_hand_side(&expected_type, right_side);
1318            }
1319        }

1331    /// updates the expected types of statements on the right side of an assignment
1332    /// e.g. x : ARRAY [0..1] OF BYTE := [2,3];
1333    pub fn update_expected_types(&mut self, expected_type: &typesystem::DataType, statement: &AstNode) {
1334        //see if we need to dive into it
1335        match statement.get_stmt() {
1336            AstStatement::Literal(AstLiteral::Array(Array { elements: Some(elements) }), ..) => {
1337                //annotate the literal-array itself
1338                self.annotation_map
1339                    .annotate_type_hint(statement, StatementAnnotation::value(expected_type.get_name()));
1340                //TODO exprssionList and MultipliedExpressions are a mess!
1341                if matches!(
1342                    elements.get_stmt(),
1343                    AstStatement::ExpressionList(..) | AstStatement::MultipliedStatement(..)
1344                ) {
1345                    self.annotation_map
1346                        .annotate_type_hint(elements, StatementAnnotation::value(expected_type.get_name()));
1347                }
1348                //annotate the array's member elements with the array's inner type
1349                if let DataTypeInformation::Array { inner_type_name, .. } =
1350                    expected_type.get_type_information()
1351                {
1352                    if let Some(inner_type) = self.index.find_effective_type_by_name(inner_type_name) {
1353                        self.update_expected_types(inner_type, elements);
1354                    }
1355                }
1356            }
1357            AstStatement::Assignment(Assignment { left, right }, ..) => {
1358                // struct initialization (left := right)
1359                // find out left's type and update a type hint for right
1360                if let (
1361                    typesystem::DataTypeInformation::Struct { name: qualifier, .. },
1362                    Some(variable_name),
1363                ) = (expected_type.get_type_information(), left.as_ref().get_flat_reference_name())
1364                {
1365                    if let Some(v) = self.index.find_member(qualifier, variable_name) {
1366                        if let Some(target_type) = self.index.find_effective_type_by_name(v.get_type_name()) {
1367                            self.annotate(left.as_ref(), to_variable_annotation(v, self.index, false));
1368                            self.annotation_map.annotate_type_hint(
1369                                right.as_ref(),
1370                                StatementAnnotation::value(v.get_type_name()),
1371                            );
1372                            self.update_expected_types(target_type, right);
1373                        }
1374                    }
1375                }
1376            }
1377            AstStatement::MultipliedStatement(data, ..) => {
1378                // n(elements)
1379                //annotate the type to all multiplied elements
1380                for ele in AstNode::get_as_list(&data.element) {
1381                    self.update_expected_types(expected_type, ele);
1382                }
1383            }
1384            AstStatement::ParenExpression(expr) => {
1385                self.update_expected_types(expected_type, expr);
1386                self.inherit_annotations(statement, expr);
1387            }
1388            AstStatement::ExpressionList(expressions, ..) => {
1389                //annotate the type to all elements
1390                for ele in expressions {
1391                    self.update_expected_types(expected_type, ele);
1392                }
1393            }
1394            AstStatement::RangeStatement(data, ..) => {
1395                self.update_expected_types(expected_type, &data.start);
1396                self.update_expected_types(expected_type, &data.end);
1397            }
1398            AstStatement::Literal(AstLiteral::Integer { .. }, ..) => {
1399                //special case -> promote a literal-Integer directly, not via type-hint
1400                // (avoid later cast)
1401                if expected_type.get_type_information().is_float() {
1402                    self.annotate(statement, StatementAnnotation::value(expected_type.get_name()))
1403                } else if let DataTypeInformation::Array { inner_type_name, .. } =
1404                    expected_type.get_type_information()
1405                {
1406                    self.annotation_map
1407                        .annotate_type_hint(statement, StatementAnnotation::value(inner_type_name))
1408                } else {
1409                    //annotate the statement, whatever it is
1410                    self.annotation_map
1411                        .annotate_type_hint(statement, StatementAnnotation::value(expected_type.get_name()))
1412                }
1413            }
1414            AstStatement::Literal(AstLiteral::String { .. }, ..) | AstStatement::BinaryExpression { .. } => {
1415                // needed if we try to initialize an array with an expression-list
1416                // without we would annotate a false type this would leed to an error in expression_generator
1417                if let DataTypeInformation::Array { inner_type_name, .. } =
1418                    expected_type.get_type_information()
1419                {
1420                    self.annotation_map
1421                        .annotate_type_hint(statement, StatementAnnotation::value(inner_type_name))
1422                } else {
1423                    //annotate the statement, whatever it is
1424                    self.annotation_map
1425                        .annotate_type_hint(statement, StatementAnnotation::value(expected_type.get_name()))
1426                }
1427            }
1428
1429            _ => {
1430                //annotate the statement, whatever it is
1431                self.annotation_map
1432                    .annotate_type_hint(statement, StatementAnnotation::value(expected_type.get_name()))
1433            }
1434        }
1435    }

1437    fn visit_variable(&mut self, ctx: &VisitorContext, variable: &Variable) {
1438        self.visit_data_type_declaration(ctx, &variable.data_type_declaration);
1439        if let Some(initializer) = variable.initializer.as_ref() {
1440            // annotate a type-hint for the initializer, it should be the same type as the variable
1441            // e.g. x : BYTE := 7 + 3;  --> 7+3 should be cast into a byte
1442            if let Some(expected_type) = self
1443                .index
1444                .find_variable(ctx.qualifier.or(ctx.pou), &[variable.name.as_str()])
1445                .and_then(|ve| self.index.find_effective_type_by_name(ve.get_type_name()))
1446            {
1447                //Create a new context with the left operator being the target variable type, and the
1448                //right side being the local context
1449                let ctx = ctx.with_lhs(expected_type.get_name());
1450
1451                if initializer.is_default_value() {
1452                    // the default-placeholder must be annotated with the correct type,
1453                    // it will be replaced by the appropriate literal later
1454                    self.annotate(initializer, StatementAnnotation::value(expected_type.get_name()));
1455                } else {
1456                    self.visit_statement(&ctx, initializer);
1457                }
1458
1459                self.type_hint_for_variable_initializer(
1460                    initializer,
1461                    expected_type,
1462                    &ctx.with_lhs(&variable.name),
1463                );
1464            }
1465        }
1466    }

1468    fn type_hint_for_variable_initializer(
1469        &mut self,
1470        initializer: &AstNode,
1471        variable_ty: &typesystem::DataType,
1472        ctx: &VisitorContext,
1473    ) {
1474        if let AstStatement::ParenExpression(expr) = &initializer.stmt {
1475            self.type_hint_for_variable_initializer(expr, variable_ty, ctx);
1476            self.inherit_annotations(initializer, expr);
1477            return;
1478        }
1479
1480        self.annotation_map.annotate_type_hint(initializer, StatementAnnotation::value(&variable_ty.name));

1482
1483        self.type_hint_for_array_of_structs(variable_ty, initializer, ctx);

1486    fn type_hint_for_array_of_structs(
1487        &mut self,
1488        expected_type: &typesystem::DataType,
1489        statement: &AstNode,
1490        ctx: &VisitorContext,
1491    ) {
1492        match expected_type.get_type_information() {
1493            DataTypeInformation::Array { inner_type_name, .. } => {
1494                let inner_data_type = self.index.get_effective_type_or_void_by_name(inner_type_name);
1495                // TODO this seems wrong
1496                let ctx = ctx.with_qualifier(inner_data_type.get_name()).with_lhs(inner_data_type.get_name());
1497
1498                if !inner_data_type.get_type_information().is_struct() {
1499                    return;
1500                }
1501
1502                match statement.get_stmt() {
1503                    AstStatement::Literal(AstLiteral::Array(array)) => match array.elements() {
1504                        Some(elements) if elements.is_expression_list() || elements.is_paren() => {
1505                            self.type_hint_for_array_of_structs(expected_type, elements, &ctx)
1506                        }
1507
1508                        _ => (),
1509                    },
1510
1511                    AstStatement::ParenExpression(expression) => {
1512                        self.type_hint_for_array_of_structs(expected_type, expression, &ctx);
1513                        self.inherit_annotations(statement, expression);
1514                    }
1515
1516                    AstStatement::ExpressionList(expressions) => {
1517                        let name = inner_data_type.get_name();
1518                        let hint = StatementAnnotation::value(name);
1519
1520                        for expression in expressions {
1521                            self.annotation_map.annotate_type_hint(expression, hint.clone());
1522
1523                            self.visit_statement(&ctx, expression);
1524                            self.type_hint_for_array_of_structs(expected_type, expression, &ctx);
1525                        }
1526
1527                        // annotate the expression list as well
1528                        self.annotation_map.annotate_type_hint(statement, hint);
1529                    }
1530
1531                    AstStatement::Assignment(Assignment { left, right, .. }) if left.is_reference() => {
1532                        if let AstStatement::Literal(AstLiteral::Array(array)) = right.as_ref().get_stmt() {
1533                            let Some(elements) = array.elements() else { return };
1534
1535                            if let Some(datatype) = self.annotation_map.get_type(left, self.index).cloned() {
1536                                self.type_hint_for_array_of_structs(&datatype, elements, &ctx);
1537                            }
1538                        }
1539
1540                        // https://github.com/PLC-lang/rusty/issues/1019
1541                        if inner_data_type.information.is_struct() {
1542                            let name = inner_data_type.get_name();
1543                            let hint = StatementAnnotation::value(name);
1544                            self.annotation_map.annotate_type_hint(statement, hint);
1545                        }
1546                    }
1547
1548                    _ => (),
1549                }
1550            }
1551
1552            // We _should_ only land here when the variable itself isn't defined as an array (e.g. `foo : STRUCT1`).
1553            // The initializer of that variable might have an array of struct defined however, hence check it's elements.
1554            DataTypeInformation::Struct { members, .. } => {
1555                let flattened = ast::flatten_expression_list(statement);
1556                for (idx, member) in members.iter().enumerate() {
1557                    let data_type = self.index.get_effective_type_or_void_by_name(member.get_type_name());
1558                    if data_type.is_array() {
1559                        let Some(AstStatement::Assignment(data)) = flattened.get(idx).map(|it| it.get_stmt())
1560                        else {
1561                            continue;
1562                        };
1563                        self.type_hint_for_array_of_structs(data_type, &data.right, ctx);
1564                    }
1565                }
1566            }
1567
1568            _ => (),
1569        }
1570    }

1582    fn get_datatype_dependencies(
1583        &self,
1584        datatype_name: &str,
1585        resolved: FxIndexSet<Dependency>,
1586    ) -> FxIndexSet<Dependency> {
1587        let mut resolved_names = resolved;
1588        let Some(datatype) = self
1589            .index
1590            .find_type(datatype_name)
1591            .or_else(|| self.annotation_map.new_index.find_type(datatype_name))
1592        else {
1593            return resolved_names;
1594        };
1595        if resolved_names.insert(Dependency::Datatype(datatype.get_name().to_string())) {
1596            match datatype.get_type_information() {
1597                DataTypeInformation::Struct { members, source, .. } => {
1598                    for member in members {
1599                        resolved_names =
1600                            self.get_datatype_dependencies(member.get_type_name(), resolved_names);
1601                    }
1602
1603                    if let StructSource::Pou(PouType::Method { parent, .. }) = source {
1604                        resolved_names = self.get_datatype_dependencies(parent, resolved_names);
1605                    }
1606
1607                    resolved_names
1608                }
1609                DataTypeInformation::Array { inner_type_name, .. }
1610                | DataTypeInformation::Pointer { inner_type_name, .. } => {
1611                    resolved_names
1612                        .insert(Dependency::Datatype(datatype.get_type_information().get_name().to_string()));
1613                    self.get_datatype_dependencies(inner_type_name, resolved_names)
1614                }
1615                _ => {
1616                    let name =
1617                        self.index.get_intrinsic_type_information(datatype.get_type_information()).get_name();
1618                    self.get_datatype_dependencies(name, resolved_names)
1619                }
1620            }
1621        } else {
1622            resolved_names
1623        }
1624    }

1626    fn visit_data_type(&mut self, ctx: &VisitorContext, data_type: &DataType) {
1627        if let Some(name) = data_type.get_name() {
1628            self.dependencies.insert(Dependency::Datatype(name.to_string()));
1629        }
1630        match data_type {
1631            DataType::StructType { name: Some(name), variables, .. } => {
1632                let ctx = ctx.with_qualifier(name);
1633                variables.iter().for_each(|v| self.visit_variable(&ctx, v))
1634            }
1635            DataType::ArrayType { referenced_type, .. } => {
1636                self.visit_data_type_declaration(ctx, referenced_type)
1637            }
1638            DataType::VarArgs { referenced_type: Some(referenced_type), .. } => {
1639                self.visit_data_type_declaration(ctx, referenced_type.as_ref())
1640            }
1641            DataType::SubRangeType { referenced_type, bounds, .. } => {
1642                if let Some(bounds) = bounds {
1643                    if let Some(expected_type) = self.index.find_effective_type_by_name(referenced_type) {
1644                        self.visit_statement(ctx, bounds);
1645                        self.update_expected_types(expected_type, bounds);
1646                    }
1647                }
1648                self.dependencies
1649                    .extend(self.get_datatype_dependencies(referenced_type, FxIndexSet::default()));
1650            }
1651            DataType::EnumType { elements, name, .. } => {
1652                let ctx = name.as_ref().map(|n| ctx.with_lhs(n)).unwrap_or(ctx.clone());
1653                self.visit_statement(&ctx, elements);
1654            }
1655            DataType::PointerType { referenced_type, .. } => {
1656                self.visit_data_type_declaration(ctx, referenced_type.as_ref())
1657            }
1658            _ => {}
1659        }
1660    }

1666    /// This function is only really useful for [`AstStatement::ParenExpression`] where we would
1667    /// like to annotate the parenthese itself with whatever annotation the inner expression got.
1668    /// For example ((1 + 2)), `1 + 2` => DINT, but also `(...)` => DINT and `((...)))` => DINT
1669    fn inherit_annotations(&mut self, paren: &AstNode, inner: &AstNode) {

1671            self.annotate(paren, StatementAnnotation::value(&annotation.name))
1672        }
1673
1674        if let Some(annotation) = self.annotation_map.get_type_hint(inner, self.index) {
1675            self.annotation_map.annotate_type_hint(paren, StatementAnnotation::value(&annotation.name))
1676        }
1677    }

1679    /// annotate a control statement
1680    fn visit_statement_control(&mut self, ctx: &VisitorContext, statement: &AstNode) {
1681        match statement.get_stmt() {
1682            AstStatement::ParenExpression(expr) => {
1683                self.visit_statement(ctx, expr);
1684                self.inherit_annotations(statement, expr);
1685            }
1686            AstStatement::ControlStatement(control) => {
1687                match control {
1688                    AstControlStatement::If(stmt) => {
1689                        stmt.blocks.iter().for_each(|b| {
1690                            self.visit_statement(&ctx.enter_control(), b.condition.as_ref());
1691                            b.body.iter().for_each(|s| self.visit_statement(ctx, s));
1692                        });
1693                        stmt.else_block.iter().for_each(|e| self.visit_statement(ctx, e));
1694                    }
1695                    AstControlStatement::ForLoop(stmt) => {
1696                        visit_all_statements!(self, ctx, &stmt.counter, &stmt.start, &stmt.end);
1697                        if let Some(by_step) = &stmt.by_step {
1698                            self.visit_statement(ctx, by_step);
1699                        }
1700                        //Hint annotate start, end and step with the counter's real type
1701                        if let Some(type_name) = self
1702                            .annotation_map
1703                            .get_type(&stmt.counter, self.index)
1704                            .map(typesystem::DataType::get_name)
1705                        {
1706                            let annotation = StatementAnnotation::value(type_name);
1707                            self.annotation_map.annotate_type_hint(&stmt.start, annotation.clone());
1708                            self.annotation_map.annotate_type_hint(&stmt.end, annotation.clone());
1709                            if let Some(by_step) = &stmt.by_step {
1710                                self.annotation_map.annotate_type_hint(by_step, annotation);
1711                            }
1712                        }
1713                        stmt.body.iter().for_each(|s| self.visit_statement(ctx, s));
1714                    }
1715                    AstControlStatement::WhileLoop(stmt) | AstControlStatement::RepeatLoop(stmt) => {
1716                        self.visit_statement(&ctx.enter_control(), &stmt.condition);
1717                        stmt.body.iter().for_each(|s| self.visit_statement(ctx, s));
1718                    }
1719                    AstControlStatement::Case(stmt) => {
1720                        self.visit_statement(ctx, &stmt.selector);
1721                        let selector_type = self.annotation_map.get_type(&stmt.selector, self.index).cloned();
1722                        stmt.case_blocks.iter().for_each(|b| {
1723                            self.visit_statement(ctx, b.condition.as_ref());
1724                            if let Some(selector_type) = &selector_type {
1725                                self.update_expected_types(selector_type, b.condition.as_ref());
1726                            }
1727                            b.body.iter().for_each(|s| self.visit_statement(ctx, s));
1728                        });
1729                        stmt.else_block.iter().for_each(|s| self.visit_statement(ctx, s));
1730                    }
1731                }
1732            }
1733
1734            AstStatement::CaseCondition(condition, ..) => self.visit_statement(ctx, condition),
1735            _ => self.visit_statement_expression(ctx, statement),
1736        }
1737    }

1739    /// annotate an expression statement
1740    fn visit_statement_expression(&mut self, ctx: &VisitorContext, statement: &AstNode) {
1741        match statement.get_stmt() {
1742            AstStatement::This => {
1743                let name = match ctx.pou.and_then(|name| self.index.find_pou(name)) {
1744                    Some(PouIndexEntry::FunctionBlock { name, .. }) => name,
1745                    Some(
1746                        PouIndexEntry::Method { parent_name: name, .. }
1747                        | PouIndexEntry::Action { parent_name: name, .. },
1748                    ) if self.index.find_pou(name).is_some_and(|it| it.is_function_block()) => name,
1749                    _ => return,
1750                };
1751                let ptr_name = format!("{}.__THIS", name);
1752                if self
1753                    .index
1754                    .find_type(&ptr_name)
1755                    .or_else(|| self.annotation_map.new_index.find_type(&ptr_name))
1756                    .is_none()
1757                {
1758                    let information = DataTypeInformation::Pointer {
1759                        name: ptr_name.clone(),
1760                        inner_type_name: name.to_string(),
1761                        auto_deref: None,
1762                        type_safe: true,
1763                        is_function: false, // TODO(vosa): In general false, but THIS^() isn't?
1764                    };
1765                    let dt = crate::typesystem::DataType {
1766                        name: ptr_name.clone(),
1767                        initial_value: None,
1768                        information,
1769                        nature: TypeNature::Any,
1770                        location: SourceLocation::internal(),
1771                    };
1772                    self.annotation_map.new_index.register_type(dt);
1773                }
1774                self.annotate(statement, StatementAnnotation::value(ptr_name));
1775            }
1776            AstStatement::DirectAccess(data, ..) => {
1777                let ctx = VisitorContext { qualifier: None, ..ctx.clone() };
1778                visit_all_statements!(self, &ctx, &data.index);
1779                let access_type = get_direct_access_type(&data.access);
1780                self.annotate(statement, StatementAnnotation::value(access_type));
1781            }
1782            AstStatement::HardwareAccess(..) => {
1783                if let Some(annotation) = self.resolve_reference_expression(statement, None, ctx) {
1784                    self.annotate(statement, annotation.clone());
1785                }
1786            }
1787            AstStatement::BinaryExpression(data, ..) => {
1788                visit_all_statements!(self, ctx, &data.left, &data.right);
1789                let statement_type = {
1790                    let left_type = self
1791                        .annotation_map
1792                        .get_type_hint(&data.left, self.index)
1793                        .or_else(|| self.annotation_map.get_type(&data.left, self.index))
1794                        .and_then(|it| self.index.find_effective_type(it))
1795                        .unwrap_or_else(|| self.index.get_void_type());
1796                    // do not use for is_pointer() check
1797                    let l_intrinsic_type =
1798                        self.index.get_intrinsic_type_by_name(left_type.get_name()).get_type_information();
1799                    let right_type = self
1800                        .annotation_map
1801                        .get_type_hint(&data.right, self.index)
1802                        .or_else(|| self.annotation_map.get_type(&data.right, self.index))
1803                        .and_then(|it| self.index.find_effective_type(it))
1804                        .unwrap_or_else(|| self.index.get_void_type());
1805                    // do not use for is_pointer() check
1806                    let r_intrinsic_type =
1807                        self.index.get_intrinsic_type_by_name(right_type.get_name()).get_type_information();
1808
1809                    if l_intrinsic_type.is_numerical() && r_intrinsic_type.is_numerical() {
1810                        let bigger_type = if l_intrinsic_type.is_bool() && r_intrinsic_type.is_bool() {
1811                            left_type
1812                        } else {
1813                            let ty = if left_type.is_bit() && right_type.is_bit() {
1814                                right_type
1815                            } else {
1816                                self.index.get_type_or_panic(DINT_TYPE)
1817                            };
1818                            get_bigger_type(
1819                                get_bigger_type(left_type, right_type, self.index),
1820                                ty,
1821                                self.index,
1822                            )
1823                        };
1824
1825                        let target_name = if data.operator.is_comparison_operator() {
1826                            BOOL_TYPE.to_string()
1827                        } else {
1828                            bigger_type.get_name().to_string()
1829                        };
1830
1831                        let bigger_is_left = bigger_type != right_type;
1832                        let bigger_is_right = bigger_type != left_type;
1833
1834                        if bigger_is_left || bigger_is_right {
1835                            // if these types are different we need to update the 'other' type's annotation
1836                            let bigger_type = bigger_type.clone(); // clone here, so we release the borrow on self
1837                            if bigger_is_right {
1838                                self.update_expected_types(&bigger_type, &data.left);
1839                            }
1840                            if bigger_is_left {
1841                                self.update_expected_types(&bigger_type, &data.right);
1842                            }
1843                        }
1844
1845                        Some(target_name)
1846                    } else if left_type.get_type_information().is_pointer()
1847                        || right_type.get_type_information().is_pointer()
1848                    {
1849                        // get the target type of the binary expression
1850                        let target_type = if data.operator.is_comparison_operator() {
1851                            // compare instructions result in BOOL
1852                            // to generate valid IR code if a pointer is beeing compared to an integer
1853                            // we need to cast the int to the pointers size
1854                            if !left_type.get_type_information().is_pointer() {
1855                                let left_type = left_type.clone(); // clone here, so we release the borrow on self
1856                                self.annotate_to_pointer_size_if_necessary(&left_type, &data.left);
1857                            } else if !right_type.get_type_information().is_pointer() {
1858                                let right_type = right_type.clone(); // clone here, so we release the borrow on self
1859                                self.annotate_to_pointer_size_if_necessary(&right_type, &data.right);
1860                            }
1861                            BOOL_TYPE
1862                        } else if left_type.get_type_information().is_pointer() {
1863                            left_type.get_name()
1864                        } else {
1865                            right_type.get_name()
1866                        };
1867                        Some(target_type.to_string())
1868                    } else if data.operator.is_comparison_operator() {
1869                        //Annotate as the function call to XXX_EQUALS/LESS/GREATER..
1870                        self.visit_compare_statement(ctx, statement);
1871                        None
1872                    } else {
1873                        None
1874                    }
1875                };
1876
1877                if let Some(statement_type) = statement_type {
1878                    self.annotate(statement, StatementAnnotation::value(statement_type.clone()));
1879
1880                    // https://github.com/PLC-lang/rusty/issues/939: We rely on type-hints in order
1881                    // to identify `=` operations that have no effect (e.g. `foo = bar;`) hence
1882                    // type-hint the conditions of control statements to eliminate false-positives.
1883                    if ctx.in_control {
1884                        self.annotation_map
1885                            .annotate_type_hint(statement, StatementAnnotation::value(statement_type))
1886                    }
1887                }
1888            }
1889            AstStatement::UnaryExpression(data, ..) => {
1890                self.visit_statement(ctx, &data.value);
1891
1892                let statement_type = if data.operator == Operator::Minus {
1893                    let inner_type =
1894                        self.annotation_map.get_type_or_void(&data.value, self.index).get_type_information();
1895
1896                    //keep the same type but switch to signed
1897                    typesystem::get_signed_type(inner_type, self.index).map(|it| it.get_name().to_string())
1898                } else {
1899                    let inner_type = self
1900                        .annotation_map
1901                        .get_type_or_void(&data.value, self.index)
1902                        .get_type_information()
1903                        .get_name()
1904                        .to_string();
1905
1906                    Some(inner_type)
1907                };
1908
1909                if let Some(statement_type) = statement_type {
1910                    self.annotate(statement, StatementAnnotation::value(statement_type));
1911                }
1912            }
1913
1914            AstStatement::ExpressionList(expressions, ..) => {
1915                expressions.iter().for_each(|e| self.visit_statement(ctx, e))
1916            }
1917
1918            AstStatement::RangeStatement(data, ..) => {
1919                visit_all_statements!(self, ctx, &data.start, &data.end);
1920            }
1921            AstStatement::Assignment(data, ..) | AstStatement::RefAssignment(data, ..) => {
1922                self.visit_statement(&ctx.enter_control(), &data.right);
1923
1924                // if the LHS of the assignment is a member access, we need to update the context - when trying to resolve
1925                // a property, this means it must be a setter, not a getter
1926                let ctx = ctx.with_property_set(data.left.is_member_access());
1927
1928                if let Some(lhs) = ctx.lhs {
1929                    //special context for left hand side
1930                    self.visit_statement(&ctx.with_pou(lhs).with_lhs(lhs), &data.left);
1931                } else {
1932                    self.visit_statement(&ctx, &data.left);
1933                }
1934
1935                // give a type hint that we want the right side to be stored in the left's type
1936                self.update_right_hand_side_expected_type(&ctx, &data.left, &data.right);
1937            }
1938            AstStatement::OutputAssignment(data, ..) => {
1939                visit_all_statements!(self, ctx, &data.left, &data.right);
1940                if let Some(lhs) = ctx.lhs {
1941                    //special context for left hand side
1942                    self.visit_statement(&ctx.with_pou(lhs), &data.left);
1943                } else {
1944                    self.visit_statement(ctx, &data.left);
1945                }
1946                self.update_right_hand_side_expected_type(ctx, &data.left, &data.right);
1947            }
1948            AstStatement::CallStatement(..) => {
1949                self.visit_call_statement(statement, ctx);
1950            }
1951            AstStatement::ReferenceExpr(data, ..) => {
1952                self.visit_reference_expr(&data.access, data.base.as_deref(), statement, ctx);
1953            }
1954            AstStatement::ReturnStatement(ReturnStatement { condition }) => {
1955                if let Some(condition) = condition {
1956                    self.visit_statement(ctx, condition)
1957                }
1958            }
1959            AstStatement::LabelStatement(..) => {
1960                if let Some(pou) = ctx.pou {
1961                    self.annotation_map.new_index.add_label(pou, statement.into());
1962                }
1963            }
1964            AstStatement::JumpStatement(JumpStatement { condition, target }) => {
1965                self.visit_statement(ctx, condition);
1966                if let Some((name, pou)) = target.get_flat_reference_name().zip(ctx.pou) {
1967                    let pou = self.jumps_to_annotate.entry(pou.to_string()).or_default();
1968                    let jumps = pou.entry(name.to_string()).or_default();
1969                    jumps.push(statement.get_id());
1970                }
1971            }
1972            AstStatement::AllocationStatement(Allocation { name, reference_type }) => {
1973                let qualified_name =
1974                    if let Some(pou) = ctx.pou { format!("{}.{}", pou, name) } else { name.to_string() };
1975                self.scopes.enter(Scope::Local(Box::new(VariableIndexEntry::new(
1976                    name,
1977                    &qualified_name,
1978                    reference_type,
1979                    ArgumentType::ByVal(VariableType::Temp),
1980                    u32::MAX,
1981                    statement.get_location(),
1982                ))))
1983            }
1984            _ => {
1985                self.visit_statement_literals(ctx, statement);
1986            }
1987        }
1988    }

1990    fn visit_reference_expr(
1991        &mut self,
1992        access: &ast::ReferenceAccess,
1993        base: Option<&AstNode>,
1994        stmt: &AstNode,
1995        ctx: &VisitorContext,
1996    ) {
1997        // first resolve base
1998        if let Some(base) = base {
1999            self.visit_statement(ctx, base);
2000        };
2001
2002        match (
2003            access,
2004            base.and_then(|it| {
2005                self.annotation_map.get_type(it, self.index).map(|it| it.get_name().to_string())
2006            }),
2007        ) {
2008            (ReferenceAccess::Member(reference), qualifier) => {
2009                // uppdate the context's const field
2010                let new_ctx = base.map(|base| ctx.with_const(self.is_const_reference(base, ctx)));
2011                let new_ctx = &new_ctx.as_ref().unwrap_or(ctx).with_property_strategy();
2012
2013                if let Some(annotation) =
2014                    self.resolve_reference_expression(reference.as_ref(), qualifier.as_deref(), new_ctx)
2015                {
2016                    self.annotate(stmt, annotation.clone());
2017                    self.annotate(reference, annotation);
2018                    // if this was a vla, we update a typehint
2019                    if self.annotation_map.get_type(stmt, self.index).filter(|it| it.is_vla()).is_some() {
2020                        self.annotate_vla_hint(new_ctx, stmt);
2021                    }
2022                }
2023            }
2024
2025            (ReferenceAccess::Global(node), _) => {
2026                let ctx = ctx.with_resolving_strategy(vec![ResolvingStrategy::Global]);
2027                if let Some(annotation) = self.resolve_reference_expression(node, None, &ctx) {
2028                    // Annotate both the identifier and the statement, i.e. `.foo` and `foo`
2029                    self.annotate(stmt, annotation.clone());
2030                    self.annotate(node, annotation);
2031                }
2032            }
2033
2034            (ReferenceAccess::Cast(target), Some(qualifier)) => {
2035                // STRING#"abc"
2036                //  base
2037                if let Some(base_type) = base.and_then(|base| self.annotation_map.get_type(base, self.index))
2038                {
2039                    // if base is an enum, we need to look for members of this specific enum
2040                    let optional_enum_qualifier = Some(qualifier.as_str()).filter(|_| base_type.is_enum());
2041                    if ctx.is_in_a_body() {
2042                        accept_cast_string_literal(&mut self.string_literals, base_type, target);
2043                    }
2044
2045                    if let Some(annotation) = self.resolve_reference_expression(
2046                        target,
2047                        optional_enum_qualifier,
2048                        &ctx.with_resolving_strategy(vec![ResolvingStrategy::Variable]),
2049                    ) {
2050                        self.annotate(target.as_ref(), annotation);
2051                        self.annotate(stmt, StatementAnnotation::value(qualifier.as_str()));
2052
2053                        if let AstStatement::Literal(..) = target.get_stmt() {
2054                            // treate casted literals as the casted type
2055                            self.annotate(target.as_ref(), StatementAnnotation::value(qualifier.as_str()));
2056                        }
2057                    }
2058                }
2059            }
2060            (ReferenceAccess::Index(index), base) => {
2061                self.visit_statement(ctx, index);
2062                let Some(base) = base else { return };
2063                if let Some(inner_type) = self
2064                    .index
2065                    .find_effective_type_info(base.as_str())
2066                    .and_then(|t| t.get_inner_array_type_name())
2067                    .and_then(|it| self.index.find_effective_type_by_name(it).map(|it| it.get_name()))
2068                {
2069                    self.annotate(stmt, StatementAnnotation::value(inner_type))
2070                }
2071            }
2072            (ReferenceAccess::Deref, _) => {
2073                if let Some(DataTypeInformation::Pointer {
2074                    inner_type_name,
2075                    auto_deref: None,
2076                    is_function,
2077                    ..
2078                }) = base
2079                    .map(|base| self.annotation_map.get_type_or_void(base, self.index))
2080                    .map(|it| it.get_type_information())
2081                {
2082                    if let Some(inner_type) = self
2083                        .index
2084                        .find_effective_type_by_name(inner_type_name)
2085                        .or(self.annotation_map.new_index.find_effective_type_by_name(inner_type_name))
2086                    {
2087                        if ctx
2088                            .resolve_strategy
2089                            .first()
2090                            .is_some_and(|opt| *opt == ResolvingStrategy::FunctionsOnly)
2091                        {
2092                            // We might be dealing with a function pointer, e.g. `ptr^(...)`
2093                            if let Some(pou) = self.index.find_pou(&inner_type.name) {
2094                                // TODO(vosa): THIS^(), needs to be handled in the polymorphism PR but is
2095                                // ignored here (!stmt.is_this_deref())
2096                                if *is_function
2097                                    && (pou.is_method() | pou.is_function_block() && !stmt.is_this_deref())
2098                                {
2099                                    let name = pou.get_name();
2100                                    let return_type = pou.get_return_type().unwrap_or(VOID_INTERNAL_NAME);
2101
2102                                    self.annotate(stmt, StatementAnnotation::fnptr(name, return_type));
2103                                    return;
2104                                }
2105                            }
2106                        }
2107
2108                        self.annotate(stmt, StatementAnnotation::value(inner_type.get_name()))
2109                    }
2110                }
2111            }
2112            (ReferenceAccess::Address, _) => {
2113                if let Some(inner_type) = base
2114                    .map(|base| self.annotation_map.get_type_or_void(base, self.index).get_name().to_string())
2115                {
2116                    let ptr_type = add_pointer_type(&mut self.annotation_map.new_index, inner_type, true);
2117                    self.annotate(stmt, StatementAnnotation::value(ptr_type))
2118                }
2119            }
2120
2121            _ => (),
2122        }
2123    }

2240    fn visit_call_statement(&mut self, statement: &AstNode, ctx: &VisitorContext) {
2241        let (operator, parameters_stmt) = if let AstStatement::CallStatement(data, ..) = statement.get_stmt()

2368                    AstLiteral::Integer(value) => {

2371                    AstLiteral::Time { .. } => {
2372                        self.annotate(statement, StatementAnnotation::value(TIME_TYPE))

2375                        self.annotate(statement, StatementAnnotation::value(TIME_OF_DAY_TYPE));
2376                    }
2377                    AstLiteral::Date { .. } => {
2378                        self.annotate(statement, StatementAnnotation::value(DATE_TYPE));

2394                self.visit_statement(ctx, &data.element)

2493fn to_variable_annotation(
2494    v: &VariableIndexEntry,
2495    index: &Index,
2496    constant_override: bool,
2497) -> StatementAnnotation {
2498    let v_type = index.get_effective_type_or_void_by_name(v.get_type_name());
2499
2500    //see if this is an auto-deref variable
2501    let (effective_type_name, kind) = match (v_type.get_type_information(), v.is_return()) {
2502        (_, true) if v_type.is_aggregate_type() => {
2503            // treat a return-aggregate variable like an auto-deref pointer since it got
2504            // passed by-ref
2505            let kind =
2506                v_type.get_type_information().get_auto_deref_type().map(|it| it.into()).unwrap_or_default();
2507            (v_type.get_name().to_string(), Some(kind))
2508        }
2509        (DataTypeInformation::Pointer { inner_type_name, auto_deref: Some(deref), name, .. }, _) => {
2510            // real auto-deref pointer
2511            let kind = match deref {
2512                ast::AutoDerefType::Default => AutoDerefType::Default,
2513                ast::AutoDerefType::Alias => AutoDerefType::Alias(name.to_owned()),
2514                ast::AutoDerefType::Reference => AutoDerefType::Reference(name.to_owned()),
2515            };
2516
2517            (inner_type_name.clone(), Some(kind))
2518        }
2519        _ => (v_type.get_name().to_string(), None),
2520    };
2521
2522    StatementAnnotation::Variable {
2523        qualified_name: v.get_qualified_name().into(),
2524        resulting_type: effective_type_name,
2525        constant: v.is_constant() || constant_override,
2526        argument_type: v.get_declaration_type(),
2527        auto_deref: kind,
2528    }
2529}

2608#[derive(Clone, Copy, PartialEq, Eq)]
2609pub enum ResolvingStrategy {
2610    /// try to resolve a variable
2611    Variable,
2612
2613    /// try to resolve a global variable
2614    Global,
2615
2616    /// try to resolve a POU
2617    POU,
2618
2619    /// try to resolve a DataType
2620    DataType,
2621
2622    /// only consider EnumTypes
2623    EnumTypeOnly,
2624
2625    /// only consider functions
2626    FunctionsOnly,
2627
2628    /// also consider references to properties before they are lowered to actual method calls
2629    Property,
2630}
2631
2632impl ResolvingStrategy {
2633    /// scopes that can be used for general references. Will resolve to local/global
2634    /// variables, Pous or datatypes
2635    pub fn default_scopes() -> Vec<Self> {
2636        vec![ResolvingStrategy::Variable, ResolvingStrategy::POU, ResolvingStrategy::DataType]
2637    }
2638
2639    /// scopes intended for call-statement-operators
2640    fn call_operator_scopes() -> Vec<ResolvingStrategy> {
2641        let mut strategy = vec![ResolvingStrategy::FunctionsOnly];
2642        strategy.extend(Self::default_scopes());
2643        strategy
2644    }
2645
2646    fn resolve_property(
2647        index: &Index,
2648        name: &str,
2649        qualifier: &str,
2650        property_set: bool,
2651    ) -> Option<StatementAnnotation> {
2652        let accessor = if property_set { "set" } else { "get" };
2653        let name = format!("__{accessor}_{name}");
2654
2655        // if our current context is a method or action, we need to look for the property in the parent
2656        let qualifier =
2657            index.find_pou(qualifier).and_then(|pou| pou.get_parent_pou_name()).unwrap_or(qualifier);
2658
2659        index.find_method(qualifier, &name).map(|_| StatementAnnotation::Property { name })
2660    }
2661
2662    /// tries to resolve the given name using the reprsented scope
2663    /// - `name` the name to resolve
2664    /// - `qualifier` an optional qualifier to prefix to the name,
2665    ///   if the qualifier is present, this method only resolves to targets with this qualifier
2666    /// - `index` the index to perform the lookups on
2667    fn resolve_name(
2668        &self,
2669        name: &str,
2670        qualifier: Option<&str>,
2671        index: &Index,
2672        ctx: &VisitorContext,
2673        scopes: &Scopes,
2674    ) -> Option<StatementAnnotation> {
2675        match self {
2676            // try to resolve the name as a variable
2677            ResolvingStrategy::Variable => {
2678                if let Some(qualifier) = qualifier {
2679                    // look for variable, enum with name "qualifier.name"
2680                    scopes
2681                        // instance.foo; // Resolver -> referencexpr ->
2682                        .find_member(index, qualifier, name)
2683                        .or_else(|| index.find_enum_variant(qualifier, name))
2684                        .map(|it| to_variable_annotation(it, index, it.is_constant() || ctx.constant))
2685                } else {
2686                    // look for member variable with name "pou.name"
2687                    // then try fopr a global variable called "name"
2688                    ctx.pou
2689                        .and_then(|pou| scopes.find_member(index, pou, name))
2690                        .or_else(|| index.find_global_variable(name))
2691                        .map(|g| to_variable_annotation(g, index, g.is_constant()))
2692                }
2693            }
2694            ResolvingStrategy::Global => {
2695                index.find_global_variable(name).map(|g| to_variable_annotation(g, index, g.is_constant()))
2696            }
2697            // try to resolve the name as POU/Action/Method
2698            ResolvingStrategy::POU => {
2699                if let Some(qualifier) = qualifier {
2700                    // look for Pou/Action with name "qualifier.name"
2701                    index
2702                        .find_pou(format!("{qualifier}.{name}").as_str())
2703                        .or_else(|| index.find_method(qualifier, name))
2704                        .map(|action| action.into())
2705                } else {
2706                    // look for Pou with name "name"
2707                    index.find_pou(name).and_then(|pou| to_pou_annotation(pou, index)).or_else(|| {
2708                        ctx.pou.and_then(|pou|
2709                                // retry with local pou as qualifier
2710                                //Use the type name of the pou in case we are resolving a
2711                                //neighboring action
2712                                index.find_pou(pou).map(|pou| pou.get_container())
2713                                .and_then(|pou|ResolvingStrategy::POU.resolve_name(name, Some(pou), index, ctx, scopes)))
2714                    })
2715                }
2716            }
2717            // try to resolve the name as a datatype
2718            ResolvingStrategy::DataType => {
2719                if qualifier.is_none() {
2720                    // look for datatype with name "name"
2721                    index
2722                        .find_type(name)
2723                        .map(|data_type| StatementAnnotation::data_type(data_type.get_name()))
2724                } else {
2725                    // there are no qualified types
2726                    None
2727                }
2728            }
2729            // try to resolve the name as an enum-type
2730            ResolvingStrategy::EnumTypeOnly => {
2731                if qualifier.is_none() {
2732                    // look for enum-tyoe with name "name"
2733                    index
2734                        .find_type(name)
2735                        .filter(|it| it.is_enum())
2736                        .map(|enum_type| StatementAnnotation::data_type(enum_type.get_name()))
2737                } else {
2738                    // there are no qualified types
2739                    None
2740                }
2741            }
2742            // try to resolve this name as a function
2743            ResolvingStrategy::FunctionsOnly => {
2744                if qualifier.is_none() {
2745                    // look for function with name "name"
2746                    index.find_pou(name).filter(|it| it.is_function()).map(|pou| pou.into())
2747                } else {
2748                    // there are no qualified functions
2749                    None
2750                }
2751            }
2752            // try to resolve this name as a property
2753            ResolvingStrategy::Property => qualifier
2754                .or(ctx.pou)
2755                .and_then(|qualifier| Self::resolve_property(index, name, qualifier, ctx.property_set)),
2756        }
2757    }
2758}

2760/// registers the resulting string-literal if the given literal is a String with a different encoding than what is
2761/// requested from given cast_type (e.g. STRING#"i am utf16", or WSTRING#'i am utf8')
2762fn accept_cast_string_literal(
2763    literals: &mut StringLiterals,
2764    cast_type: &typesystem::DataType,
2765    literal: &AstNode,
2766) {

2768    let Some(&AstLiteral::String(StringValue { ref value, is_wide })) = try_from!(literal, AstLiteral) else {
2769        return;
2770    };
2771    match (cast_type.get_type_information(), is_wide) {
2772        (DataTypeInformation::String { encoding: StringEncoding::Utf8, .. }, true)
2773        | (DataTypeInformation::String { encoding: StringEncoding::Utf16, .. }, false) => {
2774            // re-register the string-literal in the opposite encoding
2775            if is_wide {
2776                literals.utf08.insert(value.into());
2777            } else {
2778                literals.utf16.insert(value.into());
2779            }
2780        }

2782    }
