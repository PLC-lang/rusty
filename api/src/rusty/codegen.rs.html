97    /// constructs a new code-generator that generates CompilationUnits into a module with the given module_name
98    pub fn new(
99        context: &'ink CodegenContext,
100        root: Option<&Path>,
101        file_marker: FileMarker,
102        optimization_level: OptimizationLevel,
103        debug_level: DebugLevel,
104        online_change: OnlineChange,
105        target: &Target,
106    ) -> CodeGen<'ink> {
107        let module_location = file_marker.get_name().unwrap_or_default();
108        let module = context.create_module(module_location);
109        module.set_source_file_name(module_location);
110
111        // Initialize all targets
112        let initialization_config = &InitializationConfig::default();
113        inkwell::targets::Target::initialize_all(initialization_config);
114        let triple = target.get_target_triple();
115
116        // Create target from triple
117        let target_obj =
118            inkwell::targets::Target::from_triple(&triple).expect("Failed to create target from triple");
119
120        // Create a target machine with default options
121        let target_machine = target_obj
122            .create_target_machine(
123                &triple,
124                "generic", // CPU features - generic for portability
125                "",        // CPU features - empty string for default
126                optimization_level.into(),
127                inkwell::targets::RelocMode::Default,
128                inkwell::targets::CodeModel::Default,
129            )
130            .expect("Failed to create target machine");
131
132        // Get the data layout from the target machine and set the module's data layout and triple
133        let target_data = target_machine.get_target_data();
134        module.set_data_layout(&target_data.get_data_layout());
135        module.set_triple(&triple);
136
137        let debug_level = if file_marker.is_internal() { DebugLevel::None } else { debug_level };
138        let debug = debug::DebugBuilderEnum::new(context, &module, root, optimization_level, debug_level);
139        CodeGen { module, debug, module_location: module_location.to_string(), online_change }
140    }

142    pub fn generate_llvm_index(
143        &mut self,
144        context: &'ink CodegenContext,
145        annotations: &AstAnnotations,
146        literals: &StringLiterals,
147        dependencies: &FxIndexSet<Dependency>,
148        global_index: &Index,
149        got_layout: &Mutex<HashMap<String, u64>>,
150    ) -> Result<LlvmTypedIndex<'ink>, CodegenError> {
151        let llvm = Llvm::new(context, context.create_builder());
152        let mut index = LlvmTypedIndex::default();
153        //Generate types index, and any global variables associated with them.
154        let llvm_type_index = data_type_generator::generate_data_types(
155            &llvm,
156            &mut self.debug,
157            dependencies,
158            global_index,
159            annotations,
160        )?;
161        index.merge(llvm_type_index);
162
163        let mut variable_generator = VariableGenerator::new(
164            &self.module,
165            &llvm,
166            global_index,
167            annotations,
168            &index,
169            &mut self.debug,
170            &self.online_change,
171        );
172
173        //Generate global variables
174        let llvm_gv_index =
175            variable_generator.generate_global_variables(dependencies, &self.module_location)?;
176        index.merge(llvm_gv_index);
177
178        // Build our GOT layout here. We need to find all the names for globals, programs, and
179        // functions and assign them indices in the GOT, taking into account prior indices.
180        let program_globals =
181            global_index.get_program_instances().into_iter().fold(Vec::new(), |mut acc, p| {
182                acc.push(p.get_name().to_owned());
183                acc.push(p.get_qualified_name().to_owned());
184                acc.push(format!("{}_instance", p.get_name()));
185                acc
186            });
187
188        let functions = global_index.get_pous().values().filter_map(|p| match p {
189            PouIndexEntry::Function { name, linkage: LinkageType::Internal, is_generated: false, .. }
190            | PouIndexEntry::FunctionBlock { name, linkage: LinkageType::Internal, .. } => {
191                Some(String::from(name))
192            }
193            _ => None,
194        });
195        let all_names = global_index
196            .get_globals()
197            .values()
198            .map(VariableIndexEntry::get_qualified_name)
199            .map(String::from)
200            .chain(program_globals)
201            .chain(functions)
202            .map(|s| s.to_lowercase())
203            .map(|s| (crate::index::get_initializer_name(&s), s))
204            .fold(Vec::new(), |mut acc, (s, s1)| {
205                acc.push(s);
206                acc.push(s1);
207                acc
208            });
209
210        if self.online_change.is_enabled() {
211            let got_entries = &mut *got_layout.lock().unwrap();
212
213            let mut new_symbols = Vec::new();
214            let mut new_got_entries = HashMap::new();
215            let mut new_got = HashMap::new();
216
217            for name in all_names {
218                if let Some(idx) = got_entries.get(&name.to_string()) {
219                    new_got_entries.insert(name.to_string(), *idx);
220                    index.associate_got_index(&name, *idx)?;
221                    new_got.insert(*idx, name.to_string());
222                } else {
223                    new_symbols.push(name.to_string());
224                }
225            }
226
227            // Put any names that weren't there last time in any free space in the GOT.
228            let mut idx: u64 = 0;
229            for name in &new_symbols {
230                while new_got.contains_key(&idx) {
231                    idx += 1;
232                }
233                new_got_entries.insert(name.to_string(), idx);
234                index.associate_got_index(name, idx)?;
235                new_got.insert(idx, name.to_string());
236            }
237
238            // Construct our GOT as a new global array. We initialise this array in the loader code.
239            let got_size: u32 = new_got
240                .keys()
241                .max()
242                .map_or(0, |m| *m + 1)
243                .try_into()
244                .expect("the computed custom GOT size is too large");
245
246            let ptr_ty = llvm.context.i8_type().ptr_type(AddressSpace::default());
247            let empty_got = ptr_ty
248                .const_array(vec![ptr_ty.const_null(); got_size as usize].as_slice())
249                .as_basic_value_enum();
250            let custom_got_ty =
251                BasicTypeEnum::ArrayType(Llvm::get_array_type(BasicTypeEnum::PointerType(ptr_ty), got_size));
252
253            let custom_got = llvm.create_global_variable(&self.module, "__custom_got", custom_got_ty);
254            custom_got.set_linkage(inkwell::module::Linkage::WeakODR);
255            custom_got.set_initial_value(Some(empty_got), custom_got_ty);
256
257            *got_entries = new_got_entries;
258        }
259
260        //Generate opaque functions for implementations and associate them with their types
261        let llvm = Llvm::new(context, context.create_builder());
262        let llvm_impl_index = pou_generator::generate_implementation_stubs(
263            &self.module,
264            llvm,
265            dependencies,
266            global_index,
267            annotations,
268            &index,
269            &mut self.debug,
270            &self.online_change,
271        )?;
272        let llvm = Llvm::new(context, context.create_builder());
273        index.merge(llvm_impl_index);
274        let llvm_values_index = pou_generator::generate_global_constants_for_pou_members(
275            &self.module,
276            &llvm,
277            dependencies,
278            global_index,
279            annotations,
280            &index,
281            &self.module_location,
282        )?;
283        index.merge(llvm_values_index);
284
285        //Generate constants for string-literal
286        //generate literals but first sort, so we get reproducable builds
287        let mut utf08s = literals.utf08.iter().map(String::as_str).collect::<Vec<&str>>();
288        utf08s.sort_unstable();
289        for (idx, literal) in utf08s.into_iter().enumerate() {
290            let len = literal.len() + 1;
291            let data_type = llvm.context.i8_type().array_type(len as u32);
292            let literal_variable = llvm.create_global_variable(
293                &self.module,
294                format!("utf08_literal_{idx}").as_str(),
295                data_type.as_basic_type_enum(),
296            );
297            let initializer = llvm.create_const_utf8_string(literal, len)?;
298            literal_variable.make_constant().make_private().set_initializer(&initializer);
299
300            index.associate_utf08_literal(literal, literal_variable);
301        }
302        //generate literals but first sort, so we get reproducable builds
303        let mut utf16s = literals.utf16.iter().map(String::as_str).collect::<Vec<&str>>();
304        utf16s.sort_unstable();
305        for (idx, literal) in utf16s.into_iter().enumerate() {
306            let len = literal.len() + 1;
307            let data_type = llvm.context.i16_type().array_type(len as u32);
308            let literal_variable = llvm.create_global_variable(
309                &self.module,
310                format!("utf16_literal_{idx}").as_str(),
311                data_type.as_basic_type_enum(),
312            );
313            let initializer = llvm.create_const_utf16_string(literal, literal.len() + 1)?;
314            literal_variable.make_constant().make_private().set_initializer(&initializer);
315
316            index.associate_utf16_literal(literal, literal_variable);
317        }
318
319        Ok(index)
320    }

322    /// generates all TYPEs, GLOBAL-sections and POUs of the given CompilationUnit
323    pub fn generate(
324        self,
325        context: &'ink CodegenContext,
326        unit: &CompilationUnit,
327        annotations: &AstAnnotations,
328        global_index: &Index,
329        llvm_index: LlvmTypedIndex<'ink>,
330    ) -> Result<GeneratedModule<'ink>, CodegenError> {
331        //generate all pous
332        let llvm = Llvm::new(context, context.create_builder());
333        let pou_generator =
334            PouGenerator::new(llvm, global_index, annotations, &llvm_index, &self.online_change);
335
336        //Generate the POU stubs in the first go to make sure they can be referenced.
337        for implementation in &unit.implementations {
338            //Don't generate external or generic functions
339            if let Some(entry) = global_index.find_pou(implementation.name.as_str()) {
340                if !entry.is_generic() && entry.get_linkage() != &LinkageType::External {
341                    pou_generator.generate_implementation(implementation, &self.debug)?;
342                }
343            }
344        }
345
346        let location = (&unit.file).into();
347
348        self.debug.finalize();
349        log::trace!("{}", self.module.to_string());
350
351        #[cfg(feature = "verify")]
352        {
353            self.module.verify().map_err(|it| CodegenError::from(it)).map(|_| GeneratedModule {
354                module: self.module,
355                location,
356                engine: RefCell::new(None),
357            })
358        }
359
360        #[cfg(not(feature = "verify"))]
361        Ok(GeneratedModule { module: self.module, location, engine: RefCell::new(None) })
362    }

396    /// Persists the module into the disk based on output and target requirments
397    /// If an object file should be generated, all optimizations will be executed on the object
398    pub fn persist(
399        &self,
400        output_dir: Option<&Path>,
401        output_name: &str,
402        format: FormatOption,
403        target: &Target,
404        optimization_level: OptimizationLevel,
405    ) -> Result<PathBuf, CodegenError> {
406        let output = Self::get_output_file(output_dir, output_name, target);
407        //ensure output exists
408        if let Some(parent) = output.parent() {
409            std::fs::create_dir_all(parent)?;
410        }
411        match format {
412            FormatOption::Object | FormatOption::Relocatable => {
413                self.persist_as_static_obj(output, target, optimization_level)
414            }
415            FormatOption::PIC | FormatOption::Shared | FormatOption::Static => {
416                self.persist_to_shared_pic_object(output, target, optimization_level)
417            }
418            FormatOption::NoPIC => self.persist_to_shared_object(output, target, optimization_level),
419            FormatOption::Bitcode => self.persist_to_bitcode(output),
420            FormatOption::IR => self.persist_to_ir(output),
421        }
422    }
423
424    fn get_output_file(output_dir: Option<&Path>, output_name: &str, target: &Target) -> PathBuf {
425        let output_dir = output_dir.map(Path::to_path_buf).unwrap_or_else(|| PathBuf::from(""));
426        let output = if let Some(name) = target.try_get_name() {
427            output_dir.join(name).join(output_name)
428        } else {
429            output_dir.join(output_name)
430        };
431        output
432    }

438    ///
439    /// Compiles the given source into an object file and saves it in output
440    ///
441    fn persist_to_obj(
442        &self,
443        output: PathBuf,
444        reloc: RelocMode,
445        target: &Target,
446        optimization_level: OptimizationLevel,
447    ) -> Result<PathBuf, CodegenError> {
448        let initialization_config = &InitializationConfig::default();
449        inkwell::targets::Target::initialize_all(initialization_config);
450
451        let triple = target.get_target_triple();
452
453        let target = inkwell::targets::Target::from_triple(&triple)?;
454        let machine = target
455            .create_target_machine(
456                &triple,
457                //TODO : Add cpu features as optionals
458                "generic", //TargetMachine::get_host_cpu_name().to_string().as_str(),
459                "",        //TargetMachine::get_host_cpu_features().to_string().as_str(),
460                optimization_level.into(),
461                reloc,
462                CodeModel::Default,
463            )
464            .ok_or_else(|| CodegenError::new("Cannot create target machine.", SourceLocation::undefined()));
465
466        //Make sure all parents exist
467        if let Some(parent) = output.parent() {
468            std::fs::create_dir_all(parent)?;
469        }
470        ////Run the passes
471        machine
472            .and_then(|it| {
473                self.module
474                    .run_passes(optimization_level.opt_params(), &it, PassBuilderOptions::create())
475                    .map_err(Into::into)
476                    .and_then(|_| {
477                        it.write_to_file(&self.module, FileType::Object, output.as_path()).map_err(Into::into)
478                    })
479            })
480            .map(|_| output)
481    }

578    ///
579    /// Runs the function given by `name` inside the compiled module.
580    /// Returns the value returned by calling the function
581    ///
582    pub fn run<T, U>(&self, name: &str, params: &mut T) -> U {
583        let engine = self.get_execution_engine();
584
585        unsafe {
586            let main: JitFunction<MainFunction<T, U>> = engine.get_function(name).unwrap();
587            let main_t_ptr = &mut *params as *mut _;
588
589            main.call(main_t_ptr)
590        }
591    }

605    pub fn add_global_function_mapping(&self, function_name: &str, local_function: usize) {
606        let engine = self.get_execution_engine();
607        if let Some(function) = self.module.get_function(function_name) {
608            engine.add_global_mapping(&function, local_function);

610            log::debug!("Function {} does not exist", function_name);

616        if let Some(function) = self.module.get_global(global_name) {
617            engine.add_global_mapping(&function, local_var);
618        } else {
619            log::debug!("Global {} does not exist", global_name);
620        }

623    fn get_execution_engine(&self) -> ExecutionEngine<'ink> {
624        if let Some(engine) = self.engine.borrow().as_ref() {
625            return engine.clone();
626        }
627        //Create engine
628        let engine = self.module.create_jit_execution_engine(inkwell::OptimizationLevel::None).unwrap();
629        *self.engine.borrow_mut() = Some(engine.clone());
630        engine
631    }

682    }

691impl Display for CodegenError {
692    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
693        match self {
694            CodegenError::GenericError(err, location) => {
695                write!(f, "{}", err)?;
696                if !location.is_undefined() {
697                    write!(f, " at {}", location)?;
698                }
699                Ok(())
700            }
701            CodegenError::DiagnosticError(err) => write!(f, "{}", err),
702            CodegenError::BuilderError(err) => write!(f, "{}", err),
703            CodegenError::IoError(err) => write!(f, "{}", err),
704        }
705    }
706}
707
708impl From<CodegenError> for Diagnostic {
709    fn from(err: CodegenError) -> Self {
710        if let CodegenError::DiagnosticError(diagnostic) = err {
711            return diagnostic;
712        }
713        Diagnostic::new(format!("Builder error: {err}"))
714            .with_error_code("E002")
715            .with_internal_error(anyhow::anyhow!(err))
716    }
717}

719impl std::error::Error for CodegenError {
720    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
721        match self {
722            CodegenError::GenericError(_, _) => None,
723            CodegenError::BuilderError(err) => Some(err),
724            CodegenError::DiagnosticError(err) => Some(err),
725            CodegenError::IoError(err) => Some(err),
726        }
727    }
728}
