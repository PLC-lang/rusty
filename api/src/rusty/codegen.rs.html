98    /// constructs a new code-generator that generates CompilationUnits into a module with the given module_name
99    #[allow(clippy::too_many_arguments)]
100    pub fn new(
101        context: &'ink CodegenContext,
102        root: Option<&Path>,
103        file_marker: FileMarker,
104        optimization_level: OptimizationLevel,
105        debug_level: DebugLevel,
106        debug_prefix_maps: &[(PathBuf, PathBuf)],
107        debug_compilation_dir: Option<&Path>,
108        online_change: OnlineChange,
109        target: &Target,
110        build_info: Option<&str>,
111    ) -> CodeGen<'ink> {
112        let module_location = file_marker.get_name().unwrap_or_default();
113        let module = context.create_module(module_location);
114        module.set_source_file_name(module_location);
115
116        // Initialize all targets
117        let initialization_config = &InitializationConfig::default();
118        inkwell::targets::Target::initialize_all(initialization_config);
119        let triple = target.get_target_triple();
120
121        // Create target from triple
122        let target_obj =
123            inkwell::targets::Target::from_triple(&triple).expect("Failed to create target from triple");
124
125        // Create a target machine with default options
126        let target_machine = target_obj
127            .create_target_machine(
128                &triple,
129                "generic", // CPU features - generic for portability
130                "",        // CPU features - empty string for default
131                optimization_level.into(),
132                inkwell::targets::RelocMode::Default,
133                inkwell::targets::CodeModel::Default,
134            )
135            .expect("Failed to create target machine");
136
137        // Get the data layout from the target machine and set the module's data layout and triple
138        let target_data = target_machine.get_target_data();
139        module.set_data_layout(&target_data.get_data_layout());
140        module.set_triple(&triple);
141
142        // Emit `!llvm.ident` so the module self-identifies (`.comment` on ELF
143        // after link). Skipped when `None`.
144        if let Some(info) = build_info {
145            let ident_string = context.metadata_string(info);
146            let ident_node = context.metadata_node(&[ident_string.into()]);
147            module
148                .add_global_metadata("llvm.ident", &ident_node)
149                .expect("add_global_metadata on llvm.ident accepts a node value");
150        }
151
152        let debug_level = if file_marker.is_internal() { DebugLevel::None } else { debug_level };
153        let debug = debug::DebugBuilderEnum::new(
154            context,
155            &module,
156            root,
157            optimization_level,
158            debug_level,
159            debug_prefix_maps,
160            debug_compilation_dir,
161        );
162        CodeGen { module, debug, module_location: module_location.to_string(), online_change }
163    }

165    #[allow(clippy::too_many_arguments)]
166    pub fn generate_llvm_index(
167        &mut self,
168        context: &'ink CodegenContext,
169        annotations: &AstAnnotations,
170        literals: &StringLiterals,
171        dependencies: &FxIndexSet<Dependency>,
172        global_index: &Index,
173        got_layout: &Mutex<HashMap<String, u64>>,
174        constructors_only: bool,
175    ) -> Result<LlvmTypedIndex<'ink>, CodegenError> {
176        let llvm = Llvm::new(
177            context,
178            context.create_builder(),
179            self.module.get_triple().as_str().to_string_lossy().into_owned(),
180        );
181        let mut index = LlvmTypedIndex::default();
182        //Generate types index, and any global variables associated with them.
183        let llvm_type_index = data_type_generator::generate_data_types(
184            &llvm,
185            &mut self.debug,
186            dependencies,
187            global_index,
188            annotations,
189        )?;
190        index.merge(llvm_type_index);
191
192        let mut variable_generator = VariableGenerator::new(
193            &self.module,
194            &llvm,
195            global_index,
196            annotations,
197            &index,
198            &mut self.debug,
199            &self.online_change,
200        );
201
202        //Generate global variables
203        let llvm_gv_index =
204            variable_generator.generate_global_variables(dependencies, &self.module_location)?;
205        index.merge(llvm_gv_index);
206
207        // Build our GOT layout here. We need to find all the names for globals, programs, and
208        // functions and assign them indices in the GOT, taking into account prior indices.
209        let program_globals =
210            global_index.get_program_instances().into_iter().fold(Vec::new(), |mut acc, p| {
211                acc.push(p.get_name().to_owned());
212                acc.push(p.get_qualified_name().to_owned());
213                acc.push(format!("{}_instance", p.get_name()));
214                acc
215            });
216
217        let functions = global_index.get_pous().values().filter_map(|p| match p {
218            PouIndexEntry::Function { name, linkage: LinkageType::Internal, is_generated: false, .. }
219            | PouIndexEntry::FunctionBlock { name, linkage: LinkageType::Internal, .. } => {
220                Some(String::from(name))
221            }
222            _ => None,
223        });
224        let all_names = global_index
225            .get_globals()
226            .values()
227            .map(VariableIndexEntry::get_qualified_name)
228            .map(String::from)
229            .chain(program_globals)
230            .chain(functions)
231            .map(|s| s.to_lowercase())
232            .map(|s| (crate::index::get_initializer_name(&s), s))
233            .fold(Vec::new(), |mut acc, (s, s1)| {
234                acc.push(s);
235                acc.push(s1);
236                acc
237            });
238
239        if self.online_change.is_enabled() {
240            let got_entries = &mut *got_layout.lock().unwrap();
241
242            let mut new_symbols = Vec::new();
243            let mut new_got_entries = HashMap::new();
244            let mut new_got = HashMap::new();
245
246            for name in all_names {
247                if let Some(idx) = got_entries.get(&name.to_string()) {
248                    new_got_entries.insert(name.to_string(), *idx);
249                    index.associate_got_index(&name, *idx)?;
250                    new_got.insert(*idx, name.to_string());
251                } else {
252                    new_symbols.push(name.to_string());
253                }
254            }
255
256            // Put any names that weren't there last time in any free space in the GOT.
257            let mut idx: u64 = 0;
258            for name in &new_symbols {
259                while new_got.contains_key(&idx) {
260                    idx += 1;
261                }
262                new_got_entries.insert(name.to_string(), idx);
263                index.associate_got_index(name, idx)?;
264                new_got.insert(idx, name.to_string());
265            }
266
267            // Construct our GOT as a new global array. We initialise this array in the loader code.
268            let got_size: u32 = new_got
269                .keys()
270                .max()
271                .map_or(0, |m| *m + 1)
272                .try_into()
273                .expect("the computed custom GOT size is too large");
274
275            let ptr_ty = llvm.context.ptr_type(AddressSpace::default());
276            let empty_got = ptr_ty
277                .const_array(vec![ptr_ty.const_null(); got_size as usize].as_slice())
278                .as_basic_value_enum();
279            let custom_got_ty =
280                BasicTypeEnum::ArrayType(Llvm::get_array_type(BasicTypeEnum::PointerType(ptr_ty), got_size));
281
282            let custom_got = llvm.create_global_variable(&self.module, "__custom_got", custom_got_ty);
283            custom_got.set_linkage(inkwell::module::Linkage::WeakODR);
284            custom_got.set_initial_value(Some(empty_got), custom_got_ty);
285
286            *got_entries = new_got_entries;
287        }
288
289        //Generate opaque functions for implementations and associate them with their types
290        let llvm = Llvm::new(
291            context,
292            context.create_builder(),
293            self.module.get_triple().as_str().to_string_lossy().into_owned(),
294        );
295        let llvm_impl_index = pou_generator::generate_implementation_stubs(
296            &self.module,
297            llvm,
298            dependencies,
299            global_index,
300            annotations,
301            &index,
302            &mut self.debug,
303            &self.online_change,
304            &self.module_location,
305            constructors_only,
306        )?;
307        let llvm = Llvm::new(
308            context,
309            context.create_builder(),
310            self.module.get_triple().as_str().to_string_lossy().into_owned(),
311        );
312        index.merge(llvm_impl_index);
313        let llvm_values_index = pou_generator::generate_global_constants_for_pou_members(
314            &self.module,
315            &llvm,
316            dependencies,
317            global_index,
318            annotations,
319            &index,
320            &self.module_location,
321        )?;
322        index.merge(llvm_values_index);
323
324        //Generate constants for string-literal
325        //generate literals but first sort, so we get reproducable builds
326        let mut utf08s = literals.utf08.iter().map(String::as_str).collect::<Vec<&str>>();
327        utf08s.sort_unstable();
328        for (idx, literal) in utf08s.into_iter().enumerate() {
329            let len = literal.len() + 1;
330            let data_type = llvm.context.i8_type().array_type(len as u32);
331            let literal_variable = llvm.create_global_variable(
332                &self.module,
333                format!("utf08_literal_{idx}").as_str(),
334                data_type.as_basic_type_enum(),
335            );
336            let initializer = llvm.create_const_utf8_string(literal, len)?;
337            literal_variable.make_constant().make_private().set_initializer(&initializer);
338
339            index.associate_utf08_literal(literal, literal_variable);
340        }
341        //generate literals but first sort, so we get reproducable builds
342        let mut utf16s = literals.utf16.iter().map(String::as_str).collect::<Vec<&str>>();
343        utf16s.sort_unstable();
344        for (idx, literal) in utf16s.into_iter().enumerate() {
345            let len = literal.len() + 1;
346            let data_type = llvm.context.i16_type().array_type(len as u32);
347            let literal_variable = llvm.create_global_variable(
348                &self.module,
349                format!("utf16_literal_{idx}").as_str(),
350                data_type.as_basic_type_enum(),
351            );
352            let initializer = llvm.create_const_utf16_string(literal, literal.len() + 1)?;
353            literal_variable.make_constant().make_private().set_initializer(&initializer);
354
355            index.associate_utf16_literal(literal, literal_variable);
356        }
357
358        Ok(index)
359    }

361    /// generates all TYPEs, GLOBAL-sections and POUs of the given CompilationUnit
362    pub fn generate(
363        self,
364        context: &'ink CodegenContext,
365        unit: &CompilationUnit,
366        annotations: &AstAnnotations,
367        global_index: &Index,
368        llvm_index: LlvmTypedIndex<'ink>,
369        constructors_only: bool,
370    ) -> Result<GeneratedModule<'ink>, CodegenError> {
371        //generate all pous
372        let llvm = Llvm::new(
373            context,
374            context.create_builder(),
375            self.module.get_triple().as_str().to_string_lossy().into_owned(),
376        );
377        let pou_generator =
378            PouGenerator::new(llvm, global_index, annotations, &llvm_index, &self.online_change);
379
380        //Generate the POU stubs in the first go to make sure they can be referenced.
381        for implementation in &unit.implementations {
382            //Don't generate external or generic functions
383            if let Some(entry) = global_index.find_pou(implementation.name.as_str()) {
384                if !entry.is_generic() && !entry.get_linkage().is_external_or_included() {
385                    if constructors_only
386                        && !matches!(implementation.pou_type, PouType::Init | PouType::ProjectInit)
387                    {
388                        continue;
389                    }
390                    let noop_debug = DebugBuilderEnum::None;
391                    // Use noop debug ONLY for compiler-generated types (Init/ProjectInit POUs)
392                    // User-defined functions should retain full debug info even if they have internal locations
393                    // (internal locations can result from lowering/transformation of user code)
394                    let is_compiler_generated =
395                        matches!(implementation.pou_type, PouType::Init | PouType::ProjectInit);
396                    let debug = if is_compiler_generated {
397                        log::debug!(
398                            "Using noop debug for {} (compiler-generated type: {:?})",
399                            implementation.name,
400                            implementation.pou_type
401                        );
402                        &noop_debug
403                    } else {
404                        log::debug!(
405                            "Using full debug for {} (user-defined, type: {:?})",
406                            implementation.name,
407                            implementation.pou_type
408                        );
409                        &self.debug
410                    };
411                    pou_generator.generate_implementation(implementation, debug).map_err(|err| {
412                        if is_compiler_generated {
413                            // Add context about which constructor failed so the user can
414                            // trace the error back to the originating type declaration.
415                            let type_name =
416                                implementation.name.strip_suffix("__ctor").unwrap_or(&implementation.name);
417                            let note = Diagnostic::new(format!(
418                                "error occurred while generating initialization code for type '{type_name}'"
419                            ));
420                            Diagnostic::from(err).with_sub_diagnostic(note).into()
421                        } else {
422                            err
423                        }
424                    })?;
425                }
426            }
427        }
428
429        let location = (&unit.file).into();
430
431        self.debug.finalize();
432        log::trace!("{}", self.module.to_string());
433
434        #[cfg(feature = "verify")]
435        {
436            self.module.verify().map_err(|it| CodegenError::from(it)).map(|_| GeneratedModule {
437                module: self.module,
438                location,
439                engine: RefCell::new(None),
440            })
441        }
442
443        #[cfg(not(feature = "verify"))]
444        Ok(GeneratedModule { module: self.module, location, engine: RefCell::new(None) })
445    }
446}

479    /// Persists the module into the disk based on output and target requirments
480    /// If an object file should be generated, all optimizations will be executed on the object
481    pub fn persist(
482        &self,
483        output_dir: Option<&Path>,
484        output_name: &str,
485        format: FormatOption,
486        relocation_preference: RelocationPreference,
487        target: &Target,
488        optimization_level: OptimizationLevel,
489    ) -> Result<PathBuf, CodegenError> {
490        let output = Self::get_output_file(output_dir, output_name, target);
491        //ensure output exists
492        if let Some(parent) = output.parent() {
493            std::fs::create_dir_all(parent)?;
494        }
495        match format {
496            FormatOption::Object | FormatOption::Relocatable => match relocation_preference {
497                RelocationPreference::Pic => {
498                    self.persist_to_shared_pic_object(output, target, optimization_level)
499                }
500                RelocationPreference::Default | RelocationPreference::NoPic => {
501                    self.persist_as_static_obj(output, target, optimization_level)
502                }
503            },
504            FormatOption::PIC | FormatOption::Shared => match relocation_preference {
505                RelocationPreference::NoPic => {
506                    self.persist_to_shared_object(output, target, optimization_level)
507                }
508                RelocationPreference::Default | RelocationPreference::Pic => {
509                    self.persist_to_shared_pic_object(output, target, optimization_level)
510                }
511            },
512            FormatOption::Static => match relocation_preference {
513                RelocationPreference::NoPic => self.persist_as_static_obj(output, target, optimization_level),
514                RelocationPreference::Default | RelocationPreference::Pic => {
515                    self.persist_to_shared_pic_object(output, target, optimization_level)
516                }
517            },
518            FormatOption::NoPIC => match relocation_preference {
519                RelocationPreference::Pic => {
520                    self.persist_to_shared_pic_object(output, target, optimization_level)
521                }
522                RelocationPreference::Default | RelocationPreference::NoPic => {
523                    self.persist_to_shared_object(output, target, optimization_level)
524                }
525            },
526            FormatOption::Bitcode => self.persist_to_bitcode(output),
527            FormatOption::IR => self.persist_to_ir(output),
528        }
529    }

531    fn get_output_file(output_dir: Option<&Path>, output_name: &str, target: &Target) -> PathBuf {
532        let output_dir = output_dir.map(Path::to_path_buf).unwrap_or_else(|| PathBuf::from(""));
533        let output = if let Some(name) = target.try_get_name() {
534            output_dir.join(name).join(output_name)
535        } else {
536            output_dir.join(output_name)
537        };
538        output
539    }

545    ///
546    /// Compiles the given source into an object file and saves it in output
547    ///
548    fn persist_to_obj(
549        &self,
550        output: PathBuf,
551        reloc: RelocMode,
552        target: &Target,
553        optimization_level: OptimizationLevel,
554    ) -> Result<PathBuf, CodegenError> {
555        let initialization_config = &InitializationConfig::default();
556        inkwell::targets::Target::initialize_all(initialization_config);
557
558        let triple = target.get_target_triple();
559
560        let target = inkwell::targets::Target::from_triple(&triple)?;
561        let mut machine = target
562            .create_target_machine(
563                &triple,
564                //TODO : Add cpu features as optionals
565                "generic", //TargetMachine::get_host_cpu_name().to_string().as_str(),
566                "",        //TargetMachine::get_host_cpu_features().to_string().as_str(),
567                optimization_level.into(),
568                reloc,
569                CodeModel::Default,
570            )
571            .ok_or_else(|| CodegenError::new("Cannot create target machine.", SourceLocation::undefined()))?;
572
573        machine.use_init_array(true);
574
575        //Make sure all parents exist
576        if let Some(parent) = output.parent() {
577            std::fs::create_dir_all(parent)?;
578        }
579        ////Run the passes
580        self.module
581            .run_passes(optimization_level.opt_params(), &machine, PassBuilderOptions::create())
582            .and_then(|_| machine.write_to_file(&self.module, FileType::Object, output.as_path()))?;
583        Ok(output)
584    }

634    ///
635    /// Persists the given LLVM module into a bitcode file
636    ///
637    /// # Arguments
638    ///
639    /// * `codegen` - the genated LLVM module to persist
640    /// * `output` - the location on disk to save the output
641    pub fn persist_to_bitcode(&self, output: PathBuf) -> Result<PathBuf, CodegenError> {
642        if self.module.write_bitcode_to_path(&output) {
643            Ok(output)
644        } else {
645            Err(CodegenError::new("Could not write bitcode to file", SourceLocation::undefined()))
646        }

681    ///
682    /// Runs the function given by `name` inside the compiled module.
683    /// Returns the value returned by calling the function
684    ///
685    pub fn run<T, U>(&self, name: &str, params: &mut T) -> U {
686        let engine = self.get_execution_engine();
687        engine.run_static_constructors();
688        unsafe {
689            let main: JitFunction<MainFunction<T, U>> = engine.get_function(name).unwrap();
690            main.call(&mut *params as *mut _)
691        }
692    }

700        engine.run_static_constructors();
701        unsafe {
702            let main: JitFunction<MainEmptyFunction<U>> = engine.get_function(name).unwrap();
703            main.call()
704        }

726        if let Some(engine) = self.engine.borrow().as_ref() {

728        }

741#[derive(Debug)]
742pub enum CodegenError {
743    GenericError(String, SourceLocation),
744    IoError(std::io::Error),
745    DiagnosticError(Diagnostic),
746    BuilderError(BuilderError),
747}
748
749impl From<BuilderError> for CodegenError {
750    fn from(err: BuilderError) -> Self {
751        CodegenError::BuilderError(err)
752    }
753}
754
755impl From<std::io::Error> for CodegenError {
756    fn from(err: std::io::Error) -> Self {
757        CodegenError::IoError(err)
758    }
759}
760
761impl From<Diagnostic> for CodegenError {
762    fn from(err: Diagnostic) -> Self {
763        CodegenError::DiagnosticError(err)
764    }
765}
766
767impl From<String> for CodegenError {
768    fn from(err: String) -> Self {
769        log::debug!("Codegen Error(from): {err}");
770        CodegenError::GenericError(err, SourceLocation::undefined())
771    }
772}
773
774impl CodegenError {
775    pub fn new<T, U>(msg: T, location: U) -> Self
776    where
777        T: Into<String>,
778        U: Into<SourceLocation>,
779    {
780        let msg = msg.into();
781        log::debug!("Codegen Error(new): {msg}");
782        CodegenError::GenericError(msg, location.into())
783    }
784
785    fn missing_function(location: SourceLocation) -> Self {
786        CodegenError::new("Cannot generate code outside of function context.", location)
787    }
788}
789
790impl From<LLVMString> for CodegenError {
791    fn from(err: LLVMString) -> Self {
792        log::debug!("LLVM Error: {}", err.to_string_lossy());
793        CodegenError::GenericError(err.to_string_lossy().to_string(), SourceLocation::undefined())
794    }
795}
796
797impl Display for CodegenError {
798    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
799        match self {
800            CodegenError::GenericError(err, location) => {
801                write!(f, "{}", err)?;
802                if !location.is_undefined() {
803                    write!(f, " at {}", location)?;
804                }
805                Ok(())
806            }
807            CodegenError::DiagnosticError(err) => write!(f, "{}", err),
808            CodegenError::BuilderError(err) => write!(f, "{}", err),
809            CodegenError::IoError(err) => write!(f, "{}", err),
810        }
811    }
812}
813
814impl From<CodegenError> for Diagnostic {
815    fn from(err: CodegenError) -> Self {
816        if let CodegenError::DiagnosticError(diagnostic) = err {
817            return diagnostic;
818        }
819        Diagnostic::new(format!("Builder error: {err}"))
820            .with_error_code("E002")
821            .with_internal_error(anyhow::anyhow!(err))
822    }
823}
824
825impl std::error::Error for CodegenError {
826    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
827        match self {
828            CodegenError::GenericError(_, _) => None,
829            CodegenError::BuilderError(err) => Some(err),
830            CodegenError::DiagnosticError(err) => Some(err),
831            CodegenError::IoError(err) => Some(err),
832        }
833    }
834}
