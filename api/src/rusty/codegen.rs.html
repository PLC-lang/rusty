94    /// constructs a new code-generator that generates CompilationUnits into a module with the given module_name
95    pub fn new(
96        context: &'ink CodegenContext,
97        root: Option<&Path>,
98        file_marker: FileMarker,
99        optimization_level: OptimizationLevel,
100        debug_level: DebugLevel,
101        online_change: OnlineChange,
102        target: &Target,
103    ) -> CodeGen<'ink> {
104        let module_location = file_marker.get_name().unwrap_or_default();
105        let module = context.create_module(module_location);
106        module.set_source_file_name(module_location);
107
108        // Initialize all targets
109        let initialization_config = &InitializationConfig::default();
110        inkwell::targets::Target::initialize_all(initialization_config);
111        let triple = target.get_target_triple();
112
113        // Create target from triple
114        let target_obj =
115            inkwell::targets::Target::from_triple(&triple).expect("Failed to create target from triple");
116
117        // Create a target machine with default options
118        let target_machine = target_obj
119            .create_target_machine(
120                &triple,
121                "generic", // CPU features - generic for portability
122                "",        // CPU features - empty string for default
123                optimization_level.into(),
124                inkwell::targets::RelocMode::Default,
125                inkwell::targets::CodeModel::Default,
126            )
127            .expect("Failed to create target machine");
128
129        // Get the data layout from the target machine and set the module's data layout and triple
130        let target_data = target_machine.get_target_data();
131        module.set_data_layout(&target_data.get_data_layout());
132        module.set_triple(&triple);
133
134        let debug_level = if file_marker.is_internal() { DebugLevel::None } else { debug_level };
135        let debug = debug::DebugBuilderEnum::new(context, &module, root, optimization_level, debug_level);
136        CodeGen { module, debug, module_location: module_location.to_string(), online_change }
137    }

139    pub fn generate_llvm_index(
140        &mut self,
141        context: &'ink CodegenContext,
142        annotations: &AstAnnotations,
143        literals: &StringLiterals,
144        dependencies: &FxIndexSet<Dependency>,
145        global_index: &Index,
146        got_layout: &Mutex<HashMap<String, u64>>,
147    ) -> Result<LlvmTypedIndex<'ink>, Diagnostic> {
148        let llvm = Llvm::new(context, context.create_builder());
149        let mut index = LlvmTypedIndex::default();
150        //Generate types index, and any global variables associated with them.
151        let llvm_type_index = data_type_generator::generate_data_types(
152            &llvm,
153            &mut self.debug,
154            dependencies,
155            global_index,
156            annotations,
157        )?;
158        index.merge(llvm_type_index);
159
160        let mut variable_generator = VariableGenerator::new(
161            &self.module,
162            &llvm,
163            global_index,
164            annotations,
165            &index,
166            &mut self.debug,
167            &self.online_change,
168        );
169
170        //Generate global variables
171        let llvm_gv_index =
172            variable_generator.generate_global_variables(dependencies, &self.module_location)?;
173        index.merge(llvm_gv_index);
174
175        // Build our GOT layout here. We need to find all the names for globals, programs, and
176        // functions and assign them indices in the GOT, taking into account prior indices.
177        let program_globals =
178            global_index.get_program_instances().into_iter().fold(Vec::new(), |mut acc, p| {
179                acc.push(p.get_name().to_owned());
180                acc.push(p.get_qualified_name().to_owned());
181                acc.push(format!("{}_instance", p.get_name()));
182                acc
183            });
184
185        let functions = global_index.get_pous().values().filter_map(|p| match p {
186            PouIndexEntry::Function { name, linkage: LinkageType::Internal, is_generated: false, .. }
187            | PouIndexEntry::FunctionBlock { name, linkage: LinkageType::Internal, .. } => {
188                Some(String::from(name))
189            }
190            _ => None,
191        });
192        let all_names = global_index
193            .get_globals()
194            .values()
195            .map(VariableIndexEntry::get_qualified_name)
196            .map(String::from)
197            .chain(program_globals)
198            .chain(functions)
199            .map(|s| s.to_lowercase())
200            .map(|s| (crate::index::get_initializer_name(&s), s))
201            .fold(Vec::new(), |mut acc, (s, s1)| {
202                acc.push(s);
203                acc.push(s1);
204                acc
205            });
206
207        if self.online_change.is_enabled() {
208            let got_entries = &mut *got_layout.lock().unwrap();
209
210            let mut new_symbols = Vec::new();
211            let mut new_got_entries = HashMap::new();
212            let mut new_got = HashMap::new();
213
214            for name in all_names {
215                if let Some(idx) = got_entries.get(&name.to_string()) {
216                    new_got_entries.insert(name.to_string(), *idx);
217                    index.associate_got_index(&name, *idx)?;
218                    new_got.insert(*idx, name.to_string());
219                } else {
220                    new_symbols.push(name.to_string());
221                }
222            }
223
224            // Put any names that weren't there last time in any free space in the GOT.
225            let mut idx: u64 = 0;
226            for name in &new_symbols {
227                while new_got.contains_key(&idx) {
228                    idx += 1;
229                }
230                new_got_entries.insert(name.to_string(), idx);
231                index.associate_got_index(name, idx)?;
232                new_got.insert(idx, name.to_string());
233            }
234
235            // Construct our GOT as a new global array. We initialise this array in the loader code.
236            let got_size: u32 = new_got
237                .keys()
238                .max()
239                .map_or(0, |m| *m + 1)
240                .try_into()
241                .expect("the computed custom GOT size is too large");
242
243            let ptr_ty = llvm.context.i8_type().ptr_type(AddressSpace::default());
244            let empty_got = ptr_ty
245                .const_array(vec![ptr_ty.const_null(); got_size as usize].as_slice())
246                .as_basic_value_enum();
247            let custom_got_ty =
248                BasicTypeEnum::ArrayType(Llvm::get_array_type(BasicTypeEnum::PointerType(ptr_ty), got_size));
249
250            let custom_got = llvm.create_global_variable(&self.module, "__custom_got", custom_got_ty);
251            custom_got.set_linkage(inkwell::module::Linkage::WeakODR);
252            custom_got.set_initial_value(Some(empty_got), custom_got_ty);
253
254            *got_entries = new_got_entries;
255        }
256
257        //Generate opaque functions for implementations and associate them with their types
258        let llvm = Llvm::new(context, context.create_builder());
259        let llvm_impl_index = pou_generator::generate_implementation_stubs(
260            &self.module,
261            llvm,
262            dependencies,
263            global_index,
264            annotations,
265            &index,
266            &mut self.debug,
267            &self.online_change,
268        )?;
269        let llvm = Llvm::new(context, context.create_builder());
270        index.merge(llvm_impl_index);
271        let llvm_values_index = pou_generator::generate_global_constants_for_pou_members(
272            &self.module,
273            &llvm,
274            dependencies,
275            global_index,
276            annotations,
277            &index,
278            &self.module_location,
279        )?;
280        index.merge(llvm_values_index);
281
282        //Generate constants for string-literal
283        //generate literals but first sort, so we get reproducable builds
284        let mut utf08s = literals.utf08.iter().map(String::as_str).collect::<Vec<&str>>();
285        utf08s.sort_unstable();
286        for (idx, literal) in utf08s.into_iter().enumerate() {
287            let len = literal.len() + 1;
288            let data_type = llvm.context.i8_type().array_type(len as u32);
289            let literal_variable = llvm.create_global_variable(
290                &self.module,
291                format!("utf08_literal_{idx}").as_str(),
292                data_type.as_basic_type_enum(),
293            );
294            let initializer = llvm.create_const_utf8_string(literal, len)?;
295            literal_variable.make_constant().make_private().set_initializer(&initializer);
296
297            index.associate_utf08_literal(literal, literal_variable);
298        }
299        //generate literals but first sort, so we get reproducable builds
300        let mut utf16s = literals.utf16.iter().map(String::as_str).collect::<Vec<&str>>();
301        utf16s.sort_unstable();
302        for (idx, literal) in utf16s.into_iter().enumerate() {
303            let len = literal.len() + 1;
304            let data_type = llvm.context.i16_type().array_type(len as u32);
305            let literal_variable = llvm.create_global_variable(
306                &self.module,
307                format!("utf16_literal_{idx}").as_str(),
308                data_type.as_basic_type_enum(),
309            );
310            let initializer = llvm.create_const_utf16_string(literal, literal.len() + 1)?;
311            literal_variable.make_constant().make_private().set_initializer(&initializer);
312
313            index.associate_utf16_literal(literal, literal_variable);
314        }
315
316        Ok(index)
317    }

319    /// generates all TYPEs, GLOBAL-sections and POUs of the given CompilationUnit
320    pub fn generate(
321        self,
322        context: &'ink CodegenContext,
323        unit: &CompilationUnit,
324        annotations: &AstAnnotations,
325        global_index: &Index,
326        llvm_index: LlvmTypedIndex<'ink>,
327    ) -> Result<GeneratedModule<'ink>, Diagnostic> {
328        //generate all pous
329        let llvm = Llvm::new(context, context.create_builder());
330        let pou_generator =
331            PouGenerator::new(llvm, global_index, annotations, &llvm_index, &self.online_change);
332
333        //Generate the POU stubs in the first go to make sure they can be referenced.
334        for implementation in &unit.implementations {
335            //Don't generate external or generic functions
336            if let Some(entry) = global_index.find_pou(implementation.name.as_str()) {
337                if !entry.is_generic() && entry.get_linkage() != &LinkageType::External {
338                    pou_generator.generate_implementation(implementation, &self.debug)?;
339                }
340            }
341        }
342
343        let location = (&unit.file).into();
344
345        self.debug.finalize();
346        log::trace!("{}", self.module.to_string());
347
348        #[cfg(feature = "verify")]
349        {
350            self.module
351                .verify()
352                .map_err(|it| Diagnostic::new(it.to_string_lossy()).with_error_code("E071"))
353                .map(|_| GeneratedModule { module: self.module, location, engine: RefCell::new(None) })
354        }
355
356        #[cfg(not(feature = "verify"))]
357        Ok(GeneratedModule { module: self.module, location, engine: RefCell::new(None) })
358    }

399    /// Persists the module into the disk based on output and target requirments
400    /// If an object file should be generated, all optimizations will be executed on the object
401    pub fn persist(
402        &self,
403        output_dir: Option<&Path>,
404        output_name: &str,
405        format: FormatOption,
406        target: &Target,
407        optimization_level: OptimizationLevel,
408    ) -> Result<PathBuf, Diagnostic> {
409        let output = Self::get_output_file(output_dir, output_name, target);
410        //ensure output exists
411        if let Some(parent) = output.parent() {
412            std::fs::create_dir_all(parent)?;
413        }
414        match format {
415            FormatOption::Object | FormatOption::Relocatable => {
416                self.persist_as_static_obj(output, target, optimization_level)
417            }
418            FormatOption::PIC | FormatOption::Shared | FormatOption::Static => {
419                self.persist_to_shared_pic_object(output, target, optimization_level)
420            }
421            FormatOption::NoPIC => self.persist_to_shared_object(output, target, optimization_level),
422            FormatOption::Bitcode => self.persist_to_bitcode(output),
423            FormatOption::IR => self.persist_to_ir(output),
424        }
425    }

427    fn get_output_file(output_dir: Option<&Path>, output_name: &str, target: &Target) -> PathBuf {
428        let output_dir = output_dir.map(Path::to_path_buf).unwrap_or_else(|| PathBuf::from(""));
429        let output = if let Some(name) = target.try_get_name() {
430            output_dir.join(name).join(output_name)
431        } else {
432            output_dir.join(output_name)
433        };
434        output
435    }

441    ///
442    /// Compiles the given source into an object file and saves it in output
443    ///
444    fn persist_to_obj(
445        &self,
446        output: PathBuf,
447        reloc: RelocMode,
448        target: &Target,
449        optimization_level: OptimizationLevel,
450    ) -> Result<PathBuf, Diagnostic> {
451        let initialization_config = &InitializationConfig::default();
452        inkwell::targets::Target::initialize_all(initialization_config);
453
454        let triple = target.get_target_triple();
455
456        let target = inkwell::targets::Target::from_triple(&triple).map_err(|it| {
457            Diagnostic::codegen_error(
458                format!("Invalid target-tripple '{triple}' - {it:?}"),
459                SourceLocation::undefined(),
460            )
461        })?;
462        let machine = target
463            .create_target_machine(
464                &triple,
465                //TODO : Add cpu features as optionals
466                "generic", //TargetMachine::get_host_cpu_name().to_string().as_str(),
467                "",        //TargetMachine::get_host_cpu_features().to_string().as_str(),
468                optimization_level.into(),
469                reloc,
470                CodeModel::Default,
471            )
472            .ok_or_else(|| {
473                Diagnostic::codegen_error("Cannot create target machine.", SourceLocation::undefined())
474            });
475
476        //Make sure all parents exist
477        if let Some(parent) = output.parent() {
478            std::fs::create_dir_all(parent)?;
479        }
480        ////Run the passes
481        machine
482            .and_then(|it| {
483                self.module
484                    .run_passes(optimization_level.opt_params(), &it, PassBuilderOptions::create())
485                    .map_err(|it| {
486                        Diagnostic::llvm_error(output.to_str().unwrap_or_default(), &it.to_string_lossy())
487                    })
488                    .and_then(|_| {
489                        it.write_to_file(&self.module, FileType::Object, output.as_path()).map_err(|it| {
490                            Diagnostic::llvm_error(output.to_str().unwrap_or_default(), &it.to_string_lossy())
491                        })
492                    })
493            })
494            .map(|_| output)
495    }

602    ///
603    /// Runs the function given by `name` inside the compiled module.
604    /// Returns the value returned by calling the function
605    ///
606    pub fn run<T, U>(&self, name: &str, params: &mut T) -> U {
607        let engine = self.get_execution_engine();
608
609        unsafe {
610            let main: JitFunction<MainFunction<T, U>> = engine.get_function(name).unwrap();
611            let main_t_ptr = &mut *params as *mut _;
612
613            main.call(main_t_ptr)
614        }
615    }

638    pub fn add_global_variable_mapping(&self, global_name: &str, local_var: usize) {
639        let engine = self.get_execution_engine();
640        if let Some(function) = self.module.get_global(global_name) {
641            engine.add_global_mapping(&function, local_var);
642        } else {
643            log::debug!("Global {} does not exist", global_name);
644        }
645    }
646
647    fn get_execution_engine(&self) -> ExecutionEngine<'ink> {
648        if let Some(engine) = self.engine.borrow().as_ref() {
649            return engine.clone();
650        }
651        //Create engine
652        let engine = self.module.create_jit_execution_engine(inkwell::OptimizationLevel::None).unwrap();
653        *self.engine.borrow_mut() = Some(engine.clone());
654        engine
655    }
656
657    pub fn set_name(&self, name: &str) {
658        self.module.set_name(name);
659        self.module.set_source_file_name(name);
660    }
