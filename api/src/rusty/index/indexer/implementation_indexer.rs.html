1use plc_ast::ast::{Implementation, PouType, TypeNature};
2
3use crate::{
4    index::{Index, PouIndexEntry},
5    typesystem::{self, DataTypeInformation},
6};
7
8pub struct ImplementationIndexer<'i> {
9    index: &'i mut Index,
10}
11
12impl<'i> ImplementationIndexer<'i> {
13    pub fn new(index: &'i mut Index) -> Self {
14        Self { index }
15    }
16
17    pub fn index_implementation(&mut self, implementation: &Implementation) {
18        let pou_type = &implementation.pou_type;
19        let start_location = implementation
20            .statements
21            .first()
22            .map(|it| it.get_location())
23            .as_ref()
24            .or(Some(&implementation.location))
25            .cloned()
26            .unwrap();
27        self.index.register_implementation(
28            &implementation.name,
29            &implementation.type_name,
30            pou_type.get_optional_owner_class().as_ref(),
31            pou_type.into(),
32            implementation.generic,
33            start_location,
34        );
35        //if we are registering an action, also register a datatype for it
36        if pou_type == &PouType::Action {
37            let datatype = typesystem::DataType {
38                name: implementation.name.to_string(),
39                initial_value: None,
40                information: DataTypeInformation::Alias {
41                    name: implementation.name.clone(),
42                    referenced_type: implementation.type_name.clone(),
43                },
44                nature: TypeNature::Derived,
45                location: implementation.name_location.clone(),
46                linkage: implementation.linkage,
47            };
48
49            self.index.register_pou(PouIndexEntry::create_action_entry(
50                implementation.name.as_str(),
51                implementation.type_name.as_str(),
52                implementation.linkage,
53                implementation.name_location.clone(),
54            ));
55            self.index.register_pou_type(datatype);
56        }
57    }
58}
