24/// Indexer that registers all user-defined types in the index
25pub struct UserTypeIndexer<'i, 't> {
26    index: &'i mut Index,
27    user_type: &'t UserTypeDeclaration,
28    pending_initializer: Option<ConstId>,
29}
30
31impl<'i, 't> UserTypeIndexer<'i, 't> {
32    pub fn new(index: &'i mut Index, user_type: &'t UserTypeDeclaration) -> Self {
33        UserTypeIndexer { index, user_type, pending_initializer: None }
34    }
35
36    fn collect_initializer(&mut self, i: AstNode, target_type_name: String) -> ConstId {
37        let scope = self.current_scope();
38        let init =
39            self.index.get_mut_const_expressions().add_constant_expression(i, target_type_name, scope, None);
40        init
41    }
42}

44impl AstVisitor for UserTypeIndexer<'_, '_> {
45    fn visit_user_type_declaration(&mut self, user_type: &UserTypeDeclaration) {
46        //handle inner types & initializers
47        user_type.data_type.walk(self);
48        if let Some(init_index) = user_type.initializer.clone().map(|i| {
49            self.collect_initializer(i, user_type.data_type.get_name().map(|s| s.to_string()).unwrap())
50            // XXX(mihr): is unwrapping fine here?
51        }) {
52            //TODO: without field?
53            self.pending_initializer = Some(init_index);
54        };
55
56        self.visit_data_type(&user_type.data_type);
57    }
58
59    fn visit_data_type(&mut self, data_type: &DataType) {
60        match &data_type {
61            DataType::StructType { name: Some(name), variables } => {
62                self.index_struct_type(name, variables, StructSource::OriginalDeclaration)
63            }
64            DataType::EnumType { name: Some(name), numeric_type, elements } => {
65                self.index_enum_type(name, numeric_type, elements)
66            }
67            DataType::SubRangeType { name: Some(name), referenced_type, bounds } => {
68                self.index_sub_range_type(name, referenced_type, bounds.as_ref())
69            }
70            DataType::ArrayType { name: Some(name), bounds, referenced_type, is_variable_length: false } => {
71                self.index_array_type(name, bounds, referenced_type)
72            }
73            DataType::ArrayType { name: Some(name), bounds, referenced_type, is_variable_length: true } => {
74                self.index_vla_array(name, bounds, referenced_type)
75            }
76            DataType::PointerType { name, referenced_type, auto_deref, type_safe, is_function } => {
77                self.index_pointer_type(name, referenced_type, *auto_deref, *type_safe, *is_function)
78            }
79            DataType::StringType { name: Some(name), is_wide, size } => {
80                self.index_string_type(name.as_ref(), *is_wide, size.as_ref())
81            }
82            DataType::VarArgs { .. } => {
83                // do nothing
84            }
85            DataType::GenericType { name, generic_symbol, nature } => {
86                self.index_generic_type(name, generic_symbol, nature)
87            }
88            _ => {}
89        }
90    }
91}

95        self.user_type.scope.clone()

98    fn index_vla_array(&mut self, name: &str, bounds: &AstNode, referenced_type: &DataTypeDeclaration) {
99        let ndims = match bounds.get_stmt() {
100            AstStatement::VlaRangeStatement => 1,
101            AstStatement::ExpressionList(expressions) => expressions.len(),
102            _ => unreachable!("not a bounds statement"),
103        };
104
105        let referenced_type = referenced_type.get_name().expect("named datatype").to_string();
106        let struct_name = name.to_owned();
107
108        let dummy_array_name = format!("__arr_vla_{ndims}_{referenced_type}").to_lowercase();
109        let member_array_name = format!("__ptr_to_{dummy_array_name}");
110        let member_dimensions_name = format!("__bounds_{dummy_array_name}");
111
112        // check the index if a dummy-array type matching the given VLA (eg. 1 dimension, type INT) already exists.
113        // if we find a type, we can use references to the internal types. otherwise, register the array in the index
114        // and declare internal member types.
115        let (vla_arr_type_declaration, dim_arr_type_declaration) =
116            if self.index.get_effective_type_by_name(&dummy_array_name).is_ok() {
117                (
118                    DataTypeDeclaration::Reference {
119                        referenced_type: member_array_name,
120                        location: SourceLocation::internal(),
121                    },
122                    DataTypeDeclaration::Reference {
123                        referenced_type: member_dimensions_name,
124                        location: SourceLocation::internal(),
125                    },
126                )
127            } else {
128                // register dummy array type so it can later be annotated as a type hint
129                self.index.register_type(typesystem::DataType {
130                    name: dummy_array_name.clone(),
131                    initial_value: None,
132                    information: DataTypeInformation::Array {
133                        name: dummy_array_name.clone(),
134                        inner_type_name: referenced_type.clone(),
135                        // dummy dimensions that will never actually be used
136                        dimensions: (0..ndims)
137                            .map(|_| Dimension {
138                                start_offset: TypeSize::Undetermined,
139                                end_offset: TypeSize::Undetermined,
140                            })
141                            .collect::<Vec<_>>(),
142                    },
143                    nature: TypeNature::__VLA,
144                    location: SourceLocation::internal(),
145                });
146
147                // define internal vla members
148                (
149                    DataTypeDeclaration::Definition {
150                        data_type: Box::new(DataType::PointerType {
151                            name: Some(member_array_name),
152                            referenced_type: Box::new(DataTypeDeclaration::Reference {
153                                referenced_type: dummy_array_name,
154                                location: SourceLocation::internal(),
155                            }),
156                            auto_deref: None,
157                            type_safe: true,
158                            is_function: false,
159                        }),
160                        location: SourceLocation::internal(),
161                        scope: None,
162                    },
163                    DataTypeDeclaration::Definition {
164                        data_type: Box::new(DataType::ArrayType {
165                            name: Some(member_dimensions_name),
166                            bounds: AstNode::new(
167                                AstStatement::ExpressionList(
168                                    (0..ndims)
169                                        .map(|_| {
170                                            AstFactory::create_range_statement(
171                                                AstNode::new_literal(
172                                                    AstLiteral::new_integer(0),
173                                                    0,
174                                                    SourceLocation::internal(),
175                                                ),
176                                                AstNode::new_literal(
177                                                    AstLiteral::new_integer(1),
178                                                    0,
179                                                    SourceLocation::internal(),
180                                                ),
181                                                0,
182                                            )
183                                        })
184                                        .collect::<_>(),
185                                ),
186                                0,
187                                SourceLocation::internal(),
188                            ),
189                            referenced_type: Box::new(DataTypeDeclaration::Reference {
190                                referenced_type: DINT_TYPE.to_string(),
191                                location: SourceLocation::internal(),
192                            }),
193                            is_variable_length: false,
194                        }),
195                        location: SourceLocation::internal(),
196                        scope: None,
197                    },
198                )
199            };
200
201        // Create variable index entries for VLA struct members
202        let variables = vec![
203            // Pointer
204            Variable {
205                name: format!("struct_vla_{referenced_type}_{ndims}").to_lowercase(),
206                data_type_declaration: vla_arr_type_declaration,
207                initializer: None,
208                address: None,
209                location: SourceLocation::internal(),
210            },
211            // Dimensions Array
212            Variable {
213                name: "dimensions".to_string(),
214                data_type_declaration: dim_arr_type_declaration,
215                initializer: None,
216                address: None,
217                location: SourceLocation::internal(),
218            },
219        ];
220
221        self.index_struct_type(
222            &struct_name,
223            &variables,
224            StructSource::Internal(typesystem::InternalType::VariableLengthArray {
225                inner_type_name: referenced_type,
226                ndims,
227            }),
228        );
229    }

231    fn index_array_type(&mut self, name: &str, bounds: &AstNode, referenced_type: &DataTypeDeclaration) {
232        let scope = self.current_scope();
233        let dimensions: Result<Vec<Dimension>, Diagnostic> = bounds
234            .get_as_list()
235            .iter()
236            .map(|it| match it.get_stmt() {
237                AstStatement::RangeStatement(RangeStatement { start, end }) => {
238                    let constants = self.index.get_mut_const_expressions();
239                    Ok(Dimension {
240                        start_offset: TypeSize::from_expression(constants.add_constant_expression(
241                            *start.clone(),
242                            typesystem::DINT_TYPE.to_string(),
243                            scope.clone(),
244                            None,
245                        )),
246                        end_offset: TypeSize::from_expression(constants.add_constant_expression(
247                            *end.clone(),
248                            typesystem::DINT_TYPE.to_string(),
249                            scope.clone(),
250                            None,
251                        )),
252                    })
253                }
254
255                _ => Err(Diagnostic::codegen_error("Invalid array definition: RangeStatement expected", *it)),
256            })
257            .collect();
258
259        // TODO(mhasel, volsa): This unwrap will panic with `ARRAY[0..5, 5] OF DINT;`
260        let dimensions = dimensions.unwrap();
261
262        //TODO hmm we need to talk about all this unwrapping :-/
263        let referenced_type_name = referenced_type.get_name().expect("named datatype");
264        let information = DataTypeInformation::Array {
265            name: name.to_string(),
266            inner_type_name: referenced_type_name.to_string(),
267            dimensions,
268        };
269
270        self.register_type(name, information, TypeNature::Any);
271        let global_init_name = crate::index::get_initializer_name(name);
272
273        // TODO unfortunately we cannot share const-expressions between multiple
274        // index-entries
275        let init = self
276            .user_type
277            .initializer
278            .as_ref()
279            .map(|i| self.collect_initializer(i.clone(), name.to_string()));
280        if init.is_some() {
281            let variable = VariableIndexEntry::create_global(
282                global_init_name.as_str(),
283                global_init_name.as_str(),
284                name,
285                self.user_type.location.clone(),
286            )
287            .set_constant(true)
288            .set_initial_value(init);
289            self.index.register_global_initializer(&global_init_name, variable);
290        }
291    }

293    fn index_enum_type(&mut self, name: &str, numeric_type: &str, elements: &AstNode) {
294        let mut variants = Vec::new();
295
296        for ele in flatten_expression_list(elements) {
297            let variant = get_enum_element_name(ele);
298            if let AstStatement::Assignment(Assignment { right, .. }) = ele.get_stmt() {
299                let scope = self.current_scope();
300                let init = self.index.get_mut_const_expressions().add_constant_expression(
301                    right.as_ref().clone(),
302                    numeric_type.to_string(),
303                    scope,
304                    None,
305                );
306
307                variants.push(self.index.register_enum_variant(
308                    name,
309                    &variant,
310                    Some(init),
311                    ele.get_location(),
312                ))
313            } else {
314                unreachable!("the preprocessor should have provided explicit assignments for enum values")
315            }
316        }
317
318        let information = DataTypeInformation::Enum {
319            name: name.to_owned(),
320            variants,
321            referenced_type: numeric_type.to_string(),
322        };
323
324        self.register_type(name, information, TypeNature::Int);
325    }

327    fn index_sub_range_type(&mut self, name: &str, referenced_type: &str, bounds: Option<&AstNode>) {
328        let information = if let Some(AstStatement::RangeStatement(RangeStatement { start, end })) =
329            bounds.as_ref().map(|it| it.get_stmt())
330        {
331            let scope = self.current_scope();
332            let start_size = self.ast_node_to_type_size(start, &scope);
333            let end_size = self.ast_node_to_type_size(end, &scope);
334
335            DataTypeInformation::SubRange {
336                name: name.into(),
337                referenced_type: referenced_type.into(),
338                sub_range: start_size..end_size,
339            }
340        } else {
341            DataTypeInformation::Alias { name: name.into(), referenced_type: referenced_type.into() }
342        };
343
344        self.register_type(name, information, TypeNature::Int);
345    }

347    /// Converts an AstNode to a TypeSize, either as a literal or a const expression
348    fn ast_node_to_type_size(&mut self, node: &AstNode, scope: &Option<String>) -> TypeSize {
349        match &node.stmt {
350            AstStatement::Literal(AstLiteral::Integer(value)) => TypeSize::from_literal(*value as i64),
351            _ => TypeSize::from_expression(self.index.get_mut_const_expressions().add_constant_expression(
352                node.clone(),
353                DINT_TYPE.to_string(),
354                scope.clone(),
355                None,
356            )),
357        }
358    }

360    fn register_type(&mut self, name: &str, information: DataTypeInformation, nature: TypeNature) {
361        self.index.register_type(typesystem::DataType {
362            name: name.into(),
363            initial_value: self.pending_initializer,
364            information,
365            nature,
366            location: self.user_type.location.clone(),
367        });
368    }

370    fn index_generic_type(&mut self, name: &str, generic_symbol: &str, nature: &TypeNature) {
371        let information = DataTypeInformation::Generic {
372            name: name.to_string(),
373            generic_symbol: generic_symbol.to_string(),
374            nature: *nature,
375        };
376
377        self.register_type(name, information, TypeNature::Any);
378    }

380    fn index_string_type(&mut self, name: &str, is_wide: bool, size: Option<&AstNode>) {
381        let encoding = if is_wide { StringEncoding::Utf16 } else { StringEncoding::Utf8 };
382
383        //TODO: handle the case where type_name is None
384        let scope = self.current_scope();
385
386        let size = match size {
387            Some(AstNode { stmt: AstStatement::Literal(AstLiteral::Integer(value)), .. }) => {
388                TypeSize::from_literal((value + 1) as i64)
389            }
390            Some(statement) => {
391                // construct a "x + 1" expression because we need one additional character for \0 terminator
392                let len_plus_1 = AstFactory::create_plus_one_expression(
393                    statement.clone(),
394                    statement.get_location(),
395                    statement.get_id(),
396                );
397
398                TypeSize::from_expression(self.index.get_mut_const_expressions().add_constant_expression(
399                    len_plus_1,
400                    DINT_TYPE.to_string(),
401                    scope,
402                    None,
403                ))
404            }
405            None => TypeSize::from_literal((DEFAULT_STRING_LEN + 1).into()),
406        };
407        let information = DataTypeInformation::String { size, encoding };
408        self.register_type(name, information, TypeNature::String);
409
410        //TODO: can we reuse this?
411        if let Some(init) = self.pending_initializer {
412            // register a global variable with the initial value to memcopy from
413            let global_init_name = crate::index::get_initializer_name(name);
414            let initializer_global = VariableIndexEntry::create_global(
415                global_init_name.as_str(),
416                global_init_name.as_str(),
417                name,
418                self.user_type
419                    .initializer
420                    .clone()
421                    .map(|i| i.get_location())
422                    .unwrap_or_else(|| self.user_type.location.clone()),
423            )
424            .set_constant(true)
425            .set_initial_value(Some(init));
426            self.index.register_global_initializer(global_init_name.as_str(), initializer_global);
427        }
428    }

430    fn index_pointer_type(
431        &mut self,
432        name: &Option<String>,
433        referenced_type: &DataTypeDeclaration,
434        auto_deref: Option<AutoDerefType>,
435        type_safe: bool,
436        is_function: bool,
437    ) {
438        let inner_type_name = referenced_type.get_name().expect("named datatype");
439        let name = name.as_deref().unwrap();
440        let information = DataTypeInformation::Pointer {
441            name: name.to_string(),
442            inner_type_name: inner_type_name.into(),
443            auto_deref,
444            type_safe,
445            is_function,
446        };
447
448        self.index.register_type(typesystem::DataType {
449            name: name.to_string(),
450            initial_value: self.pending_initializer,
451            information,
452            nature: TypeNature::Any,
453            location: self.user_type.location.clone(),
454        });
455    }

457    fn index_struct_type(&mut self, name: &str, variables: &[Variable], source: StructSource) {
458        let scope = Some(name.to_string());
459        let members = variables
460            .iter()
461            .enumerate()
462            .map(|(count, var)| {
463                // let inner_indexer = UserTypeIndexer::new(self.index, &var.data_type_declaration);
464                self.visit_data_type_declaration(&var.data_type_declaration);
465
466                let member_type = var.data_type_declaration.get_name().expect("named variable datatype");
467                let init = self.index.get_mut_const_expressions().maybe_add_constant_expression(
468                    var.initializer.clone(),
469                    member_type,
470                    scope.clone(),
471                    Some(var.name.clone()),
472                );
473
474                let binding = var
475                    .address
476                    .as_ref()
477                    .and_then(|it| HardwareBinding::from_statement(self.index, it, scope.clone()));
478
479                self.index.create_member_variable(
480                    MemberInfo {
481                        container_name: name,
482                        variable_name: &var.name,
483                        variable_linkage: ArgumentType::ByVal(VariableType::Input), // struct members act like VAR_INPUT in terms of visibility
484                        variable_type_name: member_type,
485                        is_constant: false, //struct members are not constants //TODO thats probably not true (you can define a struct in an CONST-block?!)
486                        is_var_external: false, // see above
487                        binding,
488                        varargs: None,
489                    },
490                    init,
491                    var.location.clone(),
492                    count as u32,
493                )
494            })
495            .collect::<Vec<_>>();
496
497        let nature = source.get_type_nature();
498        let information = DataTypeInformation::Struct { name: name.to_owned(), members, source };
499
500        self.register_type(name, information, nature);
501
502        //Generate an initializer for the struct
503        let global_struct_name = crate::index::get_initializer_name(name);
504        let variable = VariableIndexEntry::create_global(
505            &global_struct_name,
506            &global_struct_name,
507            name,
508            self.user_type.location.clone(),
509        )
510        .set_initial_value(self.pending_initializer)
511        .set_constant(true);
512
513        self.index.register_global_initializer(&global_struct_name, variable);
514    }
