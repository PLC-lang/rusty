20/// Indexes all symbols found in the given Compiliation Unit
21/// and returns the resulting Index
22pub fn index(unit: &CompilationUnit) -> Index {
23    let mut indexer = SymbolIndexer::default();
24    unit.walk(&mut indexer);
25    indexer.index
26}

33impl Context {
34    pub fn replace_with_pou(&mut self, pou: impl Into<String>) -> Self {
35        std::mem::replace(self, Context { pou: pou.into() })
36    }

39/// Indexer that registers all symbols in the index
40#[derive(Default)]
41pub struct SymbolIndexer {
42    pub index: Index,
43    ctx: Context,
44}
45
46/// The SymbolIndexer is responsible for registering all delcared types and symbols in the index.
47impl AstVisitor for SymbolIndexer {
48    /// Visits a VAR_GLOBAL VariableBlock and registers all variables as globals in the index
49    fn visit_variable_block(&mut self, block: &plc_ast::ast::VariableBlock) {
50        if block.kind == VariableBlockType::Global {
51            // let the global var indexer handle the global variables
52            let mut indexer =
53                VarGlobalIndexer::new(block.constant, block.retain, block.linkage, &mut self.index);
54            for var in &block.variables {
55                indexer.visit_variable(var);
56            }
57        }
58    }
59
60    /// Visits a user type declaration
61    /// Registers the user type in the index using the UserTypeIndexer
62    fn visit_user_type_declaration(&mut self, user_type: &plc_ast::ast::UserTypeDeclaration) {
63        UserTypeIndexer::new(&mut self.index, user_type).visit_user_type_declaration(user_type);
64    }
65
66    /// Visits a pou and registers all member variables in the index
67    /// Also registers the pou's struct type in the index
68    fn visit_pou(&mut self, pou: &plc_ast::ast::Pou) {
69        PouIndexer::new(&mut self.index).visit_pou(pou);
70        let old_ctx = self.ctx.replace_with_pou(&pou.name);
71        pou.properties.iter().for_each(|property| self.visit_property(property));
72        self.ctx = old_ctx;
73    }
74
75    /// Visits an implementation and registers the implementation in the index
76    fn visit_implementation(&mut self, implementation: &Implementation) {
77        ImplementationIndexer::new(&mut self.index).index_implementation(implementation);
78    }
79
80    fn visit_config_variable(&mut self, config_variable: &plc_ast::ast::ConfigVariable) {
81        self.index.config_variables.push(config_variable.clone());
82    }
83
84    fn visit_interface(&mut self, interface: &Interface) {
85        for method in &interface.methods {
86            self.visit_pou(method);
87
88            // Register an implementation entry for each interface method so that codegen
89            // can look up the function signature when generating indirect (itable) calls.
90            // Interface methods have no body, but their LLVM function stubs are needed as
91            // type templates for `build_indirect_call`.
92            self.index.register_implementation(
93                &method.name,
94                &method.name,
95                Some(&interface.ident.name.to_string()),
96                ImplementationType::Method,
97                false,
98                method.location.clone(),
99            );
100        }
101
102        self.index.interfaces.insert(interface.ident.name.to_owned(), InterfaceIndexEntry::from(interface));
103
104        self.index.register_type(DataType {
105            name: interface.ident.name.clone(),
106            initial_value: None,
107            information: DataTypeInformation::Interface { name: interface.ident.name.clone() },
108            nature: TypeNature::Any,
109            location: interface.ident.location.clone(),
110            linkage: LinkageType::Internal,
111        });
112    }
113
114    fn visit_property(&mut self, property: &PropertyBlock) {
115        self.index.properties.insert(self.ctx.pou.clone(), property.ident.clone());
116    }
117}
