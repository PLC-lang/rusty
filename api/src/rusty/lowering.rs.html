31impl InitVisitor {
32    pub fn visit(
33        mut units: Vec<CompilationUnit>,
34        index: Index,
35        unresolvables: Vec<UnresolvableConstant>,
36        id_provider: IdProvider,
37        init_symbol_name: &'static str,
38    ) -> Vec<CompilationUnit> {
39        let mut visitor = Self::new(index, unresolvables, id_provider);
40        // before visiting, we need to collect all candidates for user-defined init functions
41        units.iter_mut().for_each(|unit| {
42            visitor.collect_user_init_candidates(unit);
43        });
44        // visit all units
45        units.iter_mut().for_each(|unit| {
46            visitor.visit_compilation_unit(unit);
47        });
48
49        visitor.extend_ast(units, init_symbol_name)
50    }

66    fn walk_with_scope<T>(&mut self, t: &mut T, pou_name: Option<impl Into<String>>)
67    where
68        T: WalkerMut,
69    {
70        let old = self.ctxt.scope(pou_name.map(Into::into));

75    fn collect_user_init_candidates(&mut self, unit: &mut CompilationUnit) {
76        // collect all candidates for user-defined init functions
77        for pou in unit.pous.iter().filter(|it| matches!(it.kind, PouType::FunctionBlock | PouType::Program))
78        {
79            // add the POU to potential `FB_INIT` candidates
80            self.user_inits
81                .insert(pou.name.to_owned(), self.index.find_method(&pou.name, "FB_INIT").is_some());
82        }
83
84        for user_type in
85            unit.user_types.iter_mut().filter(|it| matches!(it.data_type, DataType::StructType { .. }))
86        {
87            // add the struct to potential `STRUCT_INIT` candidates
88            if let Some(name) = user_type.data_type.get_name() {
89                self.user_inits.insert(name.to_string(), false);
90            };
91        }
92    }

94    fn update_initializer(&mut self, variable: &mut plc_ast::ast::Variable) {
95        // flat references to stateful pou-local variables need to have a qualifier added, so they can be resolved in the init functions
96        let scope = self.ctxt.get_scope().as_ref().map(|it| it.as_str()).unwrap_or(GLOBAL_SCOPE);
97        let needs_qualifier = |flat_ref| {
98            let rhs = self.index.find_member(scope, flat_ref);
99            let lhs = self.index.find_member(scope, variable.get_name());
100            let Some(pou) = self.index.find_pou(scope) else { return Ok(false) };
101            if !(pou.is_function() || pou.is_method())
102                && lhs.is_some_and(|it| !it.is_temp())
103                && rhs.is_some_and(|it| it.is_temp())
104            {
105                // Unable to initialize a stateful member variable with an address of a temporary value since it doesn't exist at the time of initialization
106                // On top of that, even if we were to initialize it, it would lead to a dangling pointer/potential use-after-free
107                return Err(AstFactory::create_empty_statement(
108                    SourceLocation::internal(),
109                    self.ctxt.get_id_provider().next_id(),
110                ));
111            }
112            Ok(match pou {
113                        PouIndexEntry::Program { .. }
114                        | PouIndexEntry::FunctionBlock { .. }
115                        | PouIndexEntry::Class { .. }
116                            // we only want to add qualifiers to local, non-temporary variables
117                            if rhs.is_some_and(|it| !it.is_temp()) && lhs.is_some_and(|it| !it.is_temp())=>
118                        {
119                            true
120                        }
121                        _ => false,
122                    })
123        };
124
125        if let Some(initializer) = variable.initializer.as_ref() {
126            let type_name =
127                variable.data_type_declaration.get_name().expect("Must have a type at this point");
128            let data_type = self.index.get_effective_type_or_void_by_name(type_name).get_type_information();
129            if !data_type.is_pointer() {
130                return;
131            }
132
133            let updated_initializer = match &initializer.get_stmt() {
134                // no call-statement in the initializer, so something like `a AT b` or `a : REFERENCE TO ... REF= b`
135                AstStatement::ReferenceExpr(_) => {
136                    initializer.get_flat_reference_name().and_then(|flat_ref| {
137                        needs_qualifier(flat_ref).map_or_else(Option::Some, |q| {
138                            q.then_some("self")
139                                .map(|it| create_member_reference(it, self.ctxt.get_id_provider(), None))
140                                .and_then(|base| {
141                                    initializer.get_flat_reference_name().map(|it| {
142                                        create_member_reference(it, self.ctxt.get_id_provider(), Some(base))
143                                    })
144                                })
145                        })
146                    })
147                }
148                // we found a call-statement, must be `a : REF_TO ... := REF(b) | ADR(b)`
149                AstStatement::CallStatement(CallStatement { operator, parameters }) => parameters
150                    .as_ref()
151                    .and_then(|it| it.as_ref().get_flat_reference_name())
152                    .and_then(|flat_ref| {
153                        let op = operator.as_ref().get_flat_reference_name()?;
154                        needs_qualifier(flat_ref).map_or_else(Option::Some, |q| {
155                            q.then(|| {
156                                create_call_statement(
157                                    op,
158                                    flat_ref,
159                                    Some("self"),
160                                    self.ctxt.id_provider.clone(),
161                                    &initializer.location,
162                                )
163                            })
164                        })
165                    }),
166                _ => return,
167            };
168
169            self.unresolved_initializers.insert_initializer(
170                scope,
171                Some(&variable.name),
172                &updated_initializer.or(Some(initializer.clone())),
173            );
174        }
175    }

177    fn add_init_statements(&mut self, implementation: &mut plc_ast::ast::Implementation) {
178        let predicate = |var: &VariableIndexEntry| {
179            var.is_temp()
180                || var.is_var_external()
181                || (matches!(implementation.pou_type, PouType::Function | PouType::Method { .. })
182                    && var.is_local())
183        };
184        let strip_temporaries = |inits: &mut InitAssignments| {
185            let mut temps = InitAssignments::default();
186            let ids = inits
187                .iter()
188                .filter(|(id, _)| self.index.find_member(&implementation.name, id).is_some_and(predicate))
189                .map(|(id, _)| id.to_owned())
190                .collect::<Vec<_>>();
191
192            for id in ids {
193                if let Some(init) = inits.swap_remove(&id) {
194                    temps.insert(id.to_owned(), init);
195                };
196            }
197
198            temps
199        };
200
201        // remove all initializers for the current implementation/pou
202        // XXX: this changes the order of init-statements in init-pous. but since assignments should still be ordered before call-statements,
203        // this should be fine => edge cases?
204        let Some(mut inits) = self.unresolved_initializers.swap_remove(&implementation.name) else {
205            return;
206        };
207
208        // remove all non-stateful variable entries
209        let temps = strip_temporaries(&mut inits);
210        // re-enter the remaining initializers
211        self.unresolved_initializers.insert(implementation.name.to_owned(), inits);
212
213        // collect simple assignments
214        let assignments = temps.into_iter().filter_map(|(lhs, init)| {
215            init.as_ref().map(|it| {
216                let lhs_ty = self
217                    .index
218                    .find_member(&implementation.name, &lhs)
219                    .map(|it| {
220                        self.index
221                            .get_effective_type_or_void_by_name(it.get_type_name())
222                            .get_type_information()
223                    })
224                    .unwrap();
225                if lhs_ty.is_reference_to() || lhs_ty.is_alias() {
226                    // XXX: ignore REF_TO for temp variables since they can be generated regularly
227                    let rhs = if let AstStatement::CallStatement(CallStatement {
228                        parameters: Some(parameter),
229                        ..
230                    }) = it.get_stmt()
231                    {
232                        parameter
233                    } else {
234                        it
235                    };
236                    // `REFERENCE TO` assignments in a POU body are automatically dereferenced and require the `REF=` operator to assign a pointer instead.
237                    create_ref_assignment(&lhs, None, rhs, self.ctxt.get_id_provider())
238                } else {
239                    create_assignment(&lhs, None, it, self.ctxt.get_id_provider())
240                }
241            })
242        });
243
244        // collect necessary call statements to init-functions and user-defined init-functions
245        let mut implicit_calls = Vec::new();
246        let mut user_init_calls = Vec::new();
247        self.index.get_pou_members(&implementation.name).iter().filter(|var| predicate(var)).for_each(
248            |var| {
249                let dti =
250                    self.index.get_effective_type_or_void_by_name(var.get_type_name()).get_type_information();
251                let is_external = self
252                    .index
253                    .find_pou(dti.get_name())
254                    .is_some_and(|it| it.get_linkage() == &LinkageType::External);
255                if dti.is_struct() && !is_external {
256                    implicit_calls.push(create_call_statement(
257                        &get_init_fn_name(dti.get_name()),
258                        var.get_name(),
259                        None,
260                        self.ctxt.get_id_provider(),
261                        &implementation.name_location,
262                    ));
263                }
264                if self.user_inits.contains_key(dti.get_name()) {
265                    user_init_calls.push(create_call_statement(
266                        &get_user_init_fn_name(dti.get_name()),
267                        var.get_name(),
268                        None,
269                        self.ctxt.get_id_provider(),
270                        &implementation.name_location,
271                    ));
272                }
273            },
274        );
275
276        let stmts = assignments
277            .chain(implicit_calls)
278            .chain(user_init_calls)
279            .chain(std::mem::take(&mut implementation.statements))
280            .collect::<Vec<_>>();
281        implementation.statements = stmts;
282    }

284    /// Updates the scope and initialized variable for struct types. Adds entries for each encountered struct
285    /// (this includes POU-structs, i.e. programs, ...) to the initializer map if no entry is present
286    fn update_struct_initializers(&mut self, user_type: &mut plc_ast::ast::UserTypeDeclaration) {
287        let effective_type =
288            user_type.data_type.get_name().and_then(|it| self.index.find_effective_type_by_name(it));
289        if let DataType::StructType { ref mut variables, .. } = &mut user_type.data_type {
290            let Some(ty) = effective_type else {
291                return user_type.walk(self);
292            };
293            let name = ty.get_name();
294
295            for variable in variables {
296                self.unresolved_initializers.maybe_insert_initializer(
297                    name,
298                    Some(&variable.name),
299                    &variable.initializer,
300                );
301
302                // XXX: Very duct-tapey but essentially we now have two initializers, one in the struct datatype
303                // definition itself (`DataType::StructType { initializer: Some(<arena id>), ... }`) and one in the
304                // `__init_*` function now. The former is unresolvable because it has the raw initializer, e.g.
305                // `foo := (a := (b := (c := REF(...))))` whereas the latter is resolvable because it yields something
306                // like `foo.a.b.c := REF(...)`. Thus we remove the initializer from the struct datatype definition
307                // as the codegen would otherwise fail at generating them and result in a `Cannot generate values for..`
308                // Literals and references are ignored however, since they are resolvable and / or constant.
309                if variable.initializer.as_ref().is_some_and(|opt| !opt.is_literal() && !opt.is_reference()) {
310                    variable.initializer = None;
311                }
312            }
313
314            // add container to keys if not already present
315            self.unresolved_initializers.maybe_insert_initializer(name, None, &user_type.initializer);
316        }
317    }

319    fn maybe_add_global_instance_initializer(&mut self, variable: &plc_ast::ast::Variable) {
320        let Some(global) = self.index.find_global_variable(variable.get_name()) else {
321            return;
322        };
323
324        let info =
325            self.index.get_effective_type_or_void_by_name(global.get_type_name()).get_type_information();
326
327        if !info.is_struct()
328            || self
329                .index
330                .find_pou(info.get_name())
331                .is_some_and(|it| it.get_linkage() == &LinkageType::External)
332        {
333            return;
334        }
335
336        self.unresolved_initializers.maybe_insert_initializer(GLOBAL_SCOPE, Some(variable.get_name()), &None);
337    }

339    fn collect_var_config_assignments(&mut self, var_config: &[ConfigVariable]) {

347impl AstVisitorMut for InitVisitor {
348    fn visit_compilation_unit(&mut self, unit: &mut CompilationUnit) {
349        self.collect_var_config_assignments(&unit.var_config);
350        unit.walk(self)
351    }
352
353    fn visit_implementation(&mut self, implementation: &mut plc_ast::ast::Implementation) {
354        self.add_init_statements(implementation);
355        implementation.walk(self);
356    }
357
358    fn visit_variable(&mut self, variable: &mut plc_ast::ast::Variable) {
359        self.maybe_add_global_instance_initializer(variable);
360        self.update_initializer(variable);
361        variable.walk(self);
362    }
363
364    fn visit_user_type_declaration(&mut self, user_type: &mut plc_ast::ast::UserTypeDeclaration) {
365        self.update_struct_initializers(user_type);
366        user_type.walk(self);
367    }
368
369    fn visit_data_type(&mut self, data_type: &mut DataType) {
370        if matches!(data_type, plc_ast::ast::DataType::StructType { .. }) {
371            self.walk_with_scope(data_type, data_type.get_name().map(ToOwned::to_owned))
372        } else {
373            data_type.walk(self)
374        }
375    }
376
377    fn visit_pou(&mut self, pou: &mut plc_ast::ast::Pou) {
378        if !matches!(pou.linkage, LinkageType::External | LinkageType::BuiltIn) {
379            self.unresolved_initializers.maybe_insert_initializer(&pou.name, None, &None);
380        }
381
382        self.walk_with_scope(pou, Some(&pou.name.to_owned()));
383    }
384}

435/// Takes some expression such as `bar := (baz := (qux := ADR(val)), baz2 := (qux := ADR(val)))` returning all final
436/// assignment paths such as [`bar.baz.qux := ADR(val)`, `bar.baz2.qux := ADR(val)`].
437fn create_assignment_paths(node: &AstNode, id_provider: IdProvider) -> Vec<Vec<AstNode>> {
438    match node.get_stmt() {
439        AstStatement::Assignment(Assignment { left, right }) => {
440            let mut result = create_assignment_paths(right, id_provider.clone());
441            for inner in result.iter_mut() {
442                inner.insert(0, left.as_ref().clone());
443            }
444            result
445        }
446        AstStatement::ExpressionList(nodes) => {
447            let mut result = vec![];
448            for node in nodes {
449                let inner = create_assignment_paths(node, id_provider.clone());
450                result.extend(inner);
451            }
452            result
453        }
454        AstStatement::ParenExpression(node) => create_assignment_paths(node, id_provider),
455        _ => vec![vec![node.clone()]],
456    }
457}

459/// Takes some expression such as `foo : FooStruct := (bar := (baz := (qux := ADR(val)), baz2 := (qux := ADR(val))));`
460/// and returns assignments of form [`foo.bar.baz.qux := ADR(val)`, `foo.bar.baz2.qux := ADR(val)`].
461fn create_assignments_from_initializer(
462    var_ident: &str,
463    self_ident: Option<&str>,
464    rhs: &Option<AstNode>,
465    mut id_provider: IdProvider,
466) -> Vec<AstNode> {
467    let Some(initializer) = rhs else {
468        return Vec::new();
469    };
470
471    let mut result = vec![];
472    for mut path in create_assignment_paths(initializer, id_provider.clone()) {
473        path.insert(0, create_member_reference(var_ident, id_provider.clone(), None));
474        if self_ident.is_some() {
475            path.insert(0, create_member_reference("self", id_provider.clone(), None));
476        }
477
478        let right = path.pop().expect("must have at least one node in the path");
479        let mut left = path.pop().expect("must have at least one node in the path");
480
481        for node in path.into_iter().rev() {
482            insert_base_node(&mut left, node);
483        }
484
485        result.push(AstFactory::create_assignment(left, right, id_provider.next_id()));
486    }
487
488    result
489}

491/// Inserts a new base node into the member reference chain. For example a call such as `insert_base_node("b.c", a")`
492/// will yield `a.b.c`.
493fn insert_base_node(member: &mut AstNode, new_base: AstNode) {
494    match &mut member.stmt {
495        AstStatement::ReferenceExpr(ReferenceExpr { base, .. }) => match base {
496            Some(inner) => insert_base_node(inner, new_base),
497            None => {
498                // We hit the end of the chain, simply replace the base (which must be None) with the new one
499                base.replace(Box::new(new_base));
500            }
501        },
502
503        _ => panic!("invalid function call, expected a member reference"),
504    }
505}

507fn create_ref_assignment(
508    lhs_ident: &str,
509    base_ident: Option<&str>,
510    rhs: &AstNode,
511    mut id_provider: IdProvider,
512) -> AstNode {

514        lhs_ident,
515        id_provider.clone(),
516        base_ident.map(|id| create_member_reference(id, id_provider.clone(), None)),
517    );
518    AstFactory::create_ref_assignment(lhs, rhs.to_owned(), id_provider.next_id())
519}

535pub fn create_call_statement(
536    operator: &str,
537    member_id: &str,
538    base_id: Option<&str>,
539    mut id_provider: IdProvider,
540    location: &SourceLocation,
541) -> AstNode {
542    let op = create_member_reference(operator, id_provider.clone(), None);
543    let param = create_member_reference(
544        member_id,
545        id_provider.clone(),
546        base_id.map(|it| create_member_reference(it, id_provider.clone(), None)),
547    );
548    AstFactory::create_call_statement(op, Some(param), id_provider.next_id(), location.clone())
