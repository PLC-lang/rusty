46pub fn parse_file(
47    source: &SourceCode,
48    linkage: LinkageType,
49    id_provider: IdProvider,
50    diagnostician: &mut Diagnostician,
51) -> Result<CompilationUnit, Diagnostic> {
52    let location_factory = SourceLocationFactory::for_source(source);
53    let (unit, errors) = parse(
54        lexer::lex_with_ids(&source.source, id_provider, location_factory),
55        linkage,
56        source.get_location_str(),
57    );
58    //Register the source file with the diagnostician
59    //TODO: We should reduce the clone here
60    diagnostician.register_file(source.get_location_str().to_string(), source.source.clone()); // TODO: Remove clone here, generally passing the GlobalContext instead of the actual source here or in the handle method should be sufficient
61    if diagnostician.handle(&errors) == Severity::Error {
62        Err(Diagnostic::new("Compilation aborted due to critical parse errors").with_sub_diagnostics(errors))
63    } else {
64        Ok(unit)
65    }
66}

68pub fn parse(mut lexer: ParseSession, lnk: LinkageType, file_name: &'static str) -> ParsedAst {
69    let mut unit = CompilationUnit::new(file_name);
70
71    let mut linkage = lnk;
72    let mut constant = false;
73    loop {
74        match lexer.token {
75            PropertyExternal => {
76                linkage = LinkageType::External;
77                lexer.advance();
78                //Don't reset linkage
79                continue;
80            }
81            PropertyConstant => {
82                // parse optional const pragma (only allowed in builtins for now)
83                constant = true;
84                lexer.advance();
85                continue;
86            }
87            KeywordInterface => {
88                // We ignore any method implementations in interfaces as we do not support default impls yet
89                let (interfaces, _) = parse_interface(&mut lexer);
90                unit.interfaces.push(interfaces);
91            }
92            KeywordVarGlobal => unit.global_vars.push(parse_variable_block(&mut lexer, linkage)),
93            KeywordVarConfig => unit.var_config.extend(parse_config_variables(&mut lexer)),
94
95            KeywordProgram | KeywordClass | KeywordFunction | KeywordFunctionBlock => {
96                let params = match lexer.token {
97                    KeywordProgram => (PouType::Program, KeywordEndProgram),
98                    KeywordClass => (PouType::Class, KeywordEndClass),
99                    KeywordFunction => (PouType::Function, KeywordEndFunction),
100                    _ => (PouType::FunctionBlock, KeywordEndFunctionBlock),
101                };
102
103                parse_pou(&mut lexer, &mut unit, params.0, linkage, params.1, constant);
104
105                // reset const pragma
106                constant = false;
107            }
108            KeywordAction => {
109                if let Some(implementation) = parse_action(&mut lexer, linkage, None) {
110                    unit.implementations.push(implementation);
111                }
112            }
113            KeywordActions => {
114                let last_pou = unit
115                    .pous
116                    .iter()
117                    .rfind(|it| {
118                        // Only consider the last POU that is a program, function, function block
119                        // or class
120                        matches!(
121                            it.kind,
122                            PouType::Program | PouType::Function | PouType::FunctionBlock | PouType::Class
123                        )
124                    })
125                    .map(|it| it.name.as_str())
126                    .unwrap_or("__unknown__");
127                let mut actions = parse_actions(&mut lexer, linkage, last_pou);
128                unit.implementations.append(&mut actions);
129            }
130            KeywordType => {
131                let unit_type = parse_type(&mut lexer, linkage);
132                for utype in unit_type {
133                    unit.user_types.push(utype);
134                }
135            }
136            KeywordEndActions | End => return (unit, lexer.diagnostics),
137            _ => {
138                lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
139                    "StartKeyword",
140                    lexer.slice(),
141                    lexer.location(),
142                ));
143                lexer.advance();
144            }
145        };
146        linkage = lnk;
147    }
148    //the match in the loop will always return
149}
150fn parse_actions(
151    lexer: &mut ParseSession,
152    linkage: LinkageType,
153    default_container: &str,
154) -> Vec<Implementation> {
155    parse_any_in_region(lexer, vec![KeywordEndActions], |lexer| {
156        lexer.advance();
157        let container =
158            if lexer.token == Identifier { lexer.slice_and_advance() } else { default_container.into() };
159        let mut impls = vec![];
160
161        //Go through each action
162        while lexer.token != KeywordEndActions && !lexer.is_end_of_stream() {
163            match lexer.token {
164                KeywordAction => {
165                    if let Some(implementation) = parse_action(lexer, linkage, Some(&container)) {
166                        impls.push(implementation);
167                    }
168                }
169                _ => {
170                    lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
171                        "KeywordAction",
172                        lexer.slice(),
173                        lexer.location(),
174                    ));
175                    return impls;
176                }
177            }
178        }
179        impls
180    })
181}

183/// Parses an interface and its methods / properties
184fn parse_interface(lexer: &mut ParseSession) -> (Interface, Vec<Implementation>) {
185    let location_start = lexer.range().start;
186    lexer.try_consume_or_report(KeywordInterface);
187
188    let (name, location_name) = match lexer.token {
189        Token::Identifier => parse_identifier(lexer).expect("unreachable, already matched here"),
190
191        _ => {
192            lexer.accept_diagnostic(
193                Diagnostic::new("Expected a name for the interface definition but got nothing")
194                    .with_error_code("E006")
195                    .with_location(lexer.last_location()),
196            );
197
198            // We want to keep parsing, hence we return some undefined values; the parser will yield an
199            // unrecoverable error though
200            (String::new(), SourceLocation::undefined())
201        }
202    };
203
204    let mut extensions = Vec::new();
205    let mut methods = Vec::new();
206    let mut implementations = Vec::new();
207    let mut properties = Vec::new();
208
209    if lexer.try_consume(KeywordExtends) {
210        while let Identifier = lexer.token {
211            let (name, location) = parse_identifier(lexer).expect("unreachable, already matched here");
212            extensions.push(Identifier { name, location });
213            lexer.try_consume(KeywordComma);
214        }
215    }
216    loop {
217        match lexer.token {
218            KeywordMethod => {
219                if let Some((method, imp)) =
220                    parse_method(lexer, &name, DeclarationKind::Abstract, LinkageType::Internal, false)
221                {
222                    // This is temporary? At some point we'll support them but for now it's a diagnostic
223                    if !imp.statements.is_empty() {
224                        lexer.accept_diagnostic(
225                            Diagnostic::new("Interfaces can not have a default implementation")
226                                .with_error_code("E113")
227                                .with_location(&imp.statements.first().unwrap().location),
228                        );
229                    }
230
231                    methods.push(method);
232                    implementations.push(imp);
233                }
234            }
235
236            KeywordProperty => {
237                if let Some(property) = parse_property(lexer) {
238                    for property in property.implementations.iter().filter(|imp| !imp.body.is_empty()) {
239                        lexer.accept_diagnostic(
240                            Diagnostic::new("Interfaces can not have a default implementation")
241                                .with_error_code("E113")
242                                .with_location(&property.body.first().unwrap().location),
243                        );
244                    }
245
246                    properties.push(property);
247                }
248            }
249
250            _ => break,
251        }
252    }
253
254    lexer.try_consume_or_report(KeywordEndInterface);
255    let location_end = lexer.range().start;
256
257    (
258        Interface {
259            id: lexer.next_id(),
260            ident: Identifier { name, location: location_name },
261            methods,
262            extensions,
263            location: lexer.source_range_factory.create_range(location_start..location_end),
264            properties,
265        },
266        implementations,
267    )
268}

270///
271/// parse a pou
272/// # Arguments
273///
274/// * `lexer`       - the lexer
275/// * `pou_type`    - the type of the pou currently parsed
276/// * `linkage`     - internal, external ?
277/// * `expected_end_token` - the token that ends this pou
278///
279fn parse_pou(
280    lexer: &mut ParseSession,
281    unit: &mut CompilationUnit,
282    kind: PouType,
283    linkage: LinkageType,
284    expected_end_token: lexer::Token,
285    constant: bool,
286) {
287    if constant && !matches!(linkage, LinkageType::BuiltIn) {
288        lexer.accept_diagnostic(Diagnostic::const_pragma_is_not_allowed(
289            lexer.last_location().span(&lexer.location()),
290        ));
291    }
292
293    let start = lexer.range().start;
294    lexer.advance(); //Consume ProgramKeyword
295    let closing_tokens = vec![
296        expected_end_token,
297        KeywordEndAction,
298        KeywordEndProgram,
299        KeywordEndFunction,
300        KeywordEndFunctionBlock,
301        KeywordEndClass,
302    ];
303    let result = parse_any_in_region(lexer, closing_tokens.clone(), |lexer| {
304        // parse polymorphism mode for all pou types
305        // check in validator if pou type allows polymorphism
306        let poly_mode = parse_polymorphism_mode(lexer, &kind);
307
308        let (name, name_location) =
309            parse_identifier(lexer).unwrap_or_else(|| ("".to_string(), SourceLocation::undefined())); // parse POU name
310
311        let generics = parse_generics(lexer);
312
313        with_scope(lexer, name.clone(), |lexer| {
314            // TODO: Parse USING directives
315            let super_class = parse_super_class(lexer);
316            let interfaces = parse_interface_declarations(lexer);
317
318            // parse an optional return type
319            // classes do not have a return type (check in validator)
320            let return_type = parse_return_type(lexer);
321
322            // parse variable declarations. note that var in/out/inout
323            // blocks are not allowed inside of class declarations.
324            let mut variable_blocks = vec![];
325            let allowed_var_types = [
326                KeywordVar,
327                KeywordVarInput,
328                KeywordVarOutput,
329                KeywordVarInOut,
330                KeywordVarTemp,
331                KeywordVarExternal,
332            ];
333            while allowed_var_types.contains(&lexer.token) {
334                variable_blocks.push(parse_variable_block(lexer, LinkageType::Internal));
335            }
336
337            let mut impl_pous = Vec::new();
338            let mut implementations = Vec::new();
339            let mut properties = Vec::new();
340
341            // classes and function blocks can have methods. methods consist of a Pou part
342            // and an implementation part. That's why we get another (Pou, Implementation)
343            // tuple out of parse_method() that has to be added to the list of Pous and
344            // implementations. Note that function blocks have to start with the method
345            // declarations before their implementation.
346            // all other Pous need to be checked in the validator if they can have methods.
347            while matches!(lexer.token, KeywordMethod | KeywordProperty | PropertyConstant) {
348                if !matches!(kind, PouType::FunctionBlock | PouType::Class | PouType::Program) {
349                    let location = lexer.source_range_factory.create_range(lexer.last_range.clone());
350                    let pre = if matches!(lexer.token, KeywordProperty) { "Properties" } else { "Methods" };
351
352                    lexer.accept_diagnostic(
353                        Diagnostic::new(format!("{pre} cannot be declared in a {kind}"))
354                            .with_location(location),
355                    );
356                }
357
358                if lexer.token == KeywordProperty {
359                    if let Some(property) = parse_property(lexer) {
360                        properties.push(property);
361                    }
362                } else {
363                    let is_const = lexer.try_consume(PropertyConstant);
364                    if let Some((pou, implementation)) =
365                        parse_method(lexer, &name, DeclarationKind::Concrete, linkage, is_const)
366                    {
367                        impl_pous.push(pou);
368                        implementations.push(implementation);
369                    }
370                }
371            }
372
373            // a class may not contain an implementation
374            // check in validator
375            implementations.push(parse_implementation(
376                lexer,
377                linkage,
378                kind.clone(),
379                &name,
380                &name,
381                !generics.is_empty(),
382                name_location.clone(),
383            ));
384
385            let mut pous = vec![Pou {
386                name,
387                id: lexer.next_id(),
388                kind,
389                variable_blocks,
390                return_type,
391                location: lexer.source_range_factory.create_range(start..lexer.range().end),
392                name_location,
393                poly_mode,
394                generics,
395                linkage,
396                super_class,
397                interfaces,
398                is_const: constant,
399                properties,
400            }];
401            pous.append(&mut impl_pous);
402
403            (pous, implementations)
404        })
405    });
406
407    //check if we ended on the right end-keyword
408    if closing_tokens.contains(&lexer.last_token) && lexer.last_token != expected_end_token {
409        lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
410            format!("{expected_end_token:?}").as_str(),
411            lexer.slice_region(lexer.last_range.clone()),
412            lexer.source_range_factory.create_range(lexer.last_range.clone()),
413        ));
414    }
415
416    let (mut pous, mut implementations) = result;
417    unit.pous.append(&mut pous);
418    unit.implementations.append(&mut implementations);
419}

421fn parse_generics(lexer: &mut ParseSession) -> Vec<GenericBinding> {
422    if lexer.try_consume(Token::OperatorLess) {
423        parse_any_in_region(lexer, vec![Token::OperatorGreater], |lexer| {
424            let mut generics = vec![];
425            loop {
426                //identifier
427                if let Some((name, _)) = parse_identifier(lexer) {
428                    lexer.try_consume_or_report(Token::KeywordColon);
429
430                    //Expect a type nature
431                    if let Some(nature) = parse_identifier(lexer).map(|(it, _)| parse_type_nature(lexer, &it))
432                    {
433                        generics.push(GenericBinding { name, nature });
434                    }
435                }
436
437                if !lexer.try_consume(Token::KeywordComma) || lexer.try_consume(Token::OperatorGreater) {
438                    break;
439                }
440            }
441
442            generics
443        })
444    } else {
445        vec![]
446    }
447}

449/// Parses the comma seperated identifiers after an `IMPLEMENTS` keyword, e.g. `bar` and `baz` in
450/// `INTERFACE foo IMPLEMENTS bar`
451fn parse_interface_declarations(lexer: &mut ParseSession) -> Vec<Identifier> {
452    let mut declarations = Vec::new();
453
454    if !lexer.try_consume(KeywordImplements) {
455        return declarations;
456    }
457
458    if lexer.token != Token::Identifier {
459        lexer.accept_diagnostic(
460            Diagnostic::new(
461                "Expected a comma separated list of identifiers after `IMPLEMENTS` but got nothing",
462            )
463            .with_error_code("E006")
464            .with_location(lexer.last_location()),
465        );
466
467        return declarations;
468    }
469
470    loop {
471        match lexer.token {
472            Token::Identifier => {
473                let (name, location) = parse_identifier(lexer).expect("Identifier already matched");
474                declarations.push(Identifier { name, location });
475            }
476            Token::KeywordComma => lexer.advance(),
477
478            _ => break,
479        }
480    }
481
482    declarations
483}

485fn parse_type_nature(lexer: &mut ParseSession, nature: &str) -> TypeNature {
486    match nature {
487        "ANY" => TypeNature::Any,
488        "ANY_DERIVED" => TypeNature::Derived,
489        "ANY_ELEMENTARY" => TypeNature::Elementary,
490        "ANY_MAGNITUDE" => TypeNature::Magnitude,
491        "ANY_NUM" => TypeNature::Num,
492        "ANY_REAL" => TypeNature::Real,
493        "ANY_INT" => TypeNature::Int,
494        "ANY_SIGNED" => TypeNature::Signed,
495        "ANY_UNSIGNED" => TypeNature::Unsigned,
496        "ANY_DURATION" => TypeNature::Duration,
497        "ANY_BIT" => TypeNature::Bit,
498        "ANY_CHARS" => TypeNature::Chars,
499        "ANY_STRING" => TypeNature::String,
500        "ANY_CHAR" => TypeNature::Char,
501        "ANY_DATE" => TypeNature::Date,
502        "__ANY_VLA" => TypeNature::__VLA,
503        _ => {
504            lexer.accept_diagnostic(
505                Diagnostic::new(format!("Unkown type nature `{nature}`"))
506                    .with_location(lexer.location())
507                    .with_error_code("E063"),
508            );
509            TypeNature::Any
510        }
511    }
512}

514fn parse_polymorphism_mode(lexer: &mut ParseSession, pou_type: &PouType) -> Option<PolymorphismMode> {
515    match pou_type {
516        PouType::Class | PouType::FunctionBlock | PouType::Method { .. } => {
517            Some(
518                // See if the method/pou was declared FINAL or ABSTRACT
519                if lexer.try_consume(KeywordFinal) {
520                    PolymorphismMode::Final
521                } else if lexer.try_consume(KeywordAbstract) {
522                    PolymorphismMode::Abstract
523                } else {
524                    PolymorphismMode::None
525                },
526            )
527        }
528        _ => None,
529    }
530}

532fn parse_super_class(lexer: &mut ParseSession) -> Option<Identifier> {
533    let mut extensions = vec![];
534    while lexer.try_consume(KeywordExtends) {
535        let name_and_location = parse_identifier(lexer)?;
536        extensions.push(name_and_location);
537    }
538    extensions.iter().skip(1).for_each(|(_, location)| {
539        lexer.accept_diagnostic(
540            Diagnostic::new("Multiple inheritance. POUs can only be extended once.".to_string())
541                .with_error_code("E114")
542                .with_location(location),
543        )
544    });
545
546    extensions
547        .first()
548        .map(|(name, location)| Identifier { name: name.to_string(), location: location.clone() })
549}

551fn parse_return_type(lexer: &mut ParseSession) -> Option<DataTypeDeclaration> {
552    if lexer.try_consume(KeywordColon) {
553        if let Some((declaration, initializer)) = parse_data_type_definition(lexer, None) {
554            if let Some(init) = initializer {
555                lexer.accept_diagnostic(
556                    Diagnostic::new("Return types cannot have a default value, the value will be ignored")
557                        .with_location(init.get_location())
558                        .with_error_code("E016"),
559                );
560            }
561
562            if let DataTypeDeclaration::Definition { data_type, .. } = &declaration {
563                if matches!(data_type.as_ref(), DataType::EnumType { .. } | DataType::StructType { .. }) {
564                    let datatype_name = declaration
565                        .get_location()
566                        .to_range()
567                        .map(|range| &lexer.get_src()[range])
568                        .expect("Expecing location to be a range during parsing");
569                    lexer.accept_diagnostic(
570                        ////TODO: This prints a debug version of the datatype, it should have a user readable version instead
571                        Diagnostic::new(format!(
572                            "Data Type {datatype_name} not supported as a function return type!"
573                        ))
574                        .with_error_code("E027")
575                        .with_location(&declaration),
576                    )
577                }
578            }
579
580            Some(declaration)
581        } else {
582            //missing return type
583            lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
584                "Datatype",
585                lexer.slice(),
586                lexer.source_range_factory.create_range(lexer.range()),
587            ));
588            None
589        }
590    } else {
591        None
592    }
593}

595fn parse_method(
596    lexer: &mut ParseSession,
597    parent: &str,
598    declaration_kind: DeclarationKind,
599    linkage: LinkageType,
600    constant: bool,
601) -> Option<(Pou, Implementation)> {
602    parse_any_in_region(lexer, vec![KeywordEndMethod], |lexer| {
603        // Method declarations look like this:
604        // METHOD [AccessModifier] [ABSTRACT|FINAL] [OVERRIDE] [: return_type]
605        //    ...
606        // END_METHOD
607
608        // constant pragma is only allowed in builtins for now
609        if constant {
610            lexer.accept_diagnostic(Diagnostic::const_pragma_is_not_allowed(
611                lexer.last_location().span(&lexer.location()),
612            ));
613        }
614
615        let method_start = lexer.range().start;
616        lexer.advance(); // eat METHOD keyword
617
618        let access = Some(parse_access_modifier(lexer));
619        let pou_kind = PouType::Method { parent: parent.into(), property: None, declaration_kind };
620        let poly_mode = parse_polymorphism_mode(lexer, &pou_kind);
621        let overriding = lexer.try_consume(KeywordOverride);
622        let (name, name_location) = parse_identifier(lexer)?;
623        let generics = parse_generics(lexer);
624        let return_type = parse_return_type(lexer);
625
626        let mut variable_blocks = vec![];
627        while lexer.token == KeywordVar
628            || lexer.token == KeywordVarInput
629            || lexer.token == KeywordVarOutput
630            || lexer.token == KeywordVarInOut
631            || lexer.token == KeywordVarTemp
632        {
633            variable_blocks.push(parse_variable_block(lexer, LinkageType::Internal));
634        }
635
636        let call_name = qualified_name(parent, &name);
637        let implementation = parse_implementation(
638            lexer,
639            linkage,
640            pou_kind.clone(),
641            &call_name,
642            &call_name,
643            !generics.is_empty(),
644            name_location.clone(),
645        );
646
647        // parse_implementation() will default-initialize the fields it
648        // doesn't know. thus, we have to complete the information.
649        let implementation = Implementation { overriding, access, ..implementation };
650
651        let method_end = lexer.range().end;
652        Some((
653            Pou {
654                name: call_name,
655                id: lexer.next_id(),
656                kind: pou_kind,
657                variable_blocks,
658                return_type,
659                location: lexer.source_range_factory.create_range(method_start..method_end),
660                name_location,
661                poly_mode,
662                generics,
663                linkage,
664                super_class: None,
665                interfaces: Vec::new(),
666                properties: Vec::new(),
667                is_const: constant,
668            },
669            implementation,
670        ))
671    })
672}

674fn parse_property(lexer: &mut ParseSession) -> Option<PropertyBlock> {
675    lexer.advance(); // Move past `PROPERTY` keyword
676
677    let mut has_error = false;
678
679    let identifier = parse_identifier(lexer);
680    if identifier.is_none() {
681        has_error = true;
682        lexer.accept_diagnostic(
683            Diagnostic::new("Property definition is missing a name").with_location(lexer.location()),
684        );
685    }
686
687    let datatype = parse_return_type(lexer);
688    if datatype.is_none() {
689        has_error = true;
690        lexer.accept_diagnostic(
691            Diagnostic::new("Property definition is missing a datatype").with_location(lexer.last_location()),
692        );
693    };
694
695    // This is kind of common, hence we parse invalid variable blocks to have useful error messages
696    while lexer.token.is_var() {
697        let block = parse_variable_block(lexer, LinkageType::Internal);
698        lexer.accept_diagnostic(
699            Diagnostic::new(
700                "Variable blocks may only be defined within a GET or SET block in the context of properties",
701            )
702            .with_location(&block.location)
703            .with_error_code("E007"),
704        );
705    }
706
707    let mut implementations = Vec::new();
708    while matches!(lexer.token, KeywordGet | KeywordSet) {
709        let location = lexer.location();
710        let kind = if lexer.token == KeywordGet { PropertyKind::Get } else { PropertyKind::Set };
711        lexer.advance(); // Move past `GET` or `SET` keyword
712
713        let mut variable_blocks = Vec::new();
714        while lexer.token.is_var() {
715            variable_blocks.push(parse_variable_block(lexer, LinkageType::Internal));
716        }
717
718        let statements = parse_body_in_region(
719            lexer,
720            match kind {
721                PropertyKind::Get => vec![Token::KeywordEndGet],
722                PropertyKind::Set => vec![Token::KeywordEndSet],
723            },
724        );
725        implementations.push(PropertyImplementation {
726            kind,
727            variable_blocks,
728            body: statements,
729            location,
730            end_location: lexer.last_location(),
731        });
732    }
733
734    lexer.try_consume_or_report(Token::KeywordEndProperty); // Move past `END_PROPERTY` keyword
735
736    if has_error {
737        return None;
738    };
739
740    let (name, name_location) = identifier.expect("covered above");
741    let datatype = datatype.expect("covered above");
742    Some(PropertyBlock { ident: Identifier { name, location: name_location }, datatype, implementations })
743}
744
745fn parse_access_modifier(lexer: &mut ParseSession) -> AccessModifier {
746    if lexer.try_consume(KeywordAccessPublic) {
747        AccessModifier::Public
748    } else if lexer.try_consume(KeywordAccessPrivate) {
749        AccessModifier::Private
750    } else if lexer.try_consume(KeywordAccessProtected) {
751        AccessModifier::Protected
752    } else if lexer.try_consume(KeywordAccessInternal) {
753        AccessModifier::Internal
754    } else {
755        AccessModifier::Protected
756    }
757}

759/// parse identifier and advance if successful
760/// returns the identifier as a String and the SourceRange of the parsed name
761fn parse_identifier(lexer: &mut ParseSession) -> Option<(String, SourceLocation)> {
762    let pou_name = lexer.slice().to_string();
763    if lexer.token == Identifier {
764        lexer.advance();
765        Some((pou_name, lexer.last_location()))
766    } else {
767        lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
768            "Identifier",
769            pou_name.as_str(),
770            lexer.location(),
771        ));
772        None
773    }
774}

776fn parse_implementation(
777    lexer: &mut ParseSession,
778    linkage: LinkageType,
779    pou_type: PouType,
780    call_name: &str,
781    type_name: &str,
782    generic: bool,
783    name_location: SourceLocation,
784) -> Implementation {
785    let start = lexer.range().start;
786    let statements = parse_body_standalone(lexer);
787    let end_location = lexer.location(); //Location of the current token, which shoudl be the
788                                         //end token
789    Implementation {
790        name: call_name.into(),
791        type_name: type_name.into(),
792        linkage,
793        pou_type,
794        statements,
795        location: lexer.source_range_factory.create_range(start..lexer.last_range.end),
796        name_location,
797        end_location,
798        overriding: false,
799        generic,
800        access: None,
801    }
802}

804fn parse_action(
805    lexer: &mut ParseSession,
806    linkage: LinkageType,
807    container: Option<&str>,
808) -> Option<Implementation> {
809    lexer.advance(); //Consume the Action keyword
810    let closing_tokens =
811        vec![KeywordEndAction, KeywordEndProgram, KeywordEndFunction, KeywordEndFunctionBlock];
812
813    parse_any_in_region(lexer, closing_tokens.clone(), |lexer| {
814        let name_or_container = lexer.slice_and_advance();
815
816        let (container, name, name_location) = if let Some(container) = container {
817            (container.into(), name_or_container, lexer.last_location())
818        } else {
819            let loc = lexer.last_location();
820            expect_token!(lexer, KeywordDot, None);
821
822            lexer.advance();
823
824            expect_token!(lexer, Identifier, None);
825
826            let name = lexer.slice_and_advance();
827            (name_or_container, name, loc.span(&lexer.last_location()))
828        };
829        let call_name = qualified_name(&container, &name);
830
831        let implementation = parse_implementation(
832            lexer,
833            linkage,
834            PouType::Action,
835            &call_name,
836            &container,
837            false,
838            name_location,
839        );
840        //lets see if we ended on the right END_ keyword
841        if closing_tokens.contains(&lexer.last_token) && lexer.last_token != KeywordEndAction {
842            lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
843                format!("{KeywordEndAction:?}").as_str(),
844                lexer.slice(),
845                lexer.location(),
846            ))
847        }
848        Some(implementation)
849    })
850}

852// TYPE ... END_TYPE
853fn parse_type(lexer: &mut ParseSession, linkage: LinkageType) -> Vec<UserTypeDeclaration> {
854    lexer.advance(); // consume the TYPE
855
856    parse_any_in_region(lexer, vec![KeywordEndType], |lexer| {
857        let mut declarations = vec![];
858        while !lexer.closes_open_region(&lexer.token) {
859            let name = lexer.slice_and_advance();
860            let name_location = lexer.last_location();
861            lexer.try_consume_or_report(KeywordColon);
862
863            let result = parse_full_data_type_definition(lexer, Some(name));
864
865            if let Some((DataTypeDeclaration::Definition { data_type, .. }, initializer)) = result {
866                declarations.push(UserTypeDeclaration {
867                    data_type: *data_type,
868                    initializer,
869                    location: name_location,
870                    scope: lexer.scope.clone(),
871                    linkage,
872                });
873            }
874        }
875        declarations
876    })
877}

879type DataTypeWithInitializer = (DataTypeDeclaration, Option<AstNode>);
880
881fn parse_full_data_type_definition(
882    lexer: &mut ParseSession,
883    name: Option<String>,
884) -> Option<DataTypeWithInitializer> {
885    let end_keyword = if lexer.token == KeywordStruct { KeywordEndStruct } else { KeywordSemicolon };
886    let parsed_datatype = parse_any_in_region(lexer, vec![end_keyword], |lexer| {
887        let sized = lexer.try_consume(PropertySized);
888        if lexer.try_consume(KeywordDotDotDot) {
889            Some((
890                DataTypeDeclaration::Definition {
891                    data_type: Box::new(DataType::VarArgs { referenced_type: None, sized }),
892                    location: lexer.last_location(),
893                    scope: lexer.scope.clone(),
894                },
895                None,
896            ))
897        } else {
898            parse_data_type_definition(lexer, name).map(|(type_def, initializer)| {
899                if lexer.try_consume(KeywordDotDotDot) {
900                    (
901                        DataTypeDeclaration::Definition {
902                            data_type: Box::new(DataType::VarArgs {
903                                referenced_type: Some(Box::new(type_def)),
904                                sized,
905                            }),
906                            location: lexer.last_location(),
907                            scope: lexer.scope.clone(),
908                        },
909                        None,
910                    )
911                } else {
912                    (type_def, initializer)
913                }
914            })
915        }
916    });
917
918    // The standard allows semicolons at the end of an `END_STRUCT` keyword, hence if we parsed
919    // a struct, try to also consume a semicolon if it exists
920    if end_keyword == KeywordEndStruct {
921        lexer.try_consume(KeywordSemicolon);
922    }
923
924    parsed_datatype
925}

927// TYPE xxx : 'STRUCT' | '(' | IDENTIFIER
928fn parse_data_type_definition(
929    lexer: &mut ParseSession,
930    name: Option<String>,
931) -> Option<DataTypeWithInitializer> {
932    let start = lexer.location();
933    if lexer.try_consume(KeywordStruct) {
934        // Parse struct
935        let variables = parse_variable_list(lexer);
936        Some((
937            DataTypeDeclaration::Definition {
938                data_type: Box::new(DataType::StructType { name, variables }),
939                location: start.span(&lexer.location()),
940                scope: lexer.scope.clone(),
941            },
942            None,
943        ))
944    } else if lexer.try_consume(KeywordArray) {
945        parse_array_type_definition(lexer, name)
946    } else if lexer.try_consume(KeywordPointer) {
947        let start_pos = lexer.last_range.start;
948        //Report wrong keyword
949        lexer.accept_diagnostic(
950            Diagnostic::new("`POINTER TO` is type-unsafe, consider using `REF_TO` instead")
951                .with_location(lexer.last_location())
952                .with_error_code("E015"),
953        );
954        let expect_keyword_to = |lexer: &mut ParseSession| {
955            expect_token!(lexer, KeywordTo, None);
956            Some(())
957        };
958        if expect_keyword_to(lexer).is_some() {
959            lexer.advance();
960        }
961
962        parse_pointer_definition(lexer, name, start_pos, None, false, false)
963    } else if lexer.try_consume(KeywordRef) {
964        parse_pointer_definition(lexer, name, lexer.last_range.start, None, true, false)
965    } else if lexer.try_consume(KeywordParensOpen) {
966        parse_enum_type_definition(lexer, name)
967    } else if lexer.token == KeywordString || lexer.token == KeywordWideString {
968        parse_string_type_definition(lexer, name)
969    } else if lexer.token == Identifier {
970        parse_type_reference_type_definition(lexer, name)
971    } else {
972        //no datatype?
973        lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
974            "DataTypeDefinition",
975            format!("{:?}", lexer.token).as_str(),
976            lexer.location(),
977        ));
978        None
979    }
980}

982fn parse_pointer_definition(
983    lexer: &mut ParseSession,
984    name: Option<String>,
985    start_pos: usize,
986    auto_deref: Option<AutoDerefType>,
987    type_safe: bool,
988    is_function: bool,
989) -> Option<(DataTypeDeclaration, Option<AstNode>)> {
990    parse_data_type_definition(lexer, None).map(|(decl, initializer)| {
991        (
992            DataTypeDeclaration::Definition {
993                data_type: Box::new(DataType::PointerType {
994                    name,
995                    referenced_type: Box::new(decl),
996                    auto_deref,
997                    type_safe,
998                    is_function,
999                }),
1000                // FIXME: this currently includes the initializer in the sourcelocation, resulting in 'REF_TO A := B' when creating a slice
1001                location: lexer.source_range_factory.create_range(start_pos..lexer.last_range.end),
1002                scope: lexer.scope.clone(),
1003            },
1004            initializer,
1005        )
1006    })

1009fn parse_type_reference_type_definition(
1010    lexer: &mut ParseSession,
1011    name: Option<String>,
1012) -> Option<(DataTypeDeclaration, Option<AstNode>)> {
1013    let start = lexer.range().start;
1014
1015    let mut referenced_type = lexer.slice_and_advance();
1016
1017    if lexer.try_consume(KeywordDot) {
1018        referenced_type = format!("{referenced_type}.{}", lexer.slice_and_advance());
1019    }
1020
1021    let bounds = if lexer.try_consume(KeywordParensOpen) {
1022        // INT (..) :=
1023        let bounds = parse_expression(lexer);
1024        expect_token!(lexer, KeywordParensClose, None);
1025        lexer.advance();
1026        Some(bounds)
1027    } else {
1028        None
1029    };
1030
1031    let end = lexer.last_range.end;
1032
1033    let initial_value: Option<AstNode> =
1034        if lexer.try_consume(KeywordAssignment) || lexer.try_consume(KeywordReferenceAssignment) {
1035            Some(parse_expression(lexer))
1036        } else {
1037            None
1038        };
1039
1040    if name.is_some() || bounds.is_some() {
1041        let data_type = match bounds {
1042            Some(AstNode { stmt: AstStatement::ExpressionList(expressions), id, location, .. }) => {
1043                //this is an enum
1044                DataTypeDeclaration::Definition {
1045                    data_type: Box::new(DataType::EnumType {
1046                        name,
1047                        numeric_type: referenced_type,
1048                        elements: AstFactory::create_expression_list(expressions, location, id),
1049                    }),
1050                    location: lexer.source_range_factory.create_range(start..end),
1051                    scope: lexer.scope.clone(),
1052                }
1053            }
1054            Some(AstNode {
1055                stmt: AstStatement::ReferenceExpr(ReferenceExpr { access: ReferenceAccess::Member(_), .. }),
1056                ..
1057            }) => {
1058                // a enum with just one element
1059                DataTypeDeclaration::Definition {
1060                    data_type: Box::new(DataType::EnumType {
1061                        name,
1062                        numeric_type: referenced_type,
1063                        elements: bounds.unwrap(),
1064                    }),
1065                    location: lexer.source_range_factory.create_range(start..end),
1066                    scope: lexer.scope.clone(),
1067                }
1068            }
1069            _ => DataTypeDeclaration::Definition {
1070                //something else inside the brackets -> probably a subrange?
1071                data_type: Box::new(DataType::SubRangeType { name, referenced_type, bounds }),
1072                location: lexer.source_range_factory.create_range(start..end),
1073                scope: lexer.scope.clone(),
1074            },
1075        };
1076        Some((data_type, initial_value))
1077    } else {
1078        Some((
1079            DataTypeDeclaration::Reference {
1080                referenced_type,
1081                location: lexer.source_range_factory.create_range(start..end),
1082            },
1083            initial_value,
1084        ))
1085    }
1086}

1088fn parse_string_size_expression(lexer: &mut ParseSession) -> Option<AstNode> {
1089    let opening_token = lexer.token;
1090    if lexer.try_consume(KeywordSquareParensOpen) || lexer.try_consume(KeywordParensOpen) {
1091        let opening_location = lexer.range().start;
1092        let closing_tokens = vec![KeywordSquareParensClose, KeywordParensClose];
1093        parse_any_in_region(lexer, closing_tokens, |lexer| {
1094            let size_expr = parse_expression(lexer);
1095            let error_range = lexer.source_range_factory.create_range(opening_location..lexer.range().end);
1096
1097            // Don't emit warnings if this looks like an enum (will be caught by validation).
1098            // e.g. `TYPE attemptAtStringEnum : STRING (a := 1, b := 2);` should not warn about parentheses.
1099            let is_enum_like = matches!(size_expr.get_stmt(), AstStatement::ExpressionList(_));
1100
1101            if (opening_token == KeywordParensOpen && lexer.token == KeywordSquareParensClose)
1102                || (opening_token == KeywordSquareParensOpen && lexer.token == KeywordParensClose)
1103            {
1104                lexer.accept_diagnostic(
1105                    Diagnostic::new("Mismatched types of parentheses around string size expression")
1106                        .with_location(error_range)
1107                        .with_error_code("E009"),
1108                );
1109            } else if !is_enum_like
1110                && (opening_token == KeywordParensOpen || lexer.token == KeywordParensClose)
1111            {
1112                lexer.accept_diagnostic(Diagnostic::new(
1113                    "Unusual type of parentheses around string size expression, consider using square parentheses '[]'").
1114                    with_location(error_range)
1115                    .with_error_code("E014")
1116                );
1117            }
1118
1119            Some(size_expr)
1120        })
1121    } else {
1122        None
1123    }
1124}

1126fn parse_string_type_definition(
1127    lexer: &mut ParseSession,
1128    name: Option<String>,
1129) -> Option<(DataTypeDeclaration, Option<AstNode>)> {
1130    let text = lexer.slice().to_string();
1131    let start = lexer.range().start;
1132    let is_wide = lexer.token == KeywordWideString;
1133    lexer.advance();
1134
1135    let size = parse_string_size_expression(lexer);
1136    let end = lexer.last_range.end;
1137    let location = lexer.source_range_factory.create_range(start..end);
1138
1139    // Check if this is actually an enum type (e.g., STRING (a := 1, b := 2))
1140    let is_enum_like = matches!(&size, Some(AstNode { stmt: AstStatement::ExpressionList(_), .. }));
1141
1142    match (size, &name, is_enum_like) {
1143        (Some(size), _, true) => {
1144            // This looks like an enum definition with STRING/WSTRING as the type
1145            // Create an EnumType so validation can catch it as invalid
1146            Some(DataTypeDeclaration::Definition {
1147                data_type: Box::new(DataType::EnumType { name, numeric_type: text, elements: size }),
1148                location,
1149                scope: lexer.scope.clone(),
1150            })
1151        }
1152        (Some(size), _, false) => Some(DataTypeDeclaration::Definition {
1153            data_type: Box::new(DataType::StringType { name, is_wide, size: Some(size) }),
1154            location,
1155            scope: lexer.scope.clone(),
1156        }),
1157        (None, Some(name), _) => Some(DataTypeDeclaration::Definition {
1158            data_type: Box::new(DataType::SubRangeType {
1159                name: Some(name.into()),
1160                referenced_type: text,
1161                bounds: None,
1162            }),
1163            location,
1164            scope: lexer.scope.clone(),
1165        }),
1166        _ => Some(DataTypeDeclaration::Reference { referenced_type: text, location }),
1167    }
1168    .zip(Some(
1169        (lexer.try_consume(KeywordAssignment) || lexer.try_consume(KeywordReferenceAssignment))
1170            .then(|| parse_expression(lexer)),
1171    ))
1172}

1174fn parse_enum_type_definition(
1175    lexer: &mut ParseSession,
1176    name: Option<String>,
1177) -> Option<(DataTypeDeclaration, Option<AstNode>)> {
1178    let start = lexer.last_location();
1179
1180    let elements = parse_any_in_region(lexer, vec![KeywordParensClose], parse_enum_elements)?;
1181    // Check for Codesys-style type specification after the enum list
1182    // TYPE COLOR : (...) DWORD;
1183    let numeric_type =
1184        if lexer.token == Identifier { lexer.slice_and_advance() } else { DINT_TYPE.to_string() };
1185    let initializer = lexer.try_consume(KeywordAssignment).then(|| parse_expression(lexer));
1186    Some((
1187        DataTypeDeclaration::Definition {
1188            data_type: Box::new(DataType::EnumType { name, elements, numeric_type }),
1189            location: start.span(&lexer.last_location()),
1190            scope: lexer.scope.clone(),
1191        },
1192        initializer,
1193    ))
1194}

1196/// Parse comma-separated enum elements (identifier or identifier := literal)
1197fn parse_enum_elements(lexer: &mut ParseSession) -> Option<AstNode> {
1198    let start = lexer.location();
1199    let mut elements = vec![];
1200
1201    loop {
1202        // Check if we've hit the closing token (e.g., ')') immediately
1203        // This handles empty enums like `()` for error recovery
1204        if lexer.closes_open_region(&lexer.token) {
1205            break;
1206        }
1207
1208        let element = parse_enum_element(lexer)?;
1209        elements.push(element);
1210
1211        if !lexer.try_consume(KeywordComma) {
1212            break;
1213        }
1214
1215        if lexer.closes_open_region(&lexer.token) {
1216            break;
1217        }
1218    }
1219
1220    // Handle empty enum case (no elements parsed)
1221    if elements.is_empty() {
1222        return Some(AstFactory::create_empty_statement(start, lexer.next_id()));
1223    }
1224
1225    if elements.len() == 1 {
1226        return Some(elements.into_iter().next().unwrap());
1227    }
1228
1229    Some(AstFactory::create_expression_list(elements, start.span(&lexer.last_location()), lexer.next_id()))
1230}

1232/// Parse a single enum element: identifier or identifier := literal
1233fn parse_enum_element(lexer: &mut ParseSession) -> Option<AstNode> {
1234    let start = lexer.location();
1235
1236    let idfr = parse_identifier(lexer)?;
1237
1238    let identifier_node = AstFactory::create_identifier(&idfr.0, start, lexer.next_id());
1239    let ref_expr = AstFactory::create_member_reference(identifier_node, None, lexer.next_id());
1240
1241    if lexer.try_consume(KeywordAssignment) {
1242        let value = parse_range_statement(lexer);
1243        let result = AstFactory::create_assignment(ref_expr, value, lexer.next_id());
1244        return Some(result);
1245    }
1246
1247    Some(ref_expr)
1248}

1250fn parse_array_type_definition(
1251    lexer: &mut ParseSession,
1252    name: Option<String>,
1253) -> Option<(DataTypeDeclaration, Option<AstNode>)> {
1254    let start = lexer.last_range.start;
1255    let range = parse_any_in_region(lexer, vec![KeywordOf], |lexer| {
1256        // Parse Array range
1257
1258        expect_token!(lexer, KeywordSquareParensOpen, None);
1259        lexer.advance();
1260
1261        let range_statement = parse_expression(lexer);
1262
1263        expect_token!(lexer, KeywordSquareParensClose, None);
1264        lexer.advance();
1265
1266        Some(range_statement)
1267    })?;
1268
1269    let inner_type_defintion = parse_data_type_definition(lexer, None);
1270    inner_type_defintion.map(|(reference, initializer)| {
1271        let reference_end = reference.get_location().to_range().map(|it| it.end).unwrap_or(0);
1272        let location = lexer.source_range_factory.create_range(start..reference_end);
1273
1274        let is_variable_length = match &range.get_stmt() {
1275            // Single dimensions, i.e. ARRAY[0..5] or ARRAY[*]
1276            AstStatement::RangeStatement { .. } => Some(false),
1277            AstStatement::VlaRangeStatement => Some(true),
1278
1279            // Multi dimensions, i.e. ARRAY [0..5, 5..10] or ARRAY [*, *]
1280            AstStatement::ExpressionList(expressions) => match expressions[0].get_stmt() {
1281                AstStatement::RangeStatement(..) => Some(false),
1282                AstStatement::VlaRangeStatement => Some(true),
1283                _ => None,
1284            },
1285
1286            _ => None,
1287        };
1288
1289        let is_variable_length = match is_variable_length {
1290            Some(val) => val,
1291            None => {
1292                lexer.accept_diagnostic(
1293                    Diagnostic::new(format!("Expected a range statement, got {range:?} instead"))
1294                        .with_location(range.get_location())
1295                        .with_error_code("E008"),
1296                );
1297                false
1298            }
1299        };
1300
1301        (
1302            DataTypeDeclaration::Definition {
1303                data_type: Box::new(DataType::ArrayType {
1304                    name,
1305                    bounds: range,
1306                    referenced_type: Box::new(reference),
1307                    is_variable_length,
1308                }),
1309                location,
1310                scope: lexer.scope.clone(),
1311            },
1312            initializer,
1313        )
1314    })
1315}

1317/// parse a body and recovers until the given `end_keywords`
1318fn parse_body_in_region(lexer: &mut ParseSession, end_keywords: Vec<Token>) -> Vec<AstNode> {
1319    parse_any_in_region(lexer, end_keywords, parse_body_standalone)
1320}

1322fn parse_body_standalone(lexer: &mut ParseSession) -> Vec<AstNode> {
1323    let mut statements = Vec::new();
1324    while !lexer.closes_open_region(&lexer.token) {
1325        statements.push(parse_control(lexer));
1326    }
1327    statements
1328}

1330/// parses a statement ending with a ';'
1331fn parse_statement(lexer: &mut ParseSession) -> AstNode {
1332    let result = parse_any_in_region(lexer, vec![KeywordSemicolon, KeywordColon], parse_expression);
1333    if lexer.last_token == KeywordColon {
1334        let location = result.location.span(&lexer.last_location());
1335        AstFactory::create_case_condition(result, location, lexer.next_id())
1336    } else {
1337        result
1338    }
1339}

1341pub fn with_scope<T, F: FnOnce(&mut ParseSession) -> T>(
1342    lexer: &mut ParseSession,
1343    scope: String,
1344    parse_fn: F,
1345) -> T {
1346    lexer.scope = Some(scope);
1347    let result = parse_fn(lexer);
1348    lexer.scope = None;
1349    result
1350}
1351
1352pub fn parse_any_in_region<T, F: FnOnce(&mut ParseSession) -> T>(
1353    lexer: &mut ParseSession,
1354    closing_tokens: Vec<Token>,
1355    parse_fn: F,
1356) -> T {
1357    lexer.enter_region(closing_tokens);
1358    let result = parse_fn(lexer);
1359
1360    // try to recover by eating everything until
1361    // we believe the parser is able to continue
1362    lexer.recover_until_close();
1363    lexer.close_region();
1364
1365    result
1366}

1368fn parse_reference(lexer: &mut ParseSession) -> AstNode {
1369    if let Some(statement) = expressions_parser::parse_call_statement(lexer) {
1370        statement
1371    } else {
1372        let statement = AstFactory::create_empty_statement(
1373            lexer.diagnostics.last().map_or(SourceLocation::undefined(), |d| d.get_location()),
1374            lexer.next_id(),
1375        );
1376        statement
1377    }
1378}

1384fn parse_variable_block_type(lexer: &mut ParseSession) -> VariableBlockType {
1385    let block_type = lexer.token;
1386    //Consume the type token
1387    lexer.advance();
1388    let argument_property = if lexer.try_consume(PropertyByRef) {
1389        //Report a diagnostic if blocktype is incompatible
1390        if !matches!(block_type, KeywordVarInput) {
1391            lexer.accept_diagnostic(
1392                Diagnostic::new("Invalid pragma location: Only VAR_INPUT support by ref properties")
1393                    .with_error_code("E024")
1394                    .with_location(lexer.location()),
1395            )
1396        }
1397        ArgumentProperty::ByRef
1398    } else {
1399        ArgumentProperty::ByVal
1400    };
1401    match block_type {
1402        KeywordVar => VariableBlockType::Local,
1403        KeywordVarTemp => VariableBlockType::Temp,
1404        KeywordVarInput => VariableBlockType::Input(argument_property),
1405        KeywordVarOutput => VariableBlockType::Output,
1406        KeywordVarGlobal => VariableBlockType::Global,
1407        KeywordVarInOut => VariableBlockType::InOut,
1408        KeywordVarExternal => VariableBlockType::External,
1409        _ => VariableBlockType::Local,
1410    }
1411}

1413fn parse_variable_block(lexer: &mut ParseSession, linkage: LinkageType) -> VariableBlock {
1414    let location = lexer.location();
1415    let variable_block_type = parse_variable_block_type(lexer);
1416
1417    let constant = lexer.try_consume(KeywordConstant);
1418
1419    let retain = lexer.try_consume(KeywordRetain);
1420    lexer.try_consume(KeywordNonRetain);
1421
1422    let access = parse_access_modifier(lexer);
1423
1424    let mut variables = parse_any_in_region(lexer, vec![KeywordEndVar], parse_variable_list);
1425
1426    if constant && !matches!(variable_block_type, VariableBlockType::External) {
1427        // sneak in the DefaultValue-Statements if no initializers were defined
1428        variables.iter_mut().filter(|it| it.initializer.is_none()).for_each(|it| {
1429            it.initializer = Some(AstFactory::create_default_value(it.location.clone(), lexer.next_id()));
1430        });
1431    }
1432
1433    VariableBlock { access, constant, retain, variables, kind: variable_block_type, linkage, location }
1434}

1436fn parse_variable_list(lexer: &mut ParseSession) -> Vec<Variable> {
1437    let mut variables = vec![];
1438    while lexer.token == Identifier {
1439        let mut line_vars = parse_variable_line(lexer);
1440        variables.append(&mut line_vars);
1441    }
1442    variables
1443}

1445fn parse_config_variables(lexer: &mut ParseSession) -> Vec<ConfigVariable> {
1446    parse_any_in_region(lexer, vec![KeywordEndVar], |lexer| {
1447        lexer.advance();
1448        let mut variables = vec![];
1449        while lexer.token == Identifier {
1450            if let Some(configured_var) =
1451                parse_any_in_region(lexer, vec![KeywordSemicolon], try_parse_config_var)
1452            {
1453                variables.push(configured_var);
1454            }
1455        }
1456        variables
1457    })
1458}

1460fn try_parse_config_var(lexer: &mut ParseSession) -> Option<ConfigVariable> {
1461    let start = lexer.location();
1462    let qualified_reference = parse_reference(lexer);
1463    let location = start.span(&lexer.last_location());
1464    if !lexer.try_consume(KeywordAt) {
1465        lexer.accept_diagnostic(Diagnostic::missing_token("AT", lexer.location()));
1466    }
1467
1468    let HardwareAccess((direction, access_type)) = lexer.token else {
1469        lexer.accept_diagnostic(Diagnostic::missing_token("hardware access", lexer.location()));
1470        return None;
1471    };
1472
1473    let address = parse_hardware_access(lexer, direction, access_type)?;
1474
1475    if !lexer.try_consume(KeywordColon) {
1476        lexer.accept_diagnostic(Diagnostic::missing_token(
1477            format!("{KeywordColon:?}").as_str(),
1478            lexer.location(),
1479        ));
1480    }
1481
1482    parse_data_type_definition(lexer, None).map(|(dt, init)| {
1483        if init.is_some() {
1484            lexer.accept_diagnostic(Diagnostic::unexpected_token_found(
1485                format!("{KeywordSemicolon:?}").as_str(),
1486                "Initializer",
1487                lexer.last_location().span(&lexer.location()),
1488            ))
1489        }
1490        ConfigVariable::new(qualified_reference, dt, address, location)
1491    })
1492}

1494fn parse_aliasing(lexer: &mut ParseSession, names: &(String, Range<usize>)) -> Option<Variable> {
1495    let reference = parse_reference(lexer);
1496    if !lexer.try_consume(KeywordColon) {
1497        lexer.accept_diagnostic(Diagnostic::missing_token(
1498            format!("{KeywordColon:?}").as_str(),
1499            lexer.location(),
1500        ));
1501    }
1502
1503    let start = &lexer.location().get_span().to_range().unwrap_or(lexer.last_range.clone()).start;
1504    let datatype = parse_pointer_definition(lexer, None, *start, Some(AutoDerefType::Alias), true, false);
1505    if !lexer.try_consume(KeywordSemicolon) {
1506        lexer.accept_diagnostic(Diagnostic::missing_token(
1507            format!("{KeywordSemicolon:?}").as_str(),
1508            lexer.location(),
1509        ));
1510    }
1511
1512    if let Some((data_type, _)) = datatype {
1513        return Some(Variable {
1514            name: names.0.clone(),
1515            data_type_declaration: data_type,
1516            location: lexer.source_range_factory.create_range(names.1.clone()),
1517            initializer: Some(reference),
1518            address: None,
1519        });
1520    }
1521
1522    None
1523}

1525fn parse_variable_line(lexer: &mut ParseSession) -> Vec<Variable> {
1526    // read in a comma separated list of variable names
1527    let mut var_names: Vec<(String, Range<usize>)> = vec![];
1528    while lexer.token == Identifier {
1529        let location = lexer.range();
1530        let identifier_end = location.end;
1531        var_names.push((lexer.slice_and_advance(), location));
1532
1533        if lexer.token == KeywordColon || lexer.token == KeywordAt {
1534            break;
1535        }
1536
1537        if !lexer.try_consume(KeywordComma) {
1538            let next_token_start = lexer.range().start;
1539            lexer.accept_diagnostic(Diagnostic::missing_token(
1540                format!("{KeywordColon:?} or {KeywordComma:?}").as_str(),
1541                lexer.source_range_factory.create_range(identifier_end..next_token_start),
1542            ));
1543        }
1544    }
1545
1546    //See if there's an AT keyword
1547    let mut address: Option<AstNode> = None;
1548    if lexer.try_consume(KeywordAt) {
1549        match lexer.token {
1550            HardwareAccess((direction, access_type)) => {
1551                address = parse_hardware_access(lexer, direction, access_type)
1552            }
1553
1554            Identifier => {
1555                return match parse_aliasing(lexer, &var_names[0]) {
1556                    Some(aliased_variable) => vec![aliased_variable],
1557                    None => vec![],
1558                };
1559            }
1560
1561            _ => {
1562                lexer.accept_diagnostic(Diagnostic::missing_token(
1563                    "hardware access or identifier",
1564                    lexer.location(),
1565                ));
1566            }
1567        }
1568    };
1569
1570    // colon has to come before the data type
1571    if !lexer.try_consume(KeywordColon) {
1572        lexer.accept_diagnostic(Diagnostic::missing_token(
1573            format!("{KeywordColon:?}").as_str(),
1574            lexer.location(),
1575        ));
1576    }
1577
1578    // create variables with the same data type for each of the names
1579    let mut variables = vec![];
1580
1581    let parse_definition_opt = if lexer.try_consume(KeywordReferenceTo) {
1582        parse_pointer_definition(
1583            lexer,
1584            None,
1585            lexer.last_range.start,
1586            Some(AutoDerefType::Reference),
1587            true,
1588            false,
1589        )
1590    } else if lexer.try_consume(KeywordFunctionPointer) {
1591        parse_pointer_definition(lexer, None, lexer.last_range.start, None, false, true)
1592    } else if address.is_some() {
1593        parse_pointer_definition(lexer, None, lexer.last_range.start, Some(AutoDerefType::Alias), true, false)
1594    } else {
1595        parse_full_data_type_definition(lexer, None)
1596    };
1597
1598    lexer.try_consume(KeywordSemicolon);
1599
1600    if let Some((data_type, initializer)) = parse_definition_opt {
1601        for (name, range) in var_names {
1602            variables.push(Variable {
1603                name,
1604                data_type_declaration: data_type.clone(),
1605                location: lexer.source_range_factory.create_range(range),
1606                initializer: initializer.clone(),
1607                address: address.clone(),
1608            });
1609        }
1610    }
1611
1612    variables
1613}

1615fn parse_hardware_access(
1616    lexer: &mut ParseSession,
1617    hardware_access_type: HardwareAccessType,
1618    access_type: DirectAccessType,
1619) -> Option<AstNode> {
1620    let start_location = lexer.last_location();
1621    lexer.advance();
1622    //Folowed by an integer
1623    if access_type == DirectAccessType::Template || lexer.token == LiteralInteger {
1624        let mut address = vec![];
1625        if lexer.token == LiteralInteger {
1626            loop {
1627                let int = expressions_parser::parse_strict_literal_integer(lexer)?;
1628                address.push(int);
1629                if !lexer.try_consume(KeywordDot) {
1630                    break;
1631                }
1632            }
1633        }
1634        Some(AstFactory::create_hardware_access(
1635            access_type,
1636            hardware_access_type,
1637            address,
1638            start_location.span(&lexer.last_location()),
1639            lexer.next_id(),
1640        ))
1641    } else {
1642        lexer.accept_diagnostic(Diagnostic::missing_token("LiteralInteger", lexer.location()));
1643        None
1644    }
