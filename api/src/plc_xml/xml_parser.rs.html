51pub(crate) fn visit(content: &str) -> Result<Project<'_>, Error> {
52    let mut reader = Reader::new(content);
53    reader.trim_text(true).expand_empty_elements(true);
54    let mut project = Project::default();
55
56    loop {
57        match reader.read_event()? {
58            Event::Start(tag) if tag.name().as_ref() == b"pou" => {
59                project.pous.push(Pou::visit(&mut reader, Some(tag))?)
60            }
61            Event::Start(tag) if tag.name().as_ref() == b"project" || tag.name().as_ref() == b"pous" => {
62                todo!("Project support comming in #977")
63            }
64            Event::End(tag) if tag.name().as_ref() == b"project" || tag.name().as_ref() == b"pous" => break,
65            Event::Eof => break,
66            _ => {}
67        }
68    }
69    Ok(project)
70}

72pub fn parse_file(
73    source: &SourceCode,
74    linkage: LinkageType,
75    id_provider: IdProvider,
76    diagnostician: &mut Diagnostician,
77) -> Result<CompilationUnit, Diagnostic> {
78    let (unit, errors) = parse(source, linkage, id_provider);
79    //Register the source file with the diagnostician
80    diagnostician.register_file(source.get_location_str().to_string(), source.source.clone()); // TODO: Remove clone here, generally passing the GlobalContext instead of the actual source here or in the handle method should be sufficient
81    if diagnostician.handle(&errors) == Severity::Error {
82        Err(Diagnostic::new("Compilation aborted due to parse errors"))
83    } else {
84        Ok(unit)
85    }
86}

88fn parse(
89    source: &SourceCode,
90    linkage: LinkageType,
91    id_provider: IdProvider,
92) -> (CompilationUnit, Vec<Diagnostic>) {
93    let source_location_factory = SourceLocationFactory::for_source(source);
94    // Transform the xml file to a data model.
95    // XXX: consecutive call-statements are nested in a single ast-statement. this will be broken up with temporary variables in the future
96    let mut project = match visit(&source.source) {
97        Ok(project) => project,
98        Err(why) => todo!("cfc errors need to be transformed into diagnostics; {why:?}"),
99    };
100
101    let mut diagnostics = vec![];
102    let _ = project.desugar(&source_location_factory).map_err(|e| diagnostics.extend(e));
103
104    // Create a new parse session
105    let parser =
106        ParseSession::new(&project, source.get_location_str(), id_provider, linkage, source_location_factory);
107    // Parse the declaration data field
108    let Some((unit, declaration_diagnostics)) = parser.try_parse_declaration() else {
109        unimplemented!("XML schemas without text declarations are not yet supported")
110    };
111    diagnostics.extend(declaration_diagnostics);
112
113    // Transform the data-model into an AST
114    let (implementations, parser_diagnostics) = parser.parse_model();
115    diagnostics.extend(parser_diagnostics);
116
117    (unit.with_implementations(implementations), diagnostics)
118}

129impl<'parse, 'xml> ParseSession<'parse, 'xml> {
130    fn new(
131        project: &'parse Project<'xml>,
132        file_name: &'static str,
133        id_provider: IdProvider,
134        linkage: LinkageType,
135        range_factory: SourceLocationFactory,
136    ) -> Self {
137        ParseSession { project, id_provider, linkage, file_name, range_factory, diagnostics: Vec::new() }
138    }
139
140    /// parse the compilation unit from the addData field
141    fn try_parse_declaration(&self) -> Option<(CompilationUnit, Vec<Diagnostic>)> {
142        let content = self
143            .project
144            .pous
145            .first()
146            .and_then(|it| it.interface.as_ref().and_then(|it| it.get_data_content()))?;
147
148        //TODO: if our ST parser returns a diagnostic here, we might not have a text declaration and need to rely on the XML to provide us with
149        // the necessary data. for now, we will assume to always have a text declaration
150        Some(plc::parser::parse(
151            lexer::lex_with_ids(content, self.id_provider.clone(), self.range_factory.clone()),
152            self.linkage,
153            self.file_name,
154        ))
155    }
156
157    fn parse_expression(&self, expr: &str, local_id: usize, execution_order: Option<usize>) -> AstNode {
158        let mut exp = parse_expression(&mut lexer::lex_with_ids(
159            html_escape::decode_html_entities_to_string(expr, &mut String::new()),
160            self.id_provider.clone(),
161            self.range_factory.clone(),
162        ));
163        let loc = exp.get_location();
164        exp.set_location(self.range_factory.create_block_location(local_id, execution_order).span(&loc));
165        exp
166    }
167
168    fn parse_model(mut self) -> (Vec<Implementation>, Vec<Diagnostic>) {
169        let mut implementations = vec![];
170        for pou in &self.project.pous {
171            // transform body
172            implementations.push(pou.build_implementation(&mut self));
173            // transform actions
174            pou.actions
175                .iter()
176                .for_each(|action| implementations.push(action.build_implementation(&mut self)));
177        }
178
179        (implementations, self.diagnostics)
180    }
181
182    fn next_id(&self) -> AstId {
183        self.id_provider.clone().next_id()
184    }
185
186    fn create_range(&self, range: core::ops::Range<usize>) -> SourceLocation {
187        self.range_factory.create_range(range)
188    }
189
190    fn create_block_location(&self, local_id: usize, execution_order: Option<usize>) -> SourceLocation {
191        self.range_factory.create_block_location(local_id, execution_order)
192    }
193
194    fn create_file_only_location(&self) -> SourceLocation {
195        self.range_factory.create_file_only_location()
196    }
197}
