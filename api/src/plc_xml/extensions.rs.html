1//! In this file extension traits are defined to be used within the `plc_xml` crate.
2
3use std::borrow::Cow;
4
5use quick_xml::name::QName;
6use rustc_hash::FxHashMap;
7
8use crate::error::Error;
9
10/// Trait for [`quick_xml`]s tags defined in [`quick_xml::events`]
11pub(crate) trait TryToString {
12    fn try_to_string(self) -> Result<String, Error>;
13}
14
15/// Trait to extract attribute values from XML elements wrapped inside HashMaps
16pub(crate) trait GetOrErr {
17    fn get_or_err(&self, key: &str) -> Result<String, Error>;
18}
19
20impl TryToString for &[u8] {
21    fn try_to_string(self) -> Result<String, Error> {
22        String::from_utf8(self.as_ref().to_vec()).map_err(|err| Error::Encoding(err.utf8_error()))
23    }
24}
25
26impl TryToString for QName<'_> {
27    fn try_to_string(self) -> Result<String, Error> {
28        String::from_utf8(self.into_inner().to_vec()).map_err(|err| Error::Encoding(err.utf8_error()))
29    }
30}
31
32impl TryToString for Cow<'_, [u8]> {
33    fn try_to_string(self) -> Result<String, Error> {
34        String::from_utf8(self.to_vec()).map_err(|err| Error::Encoding(err.utf8_error()))
35    }
36}
37
38impl GetOrErr for FxHashMap<String, String> {
39    fn get_or_err(&self, key: &str) -> Result<String, Error> {
40        self.get(key).map(|it| it.to_owned()).ok_or(Error::MissingAttribute(key.to_string()))
41    }
42}
