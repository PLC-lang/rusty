7pub(crate) struct Body<'xml> {

9}

11impl<'xml> Body<'xml> {
12    fn new(fbd: FunctionBlockDiagram<'xml>) -> Result<Self, Error> {
13        Ok(Self { function_block_diagram: fbd })
14    }
15
16    fn empty() -> Result<Self, Error> {
17        Ok(Self { function_block_diagram: FunctionBlockDiagram::default() })
18    }
19}
20
21impl Parseable for Body<'_> {
22    fn visit(reader: &mut Reader, _tag: Option<BytesStart>) -> Result<Self, Error> {
23        let mut body = Body::default();
24        loop {
25            match reader.read_event().map_err(Error::ReadEvent)? {
26                Event::Start(tag) if tag.name().as_ref() == b"FBD" => {
27                    body.function_block_diagram = FunctionBlockDiagram::visit(reader, Some(tag))?
28                }
29                Event::End(tag) if tag.name().as_ref() == b"body" => break,
30                Event::Eof => return Err(Error::UnexpectedEndOfFile(vec![b"body"])),
31                _ => {}
32            }
33        }
34
35        Ok(body)
36    }
37}
