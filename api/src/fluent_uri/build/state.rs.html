28/// Indicates the next possible state.

33macro_rules! impl_many {
34    ($trait:ident for $($x:ty => $($y:ty),+)*) => {
35        $($(
36            impl $trait<$y> for $x {}
37        )+)*
38    };
39}

41impl_many! { To for
42    Start => SchemeEnd, AuthorityStart, PathEnd
43    NonRefStart => SchemeEnd
44    SchemeEnd => AuthorityStart, PathEnd
45    AuthorityStart => UserinfoEnd, HostEnd
46    UserinfoEnd => HostEnd
47    HostEnd => PortEnd, AuthorityEnd
48    PortEnd => AuthorityEnd
49    AuthorityEnd => PathEnd
50    PathEnd => QueryEnd, FragmentEnd, End
51    QueryEnd => FragmentEnd, End
52    FragmentEnd => End
53}
