3#[inline]
4const fn utf8_first_byte(byte: u8, width: u32) -> u32 {
5    (byte & (0x7F >> width)) as u32
6}
7
8#[inline]
9const fn utf8_acc_cont_byte(ch: u32, byte: u8) -> u32 {
10    (ch << 6) | (byte & CONT_MASK) as u32
11}
12
13#[inline]
14pub const fn next_code_point(bytes: &[u8], i: usize) -> (u32, usize) {
15    let x = bytes[i];
16    if x < 128 {
17        return (x as u32, 1);
18    }
19
20    let init = utf8_first_byte(x, 2);
21    let y = bytes[i + 1];
22    if x < 0xE0 {
23        (utf8_acc_cont_byte(init, y), 2)
24    } else {
25        let z = bytes[i + 2];
26        let y_z = utf8_acc_cont_byte((y & CONT_MASK) as u32, z);
27        if x < 0xF0 {
28            ((init << 12) | y_z, 3)
29        } else {
30            let w = bytes[i + 3];
31            (((init & 7) << 18) | utf8_acc_cont_byte(y_z, w), 4)
32        }
33    }
34}

66pub struct Utf8Chunk<'a> {

99#[cfg(feature = "alloc")]
100impl<'a> Iterator for Utf8Chunks<'a> {
101    type Item = Utf8Chunk<'a>;
102
103    fn next(&mut self) -> Option<Utf8Chunk<'a>> {
104        if self.source.is_empty() {
105            return None;
106        }
107
108        const TAG_CONT_U8: u8 = 128;
109
110        let mut incomplete = false;
111        let mut safe_get = |i| match self.source.get(i) {
112            Some(x) => *x,
113            None => {
114                incomplete = true;
115                0
116            }
117        };
118
119        let mut i = 0;
120        let mut valid_up_to = 0;
121        while i < self.source.len() {
122            let byte = self.source[i];
123            i += 1;
124
125            if byte >= 128 {
126                let w = utf8_char_width(byte);
127
128                match w {
129                    2 => {
130                        if safe_get(i) & 192 != TAG_CONT_U8 {
131                            break;
132                        }
133                        i += 1;
134                    }
135                    3 => {
136                        match (byte, safe_get(i)) {
137                            (0xE0, 0xA0..=0xBF) => (),
138                            (0xE1..=0xEC, 0x80..=0xBF) => (),
139                            (0xED, 0x80..=0x9F) => (),
140                            (0xEE..=0xEF, 0x80..=0xBF) => (),
141                            _ => break,
142                        }
143                        i += 1;
144                        if safe_get(i) & 192 != TAG_CONT_U8 {
145                            break;
146                        }
147                        i += 1;
148                    }
149                    4 => {
150                        match (byte, safe_get(i)) {
151                            (0xF0, 0x90..=0xBF) => (),
152                            (0xF1..=0xF3, 0x80..=0xBF) => (),
153                            (0xF4, 0x80..=0x8F) => (),
154                            _ => break,
155                        }
156                        i += 1;
157                        if safe_get(i) & 192 != TAG_CONT_U8 {
158                            break;
159                        }
160                        i += 1;
161                        if safe_get(i) & 192 != TAG_CONT_U8 {
162                            break;
163                        }
164                        i += 1;
165                    }
166                    _ => break,
167                }
168            }
169
170            valid_up_to = i;
171        }
172
173        let (inspected, remaining) = self.source.split_at(i);
174        self.source = remaining;
175
176        let (valid, invalid) = inspected.split_at(valid_up_to);
177
178        Some(Utf8Chunk {
179            valid: core::str::from_utf8(valid).unwrap(),
180            invalid,
181            incomplete,
182        })
183    }
184}
