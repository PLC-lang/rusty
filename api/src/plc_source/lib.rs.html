11/// Represents the type of source a SourceContainer holds
12#[derive(Clone, Copy, Debug)]
13pub enum SourceType {
14    /// A normal text file, the content of the file could be parsed
15    Text,
16
17    /// An xml file, probably cfc
18    Xml,
19
20    /// Unknown type, probably a binary
21    Unknown,
22}
23
24/// SourceContainers offer source-code to be compiled via the load_source function.
25/// Furthermore it offers a location-String used when reporting diagnostics.
26pub trait SourceContainer: Sync + Send {
27    /// loads and returns the SourceEntry that contains the SourceCode and the path it was loaded from
28    fn load_source(&self, encoding: Option<&'static Encoding>) -> Result<SourceCode, String>;
29    /// returns the location of this source-container. Used when reporting diagnostics.
30    fn get_location(&self) -> Option<&Path>;
31
32    /// Returns the `SourceType` for the current container,
33    /// by default everything is Text unless it has the extension for a known binary/object
34    fn get_type(&self) -> SourceType {
35        if let Some(ext) = self.get_location().and_then(|it| it.extension()) {
36            match ext.to_str() {
37                Some("o") | Some("so") | Some("exe") => SourceType::Unknown,
38                //XXX: file ending vs first line? (<?xml ...)
39                Some("cfc") | Some("fbd") | Some("xml") => SourceType::Xml,
40                _ => SourceType::Text,
41            }
42        } else {
43            SourceType::Text
44        }
45    }
46
47    /// Returns a staticly available location for this source
48    fn get_location_str(&self) -> &'static str {
49        let s = self
50            .get_location()
51            .map(|it| it.to_string_lossy())
52            .map(|it| it.to_string())
53            .unwrap_or_else(|| "<internal>".into());
54        Box::leak(s.into_boxed_str())
55    }
56}

58/// The SourceCode unit is the smallest unit of compilation that can be passed to the compiler

91impl<T: AsRef<Path> + Sync + Send> SourceContainer for T {
92    fn load_source(&self, encoding: Option<&'static Encoding>) -> Result<SourceCode, String> {
93        let source_type = self.get_type();
94        if matches!(source_type, SourceType::Text | SourceType::Xml) {
95            let mut file = File::open(self).map_err(|err| err.to_string())?;
96            let source = create_source_code(&mut file, encoding)?;
97
98            Ok(SourceCode { source, path: Some(self.as_ref().to_owned()) })
99        } else {
100            Err(format!("{} is not a source file", &self.as_ref().to_string_lossy()))
101        }
102    }
103
104    fn get_location(&self) -> Option<&Path> {
105        Some(self.as_ref())
106    }
107}

109pub fn create_source_code<T: Read>(
110    reader: &mut T,
111    encoding: Option<&'static Encoding>,
112) -> Result<String, String> {
113    let mut buffer = String::new();
114    let mut decoder = DecodeReaderBytesBuilder::new().encoding(encoding).build(reader);
115    decoder.read_to_string(&mut buffer).map_err(|err| format!("{err}"))?;
116    Ok(buffer)
117}
