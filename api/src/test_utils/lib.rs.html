31pub fn parse_and_validate_buffered_ast(src: &str) -> Vec<CompilationUnit> {
32    let source: SourceCode = src.into();
33
34    match driver::parse_and_annotate_with_diagnostics("TestProject", vec![source], Diagnostician::buffered())
35    {
36        Ok((mut pipeline, project)) => {
37            project.validate(&pipeline.context, &mut pipeline.diagnostician).unwrap();
38            project.units.into_iter().map(CompilationUnit::from).collect()
39        }
40        Err(diagnostician) => panic!("{}", diagnostician.buffer().unwrap()),
41    }
42}

44fn get_debug_param(debug_level: DebugLevel) -> Option<String> {
45    match debug_level {
46        DebugLevel::None => None,
47        DebugLevel::VariablesOnly(plc::DEFAULT_DWARF_VERSION) => Some("--debug-variables".to_string()),
48        DebugLevel::VariablesOnly(version) => Some(format!("--gdwarf-variables={version}")),
49        DebugLevel::Full(plc::DEFAULT_DWARF_VERSION) => Some("--debug".to_string()),
50        DebugLevel::Full(version) => Some(format!("--gdwarf={version}")),
51    }
52}

54pub fn codegen_debug_without_unwrap(src: &str, debug_level: DebugLevel) -> Result<String, String> {
55    //Create a temp file to store the result
56    let mut output = NamedTempFile::new().map_err(|it| it.to_string())?;
57    let src: SourceCode = src.into();
58    let project = Project::new("Test".to_string())
59        .with_sources(vec![src])
60        .with_output_name(output.path().to_str().map(ToString::to_string));
61    let context =
62        GlobalContext::new().with_source(project.get_sources(), None).map_err(|it| it.to_string())?;
63    let diagnostician = Diagnostician::default();
64    let mut args =
65        vec!["plc", "--ir", "--single-module", "-O", "none", "-o", output.path().to_str().unwrap()];
66    let debug_level = get_debug_param(debug_level);
67    if let Some(debug) = &debug_level {
68        args.push(debug);
69    };
70    let params = cli::CompileParameters::parse(&args).map_err(|e| e.to_string())?;
71    let pipeline = BuildPipeline {
72        context,
73        project,
74        diagnostician,
75        compile_parameters: Some(params),
76        linker: LinkerType::Internal,
77        mutable_participants: Vec::default(),
78        participants: Vec::default(),
79        module_name: Some("<internal>".to_string()),
80    };
81
82    driver::compile_with_pipeline(pipeline).map_err(|it| it.to_string())?;
83
84    let mut res = String::new();
85    output.read_to_string(&mut res).map_err(|it| it.to_string())?;
86    Ok(res)
87}
