61/// Tokenizer options, with an impl for `Default`.
62#[derive(Clone)]
63pub struct TokenizerOpts {
64    /// Report all parse errors described in the spec, at some
65    /// performance penalty?  Default: false
66    pub exact_errors: bool,
67
68    /// Discard a `U+FEFF BYTE ORDER MARK` if we see one at the beginning
69    /// of the stream?  Default: true
70    pub discard_bom: bool,
71
72    /// Keep a record of how long we spent in each state?  Printed
73    /// when `end()` is called.  Default: false
74    pub profile: bool,
75
76    /// Initial state override.  Only the test runner should use
77    /// a non-`None` value!
78    pub initial_state: Option<states::State>,
79
80    /// Last start tag.  Only the test runner should use a
81    /// non-`None` value!
82    ///
83    /// FIXME: Can't use Tendril because we want TokenizerOpts
84    /// to be Send.
85    pub last_start_tag_name: Option<String>,
86}
87
88impl Default for TokenizerOpts {
89    fn default() -> TokenizerOpts {
90        TokenizerOpts {
91            exact_errors: false,
92            discard_bom: true,
93            profile: false,
94            initial_state: None,
95            last_start_tag_name: None,
96        }
97    }

174    /// Create a new tokenizer which feeds tokens to a particular `TokenSink`.
175    pub fn new(sink: Sink, mut opts: TokenizerOpts) -> Tokenizer<Sink> {
176        let start_tag_name = opts
177            .last_start_tag_name
178            .take()
179            .map(|s| LocalName::from(&*s));
180        let state = opts.initial_state.unwrap_or(states::Data);
181        let discard_bom = opts.discard_bom;
182        Tokenizer {
183            opts,
184            sink,
185            state,
186            char_ref_tokenizer: None,
187            at_eof: false,
188            current_char: '\0',
189            reconsume: false,
190            ignore_lf: false,
191            discard_bom,
192            current_tag_kind: StartTag,
193            current_tag_name: StrTendril::new(),
194            current_tag_self_closing: false,
195            current_tag_attrs: vec![],
196            current_attr_name: StrTendril::new(),
197            current_attr_value: StrTendril::new(),
198            current_comment: StrTendril::new(),
199            current_doctype: Doctype::default(),
200            last_start_tag_name: start_tag_name,
201            temp_buf: StrTendril::new(),
202            state_profile: BTreeMap::new(),
203            time_in_sink: 0,
204            current_line: 1,
205        }
206    }

208    /// Feed an input string into the tokenizer.
209    pub fn feed(&mut self, input: &mut BufferQueue) -> TokenizerResult<Sink::Handle> {
210        if input.is_empty() {
211            return TokenizerResult::Done;
212        }
213
214        if self.discard_bom {
215            if let Some(c) = input.peek() {
216                if c == '\u{feff}' {
217                    input.next();
218                }
219            } else {
220                return TokenizerResult::Done;
221            }
222        };
223
224        self.run(input)
225    }

231    fn process_token(&mut self, token: Token) -> TokenSinkResult<Sink::Handle> {
232        if self.opts.profile {
233            let (ret, dt) = time!(self.sink.process_token(token, self.current_line));
234            self.time_in_sink += dt;
235            ret
236        } else {
237            self.sink.process_token(token, self.current_line)
238        }
239    }

241    fn process_token_and_continue(&mut self, token: Token) {
242        assert!(matches!(

244            TokenSinkResult::Continue

248    //§ preprocessing-the-input-stream
249    // Get the next input character, which might be the character
250    // 'c' that we already consumed from the buffers.
251    fn get_preprocessed_char(&mut self, mut c: char, input: &mut BufferQueue) -> Option<char> {
252        if self.ignore_lf {
253            self.ignore_lf = false;
254            if c == '\n' {
255                c = unwrap_or_return!(input.next(), None);
256            }
257        }
258
259        if c == '\r' {
260            self.ignore_lf = true;
261            c = '\n';
262        }
263
264        if c == '\n' {
265            self.current_line += 1;
266        }
267
268        if self.opts.exact_errors
269            && match c as u32 {
270                0x01..=0x08 | 0x0B | 0x0E..=0x1F | 0x7F..=0x9F | 0xFDD0..=0xFDEF => true,
271                n if (n & 0xFFFE) == 0xFFFE => true,
272                _ => false,
273            }
274        {
275            let msg = format!("Bad character {}", c);
276            self.emit_error(Cow::Owned(msg));
277        }
278
279        trace!("got character {}", c);
280        self.current_char = c;
281        Some(c)
282    }

285    // Get the next input character, if one is available.
286    fn get_char(&mut self, input: &mut BufferQueue) -> Option<char> {
287        if self.reconsume {
288            self.reconsume = false;
289            Some(self.current_char)
290        } else {
291            input
292                .next()
293                .and_then(|c| self.get_preprocessed_char(c, input))
294        }
295    }

297    fn pop_except_from(&mut self, input: &mut BufferQueue, set: SmallCharSet) -> Option<SetResult> {
298        // Bail to the slow path for various corner cases.
299        // This means that `FromSet` can contain characters not in the set!
300        // It shouldn't matter because the fallback `FromSet` case should
301        // always do the same thing as the `NotFromSet` case.
302        if self.opts.exact_errors || self.reconsume || self.ignore_lf {
303            return self.get_char(input).map(FromSet);
304        }
305
306        let d = input.pop_except_from(set);
307        trace!("got characters {:?}", d);
308        match d {
309            Some(FromSet(c)) => self.get_preprocessed_char(c, input).map(FromSet),
310
311            // NB: We don't set self.current_char for a run of characters not
312            // in the set.  It shouldn't matter for the codepaths that use
313            // this.
314            _ => d,
315        }
316    }

318    // Check if the next characters are an ASCII case-insensitive match.  See
319    // BufferQueue::eat.
320    //
321    // NB: this doesn't set the current input character.
322    fn eat(
323        &mut self,
324        input: &mut BufferQueue,
325        pat: &str,
326        eq: fn(&u8, &u8) -> bool,
327    ) -> Option<bool> {
328        if self.ignore_lf {
329            self.ignore_lf = false;
330            if self.peek(input) == Some('\n') {
331                self.discard_char(input);
332            }
333        }
334
335        input.push_front(mem::take(&mut self.temp_buf));
336        match input.eat(pat, eq) {
337            None if self.at_eof => Some(false),
338            None => {
339                self.temp_buf.extend(input);
340                None
341            },
342            Some(matched) => Some(matched),
343        }
344    }

346    /// Run the state machine for as long as we can.
347    fn run(&mut self, input: &mut BufferQueue) -> TokenizerResult<Sink::Handle> {
348        if self.opts.profile {
349            loop {
350                let state = self.state;
351                let old_sink = self.time_in_sink;
352                let (run, mut dt) = time!(self.step(input));
353                dt -= (self.time_in_sink - old_sink);
354                let new = match self.state_profile.get_mut(&state) {
355                    Some(x) => {
356                        *x += dt;
357                        false
358                    },
359                    None => true,
360                };
361                if new {
362                    // do this here because of borrow shenanigans
363                    self.state_profile.insert(state, dt);
364                }
365                match run {
366                    ProcessResult::Continue => (),
367                    ProcessResult::Suspend => break,
368                    ProcessResult::Script(node) => return TokenizerResult::Script(node),
369                }
370            }
371        } else {
372            loop {
373                match self.step(input) {
374                    ProcessResult::Continue => (),
375                    ProcessResult::Suspend => break,
376                    ProcessResult::Script(node) => return TokenizerResult::Script(node),
377                }
378            }
379        }
380        TokenizerResult::Done
381    }

404    fn emit_char(&mut self, c: char) {
405        self.process_token_and_continue(match c {

416    fn emit_current_tag(&mut self) -> ProcessResult<Sink::Handle> {
417        self.finish_attribute();
418
419        let name = LocalName::from(&*self.current_tag_name);
420        self.current_tag_name.clear();
421
422        match self.current_tag_kind {
423            StartTag => {
424                self.last_start_tag_name = Some(name.clone());
425            },
426            EndTag => {
427                if !self.current_tag_attrs.is_empty() {
428                    self.emit_error(Borrowed("Attributes on an end tag"));
429                }
430                if self.current_tag_self_closing {
431                    self.emit_error(Borrowed("Self-closing end tag"));
432                }
433            },
434        }
435
436        let token = TagToken(Tag {
437            kind: self.current_tag_kind,
438            name,
439            self_closing: self.current_tag_self_closing,
440            attrs: std::mem::take(&mut self.current_tag_attrs),
441        });
442
443        match self.process_token(token) {
444            TokenSinkResult::Continue => ProcessResult::Continue,
445            TokenSinkResult::Plaintext => {
446                self.state = states::Plaintext;
447                ProcessResult::Continue
448            },
449            TokenSinkResult::Script(node) => {
450                self.state = states::Data;
451                ProcessResult::Script(node)
452            },
453            TokenSinkResult::RawData(kind) => {
454                self.state = states::RawData(kind);
455                ProcessResult::Continue
456            },
457        }
458    }

488    fn have_appropriate_end_tag(&self) -> bool {
489        match self.last_start_tag_name.as_ref() {
490            Some(last) => (self.current_tag_kind == EndTag) && (*self.current_tag_name == **last),
491            None => false,
492        }
493    }

495    fn create_attribute(&mut self, c: char) {
496        self.finish_attribute();
497
498        self.current_attr_name.push_char(c);
499    }
500
501    fn finish_attribute(&mut self) {
502        if self.current_attr_name.is_empty() {
503            return;
504        }
505
506        // Check for a duplicate attribute.
507        // FIXME: the spec says we should error as soon as the name is finished.
508        let dup = {
509            let name = &*self.current_attr_name;
510            self.current_tag_attrs
511                .iter()
512                .any(|a| &*a.name.local == name)
513        };
514
515        if dup {
516            self.emit_error(Borrowed("Duplicate attribute"));
517            self.current_attr_name.clear();
518            self.current_attr_value.clear();
519        } else {
520            let name = LocalName::from(&*self.current_attr_name);
521            self.current_attr_name.clear();
522            self.current_tag_attrs.push(Attribute {
523                // The tree builder will adjust the namespace if necessary.
524                // This only happens in foreign elements.
525                name: QualName::new(None, ns!(), name),
526                value: mem::take(&mut self.current_attr_value),
527            });
528        }
529    }

562    fn peek(&mut self, input: &BufferQueue) -> Option<char> {
563        if self.reconsume {
564            Some(self.current_char)

570    fn discard_char(&mut self, input: &mut BufferQueue) {
571        // peek() deals in un-processed characters (no newline normalization), while get_char()
572        // does.
573        //
574        // since discard_char is supposed to be used in combination with peek(), discard_char must
575        // discard a single raw input character, not a normalized newline.
576        if self.reconsume {
577            self.reconsume = false;
578        } else {
579            input.next();
580        }
581    }

589// Shorthand for common state machine behaviors.
590macro_rules! shorthand (
591    ( $me:ident : emit $c:expr                     ) => ( $me.emit_char($c)                                   );
592    ( $me:ident : create_tag $kind:ident $c:expr   ) => ( $me.create_tag($kind, $c)                           );
593    ( $me:ident : push_tag $c:expr                 ) => ( $me.current_tag_name.push_char($c)                  );
594    ( $me:ident : discard_tag                      ) => ( $me.discard_tag()                                   );
595    ( $me:ident : discard_char $input:expr         ) => ( $me.discard_char($input)                            );
596    ( $me:ident : push_temp $c:expr                ) => ( $me.temp_buf.push_char($c)                          );
597    ( $me:ident : emit_temp                        ) => ( $me.emit_temp_buf()                                 );
598    ( $me:ident : clear_temp                       ) => ( $me.clear_temp_buf()                                );
599    ( $me:ident : create_attr $c:expr              ) => ( $me.create_attribute($c)                            );
600    ( $me:ident : push_name $c:expr                ) => ( $me.current_attr_name.push_char($c)                 );
601    ( $me:ident : push_value $c:expr               ) => ( $me.current_attr_value.push_char($c)                );
602    ( $me:ident : append_value $c:expr             ) => ( $me.current_attr_value.push_tendril($c)             );
603    ( $me:ident : push_comment $c:expr             ) => ( $me.current_comment.push_char($c)                   );
604    ( $me:ident : append_comment $c:expr           ) => ( $me.current_comment.push_slice($c)                  );
605    ( $me:ident : emit_comment                     ) => ( $me.emit_current_comment()                          );
606    ( $me:ident : clear_comment                    ) => ( $me.current_comment.clear()                         );
607    ( $me:ident : create_doctype                   ) => ( $me.current_doctype = Doctype::default()            );
608    ( $me:ident : push_doctype_name $c:expr        ) => ( option_push(&mut $me.current_doctype.name, $c)      );
609    ( $me:ident : push_doctype_id $k:ident $c:expr ) => ( option_push($me.doctype_id($k), $c)                 );
610    ( $me:ident : clear_doctype_id $k:ident        ) => ( $me.clear_doctype_id($k)                            );
611    ( $me:ident : force_quirks                     ) => ( $me.current_doctype.force_quirks = true             );
612    ( $me:ident : emit_doctype                     ) => ( $me.emit_current_doctype()                          );
613    ( $me:ident : error                            ) => ( $me.bad_char_error()                                );
614    ( $me:ident : error_eof                        ) => ( $me.bad_eof_error()                                 );
615);

628// A little DSL for sequencing shorthand actions.
629macro_rules! go (
630    // A pattern like $($cmd:tt)* ; $($rest:tt)* causes parse ambiguity.
631    // We have to tell the parser how much lookahead we need.
632
633    ( $me:ident : $a:tt                   ; $($rest:tt)* ) => ({ sh_trace!($me: $a);          go!($me: $($rest)*); });
634    ( $me:ident : $a:tt $b:tt             ; $($rest:tt)* ) => ({ sh_trace!($me: $a $b);       go!($me: $($rest)*); });
635    ( $me:ident : $a:tt $b:tt $c:tt       ; $($rest:tt)* ) => ({ sh_trace!($me: $a $b $c);    go!($me: $($rest)*); });
636    ( $me:ident : $a:tt $b:tt $c:tt $d:tt ; $($rest:tt)* ) => ({ sh_trace!($me: $a $b $c $d); go!($me: $($rest)*); });
637
638    // These can only come at the end.
639
640    ( $me:ident : to $s:ident                    ) => ({ $me.state = states::$s; return ProcessResult::Continue;           });
641    ( $me:ident : to $s:ident $k1:expr           ) => ({ $me.state = states::$s($k1); return ProcessResult::Continue;      });
642    ( $me:ident : to $s:ident $k1:ident $k2:expr ) => ({ $me.state = states::$s($k1($k2)); return ProcessResult::Continue; });
643
644    ( $me:ident : reconsume $s:ident                    ) => ({ $me.reconsume = true; go!($me: to $s);         });
645    ( $me:ident : reconsume $s:ident $k1:expr           ) => ({ $me.reconsume = true; go!($me: to $s $k1);     });
646    ( $me:ident : reconsume $s:ident $k1:ident $k2:expr ) => ({ $me.reconsume = true; go!($me: to $s $k1 $k2); });
647
648    ( $me:ident : consume_char_ref             ) => ({ $me.consume_char_ref(); return ProcessResult::Continue;         });
649
650    // We have a default next state after emitting a tag, but the sink can override.
651    ( $me:ident : emit_tag $s:ident ) => ({
652        $me.state = states::$s;
653        return $me.emit_current_tag();
654    });
655
656    ( $me:ident : eof ) => ({ $me.emit_eof(); return ProcessResult::Suspend; });
657
658    // If nothing else matched, it's a single command
659    ( $me:ident : $($cmd:tt)+ ) => ( sh_trace!($me: $($cmd)+) );
660
661    // or nothing.
662    ( $me:ident : ) => (());
663);
664
665macro_rules! go_match ( ( $me:ident : $x:expr, $($pats:pat),+ => $($cmds:tt)* ) => (
666    match $x {
667        $($pats)|+ => go!($me: $($cmds)*),
668        _ => (),
669    }
670));

682macro_rules! pop_except_from ( ($me:expr, $input:expr, $set:expr) => (

686macro_rules! eat ( ($me:expr, $input:expr, $pat:expr) => (

694impl<Sink: TokenSink> Tokenizer<Sink> {
695    // Run the state machine for a while.
696    // Return true if we should be immediately re-invoked
697    // (this just simplifies control flow vs. break / continue).
698    #[allow(clippy::never_loop)]
699    fn step(&mut self, input: &mut BufferQueue) -> ProcessResult<Sink::Handle> {
700        if self.char_ref_tokenizer.is_some() {
701            return self.step_char_ref_tokenizer(input);
702        }
703
704        trace!("processing in state {:?}", self.state);
705        match self.state {
706            //§ data-state
707            states::Data => loop {
708                match pop_except_from!(self, input, small_char_set!('\r' '\0' '&' '<' '\n')) {
709                    FromSet('\0') => go!(self: error; emit '\0'),
710                    FromSet('&') => go!(self: consume_char_ref),
711                    FromSet('<') => go!(self: to TagOpen),
712                    FromSet(c) => go!(self: emit c),
713                    NotFromSet(b) => self.emit_chars(b),
714                }
715            },
716
717            //§ rcdata-state
718            states::RawData(Rcdata) => loop {
719                match pop_except_from!(self, input, small_char_set!('\r' '\0' '&' '<' '\n')) {
720                    FromSet('\0') => go!(self: error; emit '\u{fffd}'),
721                    FromSet('&') => go!(self: consume_char_ref),
722                    FromSet('<') => go!(self: to RawLessThanSign Rcdata),
723                    FromSet(c) => go!(self: emit c),
724                    NotFromSet(b) => self.emit_chars(b),
725                }
726            },
727
728            //§ rawtext-state
729            states::RawData(Rawtext) => loop {
730                match pop_except_from!(self, input, small_char_set!('\r' '\0' '<' '\n')) {
731                    FromSet('\0') => go!(self: error; emit '\u{fffd}'),
732                    FromSet('<') => go!(self: to RawLessThanSign Rawtext),
733                    FromSet(c) => go!(self: emit c),
734                    NotFromSet(b) => self.emit_chars(b),
735                }
736            },
737
738            //§ script-data-state
739            states::RawData(ScriptData) => loop {
740                match pop_except_from!(self, input, small_char_set!('\r' '\0' '<' '\n')) {
741                    FromSet('\0') => go!(self: error; emit '\u{fffd}'),
742                    FromSet('<') => go!(self: to RawLessThanSign ScriptData),
743                    FromSet(c) => go!(self: emit c),
744                    NotFromSet(b) => self.emit_chars(b),
745                }
746            },
747
748            //§ script-data-escaped-state
749            states::RawData(ScriptDataEscaped(Escaped)) => loop {
750                match pop_except_from!(self, input, small_char_set!('\r' '\0' '-' '<' '\n')) {
751                    FromSet('\0') => go!(self: error; emit '\u{fffd}'),
752                    FromSet('-') => go!(self: emit '-'; to ScriptDataEscapedDash Escaped),
753                    FromSet('<') => go!(self: to RawLessThanSign ScriptDataEscaped Escaped),
754                    FromSet(c) => go!(self: emit c),
755                    NotFromSet(b) => self.emit_chars(b),
756                }
757            },
758
759            //§ script-data-double-escaped-state
760            states::RawData(ScriptDataEscaped(DoubleEscaped)) => loop {
761                match pop_except_from!(self, input, small_char_set!('\r' '\0' '-' '<' '\n')) {
762                    FromSet('\0') => go!(self: error; emit '\u{fffd}'),
763                    FromSet('-') => go!(self: emit '-'; to ScriptDataEscapedDash DoubleEscaped),
764                    FromSet('<') => {
765                        go!(self: emit '<'; to RawLessThanSign ScriptDataEscaped DoubleEscaped)
766                    },
767                    FromSet(c) => go!(self: emit c),
768                    NotFromSet(b) => self.emit_chars(b),
769                }
770            },
771
772            //§ plaintext-state
773            states::Plaintext => loop {
774                match pop_except_from!(self, input, small_char_set!('\r' '\0' '\n')) {
775                    FromSet('\0') => go!(self: error; emit '\u{fffd}'),
776                    FromSet(c) => go!(self: emit c),
777                    NotFromSet(b) => self.emit_chars(b),
778                }
779            },
780
781            //§ tag-open-state
782            states::TagOpen => loop {
783                match get_char!(self, input) {
784                    '!' => go!(self: to MarkupDeclarationOpen),
785                    '/' => go!(self: to EndTagOpen),
786                    '?' => go!(self: error; clear_comment; reconsume BogusComment),
787                    c => match lower_ascii_letter(c) {
788                        Some(cl) => go!(self: create_tag StartTag cl; to TagName),
789                        None => go!(self: error; emit '<'; reconsume Data),
790                    },
791                }
792            },
793
794            //§ end-tag-open-state
795            states::EndTagOpen => loop {
796                match get_char!(self, input) {
797                    '>' => go!(self: error; to Data),
798                    c => match lower_ascii_letter(c) {
799                        Some(cl) => go!(self: create_tag EndTag cl; to TagName),
800                        None => go!(self: error; clear_comment; reconsume BogusComment),
801                    },
802                }
803            },
804
805            //§ tag-name-state
806            states::TagName => loop {
807                match get_char!(self, input) {
808                    '\t' | '\n' | '\x0C' | ' ' => go!(self: to BeforeAttributeName),
809                    '/' => go!(self: to SelfClosingStartTag),
810                    '>' => go!(self: emit_tag Data),
811                    '\0' => go!(self: error; push_tag '\u{fffd}'),
812                    c => go!(self: push_tag (c.to_ascii_lowercase())),
813                }
814            },
815
816            //§ script-data-escaped-less-than-sign-state
817            states::RawLessThanSign(ScriptDataEscaped(Escaped)) => loop {
818                match get_char!(self, input) {
819                    '/' => go!(self: clear_temp; to RawEndTagOpen ScriptDataEscaped Escaped),
820                    c => match lower_ascii_letter(c) {
821                        Some(cl) => go!(self: clear_temp; push_temp cl; emit '<'; emit c;
822                                    to ScriptDataEscapeStart DoubleEscaped),
823                        None => go!(self: emit '<'; reconsume RawData ScriptDataEscaped Escaped),
824                    },
825                }
826            },
827
828            //§ script-data-double-escaped-less-than-sign-state
829            states::RawLessThanSign(ScriptDataEscaped(DoubleEscaped)) => loop {
830                match get_char!(self, input) {
831                    '/' => go!(self: clear_temp; emit '/'; to ScriptDataDoubleEscapeEnd),
832                    _ => go!(self: reconsume RawData ScriptDataEscaped DoubleEscaped),
833                }
834            },
835
836            //§ rcdata-less-than-sign-state rawtext-less-than-sign-state script-data-less-than-sign-state
837            // otherwise
838            states::RawLessThanSign(kind) => loop {
839                match get_char!(self, input) {
840                    '/' => go!(self: clear_temp; to RawEndTagOpen kind),
841                    '!' if kind == ScriptData => {
842                        go!(self: emit '<'; emit '!'; to ScriptDataEscapeStart Escaped)
843                    },
844                    _ => go!(self: emit '<'; reconsume RawData kind),
845                }
846            },
847
848            //§ rcdata-end-tag-open-state rawtext-end-tag-open-state script-data-end-tag-open-state script-data-escaped-end-tag-open-state
849            states::RawEndTagOpen(kind) => loop {
850                let c = get_char!(self, input);
851                match lower_ascii_letter(c) {
852                    Some(cl) => go!(self: create_tag EndTag cl; push_temp c; to RawEndTagName kind),
853                    None => go!(self: emit '<'; emit '/'; reconsume RawData kind),
854                }
855            },
856
857            //§ rcdata-end-tag-name-state rawtext-end-tag-name-state script-data-end-tag-name-state script-data-escaped-end-tag-name-state
858            states::RawEndTagName(kind) => loop {
859                let c = get_char!(self, input);
860                if self.have_appropriate_end_tag() {
861                    match c {
862                        '\t' | '\n' | '\x0C' | ' ' => go!(self: clear_temp; to BeforeAttributeName),
863                        '/' => go!(self: clear_temp; to SelfClosingStartTag),
864                        '>' => go!(self: clear_temp; emit_tag Data),
865                        _ => (),
866                    }
867                }
868
869                match lower_ascii_letter(c) {
870                    Some(cl) => go!(self: push_tag cl; push_temp c),
871                    None => {
872                        go!(self: discard_tag; emit '<'; emit '/'; emit_temp; reconsume RawData kind)
873                    },
874                }
875            },
876
877            //§ script-data-double-escape-start-state
878            states::ScriptDataEscapeStart(DoubleEscaped) => loop {
879                let c = get_char!(self, input);
880                match c {
881                    '\t' | '\n' | '\x0C' | ' ' | '/' | '>' => {
882                        let esc = if &*self.temp_buf == "script" {
883                            DoubleEscaped
884                        } else {
885                            Escaped
886                        };
887                        go!(self: emit c; to RawData ScriptDataEscaped esc);
888                    },
889                    _ => match lower_ascii_letter(c) {
890                        Some(cl) => go!(self: push_temp cl; emit c),
891                        None => go!(self: reconsume RawData ScriptDataEscaped Escaped),
892                    },
893                }
894            },
895
896            //§ script-data-escape-start-state
897            states::ScriptDataEscapeStart(Escaped) => loop {
898                match get_char!(self, input) {
899                    '-' => go!(self: emit '-'; to ScriptDataEscapeStartDash),
900                    _ => go!(self: reconsume RawData ScriptData),
901                }
902            },
903
904            //§ script-data-escape-start-dash-state
905            states::ScriptDataEscapeStartDash => loop {
906                match get_char!(self, input) {
907                    '-' => go!(self: emit '-'; to ScriptDataEscapedDashDash Escaped),
908                    _ => go!(self: reconsume RawData ScriptData),
909                }
910            },
911
912            //§ script-data-escaped-dash-state script-data-double-escaped-dash-state
913            states::ScriptDataEscapedDash(kind) => loop {
914                match get_char!(self, input) {
915                    '-' => go!(self: emit '-'; to ScriptDataEscapedDashDash kind),
916                    '<' => {
917                        if kind == DoubleEscaped {
918                            go!(self: emit '<');
919                        }
920                        go!(self: to RawLessThanSign ScriptDataEscaped kind);
921                    },
922                    '\0' => go!(self: error; emit '\u{fffd}'; to RawData ScriptDataEscaped kind),
923                    c => go!(self: emit c; to RawData ScriptDataEscaped kind),
924                }
925            },
926
927            //§ script-data-escaped-dash-dash-state script-data-double-escaped-dash-dash-state
928            states::ScriptDataEscapedDashDash(kind) => loop {
929                match get_char!(self, input) {
930                    '-' => go!(self: emit '-'),
931                    '<' => {
932                        if kind == DoubleEscaped {
933                            go!(self: emit '<');
934                        }
935                        go!(self: to RawLessThanSign ScriptDataEscaped kind);
936                    },
937                    '>' => go!(self: emit '>'; to RawData ScriptData),
938                    '\0' => go!(self: error; emit '\u{fffd}'; to RawData ScriptDataEscaped kind),
939                    c => go!(self: emit c; to RawData ScriptDataEscaped kind),
940                }
941            },
942
943            //§ script-data-double-escape-end-state
944            states::ScriptDataDoubleEscapeEnd => loop {
945                let c = get_char!(self, input);
946                match c {
947                    '\t' | '\n' | '\x0C' | ' ' | '/' | '>' => {
948                        let esc = if &*self.temp_buf == "script" {
949                            Escaped
950                        } else {
951                            DoubleEscaped
952                        };
953                        go!(self: emit c; to RawData ScriptDataEscaped esc);
954                    },
955                    _ => match lower_ascii_letter(c) {
956                        Some(cl) => go!(self: push_temp cl; emit c),
957                        None => go!(self: reconsume RawData ScriptDataEscaped DoubleEscaped),
958                    },
959                }
960            },
961
962            //§ before-attribute-name-state
963            states::BeforeAttributeName => loop {
964                match get_char!(self, input) {
965                    '\t' | '\n' | '\x0C' | ' ' => (),
966                    '/' => go!(self: to SelfClosingStartTag),
967                    '>' => go!(self: emit_tag Data),
968                    '\0' => go!(self: error; create_attr '\u{fffd}'; to AttributeName),
969                    c => match lower_ascii_letter(c) {
970                        Some(cl) => go!(self: create_attr cl; to AttributeName),
971                        None => {
972                            go_match!(self: c,
973                            '"' , '\'' , '<' , '=' => error);
974                            go!(self: create_attr c; to AttributeName);
975                        },
976                    },
977                }
978            },
979
980            //§ attribute-name-state
981            states::AttributeName => loop {
982                match get_char!(self, input) {
983                    '\t' | '\n' | '\x0C' | ' ' => go!(self: to AfterAttributeName),
984                    '/' => go!(self: to SelfClosingStartTag),
985                    '=' => go!(self: to BeforeAttributeValue),
986                    '>' => go!(self: emit_tag Data),
987                    '\0' => go!(self: error; push_name '\u{fffd}'),
988                    c => match lower_ascii_letter(c) {
989                        Some(cl) => go!(self: push_name cl),
990                        None => {
991                            go_match!(self: c,
992                            '"' , '\'' , '<' => error);
993                            go!(self: push_name c);
994                        },
995                    },
996                }
997            },
998
999            //§ after-attribute-name-state
1000            states::AfterAttributeName => loop {
1001                match get_char!(self, input) {
1002                    '\t' | '\n' | '\x0C' | ' ' => (),
1003                    '/' => go!(self: to SelfClosingStartTag),
1004                    '=' => go!(self: to BeforeAttributeValue),
1005                    '>' => go!(self: emit_tag Data),
1006                    '\0' => go!(self: error; create_attr '\u{fffd}'; to AttributeName),
1007                    c => match lower_ascii_letter(c) {
1008                        Some(cl) => go!(self: create_attr cl; to AttributeName),
1009                        None => {
1010                            go_match!(self: c,
1011                            '"' , '\'' , '<' => error);
1012                            go!(self: create_attr c; to AttributeName);
1013                        },
1014                    },
1015                }
1016            },
1017
1018            //§ before-attribute-value-state
1019            // Use peek so we can handle the first attr character along with the rest,
1020            // hopefully in the same zero-copy buffer.
1021            states::BeforeAttributeValue => loop {
1022                match peek!(self, input) {
1023                    '\t' | '\n' | '\r' | '\x0C' | ' ' => go!(self: discard_char input),
1024                    '"' => go!(self: discard_char input; to AttributeValue DoubleQuoted),
1025                    '\'' => go!(self: discard_char input; to AttributeValue SingleQuoted),
1026                    '>' => go!(self: discard_char input; error; emit_tag Data),
1027                    _ => go!(self: to AttributeValue Unquoted),
1028                }
1029            },
1030
1031            //§ attribute-value-(double-quoted)-state
1032            states::AttributeValue(DoubleQuoted) => loop {
1033                match pop_except_from!(self, input, small_char_set!('\r' '"' '&' '\0' '\n')) {
1034                    FromSet('"') => go!(self: to AfterAttributeValueQuoted),
1035                    FromSet('&') => go!(self: consume_char_ref),
1036                    FromSet('\0') => go!(self: error; push_value '\u{fffd}'),
1037                    FromSet(c) => go!(self: push_value c),
1038                    NotFromSet(ref b) => go!(self: append_value b),
1039                }
1040            },
1041
1042            //§ attribute-value-(single-quoted)-state
1043            states::AttributeValue(SingleQuoted) => loop {
1044                match pop_except_from!(self, input, small_char_set!('\r' '\'' '&' '\0' '\n')) {
1045                    FromSet('\'') => go!(self: to AfterAttributeValueQuoted),
1046                    FromSet('&') => go!(self: consume_char_ref),
1047                    FromSet('\0') => go!(self: error; push_value '\u{fffd}'),
1048                    FromSet(c) => go!(self: push_value c),
1049                    NotFromSet(ref b) => go!(self: append_value b),
1050                }
1051            },
1052
1053            //§ attribute-value-(unquoted)-state
1054            states::AttributeValue(Unquoted) => loop {
1055                match pop_except_from!(
1056                    self,
1057                    input,
1058                    small_char_set!('\r' '\t' '\n' '\x0C' ' ' '&' '>' '\0')
1059                ) {
1060                    FromSet('\t') | FromSet('\n') | FromSet('\x0C') | FromSet(' ') => {
1061                        go!(self: to BeforeAttributeName)
1062                    },
1063                    FromSet('&') => go!(self: consume_char_ref),
1064                    FromSet('>') => go!(self: emit_tag Data),
1065                    FromSet('\0') => go!(self: error; push_value '\u{fffd}'),
1066                    FromSet(c) => {
1067                        go_match!(self: c,
1068                            '"' , '\'' , '<' , '=' , '`' => error);
1069                        go!(self: push_value c);
1070                    },
1071                    NotFromSet(ref b) => go!(self: append_value b),
1072                }
1073            },
1074
1075            //§ after-attribute-value-(quoted)-state
1076            states::AfterAttributeValueQuoted => loop {
1077                match get_char!(self, input) {
1078                    '\t' | '\n' | '\x0C' | ' ' => go!(self: to BeforeAttributeName),
1079                    '/' => go!(self: to SelfClosingStartTag),
1080                    '>' => go!(self: emit_tag Data),
1081                    _ => go!(self: error; reconsume BeforeAttributeName),
1082                }
1083            },
1084
1085            //§ self-closing-start-tag-state
1086            states::SelfClosingStartTag => loop {
1087                match get_char!(self, input) {
1088                    '>' => {
1089                        self.current_tag_self_closing = true;
1090                        go!(self: emit_tag Data);
1091                    },
1092                    _ => go!(self: error; reconsume BeforeAttributeName),
1093                }
1094            },
1095
1096            //§ comment-start-state
1097            states::CommentStart => loop {
1098                match get_char!(self, input) {
1099                    '-' => go!(self: to CommentStartDash),
1100                    '\0' => go!(self: error; push_comment '\u{fffd}'; to Comment),
1101                    '>' => go!(self: error; emit_comment; to Data),
1102                    c => go!(self: push_comment c; to Comment),
1103                }
1104            },
1105
1106            //§ comment-start-dash-state
1107            states::CommentStartDash => loop {
1108                match get_char!(self, input) {
1109                    '-' => go!(self: to CommentEnd),
1110                    '\0' => go!(self: error; append_comment "-\u{fffd}"; to Comment),
1111                    '>' => go!(self: error; emit_comment; to Data),
1112                    c => go!(self: push_comment '-'; push_comment c; to Comment),
1113                }
1114            },
1115
1116            //§ comment-state
1117            states::Comment => loop {
1118                match get_char!(self, input) {
1119                    c @ '<' => go!(self: push_comment c; to CommentLessThanSign),
1120                    '-' => go!(self: to CommentEndDash),
1121                    '\0' => go!(self: error; push_comment '\u{fffd}'),
1122                    c => go!(self: push_comment c),
1123                }
1124            },
1125
1126            //§ comment-less-than-sign-state
1127            states::CommentLessThanSign => loop {
1128                match get_char!(self, input) {
1129                    c @ '!' => go!(self: push_comment c; to CommentLessThanSignBang),
1130                    c @ '<' => go!(self: push_comment c),
1131                    _ => go!(self: reconsume Comment),
1132                }
1133            },
1134
1135            //§ comment-less-than-sign-bang
1136            states::CommentLessThanSignBang => loop {
1137                match get_char!(self, input) {
1138                    '-' => go!(self: to CommentLessThanSignBangDash),
1139                    _ => go!(self: reconsume Comment),
1140                }
1141            },
1142
1143            //§ comment-less-than-sign-bang-dash
1144            states::CommentLessThanSignBangDash => loop {
1145                match get_char!(self, input) {
1146                    '-' => go!(self: to CommentLessThanSignBangDashDash),
1147                    _ => go!(self: reconsume CommentEndDash),
1148                }
1149            },
1150
1151            //§ comment-less-than-sign-bang-dash-dash
1152            states::CommentLessThanSignBangDashDash => loop {
1153                match get_char!(self, input) {
1154                    '>' => go!(self: reconsume CommentEnd),
1155                    _ => go!(self: error; reconsume CommentEnd),
1156                }
1157            },
1158
1159            //§ comment-end-dash-state
1160            states::CommentEndDash => loop {
1161                match get_char!(self, input) {
1162                    '-' => go!(self: to CommentEnd),
1163                    '\0' => go!(self: error; append_comment "-\u{fffd}"; to Comment),
1164                    c => go!(self: push_comment '-'; push_comment c; to Comment),
1165                }
1166            },
1167
1168            //§ comment-end-state
1169            states::CommentEnd => loop {
1170                match get_char!(self, input) {
1171                    '>' => go!(self: emit_comment; to Data),
1172                    '!' => go!(self: to CommentEndBang),
1173                    '-' => go!(self: push_comment '-'),
1174                    _ => go!(self: append_comment "--"; reconsume Comment),
1175                }
1176            },
1177
1178            //§ comment-end-bang-state
1179            states::CommentEndBang => loop {
1180                match get_char!(self, input) {
1181                    '-' => go!(self: append_comment "--!"; to CommentEndDash),
1182                    '>' => go!(self: error; emit_comment; to Data),
1183                    '\0' => go!(self: error; append_comment "--!\u{fffd}"; to Comment),
1184                    c => go!(self: append_comment "--!"; push_comment c; to Comment),
1185                }
1186            },
1187
1188            //§ doctype-state
1189            states::Doctype => loop {
1190                match get_char!(self, input) {
1191                    '\t' | '\n' | '\x0C' | ' ' => go!(self: to BeforeDoctypeName),
1192                    '>' => go!(self: reconsume BeforeDoctypeName),
1193                    _ => go!(self: error; reconsume BeforeDoctypeName),
1194                }
1195            },
1196

1198            states::BeforeDoctypeName => loop {
1199                match get_char!(self, input) {
1200                    '\t' | '\n' | '\x0C' | ' ' => (),

1203                    },
1204                    '>' => go!(self: error; create_doctype; force_quirks; emit_doctype; to Data),

1206                                  to DoctypeName),
1207                }
1208            },
1209
1210            //§ doctype-name-state
1211            states::DoctypeName => loop {
1212                match get_char!(self, input) {
1213                    '\t' | '\n' | '\x0C' | ' ' => go!(self: clear_temp; to AfterDoctypeName),
1214                    '>' => go!(self: emit_doctype; to Data),
1215                    '\0' => go!(self: error; push_doctype_name '\u{fffd}'),
1216                    c => go!(self: push_doctype_name (c.to_ascii_lowercase())),
1217                }
1218            },
1219
1220            //§ after-doctype-name-state
1221            states::AfterDoctypeName => loop {
1222                if eat!(self, input, "public") {
1223                    go!(self: to AfterDoctypeKeyword Public);
1224                } else if eat!(self, input, "system") {
1225                    go!(self: to AfterDoctypeKeyword System);
1226                } else {
1227                    match get_char!(self, input) {
1228                        '\t' | '\n' | '\x0C' | ' ' => (),
1229                        '>' => go!(self: emit_doctype; to Data),
1230                        _ => go!(self: error; force_quirks; reconsume BogusDoctype),
1231                    }
1232                }
1233            },
1234
1235            //§ after-doctype-public-keyword-state after-doctype-system-keyword-state
1236            states::AfterDoctypeKeyword(kind) => loop {
1237                match get_char!(self, input) {
1238                    '\t' | '\n' | '\x0C' | ' ' => go!(self: to BeforeDoctypeIdentifier kind),
1239                    '"' => {
1240                        go!(self: error; clear_doctype_id kind; to DoctypeIdentifierDoubleQuoted kind)
1241                    },
1242                    '\'' => {
1243                        go!(self: error; clear_doctype_id kind; to DoctypeIdentifierSingleQuoted kind)
1244                    },
1245                    '>' => go!(self: error; force_quirks; emit_doctype; to Data),
1246                    _ => go!(self: error; force_quirks; reconsume BogusDoctype),
1247                }
1248            },
1249
1250            //§ before-doctype-public-identifier-state before-doctype-system-identifier-state
1251            states::BeforeDoctypeIdentifier(kind) => loop {
1252                match get_char!(self, input) {
1253                    '\t' | '\n' | '\x0C' | ' ' => (),
1254                    '"' => go!(self: clear_doctype_id kind; to DoctypeIdentifierDoubleQuoted kind),
1255                    '\'' => go!(self: clear_doctype_id kind; to DoctypeIdentifierSingleQuoted kind),
1256                    '>' => go!(self: error; force_quirks; emit_doctype; to Data),
1257                    _ => go!(self: error; force_quirks; reconsume BogusDoctype),
1258                }
1259            },
1260
1261            //§ doctype-public-identifier-(double-quoted)-state doctype-system-identifier-(double-quoted)-state
1262            states::DoctypeIdentifierDoubleQuoted(kind) => loop {
1263                match get_char!(self, input) {
1264                    '"' => go!(self: to AfterDoctypeIdentifier kind),
1265                    '\0' => go!(self: error; push_doctype_id kind '\u{fffd}'),
1266                    '>' => go!(self: error; force_quirks; emit_doctype; to Data),
1267                    c => go!(self: push_doctype_id kind c),
1268                }
1269            },
1270
1271            //§ doctype-public-identifier-(single-quoted)-state doctype-system-identifier-(single-quoted)-state
1272            states::DoctypeIdentifierSingleQuoted(kind) => loop {
1273                match get_char!(self, input) {
1274                    '\'' => go!(self: to AfterDoctypeIdentifier kind),
1275                    '\0' => go!(self: error; push_doctype_id kind '\u{fffd}'),
1276                    '>' => go!(self: error; force_quirks; emit_doctype; to Data),
1277                    c => go!(self: push_doctype_id kind c),
1278                }
1279            },
1280
1281            //§ after-doctype-public-identifier-state
1282            states::AfterDoctypeIdentifier(Public) => loop {
1283                match get_char!(self, input) {
1284                    '\t' | '\n' | '\x0C' | ' ' => {
1285                        go!(self: to BetweenDoctypePublicAndSystemIdentifiers)
1286                    },
1287                    '>' => go!(self: emit_doctype; to Data),
1288                    '"' => {
1289                        go!(self: error; clear_doctype_id System; to DoctypeIdentifierDoubleQuoted System)
1290                    },
1291                    '\'' => {
1292                        go!(self: error; clear_doctype_id System; to DoctypeIdentifierSingleQuoted System)
1293                    },
1294                    _ => go!(self: error; force_quirks; reconsume BogusDoctype),
1295                }
1296            },
1297
1298            //§ after-doctype-system-identifier-state
1299            states::AfterDoctypeIdentifier(System) => loop {
1300                match get_char!(self, input) {
1301                    '\t' | '\n' | '\x0C' | ' ' => (),
1302                    '>' => go!(self: emit_doctype; to Data),
1303                    _ => go!(self: error; reconsume BogusDoctype),
1304                }
1305            },
1306
1307            //§ between-doctype-public-and-system-identifiers-state
1308            states::BetweenDoctypePublicAndSystemIdentifiers => loop {
1309                match get_char!(self, input) {
1310                    '\t' | '\n' | '\x0C' | ' ' => (),
1311                    '>' => go!(self: emit_doctype; to Data),
1312                    '"' => {
1313                        go!(self: clear_doctype_id System; to DoctypeIdentifierDoubleQuoted System)
1314                    },
1315                    '\'' => {
1316                        go!(self: clear_doctype_id System; to DoctypeIdentifierSingleQuoted System)
1317                    },
1318                    _ => go!(self: error; force_quirks; reconsume BogusDoctype),
1319                }
1320            },
1321
1322            //§ bogus-doctype-state
1323            states::BogusDoctype => loop {
1324                match get_char!(self, input) {
1325                    '>' => go!(self: emit_doctype; to Data),
1326                    '\0' => go!(self: error),
1327                    _ => (),
1328                }
1329            },
1330
1331            //§ bogus-comment-state
1332            states::BogusComment => loop {
1333                match get_char!(self, input) {
1334                    '>' => go!(self: emit_comment; to Data),

1337                }
1338            },

1341            states::MarkupDeclarationOpen => loop {

1360                match get_char!(self, input) {

1363                    c => go!(self: push_temp c),
1364                }
1365            },
1366

1370                _ => go!(self: push_temp ']'; reconsume CdataSection),
1371            },
1372
1373            //§ cdata-section-end
1374            states::CdataSectionEnd => loop {
1375                match get_char!(self, input) {
1376                    ']' => go!(self: push_temp ']'),
1377                    '>' => go!(self: emit_temp; to Data),
1378                    _ => go!(self: push_temp ']'; push_temp ']'; reconsume CdataSection),
1379                }
1380            },
1381            //§ END
1382        }

1385    fn step_char_ref_tokenizer(&mut self, input: &mut BufferQueue) -> ProcessResult<Sink::Handle> {
1386        // FIXME HACK: Take and replace the tokenizer so we don't
1387        // double-mut-borrow self.  This is why it's boxed.
1388        let mut tok = self.char_ref_tokenizer.take().unwrap();
1389        let outcome = tok.step(self, input);
1390
1391        let progress = match outcome {
1392            char_ref::Done => {
1393                self.process_char_ref(tok.get_result());
1394                return ProcessResult::Continue;
1395            },
1396
1397            char_ref::Stuck => ProcessResult::Suspend,
1398            char_ref::Progress => ProcessResult::Continue,
1399        };
1400
1401        self.char_ref_tokenizer = Some(tok);
1402        progress
1403    }

1405    fn process_char_ref(&mut self, char_ref: CharRef) {
1406        let CharRef {
1407            mut chars,
1408            mut num_chars,
1409        } = char_ref;
1410
1411        if num_chars == 0 {
1412            chars[0] = '&';
1413            num_chars = 1;
1414        }
1415
1416        for i in 0..num_chars {
1417            let c = chars[i as usize];
1418            match self.state {
1419                states::Data | states::RawData(states::Rcdata) => go!(self: emit c),
1420
1421                states::AttributeValue(_) => go!(self: push_value c),
1422
1423                _ => panic!(
1424                    "state {:?} should not be reachable in process_char_ref",
1425                    self.state
1426                ),
1427            }
1428        }
1429    }

1431    /// Indicate that we have reached the end of the input.
1432    pub fn end(&mut self) {
1433        // Handle EOF in the char ref sub-tokenizer, if there is one.
1434        // Do this first because it might un-consume stuff.
1435        let mut input = BufferQueue::default();
1436        match self.char_ref_tokenizer.take() {
1437            None => (),
1438            Some(mut tok) => {
1439                tok.end_of_file(self, &mut input);
1440                self.process_char_ref(tok.get_result());
1441            },
1442        }
1443
1444        // Process all remaining buffered input.
1445        // If we're waiting for lookahead, we're not gonna get it.
1446        self.at_eof = true;
1447        assert!(matches!(self.run(&mut input), TokenizerResult::Done));
1448        assert!(input.is_empty());
1449
1450        loop {
1451            match self.eof_step() {
1452                ProcessResult::Continue => (),
1453                ProcessResult::Suspend => break,
1454                ProcessResult::Script(_) => unreachable!(),
1455            }
1456        }
1457
1458        self.sink.end();
1459
1460        if self.opts.profile {
1461            self.dump_profile();
1462        }
1463    }

1465    fn dump_profile(&self) {
1466        let mut results: Vec<(states::State, u64)> =
1467            self.state_profile.iter().map(|(s, t)| (*s, *t)).collect();
1468        results.sort_by(|&(_, x), &(_, y)| y.cmp(&x));
1469
1470        let total: u64 = results
1471            .iter()
1472            .map(|&(_, t)| t)
1473            .fold(0, ::std::ops::Add::add);
1474        println!("\nTokenizer profile, in nanoseconds");
1475        println!("\n{:12}         total in token sink", self.time_in_sink);
1476        println!("\n{:12}         total in tokenizer", total);
1477
1478        for (k, v) in results.into_iter() {
1479            let pct = 100.0 * (v as f64) / (total as f64);
1480            println!("{:12}  {:4.1}%  {:?}", v, pct, k);
1481        }
1482    }

1484    fn eof_step(&mut self) -> ProcessResult<Sink::Handle> {
1485        debug!("processing EOF in state {:?}", self.state);
1486        match self.state {
1487            states::Data
1488            | states::RawData(Rcdata)
1489            | states::RawData(Rawtext)
1490            | states::RawData(ScriptData)
1491            | states::Plaintext => go!(self: eof),
1492
1493            states::TagName
1494            | states::RawData(ScriptDataEscaped(_))
1495            | states::BeforeAttributeName
1496            | states::AttributeName
1497            | states::AfterAttributeName
1498            | states::AttributeValue(_)
1499            | states::AfterAttributeValueQuoted
1500            | states::SelfClosingStartTag
1501            | states::ScriptDataEscapedDash(_)
1502            | states::ScriptDataEscapedDashDash(_) => go!(self: error_eof; to Data),
1503
1504            states::BeforeAttributeValue => go!(self: reconsume AttributeValue Unquoted),
1505
1506            states::TagOpen => go!(self: error_eof; emit '<'; to Data),
1507
1508            states::EndTagOpen => go!(self: error_eof; emit '<'; emit '/'; to Data),
1509
1510            states::RawLessThanSign(ScriptDataEscaped(DoubleEscaped)) => {
1511                go!(self: to RawData ScriptDataEscaped DoubleEscaped)
1512            },
1513
1514            states::RawLessThanSign(kind) => go!(self: emit '<'; to RawData kind),
1515
1516            states::RawEndTagOpen(kind) => go!(self: emit '<'; emit '/'; to RawData kind),
1517
1518            states::RawEndTagName(kind) => {
1519                go!(self: emit '<'; emit '/'; emit_temp; to RawData kind)
1520            },
1521
1522            states::ScriptDataEscapeStart(kind) => go!(self: to RawData ScriptDataEscaped kind),
1523
1524            states::ScriptDataEscapeStartDash => go!(self: to RawData ScriptData),
1525
1526            states::ScriptDataDoubleEscapeEnd => {
1527                go!(self: to RawData ScriptDataEscaped DoubleEscaped)
1528            },
1529
1530            states::CommentStart
1531            | states::CommentStartDash
1532            | states::Comment
1533            | states::CommentEndDash
1534            | states::CommentEnd
1535            | states::CommentEndBang => go!(self: error_eof; emit_comment; to Data),
1536
1537            states::CommentLessThanSign | states::CommentLessThanSignBang => {
1538                go!(self: reconsume Comment)
1539            },
1540
1541            states::CommentLessThanSignBangDash => go!(self: reconsume CommentEndDash),
1542
1543            states::CommentLessThanSignBangDashDash => go!(self: reconsume CommentEnd),
1544
1545            states::Doctype | states::BeforeDoctypeName => {
1546                go!(self: error_eof; create_doctype; force_quirks; emit_doctype; to Data)
1547            },
1548
1549            states::DoctypeName
1550            | states::AfterDoctypeName
1551            | states::AfterDoctypeKeyword(_)
1552            | states::BeforeDoctypeIdentifier(_)
1553            | states::DoctypeIdentifierDoubleQuoted(_)
1554            | states::DoctypeIdentifierSingleQuoted(_)
1555            | states::AfterDoctypeIdentifier(_)
1556            | states::BetweenDoctypePublicAndSystemIdentifiers => {
1557                go!(self: error_eof; force_quirks; emit_doctype; to Data)
1558            },
1559
1560            states::BogusDoctype => go!(self: emit_doctype; to Data),
1561
1562            states::BogusComment => go!(self: emit_comment; to Data),
1563
1564            states::MarkupDeclarationOpen => go!(self: error; to BogusComment),
1565
1566            states::CdataSection => go!(self: emit_temp; error_eof; to Data),
1567
1568            states::CdataSectionBracket => go!(self: push_temp ']'; to CdataSection),
1569
1570            states::CdataSectionEnd => go!(self: push_temp ']'; push_temp ']'; to CdataSection),
1571        }
1572    }

1592    // LinesMatch implements the TokenSink trait. It is used for testing to see
1593    // if current_line is being updated when process_token is called. The lines
1594    // vector is a collection of the line numbers that each token is on.
1595    struct LinesMatch {

1600
1601    impl LinesMatch {
1602        fn new() -> LinesMatch {
1603            LinesMatch {
1604                tokens: vec![],
1605                current_str: StrTendril::new(),
1606                lines: vec![],
1607            }
1608        }
1609
1610        fn push(&mut self, token: Token, line_number: u64) {
1611            self.finish_str();
1612            self.lines.push((token, line_number));
1613        }
1614
1615        fn finish_str(&mut self) {
1616            if self.current_str.len() > 0 {
1617                let s = mem::take(&mut self.current_str);
1618                self.tokens.push(CharacterTokens(s));
1619            }
1620        }

1623    impl TokenSink for LinesMatch {
1624        type Handle = ();
1625
1626        fn process_token(
1627            &mut self,
1628            token: Token,
1629            line_number: u64,
1630        ) -> TokenSinkResult<Self::Handle> {
1631            match token {
1632                CharacterTokens(b) => {
1633                    self.current_str.push_slice(&b);
1634                },
1635
1636                NullCharacterToken => {
1637                    self.current_str.push_char('\0');
1638                },
1639
1640                ParseError(_) => {
1641                    panic!("unexpected parse error");
1642                },
1643
1644                TagToken(mut t) => {
1645                    // The spec seems to indicate that one can emit
1646                    // erroneous end tags with attrs, but the test
1647                    // cases don't contain them.
1648                    match t.kind {
1649                        EndTag => {
1650                            t.self_closing = false;
1651                            t.attrs = vec![];
1652                        },
1653                        _ => t.attrs.sort_by(|a1, a2| a1.name.cmp(&a2.name)),
1654                    }
1655                    self.push(TagToken(t), line_number);
1656                },
1657
1658                EOFToken => (),
1659
1660                _ => self.push(token, line_number),
1661            }
1662            TokenSinkResult::Continue
1663        }
1664    }
