21//! Pull parser for [CommonMark](https://commonmark.org). This crate provides a [Parser](struct.Parser.html) struct
22//! which is an iterator over [Event](enum.Event.html)s. This iterator can be used
23//! directly, or to output HTML using the [HTML module](html/index.html).
24//!
25//! By default, only CommonMark features are enabled. To use extensions like tables,
26//! footnotes or task lists, enable them by setting the corresponding flags in the
27//! [Options](struct.Options.html) struct.
28//!
29//! # Example
30//! ```rust
31//! use pulldown_cmark::{Parser, Options};
32//!
33//! let markdown_input = "Hello world, this is a ~~complicated~~ *very simple* example.";
34//!
35//! // Set up options and parser. Strikethroughs are not part of the CommonMark standard
36//! // and we therefore must enable it explicitly.
37//! let mut options = Options::empty();
38//! options.insert(Options::ENABLE_STRIKETHROUGH);
39//! let parser = Parser::new_ext(markdown_input, options);
40//!
41//! # #[cfg(feature = "html")] {
42//! // Write to String buffer.
43//! let mut html_output = String::new();
44//! pulldown_cmark::html::push_html(&mut html_output, parser);
45//!
46//! // Check that the output is what we expected.
47//! let expected_html = "<p>Hello world, this is a <del>complicated</del> <em>very simple</em> example.</p>\n";
48//! assert_eq!(expected_html, &html_output);
49//! # }
50//! ```
51//!
52//! Note that consecutive text events can happen due to the manner in which the
53//! parser evaluates the source. A utility `TextMergeStream` exists to improve
54//! the comfort of iterating the events:
55//!
56//! ```rust
57//! use pulldown_cmark::{Event, Parser, TextMergeStream};
58//!
59//! let markdown_input = "Hello world, this is a ~~complicated~~ *very simple* example.";
60//!
61//! let iterator = TextMergeStream::new(Parser::new(markdown_input));
62//!
63//! for event in iterator {
64//!     match event {
65//!         Event::Text(text) => println!("{}", text),
66//!         _ => {}
67//!     }
68//! }
69//! ```
70//!

132/// Tags for elements that can contain other elements.
133#[derive(Clone, Debug, PartialEq)]
134#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
135pub enum Tag<'a> {
136    /// A paragraph of text and other inline elements.
137    Paragraph,
138
139    /// A heading, with optional identifier, classes and custom attributes.
140    /// The identifier is prefixed with `#` and the last one in the attributes
141    /// list is chosen, classes are prefixed with `.` and custom attributes
142    /// have no prefix and can optionally have a value (`myattr` or `myattr=myvalue`).
143    Heading {
144        level: HeadingLevel,
145        id: Option<CowStr<'a>>,
146        classes: Vec<CowStr<'a>>,
147        /// The first item of the tuple is the attr and second one the value.
148        attrs: Vec<(CowStr<'a>, Option<CowStr<'a>>)>,
149    },
150
151    BlockQuote,
152    /// A code block.
153    CodeBlock(CodeBlockKind<'a>),
154
155    /// A HTML block.
156    HtmlBlock,
157
158    /// A list. If the list is ordered the field indicates the number of the first item.
159    /// Contains only list items.
160    List(Option<u64>), // TODO: add delim and tight for ast (not needed for html)
161    /// A list item.
162    Item,
163    /// A footnote definition. The value contained is the footnote's label by which it can
164    /// be referred to.
165    #[cfg_attr(feature = "serde", serde(borrow))]
166    FootnoteDefinition(CowStr<'a>),
167
168    /// A table. Contains a vector describing the text-alignment for each of its columns.
169    Table(Vec<Alignment>),
170    /// A table header. Contains only `TableCell`s. Note that the table body starts immediately
171    /// after the closure of the `TableHead` tag. There is no `TableBody` tag.
172    TableHead,
173    /// A table row. Is used both for header rows as body rows. Contains only `TableCell`s.
174    TableRow,
175    TableCell,
176
177    // span-level tags
178    Emphasis,
179    Strong,
180    Strikethrough,
181
182    /// A link.
183    Link {
184        link_type: LinkType,
185        dest_url: CowStr<'a>,
186        title: CowStr<'a>,
187        /// Identifier of reference links, e.g. `world` in the link `[hello][world]`.
188        id: CowStr<'a>,
189    },
190
191    /// An image. The first field is the link type, the second the destination URL and the third is a title,
192    /// the fourth is the link identifier.
193    Image {
194        link_type: LinkType,
195        dest_url: CowStr<'a>,
196        title: CowStr<'a>,
197        /// Identifier of reference links, e.g. `world` in the link `[hello][world]`.
198        id: CowStr<'a>,
199    },
200
201    /// A metadata block.
202    MetadataBlock(MetadataBlockKind),
203}
204
205impl<'a> Tag<'a> {
206    pub fn to_end(&self) -> TagEnd {
207        match self {
208            Tag::Paragraph => TagEnd::Paragraph,
209            Tag::Heading { level, .. } => TagEnd::Heading(*level),
210            Tag::BlockQuote => TagEnd::BlockQuote,
211            Tag::CodeBlock(_) => TagEnd::CodeBlock,
212            Tag::HtmlBlock => TagEnd::HtmlBlock,
213            Tag::List(number) => TagEnd::List(number.is_some()),
214            Tag::Item => TagEnd::Item,
215            Tag::FootnoteDefinition(_) => TagEnd::FootnoteDefinition,
216            Tag::Table(_) => TagEnd::Table,
217            Tag::TableHead => TagEnd::TableHead,
218            Tag::TableRow => TagEnd::TableRow,
219            Tag::TableCell => TagEnd::TableCell,
220            Tag::Emphasis => TagEnd::Emphasis,
221            Tag::Strong => TagEnd::Strong,
222            Tag::Strikethrough => TagEnd::Strikethrough,
223            Tag::Link { .. } => TagEnd::Link,
224            Tag::Image { .. } => TagEnd::Image,
225            Tag::MetadataBlock(kind) => TagEnd::MetadataBlock(*kind),
226        }
227    }
228}

292/// Returned when trying to convert a `usize` into a `Heading` but it fails
293/// because the usize isn't a valid heading level
294#[derive(Copy, Clone, Eq, PartialEq, Ord, PartialOrd, Hash, Debug)]
295pub struct InvalidHeadingLevel(usize);
296
297impl TryFrom<usize> for HeadingLevel {
298    type Error = InvalidHeadingLevel;
299
300    fn try_from(value: usize) -> Result<Self, Self::Error> {
301        match value {
302            1 => Ok(Self::H1),
303            2 => Ok(Self::H2),
304            3 => Ok(Self::H3),
305            4 => Ok(Self::H4),
306            5 => Ok(Self::H5),
307            6 => Ok(Self::H6),
308            _ => Err(InvalidHeadingLevel(value)),
309        }
310    }
311}

313/// Type specifier for inline links. See [the Tag::Link](enum.Tag.html#variant.Link) for more information.
314#[derive(Clone, Debug, PartialEq, Copy)]
315#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
316pub enum LinkType {
317    /// Inline link like `[foo](bar)`
318    Inline,
319    /// Reference link like `[foo][bar]`
320    Reference,
321    /// Reference without destination in the document, but resolved by the broken_link_callback
322    ReferenceUnknown,
323    /// Collapsed link like `[foo][]`
324    Collapsed,
325    /// Collapsed link without destination in the document, but resolved by the broken_link_callback
326    CollapsedUnknown,
327    /// Shortcut link like `[foo]`
328    Shortcut,
329    /// Shortcut without destination in the document, but resolved by the broken_link_callback
330    ShortcutUnknown,
331    /// Autolink like `<http://foo.bar/baz>`
332    Autolink,
333    /// Email address in autolink like `<john@example.org>`
334    Email,
335}
336
337impl LinkType {
338    /// Map the link type to an equivalent _Unknown link type.
339    fn to_unknown(self) -> Self {
340        match self {
341            LinkType::Reference => LinkType::ReferenceUnknown,
342            LinkType::Collapsed => LinkType::CollapsedUnknown,
343            LinkType::Shortcut => LinkType::ShortcutUnknown,
344            _ => unreachable!(),
345        }
346    }
347}

349/// Markdown events that are generated in a preorder traversal of the document
350/// tree, with additional `End` events whenever all of an inner node's children
351/// have been visited.
352#[derive(Clone, Debug, PartialEq)]
353#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
354pub enum Event<'a> {
355    /// Start of a tagged element. Events that are yielded after this event
356    /// and before its corresponding `End` event are inside this element.
357    /// Start and end events are guaranteed to be balanced.
358    #[cfg_attr(feature = "serde", serde(borrow))]
359    Start(Tag<'a>),
360    /// End of a tagged element.
361    End(TagEnd),
362    /// A text node.
363    #[cfg_attr(feature = "serde", serde(borrow))]
364    Text(CowStr<'a>),
365    /// An inline code node.
366    #[cfg_attr(feature = "serde", serde(borrow))]
367    Code(CowStr<'a>),
368    /// An HTML node.
369    #[cfg_attr(feature = "serde", serde(borrow))]
370    Html(CowStr<'a>),
371    /// An inline HTML node.
372    #[cfg_attr(feature = "serde", serde(borrow))]
373    InlineHtml(CowStr<'a>),
374    /// A reference to a footnote with given label, which may or may not be defined
375    /// by an event with a `Tag::FootnoteDefinition` tag. Definitions and references to them may
376    /// occur in any order.
377    #[cfg_attr(feature = "serde", serde(borrow))]
378    FootnoteReference(CowStr<'a>),
379    /// A soft line break.
380    SoftBreak,
381    /// A hard line break.
382    HardBreak,
383    /// A horizontal ruler.
384    Rule,
385    /// A task list marker, rendered as a checkbox in HTML. Contains a true when it is checked.
386    TaskListMarker(bool),
387}

401bitflags::bitflags! {
402    /// Option struct containing flags for enabling extra features
403    /// that are not part of the CommonMark spec.
404    #[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
405    pub struct Options: u32 {
406        const ENABLE_TABLES = 1 << 1;
407        /// GitHub-compatible footnote syntax.
408        ///
409        /// Footnotes are referenced with the syntax `[^IDENT]`,
410        /// and defined with an identifier followed by a colon at top level.
411        ///
412        /// ---
413        ///
414        /// ```markdown
415        /// Footnote referenced [^1].
416        ///
417        /// [^1]: footnote defined
418        /// ```
419        ///
420        /// Footnote referenced [^1].
421        ///
422        /// [^1]: footnote defined
423        const ENABLE_FOOTNOTES = 1 << 2;
424        const ENABLE_STRIKETHROUGH = 1 << 3;
425        const ENABLE_TASKLISTS = 1 << 4;
426        const ENABLE_SMART_PUNCTUATION = 1 << 5;
427        /// Extension to allow headings to have ID and classes.
428        ///
429        /// `# text { #id .class1 .class2 myattr other_attr=myvalue }`
430        /// is interpreted as a level 1 heading
431        /// with the content `text`, ID `id`, classes `class1` and `class2` and
432        /// custom attributes `myattr` (without value) and
433        /// `other_attr` with value `myvalue`.
434        /// Note that ID, classes, and custom attributes should be space-separated.
435        const ENABLE_HEADING_ATTRIBUTES = 1 << 6;
436        /// Metadata blocks in YAML style, i.e.:
437        /// - starting with a `---` line
438        /// - ending with a `---` or `...` line
439        const ENABLE_YAML_STYLE_METADATA_BLOCKS = 1 << 7;
440        /// Metadata blocks delimited by:
441        /// - `+++` line at start
442        /// - `+++` line at end
443        const ENABLE_PLUSES_DELIMITED_METADATA_BLOCKS = 1 << 8;
444        /// Older footnote syntax. This flag implies `ENABLE_FOOTNOTES`, changing it to use an
445        /// older syntax instead of the new, default, GitHub-compatible syntax.
446        ///
447        /// New syntax is different from the old syntax regarding
448        /// indentation, nesting, and footnote references with no definition:
449        ///
450        /// ```markdown
451        /// [^1]: In new syntax, this is two footnote definitions.
452        /// [^2]: In old syntax, this is a single footnote definition with two lines.
453        ///
454        /// [^3]:
455        ///
456        ///     In new syntax, this is a footnote with two paragraphs.
457        ///
458        ///     In old syntax, this is a footnote followed by a code block.
459        ///
460        /// In new syntax, this undefined footnote definition renders as
461        /// literal text [^4]. In old syntax, it creates a dangling link.
462        /// ```
463        const ENABLE_OLD_FOOTNOTES = (1 << 9) | (1 << 2);
464    }
465}
466
467impl Options {
468    pub(crate) fn has_gfm_footnotes(&self) -> bool {
469        self.contains(Options::ENABLE_FOOTNOTES) && !self.contains(Options::ENABLE_OLD_FOOTNOTES)
470    }
