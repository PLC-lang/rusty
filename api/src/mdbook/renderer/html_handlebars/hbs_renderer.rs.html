30    fn render_item(
31        &self,
32        item: &BookItem,
33        mut ctx: RenderItemContext<'_>,
34        print_content: &mut String,
35    ) -> Result<()> {
36        // FIXME: This should be made DRY-er and rely less on mutable state
37
38        let (ch, path) = match item {
39            BookItem::Chapter(ch) if !ch.is_draft_chapter() => (ch, ch.path.as_ref().unwrap()),
40            _ => return Ok(()),
41        };
42
43        if let Some(ref edit_url_template) = ctx.html_config.edit_url_template {
44            let full_path = ctx.book_config.src.to_str().unwrap_or_default().to_owned()
45                + "/"
46                + ch.source_path
47                    .clone()
48                    .unwrap_or_default()
49                    .to_str()
50                    .unwrap_or_default();
51
52            let edit_url = edit_url_template.replace("{path}", &full_path);
53            ctx.data
54                .insert("git_repository_edit_url".to_owned(), json!(edit_url));
55        }
56
57        let content = utils::render_markdown(&ch.content, ctx.html_config.smart_punctuation());
58
59        let fixed_content = utils::render_markdown_with_path(
60            &ch.content,
61            ctx.html_config.smart_punctuation(),
62            Some(path),
63        );
64        if !ctx.is_index && ctx.html_config.print.page_break {
65            // Add page break between chapters
66            // See https://developer.mozilla.org/en-US/docs/Web/CSS/break-before and https://developer.mozilla.org/en-US/docs/Web/CSS/page-break-before
67            // Add both two CSS properties because of the compatibility issue
68            print_content
69                .push_str(r#"<div style="break-before: page; page-break-before: always;"></div>"#);
70        }
71        print_content.push_str(&fixed_content);
72
73        // Update the context with data for this file
74        let ctx_path = path
75            .to_str()
76            .with_context(|| "Could not convert path to str")?;
77        let filepath = Path::new(&ctx_path).with_extension("html");
78
79        // "print.html" is used for the print page.
80        if path == Path::new("print.md") {
81            bail!("{} is reserved for internal use", path.display());
82        };
83
84        let book_title = ctx
85            .data
86            .get("book_title")
87            .and_then(serde_json::Value::as_str)
88            .unwrap_or("");
89
90        let title = if let Some(title) = ctx.chapter_titles.get(path) {
91            title.clone()
92        } else if book_title.is_empty() {
93            ch.name.clone()
94        } else {
95            ch.name.clone() + " - " + book_title
96        };
97
98        ctx.data.insert("path".to_owned(), json!(path));
99        ctx.data.insert("content".to_owned(), json!(content));
100        ctx.data.insert("chapter_title".to_owned(), json!(ch.name));
101        ctx.data.insert("title".to_owned(), json!(title));
102        ctx.data.insert(
103            "path_to_root".to_owned(),
104            json!(utils::fs::path_to_root(path)),
105        );
106        if let Some(ref section) = ch.number {
107            ctx.data
108                .insert("section".to_owned(), json!(section.to_string()));
109        }
110
111        // Render the handlebars template with the data
112        debug!("Render template");
113        let rendered = ctx.handlebars.render("index", &ctx.data)?;
114
115        let rendered = self.post_process(
116            rendered,
117            &ctx.html_config.playground,
118            &ctx.html_config.code,
119            ctx.edition,
120        );
121
122        // Write to file
123        debug!("Creating {}", filepath.display());
124        utils::fs::write_file(&ctx.destination, &filepath, rendered.as_bytes())?;
125
126        if ctx.is_index {
127            ctx.data.insert("path".to_owned(), json!("index.md"));
128            ctx.data.insert("path_to_root".to_owned(), json!(""));
129            ctx.data.insert("is_index".to_owned(), json!(true));
130            let rendered_index = ctx.handlebars.render("index", &ctx.data)?;
131            let rendered_index = self.post_process(
132                rendered_index,
133                &ctx.html_config.playground,
134                &ctx.html_config.code,
135                ctx.edition,
136            );
137            debug!("Creating index.html from {}", ctx_path);
138            utils::fs::write_file(&ctx.destination, "index.html", rendered_index.as_bytes())?;
139        }
140
141        Ok(())
142    }

144    fn render_404(
145        &self,
146        ctx: &RenderContext,
147        html_config: &HtmlConfig,
148        src_dir: &Path,
149        handlebars: &mut Handlebars<'_>,
150        data: &mut serde_json::Map<String, serde_json::Value>,
151    ) -> Result<()> {
152        let destination = &ctx.destination;
153        let content_404 = if let Some(ref filename) = html_config.input_404 {
154            let path = src_dir.join(filename);
155            std::fs::read_to_string(&path)
156                .with_context(|| format!("unable to open 404 input file {path:?}"))?
157        } else {
158            // 404 input not explicitly configured try the default file 404.md
159            let default_404_location = src_dir.join("404.md");
160            if default_404_location.exists() {
161                std::fs::read_to_string(&default_404_location).with_context(|| {
162                    format!("unable to open 404 input file {default_404_location:?}")
163                })?
164            } else {
165                "# Document not found (404)\n\nThis URL is invalid, sorry. Please use the \
166                navigation bar or search to continue."
167                    .to_string()
168            }
169        };
170        let html_content_404 =
171            utils::render_markdown(&content_404, html_config.smart_punctuation());
172
173        let mut data_404 = data.clone();
174        let base_url = if let Some(site_url) = &html_config.site_url {
175            site_url
176        } else {
177            debug!(
178                "HTML 'site-url' parameter not set, defaulting to '/'. Please configure \
179                this to ensure the 404 page work correctly, especially if your site is hosted in a \
180                subdirectory on the HTTP server."
181            );
182            "/"
183        };
184        data_404.insert("base_url".to_owned(), json!(base_url));
185        // Set a dummy path to ensure other paths (e.g. in the TOC) are generated correctly
186        data_404.insert("path".to_owned(), json!("404.md"));
187        data_404.insert("content".to_owned(), json!(html_content_404));
188
189        let mut title = String::from("Page not found");
190        if let Some(book_title) = &ctx.config.book.title {
191            title.push_str(" - ");
192            title.push_str(book_title);
193        }
194        data_404.insert("title".to_owned(), json!(title));
195        let rendered = handlebars.render("index", &data_404)?;
196
197        let rendered = self.post_process(
198            rendered,
199            &html_config.playground,
200            &html_config.code,
201            ctx.config.rust.edition,
202        );
203        let output_file = get_404_output_file(&html_config.input_404);
204        utils::fs::write_file(destination, output_file, rendered.as_bytes())?;
205        debug!("Creating 404.html ✓");
206        Ok(())
207    }

384    /// Copy across any additional CSS and JavaScript files which the book
385    /// has been configured to use.
386    fn copy_additional_css_and_js(
387        &self,
388        html: &HtmlConfig,
389        root: &Path,
390        destination: &Path,
391    ) -> Result<()> {
392        let custom_files = html.additional_css.iter().chain(html.additional_js.iter());
393
394        debug!("Copying additional CSS and JS");
395
396        for custom_file in custom_files {
397            let input_location = root.join(custom_file);
398            let output_location = destination.join(custom_file);
399            if let Some(parent) = output_location.parent() {
400                fs::create_dir_all(parent)
401                    .with_context(|| format!("Unable to create {}", parent.display()))?;
402            }
403            debug!(
404                "Copying {} -> {}",
405                input_location.display(),
406                output_location.display()
407            );
408
409            fs::copy(&input_location, &output_location).with_context(|| {
410                format!(
411                    "Unable to copy {} to {}",
412                    input_location.display(),
413                    output_location.display()
414                )
415            })?;
416        }
417
418        Ok(())
419    }

421    fn emit_redirects(
422        &self,
423        root: &Path,
424        handlebars: &Handlebars<'_>,
425        redirects: &HashMap<String, String>,
426    ) -> Result<()> {
427        if redirects.is_empty() {
428            return Ok(());
429        }
430
431        log::debug!("Emitting redirects");
432
433        for (original, new) in redirects {
434            log::debug!("Redirecting \"{}\" → \"{}\"", original, new);
435            // Note: all paths are relative to the build directory, so the
436            // leading slash in an absolute path means nothing (and would mess
437            // up `root.join(original)`).
438            let original = original.trim_start_matches('/');
439            let filename = root.join(original);
440            self.emit_redirect(handlebars, &filename, new)?;
441        }
442
443        Ok(())
444    }
445
446    fn emit_redirect(
447        &self,
448        handlebars: &Handlebars<'_>,
449        original: &Path,
450        destination: &str,
451    ) -> Result<()> {
452        if original.exists() {
453            // sanity check to avoid accidentally overwriting a real file.
454            let msg = format!(
455                "Not redirecting \"{}\" to \"{}\" because it already exists. Are you sure it needs to be redirected?",
456                original.display(),
457                destination,
458            );
459            return Err(Error::msg(msg));
460        }
461
462        if let Some(parent) = original.parent() {
463            std::fs::create_dir_all(parent)
464                .with_context(|| format!("Unable to ensure \"{}\" exists", parent.display()))?;
465        }
466
467        let ctx = json!({
468            "url": destination,
469        });
470        let f = File::create(original)?;
471        handlebars
472            .render_to_write("redirect", &ctx, f)
473            .with_context(|| {
474                format!(
475                    "Unable to create a redirect file at \"{}\"",
476                    original.display()
477                )
478            })?;
479
480        Ok(())
481    }

485    fn name(&self) -> &str {
486        "html"
487    }

489    fn render(&self, ctx: &RenderContext) -> Result<()> {
490        let book_config = &ctx.config.book;
491        let html_config = ctx.config.html_config().unwrap_or_default();
492        let src_dir = ctx.root.join(&ctx.config.book.src);
493        let destination = &ctx.destination;
494        let book = &ctx.book;
495        let build_dir = ctx.root.join(&ctx.config.build.build_dir);
496
497        if destination.exists() {
498            utils::fs::remove_dir_content(destination)
499                .with_context(|| "Unable to remove stale HTML output")?;
500        }
501
502        trace!("render");
503        let mut handlebars = Handlebars::new();
504
505        let theme_dir = match html_config.theme {
506            Some(ref theme) => {
507                let dir = ctx.root.join(theme);
508                if !dir.is_dir() {
509                    bail!("theme dir {} does not exist", dir.display());
510                }
511                dir
512            }
513            None => ctx.root.join("theme"),
514        };
515
516        let theme = theme::Theme::new(theme_dir);
517
518        debug!("Register the index handlebars template");
519        handlebars.register_template_string("index", String::from_utf8(theme.index.clone())?)?;
520
521        debug!("Register the head handlebars template");
522        handlebars.register_partial("head", String::from_utf8(theme.head.clone())?)?;
523
524        debug!("Register the redirect handlebars template");
525        handlebars
526            .register_template_string("redirect", String::from_utf8(theme.redirect.clone())?)?;
527
528        debug!("Register the header handlebars template");
529        handlebars.register_partial("header", String::from_utf8(theme.header.clone())?)?;
530
531        debug!("Register the toc handlebars template");
532        handlebars.register_template_string("toc_js", String::from_utf8(theme.toc_js.clone())?)?;
533        handlebars
534            .register_template_string("toc_html", String::from_utf8(theme.toc_html.clone())?)?;
535
536        debug!("Register handlebars helpers");
537        self.register_hbs_helpers(&mut handlebars, &html_config);
538
539        let mut data = make_data(&ctx.root, book, &ctx.config, &html_config, &theme)?;
540
541        // Print version
542        let mut print_content = String::new();
543
544        fs::create_dir_all(destination)
545            .with_context(|| "Unexpected error when constructing destination path")?;
546
547        let mut is_index = true;
548        for item in book.iter() {
549            let ctx = RenderItemContext {
550                handlebars: &handlebars,
551                destination: destination.to_path_buf(),
552                data: data.clone(),
553                is_index,
554                book_config: book_config.clone(),
555                html_config: html_config.clone(),
556                edition: ctx.config.rust.edition,
557                chapter_titles: &ctx.chapter_titles,
558            };
559            self.render_item(item, ctx, &mut print_content)?;
560            // Only the first non-draft chapter item should be treated as the "index"
561            is_index &= !matches!(item, BookItem::Chapter(ch) if !ch.is_draft_chapter());
562        }
563
564        // Render 404 page
565        if html_config.input_404 != Some("".to_string()) {
566            self.render_404(ctx, &html_config, &src_dir, &mut handlebars, &mut data)?;
567        }
568
569        // Print version
570        self.configure_print_version(&mut data, &print_content);
571        if let Some(ref title) = ctx.config.book.title {
572            data.insert("title".to_owned(), json!(title));
573        }
574
575        // Render the handlebars template with the data
576        if html_config.print.enable {
577            debug!("Render template");
578            let rendered = handlebars.render("index", &data)?;
579
580            let rendered = self.post_process(
581                rendered,
582                &html_config.playground,
583                &html_config.code,
584                ctx.config.rust.edition,
585            );
586
587            utils::fs::write_file(destination, "print.html", rendered.as_bytes())?;
588            debug!("Creating print.html ✓");
589        }
590
591        debug!("Render toc");
592        {
593            let rendered_toc = handlebars.render("toc_js", &data)?;
594            utils::fs::write_file(destination, "toc.js", rendered_toc.as_bytes())?;
595            debug!("Creating toc.js ✓");
596            data.insert("is_toc_html".to_owned(), json!(true));
597            let rendered_toc = handlebars.render("toc_html", &data)?;
598            utils::fs::write_file(destination, "toc.html", rendered_toc.as_bytes())?;
599            debug!("Creating toc.html ✓");
600            data.remove("is_toc_html");
601        }
602
603        debug!("Copy static files");
604        self.copy_static_files(destination, &theme, &html_config)
605            .with_context(|| "Unable to copy across static files")?;
606        self.copy_additional_css_and_js(&html_config, &ctx.root, destination)
607            .with_context(|| "Unable to copy across additional CSS and JS")?;
608
609        // Render search index
610        #[cfg(feature = "search")]
611        {
612            let search = html_config.search.unwrap_or_default();
613            if search.enable {
614                super::search::create_files(&search, destination, book)?;
615            }
616        }
617
618        self.emit_redirects(&ctx.destination, &handlebars, &html_config.redirect)
619            .context("Unable to emit redirects")?;
620
621        // Copy all remaining files, avoid a recursive copy from/to the book build dir
622        utils::fs::copy_files_except_ext(&src_dir, destination, true, Some(&build_dir), &["md"])?;
623
624        Ok(())
625    }
626}

628fn make_data(
629    root: &Path,
630    book: &Book,
631    config: &Config,
632    html_config: &HtmlConfig,
633    theme: &Theme,
634) -> Result<serde_json::Map<String, serde_json::Value>> {
635    trace!("make_data");
636
637    let mut data = serde_json::Map::new();
638    data.insert(
639        "language".to_owned(),
640        json!(config.book.language.clone().unwrap_or_default()),
641    );
642    data.insert(
643        "text_direction".to_owned(),
644        json!(config.book.realized_text_direction()),
645    );
646    data.insert(
647        "book_title".to_owned(),
648        json!(config.book.title.clone().unwrap_or_default()),
649    );
650    data.insert(
651        "description".to_owned(),
652        json!(config.book.description.clone().unwrap_or_default()),
653    );
654    if theme.favicon_png.is_some() {
655        data.insert("favicon_png".to_owned(), json!("favicon.png"));
656    }
657    if theme.favicon_svg.is_some() {
658        data.insert("favicon_svg".to_owned(), json!("favicon.svg"));
659    }
660    if let Some(ref live_reload_endpoint) = html_config.live_reload_endpoint {
661        data.insert(
662            "live_reload_endpoint".to_owned(),
663            json!(live_reload_endpoint),
664        );
665    }
666
667    // TODO: remove default_theme in 0.5, it is not needed.
668    let default_theme = match html_config.default_theme {
669        Some(ref theme) => theme.to_lowercase(),
670        None => "light".to_string(),
671    };
672    data.insert("default_theme".to_owned(), json!(default_theme));
673
674    let preferred_dark_theme = match html_config.preferred_dark_theme {
675        Some(ref theme) => theme.to_lowercase(),
676        None => "navy".to_string(),
677    };
678    data.insert(
679        "preferred_dark_theme".to_owned(),
680        json!(preferred_dark_theme),
681    );
682
683    // Add google analytics tag
684    if let Some(ref ga) = html_config.google_analytics {
685        data.insert("google_analytics".to_owned(), json!(ga));
686    }
687
688    if html_config.mathjax_support {
689        data.insert("mathjax_support".to_owned(), json!(true));
690    }
691
692    // This `matches!` checks for a non-empty file.
693    if html_config.copy_fonts || matches!(theme.fonts_css.as_deref(), Some([_, ..])) {
694        data.insert("copy_fonts".to_owned(), json!(true));
695    }
696
697    // Add check to see if there is an additional style
698    if !html_config.additional_css.is_empty() {
699        let mut css = Vec::new();
700        for style in &html_config.additional_css {
701            match style.strip_prefix(root) {
702                Ok(p) => css.push(p.to_str().expect("Could not convert to str")),
703                Err(_) => css.push(style.to_str().expect("Could not convert to str")),
704            }
705        }
706        data.insert("additional_css".to_owned(), json!(css));
707    }
708
709    // Add check to see if there is an additional script
710    if !html_config.additional_js.is_empty() {
711        let mut js = Vec::new();
712        for script in &html_config.additional_js {
713            match script.strip_prefix(root) {
714                Ok(p) => js.push(p.to_str().expect("Could not convert to str")),
715                Err(_) => js.push(script.to_str().expect("Could not convert to str")),
716            }
717        }
718        data.insert("additional_js".to_owned(), json!(js));
719    }
720
721    if html_config.playground.editable && html_config.playground.copy_js {
722        data.insert("playground_js".to_owned(), json!(true));
723        if html_config.playground.line_numbers {
724            data.insert("playground_line_numbers".to_owned(), json!(true));
725        }
726    }
727    if html_config.playground.copyable {
728        data.insert("playground_copyable".to_owned(), json!(true));
729    }
730
731    data.insert("print_enable".to_owned(), json!(html_config.print.enable));
732    data.insert("fold_enable".to_owned(), json!(html_config.fold.enable));
733    data.insert("fold_level".to_owned(), json!(html_config.fold.level));
734
735    let search = html_config.search.clone();
736    if cfg!(feature = "search") {
737        let search = search.unwrap_or_default();
738        data.insert("search_enabled".to_owned(), json!(search.enable));
739        data.insert(
740            "search_js".to_owned(),
741            json!(search.enable && search.copy_js),
742        );
743    } else if search.is_some() {
744        warn!("mdBook compiled without search support, ignoring `output.html.search` table");
745        warn!(
746            "please reinstall with `cargo install mdbook --force --features search`to use the \
747             search feature"
748        )
749    }
750
751    if let Some(ref git_repository_url) = html_config.git_repository_url {
752        data.insert("git_repository_url".to_owned(), json!(git_repository_url));
753    }
754
755    let git_repository_icon = match html_config.git_repository_icon {
756        Some(ref git_repository_icon) => git_repository_icon,
757        None => "fa-github",
758    };
759    data.insert("git_repository_icon".to_owned(), json!(git_repository_icon));
760
761    let mut chapters = vec![];
762
763    for item in book.iter() {
764        // Create the data to inject in the template
765        let mut chapter = BTreeMap::new();
766
767        match *item {
768            BookItem::PartTitle(ref title) => {
769                chapter.insert("part".to_owned(), json!(title));
770            }
771            BookItem::Chapter(ref ch) => {
772                if let Some(ref section) = ch.number {
773                    chapter.insert("section".to_owned(), json!(section.to_string()));
774                }
775
776                chapter.insert(
777                    "has_sub_items".to_owned(),
778                    json!((!ch.sub_items.is_empty()).to_string()),
779                );
780
781                chapter.insert("name".to_owned(), json!(ch.name));
782                if let Some(ref path) = ch.path {
783                    let p = path
784                        .to_str()
785                        .with_context(|| "Could not convert path to str")?;
786                    chapter.insert("path".to_owned(), json!(p));
787                }
788            }
789            BookItem::Separator => {
790                chapter.insert("spacer".to_owned(), json!("_spacer_"));
791            }
792        }
793
794        chapters.push(chapter);
795    }
796
797    data.insert("chapters".to_owned(), json!(chapters));
798
799    debug!("[*]: JSON constructed");
800    Ok(data)
801}

803/// Goes through the rendered HTML, making sure all header tags have
804/// an anchor respectively so people can link to sections directly.
805fn build_header_links(html: &str) -> String {
806    static BUILD_HEADER_LINKS: Lazy<Regex> = Lazy::new(|| {
807        Regex::new(r#"<h(\d)(?: id="([^"]+)")?(?: class="([^"]+)")?>(.*?)</h\d>"#).unwrap()
808    });
809    static IGNORE_CLASS: &[&str] = &["menu-title"];
810
811    let mut id_counter = HashMap::new();
812
813    BUILD_HEADER_LINKS
814        .replace_all(html, |caps: &Captures<'_>| {
815            let level = caps[1]
816                .parse()
817                .expect("Regex should ensure we only ever get numbers here");
818
819            // Ignore .menu-title because now it's getting detected by the regex.
820            if let Some(classes) = caps.get(3) {
821                for class in classes.as_str().split(" ") {
822                    if IGNORE_CLASS.contains(&class) {
823                        return caps[0].to_string();
824                    }
825                }
826            }
827
828            insert_link_into_header(
829                level,
830                &caps[4],
831                caps.get(2).map(|x| x.as_str().to_string()),
832                caps.get(3).map(|x| x.as_str().to_string()),
833                &mut id_counter,
834            )
835        })
836        .into_owned()
837}

839/// Insert a sinle link into a header, making sure each link gets its own
840/// unique ID by appending an auto-incremented number (if necessary).
841fn insert_link_into_header(
842    level: usize,
843    content: &str,
844    id: Option<String>,
845    classes: Option<String>,
846    id_counter: &mut HashMap<String, usize>,
847) -> String {
848    let id = id.unwrap_or_else(|| utils::unique_id_from_content(content, id_counter));
849    let classes = classes
850        .map(|s| format!(" class=\"{s}\""))
851        .unwrap_or_default();
852
853    format!(
854        r##"<h{level} id="{id}"{classes}><a class="header" href="#{id}">{content}</a></h{level}>"##
855    )
856}

858// The rust book uses annotations for rustdoc to test code snippets,
859// like the following:
860// ```rust,should_panic
861// fn main() {
862//     // Code here
863// }
864// ```
865// This function replaces all commas by spaces in the code block classes
866fn fix_code_blocks(html: &str) -> String {
867    static FIX_CODE_BLOCKS: Lazy<Regex> =
868        Lazy::new(|| Regex::new(r##"<code([^>]+)class="([^"]+)"([^>]*)>"##).unwrap());
869
870    FIX_CODE_BLOCKS
871        .replace_all(html, |caps: &Captures<'_>| {
872            let before = &caps[1];
873            let classes = &caps[2].replace(',', " ");
874            let after = &caps[3];
875
876            format!(r#"<code{before}class="{classes}"{after}>"#)
877        })
878        .into_owned()
879}
880
881static CODE_BLOCK_RE: Lazy<Regex> =
882    Lazy::new(|| Regex::new(r##"((?s)<code[^>]?class="([^"]+)".*?>(.*?)</code>)"##).unwrap());
883
884fn add_playground_pre(
885    html: &str,
886    playground_config: &Playground,
887    edition: Option<RustEdition>,
888) -> String {
889    CODE_BLOCK_RE
890        .replace_all(html, |caps: &Captures<'_>| {
891            let text = &caps[1];
892            let classes = &caps[2];
893            let code = &caps[3];
894
895            if classes.contains("language-rust")
896                && ((!classes.contains("ignore")
897                    && !classes.contains("noplayground")
898                    && !classes.contains("noplaypen")
899                    && playground_config.runnable)
900                    || classes.contains("mdbook-runnable"))
901            {
902                let contains_e2015 = classes.contains("edition2015");
903                let contains_e2018 = classes.contains("edition2018");
904                let contains_e2021 = classes.contains("edition2021");
905                let edition_class = if contains_e2015 || contains_e2018 || contains_e2021 {
906                    // the user forced edition, we should not overwrite it
907                    ""
908                } else {
909                    match edition {
910                        Some(RustEdition::E2015) => " edition2015",
911                        Some(RustEdition::E2018) => " edition2018",
912                        Some(RustEdition::E2021) => " edition2021",
913                        Some(RustEdition::E2024) => " edition2024",
914                        None => "",
915                    }
916                };
917
918                // wrap the contents in an external pre block
919                format!(
920                    "<pre class=\"playground\"><code class=\"{}{}\">{}</code></pre>",
921                    classes,
922                    edition_class,
923                    {
924                        let content: Cow<'_, str> = if playground_config.editable
925                            && classes.contains("editable")
926                            || text.contains("fn main")
927                            || text.contains("quick_main!")
928                        {
929                            code.into()
930                        } else {
931                            // we need to inject our own main
932                            let (attrs, code) = partition_source(code);
933
934                            format!("# #![allow(unused)]\n{attrs}#fn main() {{\n{code}#}}").into()
935                        };
936                        content
937                    }
938                )
939            } else {
940                // not language-rust, so no-op
941                text.to_owned()
942            }
943        })
944        .into_owned()
945}

947/// Modifies all `<code>` blocks to convert "hidden" lines and to wrap them in
948/// a `<span class="boring">`.
949fn hide_lines(html: &str, code_config: &Code) -> String {
950    static LANGUAGE_REGEX: Lazy<Regex> = Lazy::new(|| Regex::new(r"\blanguage-(\w+)\b").unwrap());
951    static HIDELINES_REGEX: Lazy<Regex> = Lazy::new(|| Regex::new(r"\bhidelines=(\S+)").unwrap());
952
953    CODE_BLOCK_RE
954        .replace_all(html, |caps: &Captures<'_>| {
955            let text = &caps[1];
956            let classes = &caps[2];
957            let code = &caps[3];
958
959            if classes.contains("language-rust") {
960                format!(
961                    "<code class=\"{}\">{}</code>",
962                    classes,
963                    hide_lines_rust(code)
964                )
965            } else {
966                // First try to get the prefix from the code block
967                let hidelines_capture = HIDELINES_REGEX.captures(classes);
968                let hidelines_prefix = match &hidelines_capture {
969                    Some(capture) => Some(&capture[1]),
970                    None => {
971                        // Then look up the prefix by language
972                        LANGUAGE_REGEX.captures(classes).and_then(|capture| {
973                            code_config.hidelines.get(&capture[1]).map(|p| p.as_str())
974                        })
975                    }
976                };
977
978                match hidelines_prefix {
979                    Some(prefix) => format!(
980                        "<code class=\"{}\">{}</code>",
981                        classes,
982                        hide_lines_with_prefix(code, prefix)
983                    ),
984                    None => text.to_owned(),
985                }
986            }
987        })
988        .into_owned()
989}

991fn hide_lines_rust(content: &str) -> String {
992    static BORING_LINES_REGEX: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(\s*)#(.?)(.*)$").unwrap());
993
994    let mut result = String::with_capacity(content.len());
995    let mut lines = content.lines().peekable();
996    while let Some(line) = lines.next() {
997        // Don't include newline on the last line.
998        let newline = if lines.peek().is_none() { "" } else { "\n" };
999        if let Some(caps) = BORING_LINES_REGEX.captures(line) {
1000            if &caps[2] == "#" {
1001                result += &caps[1];
1002                result += &caps[2];
1003                result += &caps[3];
1004                result += newline;
1005                continue;
1006            } else if &caps[2] != "!" && &caps[2] != "[" {
1007                result += "<span class=\"boring\">";
1008                result += &caps[1];
1009                if &caps[2] != " " {
1010                    result += &caps[2];
1011                }
1012                result += &caps[3];
1013                result += newline;
1014                result += "</span>";
1015                continue;
1016            }
1017        }
1018        result += line;
1019        result += newline;
1020    }
1021    result
1022}

1024fn hide_lines_with_prefix(content: &str, prefix: &str) -> String {
1025    let mut result = String::with_capacity(content.len());
1026    for line in content.lines() {
1027        if line.trim_start().starts_with(prefix) {
1028            let pos = line.find(prefix).unwrap();
1029            let (ws, rest) = (&line[..pos], &line[pos + prefix.len()..]);
1030
1031            result += "<span class=\"boring\">";
1032            result += ws;
1033            result += rest;
1034            result += "\n";
1035            result += "</span>";
1036            continue;
1037        }
1038        result += line;
1039        result += "\n";
1040    }
1041    result
1042}

1044fn partition_source(s: &str) -> (String, String) {
1045    let mut after_header = false;
1046    let mut before = String::new();
1047    let mut after = String::new();
1048
1049    for line in s.lines() {
1050        let trimline = line.trim();
1051        let header = trimline.chars().all(char::is_whitespace) || trimline.starts_with("#![");
1052        if !header || after_header {
1053            after_header = true;
1054            after.push_str(line);
1055            after.push('\n');
1056        } else {
1057            before.push_str(line);
1058            before.push('\n');
1059        }
1060    }
1061
1062    (before, after)
1063}
