7    Context, Handlebars, Helper, HelperDef, Output, RenderContext, RenderError, RenderErrorReason,

10// Handlebars helper to construct TOC
11#[derive(Clone, Copy)]
12pub struct RenderToc {
13    pub no_section_label: bool,
14}
15
16impl HelperDef for RenderToc {
17    fn call<'reg: 'rc, 'rc>(
18        &self,
19        _h: &Helper<'rc>,
20        _r: &'reg Handlebars<'_>,
21        ctx: &'rc Context,
22        rc: &mut RenderContext<'reg, 'rc>,
23        out: &mut dyn Output,
24    ) -> Result<(), RenderError> {
25        // get value from context data
26        // rc.get_path() is current json parent path, you should always use it like this
27        // param is the key of value you want to display
28        let chapters = rc.evaluate(ctx, "@root/chapters").and_then(|c| {
29            serde_json::value::from_value::<Vec<BTreeMap<String, String>>>(c.as_json().clone())
30                .map_err(|_| {
31                    RenderErrorReason::Other("Could not decode the JSON data".to_owned()).into()
32                })
33        })?;
34
35        let fold_enable = rc
36            .evaluate(ctx, "@root/fold_enable")?
37            .as_json()
38            .as_bool()
39            .ok_or_else(|| {
40                RenderErrorReason::Other("Type error for `fold_enable`, bool expected".to_owned())
41            })?;
42
43        let fold_level = rc
44            .evaluate(ctx, "@root/fold_level")?
45            .as_json()
46            .as_u64()
47            .ok_or_else(|| {
48                RenderErrorReason::Other("Type error for `fold_level`, u64 expected".to_owned())
49            })?;
50
51        // If true, then this is the iframe and we need target="_parent"
52        let is_toc_html = rc
53            .evaluate(ctx, "@root/is_toc_html")?
54            .as_json()
55            .as_bool()
56            .unwrap_or(false);
57
58        out.write("<ol class=\"chapter\">")?;
59
60        let mut current_level = 1;
61
62        for item in chapters {
63            let (_section, level) = if let Some(s) = item.get("section") {
64                (s.as_str(), s.matches('.').count())
65            } else {
66                ("", 1)
67            };
68
69            // Expand if folding is disabled, or if levels that are larger than this would not
70            // be folded.
71            let is_expanded = !fold_enable || level - 1 < (fold_level as usize);
72
73            match level.cmp(&current_level) {
74                Ordering::Greater => {
75                    while level > current_level {
76                        out.write("<li>")?;
77                        out.write("<ol class=\"section\">")?;
78                        current_level += 1;
79                    }
80                    write_li_open_tag(out, is_expanded, false)?;
81                }
82                Ordering::Less => {
83                    while level < current_level {
84                        out.write("</ol>")?;
85                        out.write("</li>")?;
86                        current_level -= 1;
87                    }
88                    write_li_open_tag(out, is_expanded, false)?;
89                }
90                Ordering::Equal => {
91                    write_li_open_tag(out, is_expanded, !item.contains_key("section"))?;
92                }
93            }
94
95            // Spacer
96            if item.contains_key("spacer") {
97                out.write("<li class=\"spacer\"></li>")?;
98                continue;
99            }
100
101            // Part title
102            if let Some(title) = item.get("part") {
103                out.write("<li class=\"part-title\">")?;
104                out.write(&special_escape(title))?;
105                out.write("</li>")?;
106                continue;
107            }
108
109            // Link
110            let path_exists = match item.get("path") {
111                Some(path) if !path.is_empty() => {
112                    out.write("<a href=\"")?;
113                    let tmp = Path::new(path)
114                        .with_extension("html")
115                        .to_str()
116                        .unwrap()
117                        // Hack for windows who tends to use `\` as separator instead of `/`
118                        .replace('\\', "/");
119
120                    // Add link
121                    out.write(&tmp)?;
122                    out.write(if is_toc_html {
123                        "\" target=\"_parent\">"
124                    } else {
125                        "\">"
126                    })?;
127                    true
128                }
129                _ => {
130                    out.write("<div>")?;
131                    false
132                }
133            };
134
135            if !self.no_section_label {
136                // Section does not necessarily exist
137                if let Some(section) = item.get("section") {
138                    out.write("<strong aria-hidden=\"true\">")?;
139                    out.write(section)?;
140                    out.write("</strong> ")?;
141                }
142            }
143
144            if let Some(name) = item.get("name") {
145                out.write(&special_escape(name))?
146            }
147
148            if path_exists {
149                out.write("</a>")?;
150            } else {
151                out.write("</div>")?;
152            }
153
154            // Render expand/collapse toggle
155            if let Some(flag) = item.get("has_sub_items") {
156                let has_sub_items = flag.parse::<bool>().unwrap_or_default();
157                if fold_enable && has_sub_items {
158                    out.write("<a class=\"toggle\"><div>‚ù±</div></a>")?;
159                }
160            }
161            out.write("</li>")?;
162        }
163        while current_level > 1 {
164            out.write("</ol>")?;
165            out.write("</li>")?;
166            current_level -= 1;
167        }
168
169        out.write("</ol>")?;
170        Ok(())
171    }
172}

174fn write_li_open_tag(
175    out: &mut dyn Output,
176    is_expanded: bool,
177    is_affix: bool,
178) -> Result<(), std::io::Error> {
179    let mut li = String::from("<li class=\"chapter-item ");
180    if is_expanded {
181        li.push_str("expanded ");
182    }
183    if is_affix {
184        li.push_str("affix ");
185    }
186    li.push_str("\">");
187    out.write(&li)
188}
