43/// The `Theme` struct should be used instead of the static variables because
44/// the `new()` method will look if the user has a theme directory in their
45/// source folder and use the users theme instead of the default.
46///
47/// You should only ever use the static variables directly if you want to
48/// override the user's theme with the defaults.
49#[derive(Debug, PartialEq)]

73impl Theme {
74    /// Creates a `Theme` from the given `theme_dir`.
75    /// If a file is found in the theme dir, it will override the default version.
76    pub fn new<P: AsRef<Path>>(theme_dir: P) -> Self {
77        let theme_dir = theme_dir.as_ref();
78        let mut theme = Theme::default();
79
80        // If the theme directory doesn't exist there's no point continuing...
81        if !theme_dir.exists() || !theme_dir.is_dir() {
82            return theme;
83        }
84
85        // Check for individual files, if they exist copy them across
86        {
87            let files = vec![
88                (theme_dir.join("index.hbs"), &mut theme.index),
89                (theme_dir.join("head.hbs"), &mut theme.head),
90                (theme_dir.join("redirect.hbs"), &mut theme.redirect),
91                (theme_dir.join("header.hbs"), &mut theme.header),
92                (theme_dir.join("toc.js.hbs"), &mut theme.toc_js),
93                (theme_dir.join("toc.html.hbs"), &mut theme.toc_html),
94                (theme_dir.join("book.js"), &mut theme.js),
95                (theme_dir.join("css/chrome.css"), &mut theme.chrome_css),
96                (theme_dir.join("css/general.css"), &mut theme.general_css),
97                (theme_dir.join("css/print.css"), &mut theme.print_css),
98                (
99                    theme_dir.join("css/variables.css"),
100                    &mut theme.variables_css,
101                ),
102                (theme_dir.join("highlight.js"), &mut theme.highlight_js),
103                (theme_dir.join("clipboard.min.js"), &mut theme.clipboard_js),
104                (theme_dir.join("highlight.css"), &mut theme.highlight_css),
105                (
106                    theme_dir.join("tomorrow-night.css"),
107                    &mut theme.tomorrow_night_css,
108                ),
109                (
110                    theme_dir.join("ayu-highlight.css"),
111                    &mut theme.ayu_highlight_css,
112                ),
113            ];
114
115            let load_with_warn = |filename: &Path, dest: &mut Vec<u8>| {
116                if !filename.exists() {
117                    // Don't warn if the file doesn't exist.
118                    return false;
119                }
120                if let Err(e) = load_file_contents(filename, dest) {
121                    warn!("Couldn't load custom file, {}: {}", filename.display(), e);
122                    false
123                } else {
124                    true
125                }
126            };
127
128            for (filename, dest) in files {
129                load_with_warn(&filename, dest);
130            }
131
132            let fonts_dir = theme_dir.join("fonts");
133            if fonts_dir.exists() {
134                let mut fonts_css = Vec::new();
135                if load_with_warn(&fonts_dir.join("fonts.css"), &mut fonts_css) {
136                    theme.fonts_css.replace(fonts_css);
137                }
138                if let Ok(entries) = fonts_dir.read_dir() {
139                    theme.font_files = entries
140                        .filter_map(|entry| {
141                            let entry = entry.ok()?;
142                            if entry.file_name() == "fonts.css" {
143                                None
144                            } else if entry.file_type().ok()?.is_dir() {
145                                log::info!("skipping font directory {:?}", entry.path());
146                                None
147                            } else {
148                                Some(entry.path())
149                            }
150                        })
151                        .collect();
152                }
153            }
154
155            // If the user overrides one favicon, but not the other, do not
156            // copy the default for the other.
157            let favicon_png = &mut theme.favicon_png.as_mut().unwrap();
158            let png = load_with_warn(&theme_dir.join("favicon.png"), favicon_png);
159            let favicon_svg = &mut theme.favicon_svg.as_mut().unwrap();
160            let svg = load_with_warn(&theme_dir.join("favicon.svg"), favicon_svg);
161            match (png, svg) {
162                (true, true) | (false, false) => {}
163                (true, false) => {
164                    theme.favicon_svg = None;
165                }
166                (false, true) => {
167                    theme.favicon_png = None;
168                }
169            }
170        }
171
172        theme
173    }
174}

203/// Checks if a file exists, if so, the destination buffer will be filled with
204/// its contents.
205fn load_file_contents<P: AsRef<Path>>(filename: P, dest: &mut Vec<u8>) -> Result<()> {
206    let filename = filename.as_ref();
207
208    let mut buffer = Vec::new();
209    File::open(filename)?.read_to_end(&mut buffer)?;
210
211    // We needed the buffer so we'd only overwrite the existing content if we
212    // could successfully load the file into memory.
213    dest.clear();
214    dest.append(&mut buffer);
215
216    Ok(())
217}
