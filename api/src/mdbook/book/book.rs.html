14/// Load a book into memory from its `src/` directory.
15pub fn load_book<P: AsRef<Path>>(src_dir: P, cfg: &BuildConfig) -> Result<Book> {
16    let src_dir = src_dir.as_ref();
17    let summary_md = src_dir.join("SUMMARY.md");
18
19    let mut summary_content = String::new();
20    File::open(&summary_md)
21        .with_context(|| format!("Couldn't open SUMMARY.md in {src_dir:?} directory"))?
22        .read_to_string(&mut summary_content)?;
23
24    let summary = parse_summary(&summary_content)
25        .with_context(|| format!("Summary parsing failed for file={summary_md:?}"))?;
26
27    if cfg.create_missing {
28        create_missing(src_dir, &summary).with_context(|| "Unable to create missing chapters")?;
29    }
30
31    load_book_from_disk(&summary, src_dir)
32}

34fn create_missing(src_dir: &Path, summary: &Summary) -> Result<()> {
35    let mut items: Vec<_> = summary
36        .prefix_chapters
37        .iter()
38        .chain(summary.numbered_chapters.iter())
39        .chain(summary.suffix_chapters.iter())
40        .collect();
41
42    while let Some(next) = items.pop() {
43        if let SummaryItem::Link(ref link) = *next {
44            if let Some(ref location) = link.location {
45                let filename = src_dir.join(location);
46                if !filename.exists() {
47                    if let Some(parent) = filename.parent() {
48                        if !parent.exists() {
49                            fs::create_dir_all(parent)?;
50                        }
51                    }
52                    debug!("Creating missing file {}", filename.display());
53
54                    let mut f = File::create(&filename).with_context(|| {
55                        format!("Unable to create missing file: {}", filename.display())
56                    })?;
57                    writeln!(f, "# {}", bracket_escape(&link.name))?;
58                }
59            }
60
61            items.extend(&link.nested_items);
62        }
63    }
64
65    Ok(())
66}

68/// A dumb tree structure representing a book.
69///
70/// For the moment a book is just a collection of [`BookItems`] which are
71/// accessible by either iterating (immutably) over the book with [`iter()`], or
72/// recursively applying a closure to each section to mutate the chapters, using
73/// [`for_each_mut()`].
74///
75/// [`iter()`]: #method.iter
76/// [`for_each_mut()`]: #method.for_each_mut
77#[derive(Debug, Clone, Default, PartialEq, Serialize, Deserialize)]
78pub struct Book {
79    /// The sections in this book.
80    pub sections: Vec<BookItem>,
81    __non_exhaustive: (),
82}
83
84impl Book {
85    /// Create an empty book.
86    pub fn new() -> Self {
87        Default::default()
88    }
89
90    /// Get a depth-first iterator over the items in the book.
91    pub fn iter(&self) -> BookItems<'_> {
92        BookItems {
93            items: self.sections.iter().collect(),
94        }
95    }
96
97    /// Recursively apply a closure to each item in the book, allowing you to
98    /// mutate them.
99    ///
100    /// # Note
101    ///
102    /// Unlike the `iter()` method, this requires a closure instead of returning
103    /// an iterator. This is because using iterators can possibly allow you
104    /// to have iterator invalidation errors.
105    pub fn for_each_mut<F>(&mut self, mut func: F)
106    where
107        F: FnMut(&mut BookItem),
108    {
109        for_each_mut(&mut func, &mut self.sections);
110    }
111
112    /// Append a `BookItem` to the `Book`.
113    pub fn push_item<I: Into<BookItem>>(&mut self, item: I) -> &mut Self {
114        self.sections.push(item.into());
115        self
116    }
117}
118
119pub fn for_each_mut<'a, F, I>(func: &mut F, items: I)
120where
121    F: FnMut(&mut BookItem),
122    I: IntoIterator<Item = &'a mut BookItem>,
123{
124    for item in items {
125        if let BookItem::Chapter(ch) = item {
126            for_each_mut(func, &mut ch.sub_items);
127        }
128
129        func(item);
130    }
131}

150/// The representation of a "chapter", usually mapping to a single file on
151/// disk however it may contain multiple sub-chapters.
152#[derive(Debug, Clone, Default, PartialEq, Serialize, Deserialize)]
153pub struct Chapter {
154    /// The chapter's name.
155    pub name: String,
156    /// The chapter's contents.
157    pub content: String,
158    /// The chapter's section number, if it has one.
159    pub number: Option<SectionNumber>,
160    /// Nested items.
161    pub sub_items: Vec<BookItem>,
162    /// The chapter's location, relative to the `SUMMARY.md` file.
163    ///
164    /// **Note**: After the index preprocessor runs, any README files will be
165    /// modified to be `index.md`. If you need access to the actual filename
166    /// on disk, use [`Chapter::source_path`] instead.
167    ///
168    /// This is `None` for a draft chapter.
169    pub path: Option<PathBuf>,
170    /// The chapter's source file, relative to the `SUMMARY.md` file.
171    ///
172    /// **Note**: Beware that README files will internally be treated as
173    /// `index.md` via the [`Chapter::path`] field. The `source_path` field
174    /// exists if you need access to the true file path.
175    ///
176    /// This is `None` for a draft chapter.
177    pub source_path: Option<PathBuf>,
178    /// An ordered list of the names of each chapter above this one in the hierarchy.
179    pub parent_names: Vec<String>,
180}
181
182impl Chapter {
183    /// Create a new chapter with the provided content.
184    pub fn new<P: Into<PathBuf>>(
185        name: &str,
186        content: String,
187        p: P,
188        parent_names: Vec<String>,
189    ) -> Chapter {
190        let path: PathBuf = p.into();
191        Chapter {
192            name: name.to_string(),
193            content,
194            path: Some(path.clone()),
195            source_path: Some(path),
196            parent_names,
197            ..Default::default()
198        }
199    }
200
201    /// Create a new draft chapter that is not attached to a source markdown file (and thus
202    /// has no content).
203    pub fn new_draft(name: &str, parent_names: Vec<String>) -> Self {
204        Chapter {
205            name: name.to_string(),
206            content: String::new(),
207            path: None,
208            source_path: None,
209            parent_names,
210            ..Default::default()
211        }
212    }
213
214    /// Check if the chapter is a draft chapter, meaning it has no path to a source markdown file.
215    pub fn is_draft_chapter(&self) -> bool {
216        self.path.is_none()
217    }
218}

220/// Use the provided `Summary` to load a `Book` from disk.
221///
222/// You need to pass in the book's source directory because all the links in
223/// `SUMMARY.md` give the chapter locations relative to it.
224pub(crate) fn load_book_from_disk<P: AsRef<Path>>(summary: &Summary, src_dir: P) -> Result<Book> {
225    debug!("Loading the book from disk");
226    let src_dir = src_dir.as_ref();
227
228    let prefix = summary.prefix_chapters.iter();
229    let numbered = summary.numbered_chapters.iter();
230    let suffix = summary.suffix_chapters.iter();
231
232    let summary_items = prefix.chain(numbered).chain(suffix);
233
234    let mut chapters = Vec::new();
235
236    for summary_item in summary_items {
237        let chapter = load_summary_item(summary_item, src_dir, Vec::new())?;
238        chapters.push(chapter);
239    }
240
241    Ok(Book {
242        sections: chapters,
243        __non_exhaustive: (),
244    })
245}

247fn load_summary_item<P: AsRef<Path> + Clone>(
248    item: &SummaryItem,
249    src_dir: P,
250    parent_names: Vec<String>,
251) -> Result<BookItem> {
252    match item {
253        SummaryItem::Separator => Ok(BookItem::Separator),
254        SummaryItem::Link(ref link) => {
255            load_chapter(link, src_dir, parent_names).map(BookItem::Chapter)
256        }
257        SummaryItem::PartTitle(title) => Ok(BookItem::PartTitle(title.clone())),
258    }
259}

261fn load_chapter<P: AsRef<Path>>(
262    link: &Link,
263    src_dir: P,
264    parent_names: Vec<String>,
265) -> Result<Chapter> {
266    let src_dir = src_dir.as_ref();
267
268    let mut ch = if let Some(ref link_location) = link.location {
269        debug!("Loading {} ({})", link.name, link_location.display());
270
271        let location = if link_location.is_absolute() {
272            link_location.clone()
273        } else {
274            src_dir.join(link_location)
275        };
276
277        let mut f = File::open(&location)
278            .with_context(|| format!("Chapter file not found, {}", link_location.display()))?;
279
280        let mut content = String::new();
281        f.read_to_string(&mut content).with_context(|| {
282            format!("Unable to read \"{}\" ({})", link.name, location.display())
283        })?;
284
285        if content.as_bytes().starts_with(b"\xef\xbb\xbf") {
286            content.replace_range(..3, "");
287        }
288
289        let stripped = location
290            .strip_prefix(src_dir)
291            .expect("Chapters are always inside a book");
292
293        Chapter::new(&link.name, content, stripped, parent_names.clone())
294    } else {
295        Chapter::new_draft(&link.name, parent_names.clone())
296    };
297
298    let mut sub_item_parents = parent_names;
299
300    ch.number = link.number.clone();
301
302    sub_item_parents.push(link.name.clone());
303    let sub_items = link
304        .nested_items
305        .iter()
306        .map(|i| load_summary_item(i, src_dir, sub_item_parents.clone()))
307        .collect::<Result<Vec<_>>>()?;
308
309    ch.sub_items = sub_items;
310
311    Ok(ch)
312}

314/// A depth-first iterator over the items in a book.
315///
316/// # Note
317///
318/// This struct shouldn't be created directly, instead prefer the
319/// [`Book::iter()`] method.
320pub struct BookItems<'a> {
321    items: VecDeque<&'a BookItem>,
322}
323
324impl<'a> Iterator for BookItems<'a> {
325    type Item = &'a BookItem;
326
327    fn next(&mut self) -> Option<Self::Item> {
328        let item = self.items.pop_front();
329
330        if let Some(BookItem::Chapter(ch)) = item {
331            // if we wanted a breadth-first iterator we'd `extend()` here
332            for sub_item in ch.sub_items.iter().rev() {
333                self.items.push_front(sub_item);
334            }
335        }
336
337        item
338    }
339}

341impl Display for Chapter {
342    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
343        if let Some(ref section_number) = self.number {
344            write!(f, "{section_number} ")?;
345        }
346
347        write!(f, "{}", self.name)
348    }
