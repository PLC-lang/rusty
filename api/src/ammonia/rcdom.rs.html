12//! A simple reference-counted DOM.
13//!
14//! This is sufficient as a static parse tree, but don't build a
15//! web browser using it. :)
16//!
17//! A DOM is a [tree structure] with ordered children that can be represented in an XML-like
18//! format. For example, the following graph
19//!
20//! ```text
21//! div
22//!  +- "text node"
23//!  +- span
24//! ```
25//! in HTML would be serialized as
26//!
27//! ```html
28//! <div>text node<span></span></div>
29//! ```
30//!
31//! See the [document object model article on wikipedia][dom wiki] for more information.
32//!
33//! This implementation stores the information associated with each node once, and then hands out
34//! refs to children. The nodes themselves are reference-counted to avoid copying - you can create
35//! a new ref and then a node will outlive the document. Nodes own their children, but only have
36//! weak references to their parents.
37//!
38//! [tree structure]: https://en.wikipedia.org/wiki/Tree_(data_structure)
39//! [dom wiki]: https://en.wikipedia.org/wiki/Document_Object_Model
40

61/// The different kinds of nodes in the DOM.
62#[derive(Debug)]
63pub enum NodeData {
64    /// The `Document` itself - the root node of a HTML document.
65    Document,
66
67    /// A `DOCTYPE` with name, public id, and system id. See
68    /// [document type declaration on wikipedia][dtd wiki].
69    ///
70    /// [dtd wiki]: https://en.wikipedia.org/wiki/Document_type_declaration
71    Doctype {
72        name: StrTendril,
73    },
74
75    /// A text node.
76    Text { contents: RefCell<StrTendril> },
77
78    /// A comment.
79    Comment { contents: StrTendril },
80
81    /// An element with attributes.
82    Element {
83        name: QualName,
84        attrs: RefCell<Vec<Attribute>>,
85
86        /// For HTML \<template\> elements, the [template contents].
87        ///
88        /// [template contents]: https://html.spec.whatwg.org/multipage/#template-contents
89        template_contents: RefCell<Option<Handle>>,
90
91        /// Whether the node is a [HTML integration point].
92        ///
93        /// [HTML integration point]: https://html.spec.whatwg.org/multipage/#html-integration-point
94        mathml_annotation_xml_integration_point: bool,
95    },
96
97    /// A Processing instruction.
98    ProcessingInstruction {
99        target: StrTendril,
100        contents: StrTendril,
101    },
102}

104/// A DOM node.
105pub struct Node {
106    /// Parent node.
107    pub parent: Cell<Option<WeakHandle>>,
108    /// Child nodes of this node.
109    pub children: RefCell<Vec<Handle>>,
110    /// Represents this node's data.
111    pub data: NodeData,
112}
113
114impl Node {
115    /// Create a new node from its contents
116    pub fn new(data: NodeData) -> Rc<Self> {
117        Rc::new(Node {
118            data,
119            parent: Cell::new(None),
120            children: RefCell::new(Vec::new()),
121        })
122    }
123}
124
125impl Drop for Node {
126    fn drop(&mut self) {
127        let mut nodes = mem::take(&mut *self.children.borrow_mut());
128        while let Some(node) = nodes.pop() {
129            let children = mem::take(&mut *node.children.borrow_mut());
130            nodes.extend(children.into_iter());
131            if let NodeData::Element {
132                ref template_contents,
133                ..
134            } = node.data
135            {
136                if let Some(template_contents) = template_contents.borrow_mut().take() {
137                    nodes.push(template_contents);
138                }
139            }
140        }
141    }
142}
143
144impl fmt::Debug for Node {
145    fn fmt(&self, fmt: &mut fmt::Formatter) -> fmt::Result {
146        fmt.debug_struct("Node")
147            .field("data", &self.data)
148            .field("children", &self.children)
149            .finish()
150    }
151}

153/// Reference to a DOM node.
154pub type Handle = Rc<Node>;
155
156/// Weak reference to a DOM node, used for parent pointers.
157pub type WeakHandle = Weak<Node>;
158
159/// Append a parentless node to another nodes' children
160fn append(new_parent: &Handle, child: Handle) {
161    let previous_parent = child.parent.replace(Some(Rc::downgrade(new_parent)));
162    // Invariant: child cannot have existing parent
163    assert!(previous_parent.is_none());
164    new_parent.children.borrow_mut().push(child);
165}
166
167/// If the node has a parent, get it and this node's position in its children
168fn get_parent_and_index(target: &Handle) -> Option<(Handle, usize)> {
169    if let Some(weak) = target.parent.take() {
170        let parent = weak.upgrade().expect("dangling weak pointer");
171        target.parent.set(Some(weak));
172        let i = match parent
173            .children
174            .borrow()
175            .iter()
176            .enumerate()
177            .find(|&(_, child)| Rc::ptr_eq(child, target))
178        {
179            Some((i, _)) => i,
180            None => panic!("have parent but couldn't find in parent's children!"),
181        };
182        Some((parent, i))
183    } else {
184        None
185    }
186}
187
188fn append_to_existing_text(prev: &Handle, text: &str) -> bool {
189    match prev.data {
190        NodeData::Text { ref contents } => {
191            contents.borrow_mut().push_slice(text);
192            true
193        }
194        _ => false,
195    }
196}
197
198fn remove_from_parent(target: &Handle) {
199    if let Some((parent, i)) = get_parent_and_index(target) {
200        parent.children.borrow_mut().remove(i);
201        target.parent.set(None);
202    }
203}

205/// The DOM itself; the result of parsing.
206pub struct RcDom {
207    /// The `Document` itself.
208    pub document: Handle,
209
210    /// Errors that occurred during parsing.
211    pub errors: RefCell<Vec<Cow<'static, str>>>,
212
213    /// The document's quirks mode.
214    pub quirks_mode: Cell<QuirksMode>,
215}
216
217impl TreeSink for RcDom {
218    type Output = Self;
219    fn finish(self) -> Self {
220        self
221    }
222
223    type Handle = Handle;
224
225    type ElemName<'a> = ExpandedName<'a>;
226
227    fn parse_error(&self, msg: Cow<'static, str>) {
228        self.errors.borrow_mut().push(msg);
229    }
230
231    fn get_document(&self) -> Handle {
232        self.document.clone()
233    }
234
235    fn get_template_contents(&self, target: &Handle) -> Handle {
236        if let NodeData::Element {
237            ref template_contents,
238            ..
239        } = target.data
240        {
241            template_contents
242                .borrow()
243                .as_ref()
244                .expect("not a template element!")
245                .clone()
246        } else {
247            panic!("not a template element!")
248        }
249    }
250
251    fn set_quirks_mode(&self, mode: QuirksMode) {
252        self.quirks_mode.set(mode);
253    }
254
255    fn same_node(&self, x: &Handle, y: &Handle) -> bool {
256        Rc::ptr_eq(x, y)
257    }
258
259    fn elem_name<'a>(&self, target: &'a Handle) -> ExpandedName<'a> {
260        return match target.data {
261            NodeData::Element { ref name, .. } => name.expanded(),
262            _ => panic!("not an element!"),
263        };
264    }
265
266    fn create_element(
267        &self,
268        name: QualName,
269        attrs: Vec<Attribute>,
270        flags: ElementFlags,
271    ) -> Handle {
272        Node::new(NodeData::Element {
273            name,
274            attrs: RefCell::new(attrs),
275            template_contents: RefCell::new(if flags.template {
276                Some(Node::new(NodeData::Document))
277            } else {
278                None
279            }),
280            mathml_annotation_xml_integration_point: flags.mathml_annotation_xml_integration_point,
281        })
282    }
283
284    fn create_comment(&self, text: StrTendril) -> Handle {
285        Node::new(NodeData::Comment { contents: text })
286    }
287
288    fn create_pi(&self, target: StrTendril, data: StrTendril) -> Handle {
289        Node::new(NodeData::ProcessingInstruction {
290            target,
291            contents: data,
292        })
293    }
294
295    fn append(&self, parent: &Handle, child: NodeOrText<Handle>) {
296        // Append to an existing Text node if we have one.
297        if let NodeOrText::AppendText(ref text) = child {
298            if let Some(h) = parent.children.borrow().last() {
299                if append_to_existing_text(h, text) {
300                    return;
301                }
302            }
303        }
304
305        append(
306            parent,
307            match child {
308                NodeOrText::AppendText(text) => Node::new(NodeData::Text {
309                    contents: RefCell::new(text),
310                }),
311                NodeOrText::AppendNode(node) => node,
312            },
313        );
314    }
315
316    fn append_before_sibling(&self, sibling: &Handle, child: NodeOrText<Handle>) {
317        let (parent, i) = get_parent_and_index(sibling)
318            .expect("append_before_sibling called on node without parent");
319
320        let child = match (child, i) {
321            // No previous node.
322            (NodeOrText::AppendText(text), 0) => Node::new(NodeData::Text {
323                contents: RefCell::new(text),
324            }),
325
326            // Look for a text node before the insertion point.
327            (NodeOrText::AppendText(text), i) => {
328                let children = parent.children.borrow();
329                let prev = &children[i - 1];
330                if append_to_existing_text(prev, &text) {
331                    return;
332                }
333                Node::new(NodeData::Text {
334                    contents: RefCell::new(text),
335                })
336            }
337
338            // The tree builder promises we won't have a text node after
339            // the insertion point.
340
341            // Any other kind of node.
342            (NodeOrText::AppendNode(node), _) => node,
343        };
344
345        remove_from_parent(&child);
346
347        child.parent.set(Some(Rc::downgrade(&parent)));
348        parent.children.borrow_mut().insert(i, child);
349    }
350
351    fn append_based_on_parent_node(
352        &self,
353        element: &Self::Handle,
354        prev_element: &Self::Handle,
355        child: NodeOrText<Self::Handle>,
356    ) {
357        let parent = element.parent.take();
358        let has_parent = parent.is_some();
359        element.parent.set(parent);
360
361        if has_parent {
362            self.append_before_sibling(element, child);
363        } else {
364            self.append(prev_element, child);
365        }
366    }
367
368    fn append_doctype_to_document(
369        &self,
370        name: StrTendril,
371        _public_id: StrTendril,
372        _system_id: StrTendril,
373    ) {
374        append(
375            &self.document,
376            Node::new(NodeData::Doctype {
377                name,
378            }),
379        );
380    }
381
382    fn add_attrs_if_missing(&self, target: &Handle, attrs: Vec<Attribute>) {
383        let mut existing = if let NodeData::Element { ref attrs, .. } = target.data {
384            attrs.borrow_mut()
385        } else {
386            panic!("not an element")
387        };
388
389        let existing_names = existing
390            .iter()
391            .map(|e| e.name.clone())
392            .collect::<HashSet<_>>();
393        existing.extend(
394            attrs
395                .into_iter()
396                .filter(|attr| !existing_names.contains(&attr.name)),
397        );
398    }
399
400    fn remove_from_parent(&self, target: &Handle) {
401        remove_from_parent(target);
402    }
403
404    fn reparent_children(&self, node: &Handle, new_parent: &Handle) {
405        let mut children = node.children.borrow_mut();
406        let mut new_children = new_parent.children.borrow_mut();
407        for child in children.iter() {
408            let previous_parent = child.parent.replace(Some(Rc::downgrade(new_parent)));
409            assert!(Rc::ptr_eq(
410                node,
411                &previous_parent.unwrap().upgrade().expect("dangling weak")
412            ))
413        }
414        new_children.extend(mem::take(&mut *children));
415    }
416
417    fn is_mathml_annotation_xml_integration_point(&self, target: &Handle) -> bool {
418        if let NodeData::Element {
419            mathml_annotation_xml_integration_point,
420            ..
421        } = target.data
422        {
423            mathml_annotation_xml_integration_point
424        } else {
425            panic!("not an element!")
426        }
427    }
428}
429
430impl Default for RcDom {
431    fn default() -> RcDom {
432        RcDom {
433            document: Node::new(NodeData::Document),
434            errors: vec![].into(),
435            quirks_mode: tree_builder::NoQuirks.into(),
436        }
437    }
438}

440enum SerializeOp {
441    Open(Handle),
442    Close(QualName),
443}
444
445pub struct SerializableHandle(Handle);
446
447impl From<Handle> for SerializableHandle {
448    fn from(h: Handle) -> SerializableHandle {
449        SerializableHandle(h)
450    }
451}
452
453impl Serialize for SerializableHandle {
454    fn serialize<S>(&self, serializer: &mut S, traversal_scope: TraversalScope) -> io::Result<()>
455    where
456        S: Serializer,
457    {
458        let mut ops = VecDeque::new();
459        match traversal_scope {
460            IncludeNode => ops.push_back(SerializeOp::Open(self.0.clone())),
461            ChildrenOnly(_) => ops.extend(
462                self.0
463                    .children
464                    .borrow()
465                    .iter()
466                    .map(|h| SerializeOp::Open(h.clone())),
467            ),
468        }
469
470        while let Some(op) = ops.pop_front() {
471            match op {
472                SerializeOp::Open(handle) => match handle.data {
473                    NodeData::Element {
474                        ref name,
475                        ref attrs,
476                        ..
477                    } => {
478                        serializer.start_elem(
479                            name.clone(),
480                            attrs.borrow().iter().map(|at| (&at.name, &at.value[..])),
481                        )?;
482
483                        ops.reserve(1 + handle.children.borrow().len());
484                        ops.push_front(SerializeOp::Close(name.clone()));
485
486                        for child in handle.children.borrow().iter().rev() {
487                            ops.push_front(SerializeOp::Open(child.clone()));
488                        }
489                    }
490
491                    NodeData::Doctype { ref name, .. } => serializer.write_doctype(name)?,
492
493                    NodeData::Text { ref contents } => serializer.write_text(&contents.borrow())?,
494
495                    NodeData::Comment { ref contents } => serializer.write_comment(contents)?,
496
497                    NodeData::ProcessingInstruction {
498                        ref target,
499                        ref contents,
500                    } => serializer.write_processing_instruction(target, contents)?,
501
502                    NodeData::Document => panic!("Can't serialize Document node itself"),
503                },
504
505                SerializeOp::Close(name) => {
506                    serializer.end_elem(name)?;
507                }
508            }
509        }
510
511        Ok(())
512    }
