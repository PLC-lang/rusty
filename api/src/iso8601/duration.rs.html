7/// A time duration.
8/// Durations:
9/// <https://www.rfc-editor.org/rfc/rfc3339#page-13>
10///    dur-second        = 1*DIGIT "S"
11///    dur-minute        = 1*DIGIT "M" [dur-second]
12///    dur-hour          = 1*DIGIT "H" [dur-minute]
13///    dur-time          = "T" (dur-hour / dur-minute / dur-second)
14///    dur-day           = 1*DIGIT "D"
15///    dur-week          = 1*DIGIT "W"
16///    dur-month         = 1*DIGIT "M" [dur-day]
17///    dur-year          = 1*DIGIT "Y" [dur-month]
18///    dur-date          = (dur-day / dur-month / dur-year) [dur-time]
19///    duration          = "P" (dur-date / dur-time / dur-week)
20/// ```
21///# use std::str::FromStr;
22/// assert_eq!(iso8601::Duration::from_str("P2021Y11M16DT23H26M59.123S"), Ok(iso8601::Duration::YMDHMS{ year: 2021, month: 11, day: 16, hour: 23, minute: 26, second: 59, millisecond: 123 }))
23/// ```
24#[derive(Eq, PartialEq, Debug, Copy, Clone)]
25pub enum Duration {
26    /// A duration specified by year, month, day, hour, minute and second units
27    YMDHMS {
28        /// Number of calendar years
29        year: u32,
30        /// Number of months
31        month: u32,
32        /// Number of days
33        day: u32,
34        /// Number of hours
35        hour: u32,
36        /// Number of minutes
37        minute: u32,
38        /// Number of seconds
39        second: u32,
40        /// Number of milliseconds
41        millisecond: u32,
42    },
43    /// consists of week units
44    Weeks(u32),
45}
46
47impl Duration {
48    /// Whether this duration represents a zero duration.
49    pub fn is_zero(&self) -> bool {
50        *self
51            == Duration::YMDHMS {
52                year: 0,
53                month: 0,
54                day: 0,
55                hour: 0,
56                minute: 0,
57                second: 0,
58                millisecond: 0,
59            }
60            || *self == Duration::Weeks(0)
61    }
62}

86impl From<Duration> for ::core::time::Duration {
87    fn from(duration: Duration) -> Self {
88        match duration {
89            Duration::YMDHMS {
90                year,
91                month,
92                day,
93                hour,
94                minute,
95                second,
96                millisecond,
97            } => {
98                let secs = u64::from(year) * 365 * 86_400
99                    + u64::from(month) * 30 * 86_400
100                    + u64::from(day) * 86_400
101                    + u64::from(hour) * 3600
102                    + u64::from(minute) * 60
103                    + u64::from(second);
104                let nanos = millisecond * 1_000_000;
105                Self::new(secs, nanos)
106            }
107            Duration::Weeks(week) => {
108                let secs = u64::from(week) * 7 * 86_400;
109                Self::from_secs(secs)
110            }
111        }
112    }
113}

115/// Parses a duration string.
116///
117/// A string starts with `P` and can have one of the following formats:
118///
119/// * Fully-specified duration: `P1Y2M3DT4H5M6S`
120/// * Duration in weekly intervals: `P1W`
121/// * Fully-specified duration in [`DateTime`](`crate::DateTime`) format: `P<datetime>`
122///
123/// Both fully-specified formats get parsed into the YMDHMS Duration variant.
124/// The weekly interval format gets parsed into the Weeks Duration variant.
125///
126/// The ranges for each of the individual units are not expected to exceed
127/// the next largest unit.
128///
129/// These ranges (inclusive) are as follows:
130///
131/// * Year (any valid u32)
132/// * Month 0 - 12
133/// * Week 0 - 52
134/// * Day 0 - 31
135/// * Hour 0 - 24
136/// * Minute 0 - 60
137/// * Second 0 - 60
138///
139/// ## Examples
140///
141/// ```rust
142/// let duration = iso8601::duration("P1Y2M3DT4H5M6S").unwrap();
143/// let duration = iso8601::duration("P1W").unwrap();
144/// let duration = iso8601::duration("P2015-11-03T21:56").unwrap();
145/// ```
146pub fn duration(string: &str) -> Result<Duration, String> {
147    if let Ok((_left_overs, parsed)) = parsers::parse_duration(string.as_bytes()) {
148        Ok(parsed)
149    } else {
150        Err(format!("Failed to parse duration: {}", string))
151    }
