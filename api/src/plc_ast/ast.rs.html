166impl TypeNature {
167    pub fn derives_from(self, other: TypeNature) -> bool {
168        if other == self {
169            true
170        } else {
171            match self {
172                TypeNature::Any => true,
173                TypeNature::Derived => matches!(other, TypeNature::Any),
174                TypeNature::__VLA => matches!(other, TypeNature::Any),
175                TypeNature::Elementary => matches!(other, TypeNature::Any),
176                TypeNature::Magnitude => matches!(other, TypeNature::Elementary | TypeNature::Any),
177                TypeNature::Num => {
178                    matches!(other, TypeNature::Magnitude | TypeNature::Elementary | TypeNature::Any)
179                }
180                TypeNature::Real => matches!(
181                    other,
182                    TypeNature::Num | TypeNature::Magnitude | TypeNature::Elementary | TypeNature::Any
183                ),
184                TypeNature::Int => matches!(
185                    other,
186                    TypeNature::Num | TypeNature::Magnitude | TypeNature::Elementary | TypeNature::Any
187                ),
188                TypeNature::Signed => matches!(
189                    other,
190                    TypeNature::Int
191                        | TypeNature::Num
192                        | TypeNature::Magnitude
193                        | TypeNature::Elementary
194                        | TypeNature::Any
195                ),
196                TypeNature::Unsigned => matches!(
197                    other,
198                    TypeNature::Int
199                        | TypeNature::Num
200                        | TypeNature::Magnitude
201                        | TypeNature::Elementary
202                        | TypeNature::Any
203                ),
204                TypeNature::Duration => {
205                    matches!(other, TypeNature::Magnitude | TypeNature::Elementary | TypeNature::Any)
206                }
207                TypeNature::Bit => matches!(other, TypeNature::Elementary | TypeNature::Any),
208                TypeNature::Chars => matches!(other, TypeNature::Elementary | TypeNature::Any),
209                TypeNature::String => {
210                    matches!(other, TypeNature::Chars | TypeNature::Elementary | TypeNature::Any)
211                }
212                TypeNature::Char => {
213                    matches!(other, TypeNature::Chars | TypeNature::Elementary | TypeNature::Any)
214                }
215                TypeNature::Date => matches!(other, TypeNature::Elementary | TypeNature::Any),
216            }
217        }
218    }
219
220    pub fn is_numerical(&self) -> bool {
221        self.derives_from(TypeNature::Num)
222    }
223
224    pub fn is_real(&self) -> bool {
225        self.derives_from(TypeNature::Real)
226    }
227
228    pub fn is_bit(&self) -> bool {
229        self.derives_from(TypeNature::Bit)
230    }
231}

637impl DataTypeDeclaration {
638    pub fn get_name(&self) -> Option<&str> {
639        match self {
640            Self::Aggregate { referenced_type, .. }
641            | DataTypeDeclaration::Reference { referenced_type, .. } => Some(referenced_type.as_str()),
642            DataTypeDeclaration::Definition { data_type, .. } => data_type.get_name(),
643        }
644    }
645
646    pub fn get_location(&self) -> SourceLocation {
647        match self {
648            DataTypeDeclaration::Reference { location, .. } => location.clone(),
649            DataTypeDeclaration::Definition { location, .. } => location.clone(),
650            Self::Aggregate { location, .. } => location.clone(),
651        }
652    }
653
654    pub fn get_referenced_type(&self) -> Option<String> {
655        let DataTypeDeclaration::Reference { referenced_type, .. } = self else { return None };
656        Some(referenced_type.to_owned())
657    }
658
659    pub fn get_inner_pointer_ty(&self) -> Option<DataTypeDeclaration> {
660        match self {
661            DataTypeDeclaration::Reference { .. } => Some(self.clone()),
662
663            DataTypeDeclaration::Definition { data_type, .. } => {
664                if let DataType::PointerType { referenced_type, .. } = data_type.as_ref() {
665                    return referenced_type.get_inner_pointer_ty();
666                }
667
668                None
669            }
670            DataTypeDeclaration::Aggregate { .. } => None,
671        }
672    }
673
674    pub fn is_type_safe_pointer(&self) -> bool {
675        match self {
676            DataTypeDeclaration::Definition { data_type, .. } => data_type.is_type_safe_pointer(),
677            _ => false,
678        }
679    }
680
681    pub fn is_aggregate(&self) -> bool {
682        matches!(self, DataTypeDeclaration::Aggregate { .. })
683    }

823fn replace_reference(
824    referenced_type: &mut Box<DataTypeDeclaration>,
825    type_name: String,
826    location: &SourceLocation,
827) -> Option<DataTypeDeclaration> {
828    if let DataTypeDeclaration::Reference { .. } = **referenced_type {
829        return None;
830    }
831    let new_data_type =
832        DataTypeDeclaration::Reference { referenced_type: type_name, location: location.clone() };
833    let old_data_type = std::mem::replace(referenced_type, Box::new(new_data_type));
834    Some(*old_data_type)

1561/// enum_elements should be the statement between then enum's brackets ( )

1585/// flattens expression-lists and MultipliedStatements into a vec of statements.
1586/// It can also handle nested structures like 2(3(4,5))
1587pub fn flatten_expression_list(list: &AstNode) -> Vec<&AstNode> {
1588    match &list.stmt {
1589        AstStatement::ExpressionList(expressions, ..) => {
1590            expressions.iter().by_ref().flat_map(flatten_expression_list).collect()
1591        }
1592        AstStatement::MultipliedStatement(MultipliedStatement { multiplier, element }, ..) => {
1593            std::iter::repeat_n(flatten_expression_list(element), *multiplier as usize).flatten().collect()
1594        }
1595        AstStatement::ParenExpression(expression) => flatten_expression_list(expression),
1596        _ => vec![list],
1597    }
1598}
1599
1600pub fn steal_expression_list(list: &mut AstNode) -> Vec<AstNode> {
1601    match &mut list.stmt {
1602        AstStatement::ExpressionList(expressions, ..) => std::mem::take(expressions),
1603        AstStatement::ParenExpression(expression) => steal_expression_list(expression),
1604        _ => {
1605            let node = std::mem::take(list);
1606            vec![node]
1607        }
1608    }
1609}
