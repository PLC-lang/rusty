19pub fn pre_process(unit: &mut CompilationUnit, mut id_provider: IdProvider) {
20    //process all local variables from POUs
21    for pou in unit.pous.iter_mut() {
22        //Find all generic types in that pou
23        let generic_types = preprocess_generic_structs(pou);
24        unit.user_types.extend(generic_types);
25
26        process_pou_variables(pou, &mut unit.user_types);
27    }
28
29    for interface in unit.interfaces.iter_mut().flat_map(|it| &mut it.methods) {
30        process_pou_variables(interface, &mut unit.user_types);
31    }
32
33    //process all variables from GVLs
34    process_global_variables(unit, &mut id_provider);
35    process_var_config_variables(unit);
36
37    //process all variables in dataTypes
38    let mut new_types = vec![];
39    for dt in unit.user_types.iter_mut() {
40        {
41            match &mut dt.data_type {
42                DataType::StructType { name, variables, .. } => {
43                    let name: &str = name.as_ref().map(|it| it.as_str()).unwrap_or("undefined");
44                    variables
45                        .iter_mut()
46                        .filter(|it| should_generate_implicit_type(it))
47                        .for_each(|var| pre_process_variable_data_type(name, var, &mut new_types));
48                }
49                DataType::ArrayType { name, referenced_type, .. }
50                | DataType::PointerType { name, referenced_type, .. }
51                    if should_generate_implicit(referenced_type) =>
52                {
53                    let name: &str = name.as_ref().map(|it| it.as_str()).unwrap_or("undefined");
54
55                    let type_name = internal_type_name("", name);
56                    let type_ref = DataTypeDeclaration::Reference {
57                        referenced_type: type_name.clone(),
58                        location: SourceLocation::internal(), //return_type.get_location(),
59                    };
60                    let datatype = std::mem::replace(referenced_type, Box::new(type_ref));
61                    if let DataTypeDeclaration::Definition { mut data_type, location, scope } = *datatype {
62                        data_type.set_name(type_name);
63                        add_nested_datatypes(name, &mut data_type, &mut new_types, &location);
64                        let data_type =
65                            UserTypeDeclaration { data_type: *data_type, initializer: None, location, scope };
66                        new_types.push(data_type);
67                    }
68                }
69                DataType::EnumType { elements, .. }
70                    if matches!(elements.stmt, AstStatement::EmptyStatement { .. }) =>
71                {
72                    //avoid empty statements, just use an empty expression list to make it easier to work with
73                    let _ = std::mem::replace(&mut elements.stmt, AstStatement::ExpressionList(vec![]));
74                }
75                DataType::EnumType { elements: original_elements, name: Some(enum_name), .. }
76                    if !matches!(original_elements.stmt, AstStatement::EmptyStatement { .. }) =>
77                {
78                    let mut last_name: Option<String> = None;
79
80                    fn extract_flat_ref_name(statement: &AstNode) -> &str {
81                        statement.get_flat_reference_name().expect("expected assignment")
82                    }
83
84                    let initialized_enum_elements = flatten_expression_list(original_elements)
85                        .iter()
86                        .map(|it| {
87                            try_from!(it, Assignment).map_or_else(
88                                || (extract_flat_ref_name(it), None, it.get_location()),
89                                |Assignment { left, right }| {
90                                    (
91                                        extract_flat_ref_name(left.as_ref()),
92                                        Some(*right.clone()),
93                                        it.get_location(),
94                                    )
95                                },
96                            )
97                        })
98                        .map(|(element_name, initializer, location)| {
99                            let enum_literal = initializer.unwrap_or_else(|| {
100                                build_enum_initializer(&last_name, &location, &mut id_provider, enum_name)
101                            });
102                            last_name = Some(element_name.to_string());
103                            AstFactory::create_assignment(
104                                AstFactory::create_member_reference(
105                                    AstFactory::create_identifier(
106                                        element_name,
107                                        &location,
108                                        id_provider.next_id(),
109                                    ),
110                                    None,
111                                    id_provider.next_id(),
112                                ),
113                                enum_literal,
114                                id_provider.next_id(),
115                            )
116                        })
117                        .collect::<Vec<AstNode>>();
118                    // if the enum is empty, we dont change anything
119                    if !initialized_enum_elements.is_empty() {
120                        // we can safely unwrap because we checked the vec
121                        let start_loc =
122                            initialized_enum_elements.first().expect("non empty vec").get_location();
123                        let end_loc =
124                            initialized_enum_elements.iter().last().expect("non empty vec").get_location();
125                        //swap the expression list with our new Assignments
126                        let expression = AstFactory::create_expression_list(
127                            initialized_enum_elements,
128                            start_loc.span(&end_loc),
129                            id_provider.next_id(),
130                        );
131                        let _ = std::mem::replace(original_elements, expression);
132                    }
133                }
134                _ => {}
135            }
136        }
137    }
138    unit.user_types.append(&mut new_types);
139}

141fn process_pou_variables(pou: &mut Pou, user_types: &mut Vec<UserTypeDeclaration>) {
142    let local_variables = pou
143        .variable_blocks
144        .iter_mut()
145        .flat_map(|it| it.variables.iter_mut())
146        .filter(|it| should_generate_implicit_type(it));
147
148    for var in local_variables {
149        pre_process_variable_data_type(pou.name.as_str(), var, user_types)
150    }
151
152    //Generate implicit type for returns
153    preprocess_return_type(pou, user_types);
154}

156fn process_global_variables(unit: &mut CompilationUnit, id_provider: &mut IdProvider) {
157    let mut mangled_globals = Vec::new();
158
159    for global_var in unit.global_vars.iter_mut().flat_map(|block| block.variables.iter_mut()) {
160        let ref_ty = global_var.data_type_declaration.get_inner_pointer_ty();
161
162        if should_generate_implicit_type(global_var) {
163            pre_process_variable_data_type("global", global_var, &mut unit.user_types)
164        }
165
166        // In any case, we have to inject initializers into aliased hardware access variables
167        if let Some(ref node) = global_var.address {
168            if let AstStatement::HardwareAccess(hardware) = &node.stmt {
169                let name = hardware.get_mangled_variable_name();
170
171                // %I*: DWORD; should not be declared at this stage, it is just skipped
172                if hardware.is_template() {
173                    continue;
174                }
175
176                let mangled_initializer = AstFactory::create_member_reference(
177                    AstFactory::create_identifier(&name, SourceLocation::internal(), id_provider.next_id()),
178                    None,
179                    id_provider.next_id(),
180                );
181
182                global_var.initializer = Some(mangled_initializer);
183
184                let internal_mangled_var = Variable {
185                    name,
186                    data_type_declaration: ref_ty.unwrap_or(global_var.data_type_declaration.clone()),
187                    initializer: None,
188                    address: None,
189                    location: node.location.clone(),
190                };
191                mangled_globals.push(internal_mangled_var);
192            }
193        }
194    }
195
196    update_generated_globals(unit, mangled_globals);
197}

199fn process_var_config_variables(unit: &mut CompilationUnit) {
200    let variables = unit.var_config.iter().filter_map(|ConfigVariable { data_type, address, .. }| {
201        let AstStatement::HardwareAccess(hardware) = &address.stmt else {
202            unreachable!("Must be parsed as hardware access")
203        };
204
205        if hardware.is_template() {
206            return None;
207        }
208
209        // Check if the mangled variable already exists in any of the global variable blocks
210        // XXX: Not a fan of this, we should fix the underlying issue with variable block creation here...
211        let name = hardware.get_mangled_variable_name();
212        if find_mangled_variable(unit, &name) {
213            return None; // Already exists, skip
214        }
215
216        Some(Variable {
217            name,
218            data_type_declaration: data_type.get_inner_pointer_ty().unwrap_or(data_type.clone()),
219            initializer: None,
220            address: None,
221            location: address.get_location(),
222        })
223    });
224
225    update_generated_globals(unit, variables.collect());
226}

228fn update_generated_globals(unit: &mut CompilationUnit, mangled_globals: Vec<Variable>) {
229    let mut block = if let Some(index) = unit
230        .global_vars
231        .iter()
232        .position(|block| block.kind == VariableBlockType::Global && block.location.is_builtin_internal())
233    {
234        unit.global_vars.remove(index)
235    } else {
236        VariableBlock::default().with_block_type(VariableBlockType::Global)
237    };
238    for var in mangled_globals {
239        if !block.variables.contains(&var) {
240            block.variables.push(var);
241        }
242    }
243
244    unit.global_vars.push(block);
245}

247fn find_mangled_variable(unit: &CompilationUnit, name: &str) -> bool {
248    unit.global_vars.iter().flat_map(|block| &block.variables).any(|var| var.name == name)
249}

251fn build_enum_initializer(
252    last_name: &Option<String>,
253    location: &SourceLocation,
254    id_provider: &mut IdProvider,
255    enum_name: &mut str,
256) -> AstNode {
257    if let Some(last_element) = last_name.as_ref() {
258        // generate a `enum#last + 1` statement
259        let enum_ref = AstFactory::create_identifier(last_element, location, id_provider.next_id());
260        let type_element = AstFactory::create_member_reference(
261            AstFactory::create_identifier(enum_name, location, id_provider.next_id()),
262            None,
263            id_provider.next_id(),
264        );
265        AstFactory::create_binary_expression(
266            AstFactory::create_cast_statement(type_element, enum_ref, location, id_provider.next_id()),
267            Operator::Plus,
268            AstNode::new_literal(AstLiteral::new_integer(1), id_provider.next_id(), location.clone()),
269            id_provider.next_id(),
270        )
271    } else {
272        AstNode::new_literal(AstLiteral::new_integer(0), id_provider.next_id(), location.clone())
273    }
274}

276fn preprocess_generic_structs(pou: &mut Pou) -> Vec<UserTypeDeclaration> {
277    let mut generic_types = FxHashMap::default();
278    let mut types = vec![];
279    for binding in &pou.generics {
280        let new_name = format!("__{}__{}", pou.name, binding.name); // TODO: Naming convention (see plc_util/src/convention.rs)
281
282        //Generate a type for the generic
283        let data_type = UserTypeDeclaration {
284            data_type: DataType::GenericType {
285                name: new_name.clone(),
286                generic_symbol: binding.name.clone(),
287                nature: binding.nature,
288            },
289            initializer: None,
290            scope: Some(pou.name.clone()),
291            location: pou.location.clone(),
292        };
293        types.push(data_type);
294        generic_types.insert(binding.name.clone(), new_name);
295    }
296    for var in pou.variable_blocks.iter_mut().flat_map(|it| it.variables.iter_mut()) {
297        replace_generic_type_name(&mut var.data_type_declaration, &generic_types);
298    }
299    if let Some(datatype) = pou.return_type.as_mut() {
300        replace_generic_type_name(datatype, &generic_types);
301    };
302    types
303}

305fn preprocess_return_type(pou: &mut Pou, types: &mut Vec<UserTypeDeclaration>) {
306    if let Some(return_type) = &pou.return_type {
307        if should_generate_implicit(return_type) {
308            let type_name = format!("__{}_return", &pou.name); // TODO: Naming convention (see plc_util/src/convention.rs)
309            let type_ref = DataTypeDeclaration::Reference {
310                referenced_type: type_name.clone(),
311                location: return_type.get_location(),
312            };
313            let datatype = pou.return_type.replace(type_ref);
314            if let Some(DataTypeDeclaration::Definition { mut data_type, location, scope }) = datatype {
315                data_type.set_name(type_name);
316                add_nested_datatypes(pou.name.as_str(), &mut data_type, types, &location);
317                let data_type =
318                    UserTypeDeclaration { data_type: *data_type, initializer: None, location, scope };
319                types.push(data_type);
320            }
321        }
322    }
323}

325fn should_generate_implicit(datatype: &DataTypeDeclaration) -> bool {
326    match datatype {
327        DataTypeDeclaration::Reference { .. } | DataTypeDeclaration::Aggregate { .. } => false,
328        DataTypeDeclaration::Definition { data_type, .. }
329            if matches!(data_type.as_ref(), DataType::VarArgs { .. }) =>
330        {
331            false
332        }
333        DataTypeDeclaration::Definition { .. } => true,
334    }
335}
336
337fn should_generate_implicit_type(variable: &Variable) -> bool {
338    should_generate_implicit(&variable.data_type_declaration)
339}

341fn pre_process_variable_data_type(
342    container_name: &str,
343    variable: &mut Variable,
344    types: &mut Vec<UserTypeDeclaration>,
345) {
346    let new_type_name = internal_type_name(&format!("{container_name}_"), &variable.name);
347    if let DataTypeDeclaration::Definition { mut data_type, location, scope } =
348        variable.replace_data_type_with_reference_to(new_type_name.clone())
349    {
350        // create index entry
351        add_nested_datatypes(new_type_name.as_str(), &mut data_type, types, &location);
352        data_type.set_name(new_type_name);
353        types.push(UserTypeDeclaration { data_type: *data_type, initializer: None, location, scope });
354    }
355    //make sure it gets generated
356}

358fn add_nested_datatypes(
359    container_name: &str,
360    datatype: &mut DataType,
361    types: &mut Vec<UserTypeDeclaration>,
362    location: &SourceLocation,
363) {
364    // TODO: Naming convention (see plc_util/src/convention.rs)
365    let new_type_name = format!("{container_name}_");
366    // FIXME: When processing pointer-to-pointer types (e.g., alias variables pointing to existing pointers),
367    // the inner type is already a DataTypeDeclaration::Reference, so replace_data_type_with_reference_to
368    // returns None and nested type processing is skipped. This results in incomplete names like "__global_alias_var_"
369    // (with trailing underscore but no suffix) when the inner pointer type should be processed.
370    // We need to distinguish between pointer references (which should be processed) and other references
371    // (which should return None) to properly handle nested pointer structures.
372    if let Some(DataTypeDeclaration::Definition { mut data_type, location: inner_location, scope }) =
373        datatype.replace_data_type_with_reference_to(new_type_name.clone(), location)
374    {
375        data_type.set_name(new_type_name.clone());
376        add_nested_datatypes(new_type_name.as_str(), &mut data_type, types, &inner_location);
377        types.push(UserTypeDeclaration {
378            data_type: *data_type,
379            initializer: None,
380            location: location.clone(),
381            scope,
382        });
383    }
384}

386fn replace_generic_type_name(dt: &mut DataTypeDeclaration, generics: &FxHashMap<String, String>) {
387    match dt {
388        DataTypeDeclaration::Definition { data_type, .. } => match data_type.as_mut() {
389            DataType::ArrayType { referenced_type, .. }
390            | DataType::PointerType { referenced_type, .. }
391            | DataType::VarArgs { referenced_type: Some(referenced_type), .. } => {
392                replace_generic_type_name(referenced_type.as_mut(), generics)
393            }
394            _ => {}
395        },
396        DataTypeDeclaration::Reference { referenced_type, .. } => {
397            if let Some(type_name) = generics.get(referenced_type) {
398                referenced_type.clone_from(type_name);
399            }
400        }
401        DataTypeDeclaration::Aggregate { .. } => {} //todo!(),
402    }
