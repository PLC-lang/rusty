19pub fn pre_process(unit: &mut CompilationUnit, mut id_provider: IdProvider) {
20    //process all local variables from POUs
21    for pou in unit.pous.iter_mut() {
22        //Find all generic types in that pou
23        let generic_types = preprocess_generic_structs(pou);
24        unit.user_types.extend(generic_types);
25
26        process_pou_variables(pou, &mut unit.user_types);
27    }
28
29    for interface in unit.interfaces.iter_mut().flat_map(|it| &mut it.methods) {
30        process_pou_variables(interface, &mut unit.user_types);
31    }
32
33    // XXX: Track which hardware-backing globals (__PI_*, __M_*, __G_*) have already been created.
34    // Seeded from existing globals so re-runs of the pipeline (e.g. lowering re-indexes) don't
35    // produce duplicates. This is a workaround for the fact that pre_process is called multiple
36    // times during the pipeline and should be removed when the lowering phase is refactored.
37    let mut known_hw_globals: rustc_hash::FxHashSet<String> = unit
38        .global_vars
39        .iter()
40        .flat_map(|block| &block.variables)
41        .filter(|var| {
42            var.name.starts_with("__PI_") || var.name.starts_with("__M_") || var.name.starts_with("__G_")
43        })
44        .map(|var| var.name.clone())
45        .collect();
46
47    //process all variables from GVLs
48    process_global_variables(unit, &mut id_provider, &mut known_hw_globals);
49    process_var_config_variables(unit, &mut known_hw_globals);
50
51    // Same as above but for struct members with hardware addresses (e.g. `c AT %IX1.2 : BOOL`
52    // inside a STRUCT). Creates the backing __PI_* global and sets the member's initializer.
53    process_struct_hardware_variables(unit, &mut id_provider, &mut known_hw_globals);
54
55    //process all variables in dataTypes
56    let mut new_types = vec![];
57    for dt in unit.user_types.iter_mut() {
58        {
59            match &mut dt.data_type {
60                DataType::StructType { name, variables, .. } => {
61                    let name: &str = name.as_ref().map(|it| it.as_str()).unwrap_or("undefined");
62                    variables.iter_mut().filter(|it| should_generate_implicit_type(it)).for_each(|var| {
63                        pre_process_variable_data_type(name, var, &mut new_types, dt.linkage)
64                    });
65                }
66                DataType::ArrayType { name, referenced_type, .. }
67                | DataType::PointerType { name, referenced_type, .. }
68                    if should_generate_implicit(referenced_type) =>
69                {
70                    let name: &str = name.as_ref().map(|it| it.as_str()).unwrap_or("undefined");
71
72                    let type_name = internal_type_name("", name);
73                    let type_ref = DataTypeDeclaration::Reference {
74                        referenced_type: type_name.clone(),
75                        location: SourceLocation::internal(), //return_type.get_location(),
76                    };
77                    let datatype = std::mem::replace(referenced_type, Box::new(type_ref));
78                    if let DataTypeDeclaration::Definition { mut data_type, location, scope } = *datatype {
79                        data_type.set_name(type_name);
80                        add_nested_datatypes(name, &mut data_type, &mut new_types, &location, dt.linkage);
81                        let data_type = UserTypeDeclaration {
82                            data_type: *data_type,
83                            initializer: None,
84                            location,
85                            scope,
86                            linkage: dt.linkage,
87                        };
88                        new_types.push(data_type);
89                    }
90                }
91                DataType::EnumType { elements, .. }
92                    if matches!(elements.stmt, AstStatement::EmptyStatement { .. }) =>
93                {
94                    //avoid empty statements, just use an empty expression list to make it easier to work with
95                    let _ = std::mem::replace(&mut elements.stmt, AstStatement::ExpressionList(vec![]));
96                }
97                DataType::EnumType { elements: original_elements, name: Some(enum_name), .. }
98                    if !matches!(original_elements.stmt, AstStatement::EmptyStatement { .. }) =>
99                {
100                    let mut last_name: Option<String> = None;
101
102                    fn extract_flat_ref_name(statement: &AstNode) -> &str {
103                        statement.get_flat_reference_name().expect("expected assignment")
104                    }
105
106                    let initialized_enum_elements = flatten_expression_list(original_elements)
107                        .iter()
108                        .map(|it| {
109                            try_from!(it, Assignment).map_or_else(
110                                || (extract_flat_ref_name(it), None, it.get_location()),
111                                |Assignment { left, right }| {
112                                    (
113                                        extract_flat_ref_name(left.as_ref()),
114                                        Some(*right.clone()),
115                                        it.get_location(),
116                                    )
117                                },
118                            )
119                        })
120                        .map(|(element_name, initializer, location)| {
121                            let enum_literal = initializer.unwrap_or_else(|| {
122                                build_enum_initializer(&last_name, &location, &mut id_provider, enum_name)
123                            });
124                            last_name = Some(element_name.to_string());
125                            AstFactory::create_assignment(
126                                AstFactory::create_member_reference(
127                                    AstFactory::create_identifier(
128                                        element_name,
129                                        &location,
130                                        id_provider.next_id(),
131                                    ),
132                                    None,
133                                    id_provider.next_id(),
134                                ),
135                                enum_literal,
136                                id_provider.next_id(),
137                            )
138                        })
139                        .collect::<Vec<AstNode>>();
140                    // if the enum is empty, we dont change anything
141                    if !initialized_enum_elements.is_empty() {
142                        // we can safely unwrap because we checked the vec
143                        let start_loc =
144                            initialized_enum_elements.first().expect("non empty vec").get_location();
145                        let end_loc =
146                            initialized_enum_elements.iter().last().expect("non empty vec").get_location();
147                        //swap the expression list with our new Assignments
148                        let expression = AstFactory::create_expression_list(
149                            initialized_enum_elements,
150                            start_loc.span(&end_loc),
151                            id_provider.next_id(),
152                        );
153                        let _ = std::mem::replace(original_elements, expression);
154                    }
155                }
156                _ => {}
157            }
158        }
159    }
160    unit.user_types.append(&mut new_types);
161}

163fn process_pou_variables(pou: &mut Pou, user_types: &mut Vec<UserTypeDeclaration>) {
164    let local_variables = pou
165        .variable_blocks
166        .iter_mut()
167        .flat_map(|it| it.variables.iter_mut())
168        .filter(|it| should_generate_implicit_type(it));
169
170    for var in local_variables {
171        pre_process_variable_data_type(pou.name.as_str(), var, user_types, pou.linkage)
172    }
173
174    //Generate implicit type for returns
175    preprocess_return_type(pou, user_types);
176}

178fn process_global_variables(
179    unit: &mut CompilationUnit,
180    id_provider: &mut IdProvider,
181    known_hw_globals: &mut rustc_hash::FxHashSet<String>,
182) {
183    let mut mangled_globals = Vec::new();
184
185    for (linkage, global_var) in
186        unit.global_vars.iter_mut().flat_map(|block| block.variables.iter_mut().map(|it| (block.linkage, it)))
187    {
188        let ref_ty = global_var.data_type_declaration.get_inner_pointer_ty();
189
190        if should_generate_implicit_type(global_var) {
191            pre_process_variable_data_type("global", global_var, &mut unit.user_types, linkage)
192        }
193
194        // In any case, we have to inject initializers into aliased hardware access variables
195        if let Some(ref node) = global_var.address {
196            if let AstStatement::HardwareAccess(hardware) = &node.stmt {
197                let name = hardware.get_mangled_variable_name();
198
199                // %I*: DWORD; should not be declared at this stage, it is just skipped
200                if hardware.is_template() {
201                    continue;
202                }
203
204                let mangled_initializer = AstFactory::create_member_reference(
205                    AstFactory::create_identifier(&name, SourceLocation::internal(), id_provider.next_id()),
206                    None,
207                    id_provider.next_id(),
208                );
209
210                global_var.initializer = Some(mangled_initializer);
211
212                if known_hw_globals.insert(name.clone()) {
213                    let internal_mangled_var = Variable {
214                        name,
215                        data_type_declaration: ref_ty.unwrap_or(global_var.data_type_declaration.clone()),
216                        initializer: None,
217                        address: None,
218                        location: node.location.clone(),
219                    };
220                    mangled_globals.push(internal_mangled_var);
221                }
222            }
223        }
224    }
225
226    update_generated_globals(unit, mangled_globals);
227}

229fn process_var_config_variables(
230    unit: &mut CompilationUnit,
231    known_hw_globals: &mut rustc_hash::FxHashSet<String>,
232) {
233    let variables = unit.var_config.iter().filter_map(|ConfigVariable { data_type, address, .. }| {
234        let AstStatement::HardwareAccess(hardware) = &address.stmt else {
235            unreachable!("Must be parsed as hardware access")
236        };
237
238        if hardware.is_template() {
239            return None;
240        }
241
242        let name = hardware.get_mangled_variable_name();
243        if !known_hw_globals.insert(name.clone()) {
244            return None; // Already exists, skip
245        }
246
247        Some(Variable {
248            name,
249            data_type_declaration: data_type.get_inner_pointer_ty().unwrap_or(data_type.clone()),
250            initializer: None,
251            address: None,
252            location: address.get_location(),
253        })
254    });
255
256    update_generated_globals(unit, variables.collect());
257}

259/// Processes struct members declared with IEC hardware addresses (e.g. `c AT %IX1.2 : BOOL`).
260/// For each such member, creates a backing global variable (`__PI_1_2`) and sets the member's
261/// initializer to reference it so the lowering emits a `REF=` assignment in the struct's constructor.
262fn process_struct_hardware_variables(
263    unit: &mut CompilationUnit,
264    id_provider: &mut IdProvider,
265    known_hw_globals: &mut rustc_hash::FxHashSet<String>,
266) {
267    let mut mangled_globals = Vec::new();
268
269    for dt in unit.user_types.iter_mut() {
270        if let DataType::StructType { variables, .. } = &mut dt.data_type {
271            for var in variables.iter_mut() {
272                if let Some(ref node) = var.address {
273                    if let AstStatement::HardwareAccess(hardware) = &node.stmt {
274                        if hardware.is_template() {
275                            continue;
276                        }
277
278                        let name = hardware.get_mangled_variable_name();
279                        let ref_ty = var.data_type_declaration.get_inner_pointer_ty();
280
281                        let mangled_initializer = AstFactory::create_member_reference(
282                            AstFactory::create_identifier(
283                                &name,
284                                SourceLocation::internal(),
285                                id_provider.next_id(),
286                            ),
287                            None,
288                            id_provider.next_id(),
289                        );
290
291                        var.initializer = Some(mangled_initializer);
292
293                        if known_hw_globals.insert(name.clone()) {
294                            let internal_mangled_var = Variable {
295                                name,
296                                data_type_declaration: ref_ty.unwrap_or(var.data_type_declaration.clone()),
297                                initializer: None,
298                                address: None,
299                                location: node.location.clone(),
300                            };
301                            mangled_globals.push(internal_mangled_var);
302                        }
303                    }
304                }
305            }
306        }
307    }
308
309    update_generated_globals(unit, mangled_globals);
310}

312fn update_generated_globals(unit: &mut CompilationUnit, mangled_globals: Vec<Variable>) {
313    if mangled_globals.is_empty() {
314        return;
315    }
316    let mut block = if let Some(index) = unit
317        .global_vars
318        .iter()
319        .position(|block| block.kind == VariableBlockType::Global && block.location.is_builtin_internal())
320    {
321        unit.global_vars.remove(index)
322    } else {
323        VariableBlock::default().with_block_type(VariableBlockType::Global)
324    };
325    for var in mangled_globals {
326        if !block.variables.contains(&var) {
327            block.variables.push(var);
328        }
329    }
330
331    unit.global_vars.push(block);
332}

334fn build_enum_initializer(
335    last_name: &Option<String>,
336    location: &SourceLocation,
337    id_provider: &mut IdProvider,
338    enum_name: &mut str,
339) -> AstNode {
340    if let Some(last_element) = last_name.as_ref() {
341        // generate a `enum#last + 1` statement
342        let enum_ref = AstFactory::create_identifier(last_element, location, id_provider.next_id());
343        let type_element = AstFactory::create_member_reference(
344            AstFactory::create_identifier(enum_name, location, id_provider.next_id()),
345            None,
346            id_provider.next_id(),
347        );
348        AstFactory::create_binary_expression(
349            AstFactory::create_cast_statement(type_element, enum_ref, location, id_provider.next_id()),
350            Operator::Plus,
351            AstNode::new_literal(AstLiteral::new_integer(1), id_provider.next_id(), location.clone()),
352            id_provider.next_id(),
353        )
354    } else {
355        AstNode::new_literal(AstLiteral::new_integer(0), id_provider.next_id(), location.clone())
356    }
357}

359fn preprocess_generic_structs(pou: &mut Pou) -> Vec<UserTypeDeclaration> {
360    let mut generic_types = FxHashMap::default();
361    let mut types = vec![];
362    for binding in &pou.generics {
363        let new_name = format!("__{}__{}", pou.name, binding.name); // TODO: Naming convention (see plc_util/src/convention.rs)
364
365        //Generate a type for the generic
366        let data_type = UserTypeDeclaration {
367            data_type: DataType::GenericType {
368                name: new_name.clone(),
369                generic_symbol: binding.name.clone(),
370                nature: binding.nature,
371            },
372            initializer: None,
373            scope: Some(pou.name.clone()),
374            location: pou.location.clone(),
375            linkage: pou.linkage,
376        };
377        types.push(data_type);
378        generic_types.insert(binding.name.clone(), new_name);
379    }
380    for var in pou.variable_blocks.iter_mut().flat_map(|it| it.variables.iter_mut()) {
381        replace_generic_type_name(&mut var.data_type_declaration, &generic_types);
382    }
383    if let Some(datatype) = pou.return_type.as_mut() {
384        replace_generic_type_name(datatype, &generic_types);
385    };
386    types
387}

389fn preprocess_return_type(pou: &mut Pou, types: &mut Vec<UserTypeDeclaration>) {
390    let linkage = pou.linkage;
391    if let Some(return_type) = &pou.return_type {
392        if should_generate_implicit(return_type) {
393            let type_name = format!("__{}_return", &pou.name); // TODO: Naming convention (see plc_util/src/convention.rs)
394            let type_ref = DataTypeDeclaration::Reference {
395                referenced_type: type_name.clone(),
396                location: return_type.get_location(),
397            };
398            let datatype = pou.return_type.replace(type_ref);
399            if let Some(DataTypeDeclaration::Definition { mut data_type, location, scope }) = datatype {
400                data_type.set_name(type_name);
401                add_nested_datatypes(pou.name.as_str(), &mut data_type, types, &location, linkage);
402                let data_type = UserTypeDeclaration {
403                    data_type: *data_type,
404                    initializer: None,
405                    location,
406                    scope,
407                    linkage,
408                };
409                types.push(data_type);
410            }
411        }
412    }
413}

415fn should_generate_implicit(datatype: &DataTypeDeclaration) -> bool {
416    match datatype {
417        DataTypeDeclaration::Reference { .. } | DataTypeDeclaration::Aggregate { .. } => false,
418        DataTypeDeclaration::Definition { data_type, .. }
419            if matches!(data_type.as_ref(), DataType::VarArgs { .. }) =>
420        {
421            false
422        }
423        DataTypeDeclaration::Definition { .. } => true,
424    }
425}
426
427fn should_generate_implicit_type(variable: &Variable) -> bool {
428    should_generate_implicit(&variable.data_type_declaration)
429}

431fn pre_process_variable_data_type(
432    container_name: &str,
433    variable: &mut Variable,
434    types: &mut Vec<UserTypeDeclaration>,
435    linkage: LinkageType,
436) {
437    let new_type_name = internal_type_name(&format!("{container_name}_"), &variable.name);
438    if let DataTypeDeclaration::Definition { mut data_type, location, scope } =
439        variable.replace_data_type_with_reference_to(new_type_name.clone())
440    {
441        // create index entry
442        add_nested_datatypes(new_type_name.as_str(), &mut data_type, types, &location, linkage);
443        data_type.set_name(new_type_name);
444        types.push(UserTypeDeclaration {
445            data_type: *data_type,
446            initializer: None,
447            location,
448            scope,
449            linkage,
450        });
451    }
452    //make sure it gets generated
453}

455fn add_nested_datatypes(
456    container_name: &str,
457    datatype: &mut DataType,
458    types: &mut Vec<UserTypeDeclaration>,
459    location: &SourceLocation,
460    linkage: LinkageType,
461) {
462    // TODO: Naming convention (see plc_util/src/convention.rs)
463    let new_type_name = format!("{container_name}_");
464    // FIXME: When processing pointer-to-pointer types (e.g., alias variables pointing to existing pointers),
465    // the inner type is already a DataTypeDeclaration::Reference, so replace_data_type_with_reference_to
466    // returns None and nested type processing is skipped. This results in incomplete names like "__global_alias_var_"
467    // (with trailing underscore but no suffix) when the inner pointer type should be processed.
468    // We need to distinguish between pointer references (which should be processed) and other references
469    // (which should return None) to properly handle nested pointer structures.
470    if let Some(DataTypeDeclaration::Definition { mut data_type, location: inner_location, scope }) =
471        datatype.replace_data_type_with_reference_to(new_type_name.clone(), location)
472    {
473        data_type.set_name(new_type_name.clone());
474        add_nested_datatypes(new_type_name.as_str(), &mut data_type, types, &inner_location, linkage);
475        types.push(UserTypeDeclaration {
476            data_type: *data_type,
477            initializer: None,
478            location: location.clone(),
479            scope,
480            linkage,
481        });
482    }
483}

485fn replace_generic_type_name(dt: &mut DataTypeDeclaration, generics: &FxHashMap<String, String>) {
486    match dt {
487        DataTypeDeclaration::Definition { data_type, .. } => match data_type.as_mut() {
488            DataType::ArrayType { referenced_type, .. }
489            | DataType::PointerType { referenced_type, .. }
490            | DataType::VarArgs { referenced_type: Some(referenced_type), .. } => {
491                replace_generic_type_name(referenced_type.as_mut(), generics)
492            }
493            _ => {}
494        },
495        DataTypeDeclaration::Reference { referenced_type, .. } => {
496            if let Some(type_name) = generics.get(referenced_type) {
497                referenced_type.clone_from(type_name);
498            }
499        }
500        DataTypeDeclaration::Aggregate { .. } => {} //todo!(),
501    }
