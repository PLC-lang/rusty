14/// Macro that calls the visitor's `visit` method for every AstNode in the passed iterator `iter`.
15macro_rules! visit_all_nodes {
16    ($visitor:expr, $iter:expr) => {
17        // Note: The `allow` is needed to suppress warnings about `while let Some(...)` warnings
18        // because `visit_all_nodes!` is used for both Option and Non-Option types
19        #[allow(warnings)]
20        {
21            for node in $iter {
22                $visitor.visit(node);
23            }
24        }
25    };
26}
27
28/// Macro that calls the visitor's `visit` method for every AstNode in the passed sequence of nodes.
29macro_rules! visit_nodes {
30    ($visitor:expr, $($node:expr),*) => {
31        $(
32            $visitor.visit($node);
33        )*
34    };
35}

37/// The `Walker` implements the traversal of the AST nodes and Ast-related objects (e.g. CompilationUnit).
38/// The `walk` method is called on the object to visit its children.
39/// If the object passed to a `AstVisitor`'s `visit` method implements the `Walker` trait,
40/// a call to the it's walk function continues the visiting process on its children.
41///
42/// Spliting the traversal logic into a separate trait allows to call the default traversal logic
43/// from the visitor while overriding the visitor's `visit` method for specific nodes.
44///
45/// # Example
46/// ```
47/// use plc_ast::ast::AstNode;
48/// use plc_ast::visitor::Walker;
49/// use plc_ast::visitor::AstVisitor;
50///
51/// struct MyAssignment {
52///   left: AstNode,
53///  right: AstNode,
54/// }
55///
56/// impl Walker for MyAssignment {
57///     fn walk<V>(&self, visitor: &mut V)
58///     where
59///         V: AstVisitor,
60///     {
61///         visitor.visit(&self.right);
62///         visitor.visit(&self.left);
63///     }
64/// }
65/// ```
66///
67pub trait Walker {
68    fn walk<V>(&self, visitor: &mut V)
69    where
70        V: AstVisitor;
71}

73/// The `AstVisitor` trait provides a set of methods for visiting different types of AST nodes.
74/// Implementors can individually override the methods they are interested in. When overriding a method,
75/// make sure to call `walk` on the visited statement to visit its children. DO NOT call walk on
76/// the node itself to avoid a recursion (last parameter). Implementors may also decide to not call
77/// the statement's `walk` method to avoid visiting the children of the statement.
78///
79/// The visitor offers strongly typed `visit_X` functions for every node type. The function's signature
80/// is `fn visit_X(&mut self, stmt: &X, node: &AstNode)`. The `stmt` parameter is the unwrapped, typed
81/// node and the `node` parameter is the `AstNode` wrapping the stmt. The `AstNode` node offers access to location
82/// information and the AstId. Note that some nodes are not wrapped in an `AstNode` node (e.g. `CompilationUnit`)
83/// and therefore only the strongly typed node is passed to the `visit_X` function.
84///
85/// # Example
86/// ```
87/// use plc_ast::{
88///    ast::{Assignment, AstNode},
89///    visitor::{AstVisitor, Walker},
90/// };
91///
92/// struct AssignmentCounter {
93///    count: usize,
94/// }
95///
96/// impl AstVisitor for AssignmentCounter {
97///    fn visit_assignment(&mut self, stmt: &Assignment, _node: &AstNode) {
98///        self.count += 1;
99///        // visit child nodes
100///        stmt.walk(self);
101///    }
102///
103///    fn visit_output_assignment(&mut self, stmt: &Assignment, _node: &AstNode) {
104///        self.count += 1;
105///        // visit child nodes
106///        stmt.walk(self);
107///    }
108/// }
109/// ```

433/// Helper method that walks through a slice of `ConditionalBlock` and applies the visitor's `walk` method to each node.
434fn walk_conditional_blocks<V>(visitor: &mut V, blocks: &[ConditionalBlock])
435where
436    V: AstVisitor,
437{
438    for b in blocks {
439        visit_nodes!(visitor, &b.condition);
440        visit_all_nodes!(visitor, &b.body);
441    }
442}

546impl Walker for AstControlStatement {
547    fn walk<V>(&self, visitor: &mut V)
548    where
549        V: AstVisitor,
550    {
551        match self {
552            AstControlStatement::If(stmt) => {
553                walk_conditional_blocks(visitor, &stmt.blocks);
554                visit_all_nodes!(visitor, &stmt.else_block);
555            }
556            AstControlStatement::WhileLoop(stmt) | AstControlStatement::RepeatLoop(stmt) => {
557                visit_nodes!(visitor, &stmt.condition);
558                visit_all_nodes!(visitor, &stmt.body);
559            }
560            AstControlStatement::ForLoop(stmt) => {
561                visit_nodes!(visitor, &stmt.counter, &stmt.start, &stmt.end);
562                visit_all_nodes!(visitor, &stmt.by_step);
563                visit_all_nodes!(visitor, &stmt.body);
564            }
565            AstControlStatement::Case(stmt) => {
566                visit_nodes!(visitor, &stmt.selector);
567                walk_conditional_blocks(visitor, &stmt.case_blocks);
568                visit_all_nodes!(visitor, &stmt.else_block);
569            }
570        }
571    }
572}

592impl Walker for AstNode {
593    fn walk<V>(&self, visitor: &mut V)
594    where
595        V: AstVisitor,
596    {
597        let node = self;
598        match &self.stmt {
599            AstStatement::EmptyStatement(stmt) => visitor.visit_empty_statement(stmt, node),
600            AstStatement::DefaultValue(stmt) => visitor.visit_default_value(stmt, node),
601            AstStatement::Literal(stmt) => visitor.visit_literal(stmt, node),
602            AstStatement::MultipliedStatement(stmt) => visitor.visit_multiplied_statement(stmt, node),

608            AstStatement::UnaryExpression(stmt) => visitor.visit_unary_expression(stmt, node),
609            AstStatement::ExpressionList(stmt) => visitor.visit_expression_list(stmt, node),
610            AstStatement::ParenExpression(stmt) => visitor.visit_paren_expression(stmt, node),
611            AstStatement::RangeStatement(stmt) => visitor.visit_range_statement(stmt, node),

616            AstStatement::CallStatement(stmt) => visitor.visit_call_statement(stmt, node),
617            AstStatement::ControlStatement(stmt) => visitor.visit_control_statement(stmt, node),
618            AstStatement::CaseCondition(stmt) => visitor.visit_case_condition(stmt, node),
619            AstStatement::ExitStatement(_stmt) => visitor.visit_exit_statement(node),

628    }

712impl Walker for DataType {
713    fn walk<V>(&self, visitor: &mut V)
714    where
715        V: AstVisitor,
716    {
717        match self {
718            DataType::StructType { variables, .. } => {
719                for v in variables.iter() {
720                    visitor.visit_variable(v);
721                }
722            }
723            DataType::EnumType { elements, .. } => {
724                for ele in flatten_expression_list(elements) {
725                    visitor.visit_enum_element(ele);
726                }
727            }
728            DataType::SubRangeType { bounds, .. } => {
729                visit_all_nodes!(visitor, bounds);
730            }
731            DataType::ArrayType { bounds, referenced_type, .. } => {
732                visitor.visit(bounds);
733                visitor.visit_data_type_declaration(referenced_type);
734            }
735            DataType::PointerType { referenced_type, .. } => {
736                visitor.visit_data_type_declaration(referenced_type);
737            }
738            DataType::StringType { size, .. } => {
739                visit_all_nodes!(visitor, size);
740            }
741            DataType::VarArgs { referenced_type, .. } => {
742                if let Some(data_type_declaration) = referenced_type {
743                    visitor.visit_data_type_declaration(data_type_declaration);
744                }
745            }
746            DataType::GenericType { .. } => {
747                //no further visits
748            }
749        }
750    }
751}
