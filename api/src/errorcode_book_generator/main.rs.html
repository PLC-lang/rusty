13/// A mdbook preprocessor to generate error code pages
14pub fn make_app() -> Command {
15    Command::new("error_code_generator")
16        .about("A mdbook preprocessor to generate error code pages")
17        .subcommand(
18            Command::new("supports")
19                .arg(Arg::new("renderer").required(true))
20                .about("Check whether a renderer is present by this preprocessor"),
21        )
22}
23
24fn main() {
25    let matches = make_app().get_matches();
26
27    let preprocessor = ErrorCodeGenerator {};
28
29    if let Some(sub_args) = matches.subcommand_matches("supports") {
30        handle_supports(&preprocessor, sub_args);
31    } else if let Err(e) = handle_preprocessing(&preprocessor) {
32        eprintln!("{}", e);
33        std::process::exit(1);
34    }
35}

37fn handle_preprocessing(preprocessor: &ErrorCodeGenerator) -> Result<(), Error> {
38    let (ctx, book) = CmdPreprocessor::parse_input(io::stdin())?;
39
40    let book_version = Version::parse(&ctx.mdbook_version)?;
41    let version_req = VersionReq::parse(mdbook::MDBOOK_VERSION)?;
42
43    if !version_req.matches(&book_version) {
44        eprintln!(
45            "Warning: The {} plugin was built against version {} of mdbook, \
46             but we're being called from version {}",
47            preprocessor.name(),
48            mdbook::MDBOOK_VERSION,
49            ctx.mdbook_version
50        );
51    }
52
53    let processed_book = preprocessor.run(&ctx, book)?;
54    serde_json::to_writer(io::stdout(), &processed_book)?;
55
56    Ok(())
57}
58
59fn handle_supports(preprocessor: &dyn Preprocessor, sub_args: &clap::ArgMatches) -> ! {
60    let renderer = sub_args.get_one::<String>("renderer").expect("Required argument");
61    let supported = preprocessor.supports_renderer(renderer);
62
63    if supported {
64        std::process::exit(0);
65    } else {
66        eprintln!("{renderer} is not supported");
67        std::process::exit(1);
68    }
69}

71struct ErrorCodeGenerator {}
72
73impl Preprocessor for ErrorCodeGenerator {
74    fn name(&self) -> &str {
75        "errors"
76    }
77
78    fn run(
79        &self,
80        ctx: &mdbook::preprocess::PreprocessorContext,
81        mut book: mdbook::book::Book,
82    ) -> mdbook::errors::Result<mdbook::book::Book> {
83        //Get a config location on where to read the errors from
84        //Get a config location on where to place the error chapters in the book
85        let preprocessor = ctx.config.get_preprocessor(self.name()).expect("Preprocessor configured");
86        let Some((source, target)) = preprocessor
87            .get("err_location")
88            .and_then(Value::as_str)
89            .zip(preprocessor.get("err_target").and_then(Value::as_str))
90        else {
91            return Err(Error::msg(
92                "Invalid configuration, the keys \"err_location\" and \"err_target\" must be set",
93            ));
94        };
95
96        //Read the config source for all md files
97        let source = ctx.root.join(PathBuf::from_str(source)?);
98        let target = PathBuf::from_str(target)?;
99        if !source.exists() {
100            return Err(Error::msg(format!("{}: Invalid Path", source.to_string_lossy())));
101        }
102
103        //Find the section in the book where to add the chapters
104        let mut files: Vec<_> = fs::read_dir(source)?
105            .flatten()
106            .map(|it| it.path())
107            .filter(|it| it.extension().unwrap_or_default() == "md")
108            .collect();
109        files.sort();
110
111        // This goes through the entire book to find the correct chapter to add the error codes
112        // It then creates sub chapters for each `.md` file it finds
113        // The title of the chapter would be the name of the file.
114        // The chapters are sorted alphabetically.
115        book.for_each_mut(|item| {
116            if let BookItem::Chapter(chapter) = item {
117                if chapter.source_path == Some(target.clone()) {
118                    let mut parents = chapter.parent_names.clone();
119                    parents.push(chapter.name.clone());
120                    let mut number = 1;
121                    for file in &files {
122                        //Add each file to the book
123                        let name = file.file_stem().and_then(|it| it.to_str()).unwrap_or_default();
124                        let content = std::fs::read_to_string(file).unwrap();
125                        let mut error_chapter = Chapter::new(
126                            name,
127                            content,
128                            target.join(name).with_extension("md"),
129                            parents.clone(),
130                        );
131                        let mut chapter_number = chapter.number.clone().unwrap_or_default();
132                        chapter_number.push(number);
133                        error_chapter.number = Some(chapter_number);
134                        chapter.sub_items.push(error_chapter.into());
135                        number += 1;
136                    }
137                }
138            }
139        });
140        Ok(book)
141    }
