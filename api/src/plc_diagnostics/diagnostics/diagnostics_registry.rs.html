13macro_rules! add_diagnostic {
14    ($($number:ident, $severity:expr, $desc:expr,)*) => {
15        { let mut m : FxHashMap<&str, DiagnosticEntry> = FxHashMap::default();
16            $(
17                {
18                    let code = stringify!($number);
19                    m.insert(code, DiagnosticEntry{ code, severity: $severity, description: $desc});
20                }
21            )*
22            m
23        }}
24}
25
26pub struct DiagnosticsRegistry(FxHashMap<&'static str, DiagnosticEntry>);
27
28#[derive(Default, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
29pub struct DiagnosticEntry {
30    code: &'static str,
31    severity: Severity,
32    description: &'static str,
33}
34
35impl Default for DiagnosticsRegistry {
36    fn default() -> Self {
37        Self::new(DIAGNOSTICS.clone())
38    }
39}
40
41impl DiagnosticsRegistry {
42    /// Creates an empty registry.
43    fn new(map: FxHashMap<&'static str, DiagnosticEntry>) -> Self {
44        DiagnosticsRegistry(map)
45    }
46
47    pub fn with_configuration(mut self, config: DiagnosticsConfiguration) -> Self {
48        for (severity, codes) in config.0 {
49            for code in &codes {
50                if let Some(entry) = self.0.get_mut(code.as_str()) {
51                    entry.severity = severity
52                }
53            }
54        }
55        self
56    }
57}
58
59impl DiagnosticAssessor for DiagnosticsRegistry {
60    /// Assesses the diagnostic based on the current registered map. If no entry is found, the
61    /// default diagnostic is returned
62    fn assess(&self, d: &Diagnostic) -> Severity {
63        self.0.get(d.get_error_code()).map(|it| it.severity).unwrap_or_else(|| {
64            log::warn!("Unrecognized error code {}. Using default severity", d.get_error_code());
65            Severity::default()
66        })
67    }
68
69    fn explain(&self, error: &str) -> String {
70        let Some(info) = self.0.get(error) else {
71            return format!("Unknown error {error}");
72        };
73        format!(
74            r"Explanation for error {error}:
75{}
76",
77            info.description
78        )
79    }
80
81    fn get_diagnostic_configuration(&self) -> String {
82        let config: DiagnosticsConfiguration = self.into();
83        serde_json::ser::to_string(&config).expect("Cannot fail")
84    }
85}
86
87#[derive(Serialize, Deserialize, PartialEq, Eq, Default)]
88#[serde(transparent)]
89pub struct DiagnosticsConfiguration(FxHashMap<Severity, Vec<String>>);
90
91impl From<&DiagnosticsRegistry> for DiagnosticsConfiguration {
92    fn from(registry: &DiagnosticsRegistry) -> Self {
93        let mut res = DiagnosticsConfiguration::default();
94        for val in registry.0.values() {
95            let entry = res.0.entry(val.severity).or_default();
96            entry.push(val.code.into());
97        }
98        res
99    }
100}

229#[cfg(test)]
230mod tests {
231    use crate::{
232        diagnostician::DiagnosticAssessor,
233        diagnostics::{diagnostics_registry::DiagnosticsConfiguration, Diagnostic, Severity},
234    };
235
236    use super::DiagnosticsRegistry;
237
238    #[test]
239    fn deserialize_empty_json() {
240        let error_config = "{}";
241
242        let DiagnosticsConfiguration(configuration) = serde_json::de::from_str(error_config).unwrap();
243        assert!(configuration.is_empty());
244    }
245
246    #[test]
247    fn deserialize_json() {
248        let error_config = r#"{
249            "error": ["E001", "E002"],
250            "warning": ["E003", "E004"],
251            "info": ["E005"],
252            "ignore": ["E010"]
253        }"#;
254
255        let DiagnosticsConfiguration(configuration) = serde_json::de::from_str(error_config).unwrap();
256        assert_eq!(configuration.len(), 4);
257        assert_eq!(configuration.get(&Severity::Error).unwrap(), &["E001", "E002"]);
258        assert_eq!(configuration.get(&Severity::Warning).unwrap(), &["E003", "E004"]);
259        assert_eq!(configuration.get(&Severity::Info).unwrap(), &["E005"]);
260        assert_eq!(configuration.get(&Severity::Ignore).unwrap(), &["E010"]);
261    }
262
263    #[test]
264    fn overridden_errors_are_assessed_correctly() {
265        let error_config = r#"{
266            "error": ["E090"],
267            "warning": ["E001"],
268            "info": ["E002"],
269            "ignore": ["E003"]
270        }"#;
271
272        let configuration = serde_json::de::from_str(error_config).unwrap();
273        let diagnostics_registry = DiagnosticsRegistry::default().with_configuration(configuration);
274
275        let e090 = Diagnostic::new("Warning->Error").with_error_code("E090");
276        let e001 = Diagnostic::new("Error->Warning").with_error_code("E001");
277        let e002 = Diagnostic::new("Error->Info").with_error_code("E002");
278        let e003 = Diagnostic::new("Error->Ignore").with_error_code("E003");
279
280        assert_eq!(diagnostics_registry.assess(&e090), Severity::Error);
281        assert_eq!(diagnostics_registry.assess(&e001), Severity::Warning);
282        assert_eq!(diagnostics_registry.assess(&e002), Severity::Info);
283        assert_eq!(diagnostics_registry.assess(&e003), Severity::Ignore);
284    }
