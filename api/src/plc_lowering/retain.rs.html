1//! # Retain Variable Lowering
2//!
3//! This module handles the lowering of `RETAIN` variables as defined in IEC 61131-3.
4//! Retain variables persist their values across power cycles by being placed in a
5//! dedicated `.retain` linker section, which a PLC runtime maps to non-volatile memory.
6//!
7//! The lowering pass handles three cases:
8//!
9//! ## 1. PROGRAM retain variables
10//! Variables declared as `VAR RETAIN` inside a `PROGRAM` are extracted into global
11//! retain variables and replaced with auto-dereferenced pointers. For example:
12//!
13//! ```st
14//! PROGRAM Main
15//!     VAR RETAIN
16//!         counter : INT := 0;
17//!     END_VAR
18//! END_PROGRAM
19//! ```
20//!
21//! Is transformed into:
22//! - A global variable `__Main_counter__retain : INT := 0` in a retain block
23//! - The original `counter` becomes an auto-deref pointer to `__Main_counter__retain`
24//!
25//! This indirection ensures the PROGRAM's instance struct contains pointers to the
26//! retain globals, while the actual values live in the `.retain` section.
27//!
28//! ## 2. Global variables with transitive retain
29//! If a global variable's type transitively contains retain members (e.g. an FB with
30//! `VAR RETAIN`), the variable is moved to a retain block even if it was not explicitly
31//! declared in a `VAR_GLOBAL RETAIN` block.
32//!
33//! ## 3. FUNCTION_BLOCK retain variables
34//! Retain variables inside FBs stay in-place within the FB's struct definition. The
35//! entire FB instance receives the `.retain` section treatment when it is instantiated
36//! at the global or program level, handled by the transitive `should_retain()` check
37//! during code generation.
38
39use plc_ast::{
40    ast::{
41        AccessModifier, AstFactory, AutoDerefType, CompilationUnit, DataType, DataTypeDeclaration, Variable,
42    },
43    mut_visitor::{AstVisitorMut, WalkerMut},
44    provider::IdProvider,
45};
46use plc_source::source_location::SourceLocation;

48pub struct RetainParticipant {
49    pub ids: IdProvider,
50}
51
52impl RetainParticipant {
53    pub fn new(ids: IdProvider) -> Self {
54        Self { ids }
55    }
56
57    pub fn lower_retain(&mut self, units: &mut [CompilationUnit], index: plc::index::Index) {
58        let mut lowerer = RetainLowerer { ids: self.ids.clone(), index, context: Context::default() };
59        for unit in units {
60            lowerer.visit_compilation_unit(unit);
61        }
62    }
63}
64
65struct RetainLowerer {
66    ids: IdProvider,
67    index: plc::index::Index,
68    context: Context,
69}
70
71#[derive(Debug, Default)]
72struct Context {
73    container_name: Option<String>,
74    in_program: bool,
75    retain_variables: Vec<Variable>,
76}
77
78impl AstVisitorMut for RetainLowerer {
79    fn visit_compilation_unit(&mut self, unit: &mut CompilationUnit) {
80        unit.walk(self);
81        // After visiting the compilation unit, add all retain variables to the global vars
82        if !self.context.retain_variables.is_empty() {
83            // Find an existing retain global variable block or create a new one if it doesn't exist
84            unit.global_vars
85                .iter_mut()
86                .find(|block| block.retain)
87                .map(|block| {
88                    block.variables.append(&mut self.context.retain_variables);
89                })
90                .unwrap_or_else(|| {
91                    let retain_block = plc_ast::ast::VariableBlock {
92                        variables: self.context.retain_variables.drain(..).collect(),
93                        kind: plc_ast::ast::VariableBlockType::Global,
94                        constant: false,
95                        retain: true,
96                        linkage: plc_ast::ast::LinkageType::Internal,
97                        location: SourceLocation::internal(),
98                        access: AccessModifier::Public,
99                    };
100                    unit.global_vars.push(retain_block);
101                });
102        }
103    }
104    fn visit_pou(&mut self, pou: &mut plc_ast::ast::Pou) {
105        self.context.in_program = matches!(pou.kind, plc_ast::ast::PouType::Program);
106        self.context.container_name = Some(pou.name.clone());
107        pou.walk(self);
108        self.context.in_program = false;
109        self.context.container_name = None;
110    }
111
112    fn visit_variable_block(&mut self, block: &mut plc_ast::ast::VariableBlock) {
113        let variables = std::mem::take(&mut block.variables);
114        //If the block is retain but we are in a program, mark the block as non-retain
115        if block.retain && self.context.in_program {
116            block.retain = false;
117        }
118        for variable in variables {
119            let Some(variable_index) =
120                self.index.find_variable(self.context.container_name.as_deref(), &[variable.get_name()])
121            else {
122                block.variables.push(variable);
123                continue;
124            };
125
126            if !variable_index.should_retain(&self.index) {
127                block.variables.push(variable);
128                continue;
129            }
130
131            if self.context.in_program {
132                let (old_variable, new_var) = self.replace_with_retain_variable(variable);
133                self.context.retain_variables.push(new_var);
134                block.variables.push(old_variable);
135            } else if matches!(block.kind, plc_ast::ast::VariableBlockType::Global) && !block.retain {
136                // Global variable in a non-retain block whose type transitively contains retain
137                // members (e.g. an FB with VAR RETAIN). Move it to a retain block.
138                self.context.retain_variables.push(variable);
139            } else {
140                // FB retain variables stay in-place within the FB's struct. The entire FB instance
141                // gets placed in the .retain section when instantiated at the global/program level,
142                // handled by the transitive should_retain() check during codegen.
143                block.variables.push(variable);
144            }
145        }
146    }
147}
148
149impl RetainLowerer {
150    /// Replaces a retain variable in a program with a global retain variable and replaces the original variable with an auto reference to the global variable
151    fn replace_with_retain_variable(&mut self, mut variable: Variable) -> (Variable, Variable) {
152        let new_name = format!(
153            "__{}_{}__retain",
154            self.context.container_name.as_deref().unwrap_or_default(),
155            variable.get_name()
156        );
157        // Create a global variable called __<pou_name>_<var_name> and move the initializer and datatype to the global variable
158        let new_var = Variable {
159            name: new_name,
160            data_type_declaration: variable.data_type_declaration.clone(),
161            initializer: variable.initializer.take(),
162            location: variable.location.clone(),
163            address: None,
164        };
165        variable.data_type_declaration = DataTypeDeclaration::Definition {
166            data_type: Box::new(DataType::PointerType {
167                name: None,
168                referenced_type: Box::new(variable.data_type_declaration.clone()),
169                auto_deref: Some(AutoDerefType::Alias),
170                type_safe: true,
171                is_function: false,
172            }),
173            location: variable.data_type_declaration.get_location(),
174            scope: self.context.container_name.clone(),
175        };
176        variable.initializer = Some(AstFactory::create_member_reference(
177            AstFactory::create_identifier(&new_var.name, variable.location.clone(), self.ids.next_id()),
178            None,
179            self.ids.next_id(),
180        ));
181        (variable, new_var)
182    }
183}

185#[cfg(test)]
186mod tests {
187    use plc_ast::ast::{CompilationUnit, DataTypeDeclaration, Variable, VariableBlock};
188    use plc_driver::parse_and_annotate;
189    use plc_source::SourceCode;

191    /// Finds the first global variable block with `retain: true` in a compilation unit.
192    fn find_retain_block(unit: &CompilationUnit) -> &VariableBlock {
193        unit.global_vars.iter().find(|b| b.retain).expect("expected a global retain block")
194    }
195
196    /// Finds a variable by name in a given list of variable blocks.
197    fn find_variable_in_blocks<'a>(blocks: &'a [VariableBlock], name: &str) -> &'a Variable {
198        blocks
199            .iter()
200            .flat_map(|b| &b.variables)
201            .find(|v| v.get_name() == name)
202            .unwrap_or_else(|| panic!("expected variable '{name}' in blocks"))
203    }
204
205    /// Finds a POU by name in a compilation unit and returns a variable from it.
206    fn find_pou_variable<'a>(unit: &'a CompilationUnit, pou_name: &str, var_name: &str) -> &'a Variable {
207        let pou = unit
208            .pous
209            .iter()
210            .find(|p| p.name == pou_name)
211            .unwrap_or_else(|| panic!("POU '{pou_name}' not found"));
212        find_variable_in_blocks(&pou.variable_blocks, var_name)
213    }
214
215    /// Returns the referenced type name from a DataTypeDeclaration, if it is a Reference variant.
216    fn referenced_type_name(decl: &DataTypeDeclaration) -> &str {
217        match decl {
218            DataTypeDeclaration::Reference { referenced_type, .. } => referenced_type,
219            other => panic!("expected DataTypeDeclaration::Reference, got {other:?}"),
220        }
221    }

223    #[test]
224    fn test_retain_lowering_on_program() {
225        let source: SourceCode = r#"
226        PROGRAM Test
227        VAR RETAIN
228            x: INT := 5;
229        END_VAR
230        END_PROGRAM
231        "#
232        .into();
233
234        let (_, project) =
235            parse_and_annotate("test", vec![source]).expect("Failed to parse compilation unit");
236        let unit = project.units[0].get_unit();
237
238        // A global retain block should exist with the extracted retain variable
239        let retain_block = find_retain_block(unit);
240        assert_eq!(retain_block.variables.len(), 1);
241        let retain_var = &retain_block.variables[0];
242        assert_eq!(retain_var.get_name(), "__Test_x__retain");
243        assert_eq!(referenced_type_name(&retain_var.data_type_declaration), "INT");
244        assert!(retain_var.initializer.is_some(), "retain variable should carry the original initializer");
245
246        // The program's variable should be replaced with an auto-deref pointer
247        let pou_var = find_pou_variable(unit, "Test", "x");
248        // The type should now reference the generated pointer type, not INT directly
249        assert_eq!(referenced_type_name(&pou_var.data_type_declaration), "__Test_x");
250        assert!(pou_var.initializer.is_some(), "program variable should have a reference initializer");
251
252        // The program's variable block should no longer be marked retain
253        let test_pou = unit.pous.iter().find(|p| p.name == "Test").unwrap();
254        assert!(
255            test_pou.variable_blocks.iter().all(|b| !b.retain),
256            "program variable blocks should not be retain after lowering"
257        );
258    }

260    #[test]
261    fn test_retain_lowering_on_program_nested() {
262        let source: SourceCode = r#"
263        FUNCTION_BLOCK FB
264        VAR RETAIN
265            a: INT := 5;
266        END_VAR
267        END_FUNCTION_BLOCK
268        PROGRAM Test
269        VAR
270            x: FB;
271        END_VAR
272        END_PROGRAM
273        "#
274        .into();
275
276        let (_, project) =
277            parse_and_annotate("test", vec![source]).expect("Failed to parse compilation unit");
278        let unit = project.units[0].get_unit();
279
280        // The FB instance should be extracted to a global retain variable
281        let retain_block = find_retain_block(unit);
282        assert_eq!(retain_block.variables.len(), 1);
283        let retain_var = &retain_block.variables[0];
284        assert_eq!(retain_var.get_name(), "__Test_x__retain");
285        assert_eq!(referenced_type_name(&retain_var.data_type_declaration), "FB");
286
287        // The FB's own retain block should remain intact (retain stays in-place for FBs)
288        let fb_pou = unit.pous.iter().find(|p| p.name == "FB").unwrap();
289        let fb_retain_block =
290            fb_pou.variable_blocks.iter().find(|b| b.retain).expect("FB should still have a retain block");
291        let fb_retain_blocks = [fb_retain_block.clone()];
292        let fb_var = find_variable_in_blocks(&fb_retain_blocks, "a");
293        assert_eq!(referenced_type_name(&fb_var.data_type_declaration), "INT");
294
295        // The program's variable should be replaced with an auto-deref pointer to the retain global
296        let pou_var = find_pou_variable(unit, "Test", "x");
297        assert_eq!(referenced_type_name(&pou_var.data_type_declaration), "__Test_x");
298        assert!(pou_var.initializer.is_some(), "program variable should have a reference initializer");
299    }

301    #[test]
302    fn test_retain_in_global_nested_should_move_to_retain_block() {
303        let source: SourceCode = r#"
304        FUNCTION_BLOCK FB
305        VAR RETAIN
306            a: INT := 5;
307        END_VAR
308        END_FUNCTION_BLOCK
309        VAR_GLOBAL RETAIN
310            explicit_retain: FB;
311            y : INT;
312        END_VAR
313        VAR_GLOBAL
314            implicit_retain: FB;
315            x : INT;
316        END_VAR
317        "#
318        .into();
319
320        let (_, project) =
321            parse_and_annotate("test", vec![source]).expect("Failed to parse compilation unit");
322        let unit = project.units[0].get_unit();
323
324        // The retain block should contain the explicitly retained variables AND the
325        // implicitly retained one (implicit_retain is FB which transitively contains retain members)
326        let retain_block = find_retain_block(unit);
327        let retain_var_names: Vec<&str> = retain_block.variables.iter().map(|v| v.get_name()).collect();
328        assert!(retain_var_names.contains(&"explicit_retain"), "explicit_retain should be in retain block");
329        assert!(
330            retain_var_names.contains(&"y"),
331            "y should stay in retain block (declared in VAR_GLOBAL RETAIN)"
332        );
333        assert!(
334            retain_var_names.contains(&"implicit_retain"),
335            "implicit_retain should be moved to retain block (transitive retain)"
336        );
337
338        // The non-retain global block should only contain x (a plain INT with no retain)
339        let non_retain_globals: Vec<&str> = unit
340            .global_vars
341            .iter()
342            .filter(|b| !b.retain)
343            .flat_map(|b| &b.variables)
344            .filter(|v| !v.get_name().starts_with("__")) // exclude compiler-generated variables
345            .map(|v| v.get_name())
346            .collect();
347        assert_eq!(non_retain_globals, vec!["x"], "only x should remain in the non-retain global block");
348    }
