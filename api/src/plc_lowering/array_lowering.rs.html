1//! # Array Initializer Lowering
2//!
3//! This pass rewrites array initializer assignments whose elements cannot be
4//! evaluated at compile time into sequences of indexed assignments and/or FOR
5//! loops that the codegen backend can handle.
6//!
7//! ## Why this exists
8//!
9//! The initializer pass (`initializer.rs`) emits constructor bodies that assign
10//! array literals to their targets: `self.arr := [init];`.  When every element
11//! of the literal is a compile-time constant (integers, reals, bools, …) codegen
12//! materializes the whole array as an anonymous global constant and copies it
13//! with `memcpy` — no lowering needed.
14//!
15//! However, some element expressions are **not** compile-time constants:
16//!
17//! - **Function calls** such as `ADR(x)` produce addresses that depend on
18//!   runtime memory layout.  These can never appear in a global constant.
19//!
20//! - **Struct literal initializers** like `(a := 5, b := 10)` are semantically
21//!   constant but codegen's `generate_literal_array_value` currently cannot
22//!   evaluate them inside an array context (it stack-overflows in the
23//!   context-free expression generator).  This is a codegen limitation, not a
24//!   fundamental one — once fixed, struct literals can be removed from the
25//!   trigger list.
26//!
27//! For these cases the assignment must be decomposed into individual element
28//! stores that the expression generator can handle one at a time.
29//!
30//! ## When this pass triggers
31//!
32//! The pass walks every implementation body looking for assignments whose RHS
33//! is a `LiteralArray`.  It checks the array's element tree for non-constant
34//! expressions via [`contains_non_constant_expression`]:
35//!
36//! | Element kind             | Constant? | Triggers lowering?                   |
37//! |--------------------------|-----------|--------------------------------------|
38//! | Literal (int, real, …)   | yes       | no                                   |
39//! | String literal           | yes       | no                                   |
40//! | Variable reference       | **no**    | **yes** — runtime value              |
41//! | Function call (`ADR(x)`) | **no**    | **yes** — runtime value              |
42//! | Struct literal `(a:=1)`  | **no***   | **yes** — codegen limitation         |
43//!
44//! *Struct literals are semantically constant but are treated as non-constant
45//! because codegen's `generate_literal_array_value` cannot currently evaluate
46//! them inside an array context.
47//!
48//! If no non-constant expression is found, the literal is left as-is for
49//! codegen's memcpy path.
50//!
51//! ## Lowering rules
52//!
53//! Given `arr : ARRAY[start..end] OF T := [init];` where `init` contains
54//! non-constant expressions:
55//!
56//! | Pattern                      | Lowered form                                            |
57//! |------------------------------|---------------------------------------------------------|
58//! | `[N(val)]`  (N ≥ threshold)  | `FOR __idx := start TO start+N-1 DO arr[__idx] := val`  |
59//! | `[N(val)]`  (N < threshold)  | `arr[start] := val; arr[start+1] := val; …`             |
60//! | `[v1, v2, v3]`               | `arr[start] := v1; arr[start+1] := v2; …`               |
61//! | `[3(v1), v2, 2(v3)]`         | mixed — each segment lowered independently              |
62//!
63//! Multi-dimensional arrays are also handled:
64//! - Individual assignments use computed multi-dimensional indices via
65//!   [`ArrayInfo::flat_to_indices`] (e.g. flat index 5 on `ARRAY[0..2, 0..2]`
66//!   becomes `arr[1, 2]`).
67//! - A single `MultipliedStatement` filling the entire array uses nested FOR
68//!   loops (one per dimension).
69//! - Partial multiplied segments on multi-dim arrays are unrolled to individual
70//!   assignments with computed indices.
71//!
72//! ## Side effects
73//!
74//! After lowering, the pass:
75//! 1. **Strips the original initializer** from the variable / user-type
76//!    declaration so the data-type generator does not attempt to create a
77//!    `__init` global for an expression it cannot evaluate.
78//! 2. **Adds `VAR_TEMP` counter variables** (e.g. `__literal_idx`) to POUs
79//!    that contain generated FOR loops.

102// ── Array dimension info ────────────────────────────────────────────────────
103
104/// Information about a single dimension of an array.
105pub(crate) struct DimInfo {
106    pub(crate) start: i64,
107    pub(crate) size: i64,
108}
109
110/// Collected dimension information for an array type.
111pub(crate) struct ArrayInfo {
112    pub(crate) dims: Vec<DimInfo>,
113}
114
115impl ArrayInfo {
116    fn is_single_dim(&self) -> bool {
117        self.dims.len() == 1
118    }
119
120    /// Total number of elements across all dimensions.
121    fn total_elements(&self) -> i64 {
122        self.dims.iter().map(|d| d.size).product()
123    }
124
125    /// Converts a 0-based flat index into actual multi-dimensional indices
126    /// (with per-dimension start offsets applied).  Row-major order: the
127    /// last dimension varies fastest.
128    pub(crate) fn flat_to_indices(&self, flat: i64) -> Vec<i64> {
129        let mut indices = vec![0i64; self.dims.len()];
130        let mut remaining = flat;
131        for i in (0..self.dims.len()).rev() {
132            indices[i] = (remaining % self.dims[i].size) + self.dims[i].start;
133            remaining /= self.dims[i].size;
134        }
135        indices
136    }
137
138    /// Returns counter variable names for multi-dim nested FOR loops.
139    fn nested_counter_names(&self) -> Vec<String> {
140        (0..self.dims.len()).map(|i| format!("__literal_idx_{i}")).collect()
141    }
142}

154/// Walks every implementation in the compilation unit and rewrites assignments whose
155/// right-hand side is an array literal (`LiteralArray`) into indexed assignments
156/// and/or FOR loops.
157pub fn lower_literal_arrays(unit: &mut CompilationUnit, index: &Index, id_provider: &mut IdProvider) {
158    // Track which POUs need which counter variables for FOR loops.
159    let mut pou_counters: HashMap<String, BTreeSet<String>> = HashMap::new();
160
161    // First pass: rewrite `(CONST)(value)` call statements inside array literal
162    // initializers into proper `MultipliedStatement` nodes.  The parser produces a
163    // `CallStatement { operator: ParenExpression(ref), parameters: value }` for the
164    // IEC 61131-3 syntax `[(NB)(value)]` because it cannot distinguish this from a
165    // function call at parse time.  With the index available we can resolve the
166    // constant and emit the correct AST.
167    rewrite_const_multiplied_initializers(unit, index);
168
169    for implementation in &mut unit.implementations {
170        let mut new_statements = Vec::new();
171        let mut counters: BTreeSet<String> = BTreeSet::new();
172
173        for stmt in std::mem::take(&mut implementation.statements) {
174            if let Some(lowered) =
175                try_lower_array_assignment(&stmt, index, &implementation.type_name, id_provider)
176            {
177                counters.extend(lowered.counter_names);
178                new_statements.extend(lowered.statements);
179            } else {
180                new_statements.push(stmt);
181            }
182        }
183
184        if !counters.is_empty() {
185            pou_counters.insert(implementation.name.clone(), counters);
186        }
187
188        implementation.statements = new_statements;
189    }
190
191    // Add VAR_TEMP counter variables to POUs that need them for FOR loops.
192    if !pou_counters.is_empty() {
193        add_counter_variables(unit, &pou_counters);
194    }
195}

197// ── Constant-multiplier rewriting ───────────────────────────────────────────
198
199/// Rewrites `CallStatement` nodes inside `LiteralArray` initializers that represent
200/// the IEC 61131-3 constant-multiplied syntax `[(CONST)(value)]`.
201///
202/// The parser cannot distinguish `(NB_BOOL)(0.0033)` from a call on a parenthesized
203/// expression, so it produces a `CallStatement { operator: ParenExpression(NB_BOOL),
204/// parameters: 0.0033 }`.  With the index available we can resolve the constant to
205/// its integer value and rewrite the node into a proper `MultipliedStatement`.
206fn rewrite_const_multiplied_initializers(unit: &mut CompilationUnit, index: &Index) {
207    // Rewrite in POU variable block initializers.
208    for pou in &mut unit.pous {
209        let pou_name = pou.name.clone();
210        for block in &mut pou.variable_blocks {
211            for var in &mut block.variables {
212                if let Some(init) = &mut var.initializer {
213                    rewrite_array_literal_elements(init, index, Some(&pou_name));
214                }
215            }
216        }
217    }
218    // Rewrite in global variable block initializers (no POU context).
219    for block in &mut unit.global_vars {
220        for var in &mut block.variables {
221            if let Some(init) = &mut var.initializer {
222                rewrite_array_literal_elements(init, index, None);
223            }
224        }
225    }
226    // Rewrite in user-type initializers (no POU context).
227    for udt in &mut unit.user_types {
228        if let Some(init) = &mut udt.initializer {
229            rewrite_array_literal_elements(init, index, None);
230        }
231    }
232    // Rewrite in implementation body statements (assignments generated by the initializer pass).
233    for implementation in &mut unit.implementations {
234        let type_name = implementation.type_name.clone();
235        for stmt in &mut implementation.statements {
236            if let AstStatement::Assignment(data) = stmt.get_stmt_mut() {
237                rewrite_array_literal_elements(&mut data.right, index, Some(&type_name));
238            }
239        }
240    }
241}

243/// If `node` is a `LiteralArray`, rewrites any constant-multiplied call statements
244/// in its element tree into `MultipliedStatement` nodes.
245fn rewrite_array_literal_elements(node: &mut AstNode, index: &Index, pou_name: Option<&str>) {
246    if let AstStatement::Literal(AstLiteral::Array(Array { elements: Some(elements) })) = node.get_stmt_mut()
247    {
248        rewrite_element_tree(elements, index, pou_name);
249    }
250}
251
252/// Recursively walks the element tree of an array literal and rewrites
253/// constant-multiplied call statements.
254fn rewrite_element_tree(node: &mut AstNode, index: &Index, pou_name: Option<&str>) {
255    match node.get_stmt_mut() {
256        AstStatement::ExpressionList(exprs) => {
257            for expr in exprs.iter_mut() {
258                rewrite_element_tree(expr, index, pou_name);
259            }
260        }
261        AstStatement::CallStatement(_) => {
262            if let Some(multiplied) = try_rewrite_call_as_multiplied(node, index, pou_name) {
263                *node = multiplied;
264            }
265        }
266        _ => {}
267    }
268}

270/// Attempts to rewrite a `CallStatement` into a `MultipliedStatement`.
271/// Matches the pattern: `CallStatement { operator: ParenExpression(ref), parameters: value }`
272/// where `ref` resolves to a constant integer variable in the index.
273///
274/// NOTE: Non-constant variables used as multipliers (e.g. `[(n)(val)]` where `n`
275/// is a `VAR` rather than `VAR CONSTANT`) cannot be supported without changing
276/// `MultipliedStatement.multiplier` from `u32` to `Box<AstNode>`.  For now only
277/// constant integer references are rewritten; non-constant references are left as
278/// `CallStatement` nodes and will produce a codegen error.
279fn try_rewrite_call_as_multiplied(node: &AstNode, index: &Index, pou_name: Option<&str>) -> Option<AstNode> {
280    let AstStatement::CallStatement(CallStatement { operator, parameters }) = node.get_stmt() else {
281        return None;
282    };
283    let AstStatement::ParenExpression(inner) = operator.get_stmt() else {
284        return None;
285    };
286    let ref_name = inner.get_flat_reference_name()?;
287    let parameters = parameters.as_ref()?;
288
289    // Look up the variable: first as a POU-local member (if we have a POU context),
290    // then as a global variable.
291    let variable = pou_name
292        .and_then(|pou| index.find_member(pou, ref_name))
293        .or_else(|| index.find_global_variable(ref_name))?;
294
295    if !variable.is_constant() {
296        return None;
297    }
298
299    // Resolve the initial value to a literal integer.
300    let initial_value = index.get_initial_value(&variable.initial_value)?;
301    let AstStatement::Literal(AstLiteral::Integer(value)) = initial_value.get_stmt() else {
302        return None;
303    };
304
305    let multiplier = u32::try_from(*value).ok()?;
306    let location = operator.get_location().span(&parameters.get_location());
307
308    Some(AstFactory::create_multiplied_statement(multiplier, *parameters.clone(), location, node.get_id()))
309}

313/// Adds `VAR_TEMP` counter variables to POUs that have generated FOR loops.
314fn add_counter_variables(unit: &mut CompilationUnit, pou_counters: &HashMap<String, BTreeSet<String>>) {
315    for pou in &mut unit.pous {
316        if let Some(counter_names) = pou_counters.get(&pou.name) {
317            let variables: Vec<Variable> = counter_names
318                .iter()
319                .map(|name| Variable {
320                    name: name.clone(),
321                    data_type_declaration: DataTypeDeclaration::Reference {
322                        referenced_type: "DINT".to_string(),
323                        location: SourceLocation::internal(),
324                    },
325                    initializer: None,
326                    address: None,
327                    location: SourceLocation::internal(),
328                })
329                .collect();
330
331            pou.variable_blocks.push(VariableBlock {
332                variables,
333                kind: VariableBlockType::Temp,
334                ..Default::default()
335            });
336        }
337    }
338}

342/// Checks whether `stmt` is an assignment whose RHS is a `LiteralArray`. If so,
343/// returns the resolved array type name and the lowered statements.
344///
345/// Only arrays whose elements contain non-constant expressions (e.g. function calls)
346/// are lowered. Constant array literals are handled efficiently at codegen level via
347/// memcpy from a materialized global constant.
348fn try_lower_array_assignment(
349    stmt: &AstNode,
350    index: &Index,
351    pou_type_name: &str,
352    id_provider: &mut IdProvider,
353) -> Option<LoweredResult> {
354    let AstStatement::Assignment(data) = stmt.get_stmt() else {
355        return None;
356    };
357
358    let AstStatement::Literal(AstLiteral::Array(Array { elements: Some(elements) })) = data.right.get_stmt()
359    else {
360        return None;
361    };
362
363    // Only lower if the array contains non-constant expressions (function calls).
364    // Constant array literals are handled at codegen via memcpy from a global.
365    if !contains_non_constant_expression(elements) {
366        return None;
367    }
368
369    // Look up the LHS variable type to get array dimensions.
370    let lhs_type_name = find_lhs_type_name(data.left.as_ref(), index, pou_type_name)?;
371    let type_info = index.find_effective_type_info(&lhs_type_name)?;
372
373    let plc::typesystem::DataTypeInformation::Array { dimensions, .. } = type_info else {
374        return None;
375    };
376
377    // Build ArrayInfo from all dimensions.
378    let mut dims = Vec::new();
379    for dim in dimensions {
380        let start = dim.start_offset.as_int_value(index).ok()?;
381        let end = dim.end_offset.as_int_value(index).ok()?;
382        let size = end - start + 1;
383        if size <= 0 {
384            return None;
385        }
386        dims.push(DimInfo { start, size });
387    }
388
389    let array_info = ArrayInfo { dims };
390    let lowered = lower_array_elements(data.left.as_ref(), elements, &array_info, id_provider);
391
392    Some(lowered)
393}

395/// Returns `true` if the expression tree contains any non-constant expression
396/// that cannot be evaluated at compile time, such as function calls or struct
397/// literal initializers `(a := 1, b := 2)`.
398fn contains_non_constant_expression(node: &AstNode) -> bool {
399    !crate::helper::is_const_expression(node, None, None)
400}

402/// Determines the type name of the LHS of an assignment by consulting the index.
403/// Handles `self.field`, `field`, and plain identifiers.
404fn find_lhs_type_name(lhs: &AstNode, index: &Index, pou_type_name: &str) -> Option<String> {
405    match lhs.get_stmt() {
406        AstStatement::ReferenceExpr(plc_ast::ast::ReferenceExpr {
407            access: plc_ast::ast::ReferenceAccess::Member(member),
408            base,
409        }) => {
410            let member_name = member.get_flat_reference_name()?;
411
412            if let Some(base) = base {
413                // e.g. `self.arr` — resolve base type then find member
414                let base_type = find_lhs_type_name(base, index, pou_type_name)?;
415                let base_type_info = index.find_effective_type_info(&base_type)?;
416                if let plc::typesystem::DataTypeInformation::Struct { name, .. } = base_type_info {
417                    let member_entry = index.find_member(name, member_name)?;
418                    Some(member_entry.get_type_name().to_string())
419                } else {
420                    None
421                }
422            } else if member_name == "self" {
423                // `self` in a constructor refers to the POU/type being constructed.
424                // It's a VAR_IN_OUT so its indexed type is a pointer
425                // (`__auto_pointer_to_tarr`).  Dereference through the pointer to
426                // get the actual type name (e.g. `tarr`).
427                index.find_member(pou_type_name, "self").map(|v| {
428                    let type_name = v.get_type_name();
429                    match index.find_effective_type_info(type_name) {
430                        Some(plc::typesystem::DataTypeInformation::Pointer { inner_type_name, .. }) => {
431                            inner_type_name.to_string()
432                        }
433                        _ => type_name.to_string(),
434                    }
435                })
436            } else {
437                // Bare member reference (e.g. `arr` in function body) — the
438                // initializer produces `ReferenceExpr { Member("arr"), base: None }`
439                // for local/global variables. Resolve as a member of the current POU.
440                index.find_member(pou_type_name, member_name).map(|v| v.get_type_name().to_string())
441            }
442        }
443        AstStatement::Identifier(name) => {
444            if name == "self" {
445                Some(pou_type_name.to_string())
446            } else {
447                index
448                    .find_member(pou_type_name, name)
449                    .map(|v| v.get_type_name().to_string())
450                    .or_else(|| index.find_effective_type_info(name).map(|_| name.clone()))
451            }
452        }
453        _ => None,
454    }
455}

457/// The core lowering: given the LHS reference, the array literal elements,
458/// and the array dimension info, produces a list of statements.
459fn lower_array_elements(
460    lhs: &AstNode,
461    elements: &AstNode,
462    array_info: &ArrayInfo,
463    id_provider: &mut IdProvider,
464) -> LoweredResult {
465    let mut statements = Vec::new();
466    let mut counter_names = BTreeSet::new();
467    let mut current_flat_offset: i64 = 0;
468
469    match elements.get_stmt() {
470        // [N(val)] — single multiplied segment
471        AstStatement::MultipliedStatement(MultipliedStatement { multiplier, element }) => {
472            let segment = lower_multiplied_segment(
473                lhs,
474                element,
475                *multiplier,
476                current_flat_offset,
477                array_info,
478                id_provider,
479            );
480            counter_names.extend(segment.counter_names);
481            statements.extend(segment.statements);
482        }
483
484        // [e1, e2, ...] — expression list, may contain multiplied segments
485        AstStatement::ExpressionList(expressions) => {
486            for expr in expressions {
487                match expr.get_stmt() {
488                    AstStatement::MultipliedStatement(MultipliedStatement { multiplier, element }) => {
489                        let segment = lower_multiplied_segment(
490                            lhs,
491                            element,
492                            *multiplier,
493                            current_flat_offset,
494                            array_info,
495                            id_provider,
496                        );
497                        counter_names.extend(segment.counter_names);
498                        statements.extend(segment.statements);
499                        current_flat_offset += i64::from(*multiplier);
500                    }
501                    _ => {
502                        let indexed_lhs =
503                            make_array_element_access(lhs, current_flat_offset, array_info, id_provider);
504                        let assignment =
505                            AstFactory::create_assignment(indexed_lhs, expr.clone(), id_provider.next_id());
506                        statements.push(assignment);
507                        current_flat_offset += 1;
508                    }
509                }
510            }
511        }
512
513        // Single value (unusual but valid: `[val]`)
514        _ => {
515            let indexed_lhs = make_array_element_access(lhs, current_flat_offset, array_info, id_provider);
516            let assignment =
517                AstFactory::create_assignment(indexed_lhs, elements.clone(), id_provider.next_id());
518            statements.push(assignment);
519        }
520    }
521
522    LoweredResult { statements, counter_names }
523}

525/// Lowers a `MultipliedStatement(count, element)` segment.
526/// Small counts → unrolled individual assignments.
527/// Large counts on single-dim → a FOR loop.
528/// Large counts on multi-dim filling the whole array → nested FOR loops.
529/// Large counts on multi-dim (partial) → unrolled individual assignments.
530fn lower_multiplied_segment(
531    lhs: &AstNode,
532    element: &AstNode,
533    count: u32,
534    flat_offset: i64,
535    array_info: &ArrayInfo,
536    id_provider: &mut IdProvider,
537) -> LoweredResult {
538    let use_for_loop = count >= FOR_LOOP_THRESHOLD;
539
540    if !use_for_loop {
541        // Small count: unroll to individual indexed assignments.
542        return unroll_to_assignments(lhs, element, count, flat_offset, array_info, id_provider);
543    }
544
545    if array_info.is_single_dim() {
546        // Single-dim, large: emit a FOR loop.
547        let start = array_info.dims[0].start + flat_offset;
548        let end = start + i64::from(count) - 1;
549        let for_loop = make_for_loop(lhs, element, start, end, id_provider);
550        LoweredResult { statements: vec![for_loop], counter_names: BTreeSet::from([IDX_VAR.to_string()]) }
551    } else if flat_offset == 0 && i64::from(count) == array_info.total_elements() {
552        // Multi-dim, full fill: emit nested FOR loops (one per dimension).
553        let for_loop = make_nested_for_loops(lhs, element, array_info, id_provider);
554        let counter_names = array_info.nested_counter_names().into_iter().collect();
555        LoweredResult { statements: vec![for_loop], counter_names }
556    } else {
557        // Multi-dim, partial large: unroll with computed multi-dim indices.
558        // This is rare but correct — each element gets its own assignment.
559        unroll_to_assignments(lhs, element, count, flat_offset, array_info, id_provider)
560    }
561}

563/// Unrolls a multiplied segment into individual indexed assignments.
564fn unroll_to_assignments(
565    lhs: &AstNode,
566    element: &AstNode,
567    count: u32,
568    flat_offset: i64,
569    array_info: &ArrayInfo,
570    id_provider: &mut IdProvider,
571) -> LoweredResult {
572    let statements = (0..count)
573        .map(|i| {
574            let indexed_lhs =
575                make_array_element_access(lhs, flat_offset + i64::from(i), array_info, id_provider);
576            AstFactory::create_assignment(indexed_lhs, element.clone(), id_provider.next_id())
577        })
578        .collect();
579    LoweredResult { statements, counter_names: BTreeSet::new() }
580}

584/// Creates the appropriate indexed access for a given flat offset.
585/// For single-dim arrays: `lhs[start + flat_offset]`.
586/// For multi-dim arrays: `lhs[i0, i1, ...]` with computed indices.
587fn make_array_element_access(
588    lhs: &AstNode,
589    flat_offset: i64,
590    array_info: &ArrayInfo,
591    id_provider: &mut IdProvider,
592) -> AstNode {
593    if array_info.is_single_dim() {
594        make_indexed_access(lhs, array_info.dims[0].start + flat_offset, id_provider)
595    } else {
596        let indices = array_info.flat_to_indices(flat_offset);
597        make_multi_dim_indexed_access(lhs, &indices, id_provider)
598    }
599}

601/// Creates `lhs[i0, i1, ...]` for multi-dimensional array access.
602fn make_multi_dim_indexed_access(lhs: &AstNode, indices: &[i64], id_provider: &mut IdProvider) -> AstNode {
603    let loc = SourceLocation::internal();
604    let idx_nodes: Vec<AstNode> = indices.iter().map(|&i| make_int_literal(i, id_provider)).collect();
605    let idx_expr = AstFactory::create_expression_list(idx_nodes, loc.clone(), id_provider.next_id());
606    AstFactory::create_index_reference(idx_expr, Some(lhs.clone()), id_provider.next_id(), loc)
607}

609/// Creates `lhs[index_literal]` for single-dimension array access.
610fn make_indexed_access(lhs: &AstNode, index: i64, id_provider: &mut IdProvider) -> AstNode {
611    let idx_node = make_int_literal(index, id_provider);
612    AstFactory::create_index_reference(
613        idx_node,
614        Some(lhs.clone()),
615        id_provider.next_id(),
616        SourceLocation::internal(),
617    )
618}

620/// Creates: `FOR __literal_idx := start TO end DO lhs[__literal_idx] := element; END_FOR`
621/// Used for single-dimension arrays.
622fn make_for_loop(
623    lhs: &AstNode,
624    element: &AstNode,
625    start: i64,
626    end: i64,
627    id_provider: &mut IdProvider,
628) -> AstNode {
629    let loc = SourceLocation::internal();
630
631    let counter = make_member_reference(IDX_VAR, id_provider);
632    let start_node = make_int_literal(start, id_provider);
633    let end_node = make_int_literal(end, id_provider);
634
635    // lhs[__literal_idx]
636    let idx_ref = make_member_reference(IDX_VAR, id_provider);
637    let indexed_lhs =
638        AstFactory::create_index_reference(idx_ref, Some(lhs.clone()), id_provider.next_id(), loc.clone());
639
640    let body_assignment = AstFactory::create_assignment(indexed_lhs, element.clone(), id_provider.next_id());
641
642    AstFactory::create_for_loop(
643        ForLoopStatement {
644            counter: Box::new(counter),
645            start: Box::new(start_node),
646            end: Box::new(end_node),
647            by_step: None,
648            body: vec![body_assignment],
649            end_location: loc.clone(),
650        },
651        loc,
652        id_provider.next_id(),
653    )
654}

656/// Creates nested FOR loops for multi-dimensional arrays:
657/// ```text
658/// FOR __literal_idx_0 := dim0_start TO dim0_end DO
659///   FOR __literal_idx_1 := dim1_start TO dim1_end DO
660///     ...
661///       lhs[__literal_idx_0, __literal_idx_1, ...] := element;
662///     ...
663///   END_FOR
664/// END_FOR
665/// ```
666fn make_nested_for_loops(
667    lhs: &AstNode,
668    element: &AstNode,
669    array_info: &ArrayInfo,
670    id_provider: &mut IdProvider,
671) -> AstNode {
672    let loc = SourceLocation::internal();
673    let counter_names = array_info.nested_counter_names();
674
675    // Create the indexed access: lhs[__literal_idx_0, __literal_idx_1, ...]
676    let idx_refs: Vec<AstNode> =
677        counter_names.iter().map(|name| make_member_reference(name, id_provider)).collect();
678    let idx_expr = AstFactory::create_expression_list(idx_refs, loc.clone(), id_provider.next_id());
679    let indexed_lhs =
680        AstFactory::create_index_reference(idx_expr, Some(lhs.clone()), id_provider.next_id(), loc.clone());
681
682    let assignment = AstFactory::create_assignment(indexed_lhs, element.clone(), id_provider.next_id());
683
684    // Build nested FOR loops from innermost dimension to outermost.
685    let mut body = vec![assignment];
686    for i in (0..array_info.dims.len()).rev() {
687        let counter = make_member_reference(&counter_names[i], id_provider);
688        let start_node = make_int_literal(array_info.dims[i].start, id_provider);
689        let end_node = make_int_literal(array_info.dims[i].start + array_info.dims[i].size - 1, id_provider);
690
691        let for_loop = AstFactory::create_for_loop(
692            ForLoopStatement {
693                counter: Box::new(counter),
694                start: Box::new(start_node),
695                end: Box::new(end_node),
696                by_step: None,
697                body,
698                end_location: loc.clone(),
699            },
700            loc.clone(),
701            id_provider.next_id(),
702        );
703        body = vec![for_loop];
704    }
705
706    body.pop().expect("nested FOR loop construction should produce exactly one statement")
707}
