1//! Shared helpers for determining whether an expression tree is compile-time constant.
2
3use plc::index::Index;
4use plc_ast::ast::{AstNode, AstStatement, MultipliedStatement};
5
6/// Returns `true` if every leaf in the expression tree can be evaluated at
7/// compile time. Literals and references to constant variables (when `index`
8/// and `pou_name` are provided) are considered constant. Function calls,
9/// struct literal assignments, and non-constant variable references are not.
10pub fn is_const_expression(node: &AstNode, index: Option<&Index>, pou_name: Option<&str>) -> bool {
11    match node.get_stmt() {
12        AstStatement::Literal(..) => true,
13        AstStatement::ExpressionList(exprs) => exprs.iter().all(|e| is_const_expression(e, index, pou_name)),
14        AstStatement::MultipliedStatement(MultipliedStatement { element, .. }) => {
15            is_const_expression(element, index, pou_name)
16        }
17        AstStatement::ParenExpression(inner) => is_const_expression(inner, index, pou_name),
18        AstStatement::Identifier(..) | AstStatement::ReferenceExpr(..) => {
19            // If we have an index, check whether this is a reference to a constant variable.
20            index.is_some_and(|idx| is_const_reference(node, idx, pou_name))
21        }
22        // Everything else: function calls, struct literal assignments, etc.
23        _ => false,
24    }
25}
26
27/// Returns `true` if the node is a reference to a constant variable.
28fn is_const_reference(node: &AstNode, index: &Index, pou_name: Option<&str>) -> bool {
29    let Some(name) = node.get_flat_reference_name() else { return false };
30
31    // Check as POU-local member first, then as global.
32    let variable =
33        pou_name.and_then(|pou| index.find_member(pou, name)).or_else(|| index.find_global_variable(name));
34
35    variable.is_some_and(|v| v.is_constant())
36}
