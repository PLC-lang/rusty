1use std::time::Instant;
2
3use plc::lexer;
4use plc_ast::provider::IdProvider;
5use plc_source::source_location::SourceLocationFactory;
6
7use crate::reporter::DurationWrapper;
8
9use super::Task;
10
11pub struct Lexer(pub &'static str);
12impl Task for Lexer {
13    fn get_name(&self) -> String {
14        format!("lexer/{}", self.0)
15    }
16
17    fn execute(&self) -> anyhow::Result<std::time::Duration> {
18        let content = std::fs::read_to_string(format!("./xtask/res/{}", self.0)).unwrap();
19        let mut lexer =
20            lexer::lex_with_ids(&content, IdProvider::default(), SourceLocationFactory::internal(&content));
21
22        let now = Instant::now();
23        while !lexer.is_end_of_stream() {
24            lexer.advance();
25        }
26        let elapsed = now.elapsed();
27
28        debug_assert_eq!(lexer.token, plc::lexer::Token::End);
29        debug_assert!(lexer.last_range.end >= 143100);
30        Ok(elapsed)
31    }
32
33    fn get_wrapped(&self, duration: std::time::Duration) -> DurationWrapper {
34        DurationWrapper::Micros(duration)
35    }
36}
