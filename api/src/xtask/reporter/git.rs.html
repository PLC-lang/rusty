10impl Reporter for GitReporter {
11    fn persist(&self, report: super::BenchmarkReport) -> anyhow::Result<()> {
12        let sh = Shell::new()?;
13        let branch = "metrics-data";
14        let filename = "metrics.json";
15        let message = format!("Update {}", report.commit);
16        let user_name = cmd!(sh, "git log -1 --pretty=format:'%an'").read()?;
17        let user_mail = cmd!(sh, "git log -1 --pretty=format:'%ae'").read()?;
18
19        cmd!(sh, "git pull").run()?;
20        cmd!(sh, "git config user.name \"{user_name}\"").run()?;
21        cmd!(sh, "git config user.email \"{user_mail}\"").run()?;
22        cmd!(sh, "git checkout {branch}").run()?;
23
24        let mut file = fs::File::options().create(true).append(true).open(filename)?;
25        writeln!(file, "{}", serde_json::to_string(&report)?)?;
26
27        cmd!(sh, "git add {filename}").run()?;
28        cmd!(sh, "git commit -m {message}").run()?;
29        cmd!(sh, "git push origin {branch}").run()?;
30
31        Ok(())
32    }
33}
