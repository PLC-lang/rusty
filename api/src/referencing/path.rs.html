77/// Escape a key into a JSON Pointer segment: `~` -> `~0`, `/` -> `~1`.
78///
79/// Appends the escaped form of `value` directly to `buffer`.
80pub fn write_escaped_str(buffer: &mut String, value: &str) {
81    match value.find(['~', '/']) {
82        Some(mut escape_idx) => {
83            let mut remaining = value;
84
85            // Loop through the string to replace `~` and `/`
86            loop {
87                let (before, after) = remaining.split_at(escape_idx);
88                // Copy everything before the escape char
89                buffer.push_str(before);
90
91                // Append the appropriate escape sequence
92                match after.as_bytes()[0] {
93                    b'~' => buffer.push_str("~0"),
94                    b'/' => buffer.push_str("~1"),
95                    _ => unreachable!(),
96                }
97
98                // Move past the escaped character
99                remaining = &after[1..];
100
101                // Find the next `~` or `/` to continue escaping
102                if let Some(next_escape_idx) = remaining.find(['~', '/']) {
103                    escape_idx = next_escape_idx;
104                } else {
105                    // Append any remaining part of the string
106                    buffer.push_str(remaining);
107                    break;
108                }
109            }
110        }
111        None => {
112            // If no escape characters are found, append the segment as is
113            buffer.push_str(value);
114        }
115    }
116}
