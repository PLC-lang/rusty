12type CacheMap = HashMap<(String, String), Arc<Uri<String>>>;

14struct StrPair<'a>(&'a str, &'a str);
15
16impl Hash for StrPair<'_> {
17    fn hash<H: Hasher>(&self, state: &mut H) {
18        self.0.hash(state);
19        self.1.hash(state);
20    }
21}
22
23impl Equivalent<(String, String)> for StrPair<'_> {
24    fn equivalent(&self, (a, b): &(String, String)) -> bool {
25        self.0 == a && self.1 == b
26    }
27}

30pub(crate) struct UriCache {
31    cache: CacheMap,
32}
33
34impl UriCache {
35    pub(crate) fn new() -> Self {
36        Self {
37            cache: HashMap::new(),
38        }
39    }
40
41    pub(crate) fn with_capacity(capacity: usize) -> Self {
42        Self {
43            cache: HashMap::with_capacity(capacity),
44        }
45    }
46
47    pub(crate) fn resolve_against(
48        &mut self,
49        base: &Uri<&str>,
50        uri: impl AsRef<str>,
51    ) -> Result<Arc<Uri<String>>, Error> {
52        let base_str = base.as_str();
53        let reference = uri.as_ref();
54        if let Some(cached) = self.cache.get(&StrPair(base_str, reference)) {
55            return Ok(Arc::clone(cached));
56        }
57        let resolved = Arc::new(uri::resolve_against(base, reference)?);
58        self.cache.insert(
59            (base_str.to_owned(), reference.to_owned()),
60            Arc::clone(&resolved),
61        );
62        Ok(resolved)
63    }
64
65    pub(crate) fn into_shared(self) -> SharedUriCache {
66        SharedUriCache {
67            cache: RwLock::new(self.cache),
68        }
69    }

72/// A dedicated type for URI resolution caching.
73#[derive(Debug)]
74pub(crate) struct SharedUriCache {
75    cache: RwLock<CacheMap>,
76}
77
78impl Clone for SharedUriCache {
79    fn clone(&self) -> Self {
80        Self {
81            cache: RwLock::new(
82                self.cache
83                    .read()
84                    .iter()
85                    .map(|(key, value)| (key.clone(), Arc::clone(value)))
86                    .collect(),
87            ),
88        }
89    }
90}
91
92impl SharedUriCache {
93    pub(crate) fn resolve_against(
94        &self,
95        base: &Uri<&str>,
96        uri: impl AsRef<str>,
97    ) -> Result<Arc<Uri<String>>, Error> {
98        let base_str = base.as_str();
99        let reference = uri.as_ref();
100        let lookup = StrPair(base_str, reference);
101
102        if let Some(cached) = self.cache.read().get(&lookup) {
103            return Ok(Arc::clone(cached));
104        }
105
106        let cache = self.cache.upgradable_read();
107        if let Some(cached) = cache.get(&lookup) {
108            return Ok(Arc::clone(cached));
109        }
110
111        let resolved = Arc::new(uri::resolve_against(base, reference)?);
112        let mut cache = RwLockUpgradableReadGuard::upgrade(cache);
113        cache.insert(
114            (base_str.to_owned(), reference.to_owned()),
115            Arc::clone(&resolved),
116        );
117        Ok(resolved)
118    }
