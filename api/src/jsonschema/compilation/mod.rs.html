27/// The structure that holds a JSON Schema compiled into a validation tree
28#[derive(Debug)]
29pub struct JSONSchema {
30    pub(crate) node: SchemaNode,
31    config: Arc<CompilationOptions>,
32}
33
34pub(crate) static DEFAULT_SCOPE: Lazy<Url> =
35    Lazy::new(|| url::Url::parse(DEFAULT_ROOT_URL).expect("Is a valid URL"));

37impl JSONSchema {
38    /// Return a default `CompilationOptions` that can configure
39    /// `JSONSchema` compilaton flow.
40    ///
41    /// Using options you will be able to configure the draft version
42    /// to use during `JSONSchema` compilation
43    ///
44    /// Example of usage:
45    /// ```rust
46    /// # use crate::jsonschema::{Draft, JSONSchema};
47    /// # let schema = serde_json::json!({});
48    /// let maybe_jsonschema: Result<JSONSchema, _> = JSONSchema::options()
49    ///     .with_draft(Draft::Draft7)
50    ///     .compile(&schema);
51    /// ```
52    #[must_use]
53    pub fn options() -> CompilationOptions {
54        CompilationOptions::default()
55    }
56
57    /// Compile the input schema into a validation tree.
58    ///
59    /// The method is equivalent to `JSONSchema::options().compile(schema)`
60    pub fn compile(schema: &Value) -> Result<JSONSchema, ValidationError> {
61        Self::options().compile(schema)
62    }
63
64    /// Run validation against `instance` and return an iterator over `ValidationError` in the error case.
65    #[inline]
66    pub fn validate<'instance>(
67        &'instance self,
68        instance: &'instance Value,
69    ) -> Result<(), ErrorIterator<'instance>> {
70        let instance_path = InstancePath::new();
71        let mut errors = self.node.validate(instance, &instance_path).peekable();
72        if errors.peek().is_none() {
73            Ok(())
74        } else {
75            Err(Box::new(errors))
76        }
77    }
78
79    /// Run validation against `instance` but return a boolean result instead of an iterator.
80    /// It is useful for cases, where it is important to only know the fact if the data is valid or not.
81    /// This approach is much faster, than `validate`.
82    #[must_use]
83    #[inline]
84    pub fn is_valid(&self, instance: &Value) -> bool {
85        self.node.is_valid(instance)
86    }
87
88    /// Apply the schema and return an `Output`. No actual work is done at this point, the
89    /// evaluation of the schema is deferred until a method is called on the `Output`. This is
90    /// because different output formats will have different performance characteristics.
91    ///
92    /// # Examples
93    ///
94    /// "basic" output format
95    ///
96    /// ```rust
97    /// # use crate::jsonschema::{Draft, JSONSchema, output::{Output, BasicOutput}};
98    /// let schema_json = serde_json::json!({
99    ///     "title": "string value",
100    ///     "type": "string"
101    /// });
102    /// let instance = serde_json::json!{"some string"};
103    /// let schema = JSONSchema::options().compile(&schema_json).unwrap();
104    /// let output: BasicOutput = schema.apply(&instance).basic();
105    /// let output_json = serde_json::to_value(output).unwrap();
106    /// assert_eq!(output_json, serde_json::json!({
107    ///     "valid": true,
108    ///     "annotations": [
109    ///         {
110    ///             "keywordLocation": "",
111    ///             "instanceLocation": "",
112    ///             "annotations": {
113    ///                 "title": "string value"
114    ///             }
115    ///         }
116    ///     ]
117    /// }));
118    /// ```
119    #[must_use]
120    pub const fn apply<'a, 'b>(&'a self, instance: &'b Value) -> Output<'a, 'b> {
121        Output::new(self, &self.node, instance)
122    }
123
124    /// The [`Draft`] which this schema was compiled against
125    #[must_use]
126    pub fn draft(&self) -> Draft {
127        self.config.draft()
128    }
129
130    /// The [`CompilationOptions`] that were used to compile this schema
131    #[must_use]
132    pub fn config(&self) -> Arc<CompilationOptions> {
133        Arc::clone(&self.config)
134    }
135}

137/// Compile JSON schema into a tree of validators.
138#[inline]
139pub(crate) fn compile_validators<'a>(
140    schema: &'a Value,
141    context: &CompilationContext,
142) -> Result<SchemaNode, ValidationError<'a>> {
143    let context = context.push(schema)?;
144    let relative_path = context.clone().into_pointer();
145    match schema {
146        Value::Bool(value) => match value {
147            true => Ok(SchemaNode::new_from_boolean(&context, None)),
148            false => Ok(SchemaNode::new_from_boolean(
149                &context,
150                Some(
151                    keywords::boolean::FalseValidator::compile(relative_path)
152                        .expect("Should always compile"),
153                ),
154            )),
155        },
156        Value::Object(object) => {
157            // In Draft 2019-09 and later, `$ref` can be evaluated alongside other attribute aka
158            // adjacent validation. We check here to see if adjacent validation is supported, and if
159            // so, we use the normal keyword validator collection logic.
160            //
161            // Otherwise, we isolate `$ref` and generate a schema reference validator directly.
162            let maybe_reference = object
163                .get("$ref")
164                .filter(|_| !keywords::ref_::supports_adjacent_validation(context.config.draft()));
165            if let Some(reference) = maybe_reference {
166                let unmatched_keywords = object
167                    .iter()
168                    .filter_map(|(k, v)| {
169                        if k.as_str() == "$ref" {
170                            None
171                        } else {
172                            Some((k.clone(), v.clone()))
173                        }
174                    })
175                    .collect();
176
177                let validator = keywords::ref_::compile(object, reference, &context)
178                    .expect("should always return Some")?;
179
180                let validators = vec![("$ref".to_string(), validator)];
181                Ok(SchemaNode::new_from_keywords(
182                    &context,
183                    validators,
184                    Some(unmatched_keywords),
185                ))
186            } else {
187                let mut validators = Vec::with_capacity(object.len());
188                let mut unmatched_keywords = AHashMap::new();
189                let mut is_if = false;
190                let mut is_props = false;
191                for (keyword, subschema) in object {
192                    if keyword == "if" {
193                        is_if = true;
194                    }
195                    if keyword == "properties"
196                        || keyword == "additionalProperties"
197                        || keyword == "patternProperties"
198                    {
199                        is_props = true;
200                    }
201                    if let Some(validator) = context
202                        .config
203                        .draft()
204                        .get_validator(keyword)
205                        .and_then(|f| f(object, subschema, &context))
206                    {
207                        validators.push((keyword.clone(), validator?));
208                    } else {
209                        unmatched_keywords.insert(keyword.to_string(), subschema.clone());
210                    }
211                }
212                if is_if {
213                    unmatched_keywords.remove("then");
214                    unmatched_keywords.remove("else");
215                }
216                if is_props {
217                    unmatched_keywords.remove("additionalProperties");
218                    unmatched_keywords.remove("patternProperties");
219                    unmatched_keywords.remove("properties");
220                }
221                let unmatched_keywords = if unmatched_keywords.is_empty() {
222                    None
223                } else {
224                    Some(unmatched_keywords)
225                };
226                Ok(SchemaNode::new_from_keywords(
227                    &context,
228                    validators,
229                    unmatched_keywords,
230                ))
231            }
232        }
233        _ => Err(ValidationError::multiple_type_error(
234            JSONPointer::default(),
235            relative_path,
236            schema,
237            PrimitiveTypesBitMap::new()
238                .add_type(PrimitiveType::Boolean)
239                .add_type(PrimitiveType::Object),
240        )),
241    }

244#[cfg(test)]
245mod tests {
246    use super::JSONSchema;
247    use crate::error::ValidationError;
248    use serde_json::{from_str, json, Value};
249    use std::{fs::File, io::Read, path::Path};

251    fn load(path: &str, idx: usize) -> Value {
252        let path = Path::new(path);
253        let mut file = File::open(path).unwrap();
254        let mut content = String::new();
255        file.read_to_string(&mut content).ok().unwrap();
256        let data: Value = from_str(&content).unwrap();
257        let case = &data.as_array().unwrap()[idx];
258        case.get("schema").unwrap().clone()
259    }
260
261    #[test]
262    fn only_keyword() {
263        // When only one keyword is specified
264        let schema = json!({"type": "string"});
265        let compiled = JSONSchema::compile(&schema).unwrap();
266        let value1 = json!("AB");
267        let value2 = json!(1);
268        // And only this validator
269        assert_eq!(compiled.node.validators().len(), 1);
270        assert!(compiled.validate(&value1).is_ok());
271        assert!(compiled.validate(&value2).is_err());
272    }
273
274    #[test]
275    fn validate_ref() {
276        let schema = load("tests/suite/tests/draft7/ref.json", 1);
277        let value = json!({"bar": 3});
278        let compiled = JSONSchema::compile(&schema).unwrap();
279        assert!(compiled.validate(&value).is_ok());
280        let value = json!({"bar": true});
281        assert!(compiled.validate(&value).is_err());
282    }
283
284    #[test]
285    fn wrong_schema_type() {
286        let schema = json!([1]);
287        let compiled = JSONSchema::compile(&schema);
288        assert!(compiled.is_err());
289    }
290
291    #[test]
292    fn multiple_errors() {
293        let schema = json!({"minProperties": 2, "propertyNames": {"minLength": 3}});
294        let value = json!({"a": 3});
295        let compiled = JSONSchema::compile(&schema).unwrap();
296        let result = compiled.validate(&value);
297        let errors: Vec<ValidationError> = result.unwrap_err().collect();
298        assert_eq!(errors.len(), 2);
299        assert_eq!(
300            errors[0].to_string(),
301            r#"{"a":3} has less than 2 properties"#
302        );
303        assert_eq!(errors[1].to_string(), r#""a" is shorter than 3 characters"#);
304    }
