12static DEFAULT_SCHEME: &str = "json-schema";
13
14/// Context holds information about used draft and current scope.
15#[derive(Debug, Clone)]
16pub(crate) struct CompilationContext<'a> {
17    base_uri: BaseUri<'a>,
18    pub(crate) config: Arc<CompilationOptions>,
19    pub(crate) resolver: Arc<Resolver>,
20    pub(crate) schema_path: InstancePath<'a>,
21}
22
23#[derive(Debug, Clone)]
24pub(crate) enum BaseUri<'a> {
25    /// A base URL was specified, either because we know a reasonable base URI where we retrieved the
26    /// schema or because a URI was specified via $id
27    Known(Cow<'a, Url>),
28    /// We don't know a base URI for this schema
29    Unknown,
30}
31
32impl<'a> BaseUri<'a> {
33    pub(crate) fn with_new_scope(&self, new_scope: &str) -> Result<Self, ParseError> {
34        let options = match self {
35            BaseUri::Known(u) => Url::options().base_url(Some(u)),
36            BaseUri::Unknown => Url::options().base_url(Some(&DEFAULT_SCOPE)),
37        };
38        Ok(options.parse(new_scope)?.into())
39    }
40}
41
42impl<'a> From<Option<Url>> for BaseUri<'a> {
43    fn from(u: Option<Url>) -> Self {
44        u.map_or(BaseUri::Unknown, Into::into)
45    }
46}
47
48impl<'a> From<&'a Url> for BaseUri<'a> {
49    fn from(u: &'a Url) -> Self {
50        if u.scheme() == DEFAULT_SCHEME {
51            BaseUri::Unknown
52        } else {
53            BaseUri::Known(Cow::Borrowed(u))
54        }
55    }
56}
57
58impl<'a> From<Url> for BaseUri<'a> {
59    fn from(u: Url) -> Self {
60        if u.scheme() == DEFAULT_SCHEME {
61            BaseUri::Unknown
62        } else {
63            BaseUri::Known(Cow::Owned(u))
64        }
65    }
66}
67
68impl<'a> From<&'a BaseUri<'a>> for Cow<'a, Url> {
69    fn from(uri: &BaseUri<'a>) -> Self {
70        match uri {
71            BaseUri::Unknown => Cow::Borrowed(&DEFAULT_SCOPE),
72            BaseUri::Known(u) => u.clone(),
73        }
74    }
75}

77impl<'a> CompilationContext<'a> {
78    pub(crate) const fn new(
79        scope: BaseUri<'a>,
80        config: Arc<CompilationOptions>,
81        resolver: Arc<Resolver>,
82    ) -> Self {
83        CompilationContext {
84            base_uri: scope,
85            config,
86            resolver,
87            schema_path: InstancePath::new(),
88        }
89    }
90
91    #[allow(clippy::doc_markdown)]
92    /// Push a new scope. All URLs built from the new context will have this scope in them.
93    /// Before push:
94    ///    scope = http://example.com/
95    ///    build_url("#/definitions/foo") -> "http://example.com/#/definitions/foo"
96    /// After push this schema - {"$id": "folder/", ...}
97    ///    scope = http://example.com/folder/
98    ///    build_url("#/definitions/foo") -> "http://example.com/folder/#/definitions/foo"
99    ///
100    /// In other words it keeps track of sub-folders during compilation.
101    #[inline]
102    pub(crate) fn push(&'a self, schema: &Value) -> Result<Self, ParseError> {
103        if let Some(id) = schemas::id_of(self.config.draft(), schema) {
104            Ok(CompilationContext {
105                base_uri: self.base_uri.with_new_scope(id)?,
106                config: Arc::clone(&self.config),
107                resolver: Arc::clone(&self.resolver),
108                schema_path: self.schema_path.clone(),
109            })
110        } else {
111            Ok(CompilationContext {
112                base_uri: self.base_uri.clone(),
113                config: Arc::clone(&self.config),
114                resolver: Arc::clone(&self.resolver),
115                schema_path: self.schema_path.clone(),
116            })
117        }
118    }
119
120    #[inline]
121    pub(crate) fn with_path(&'a self, chunk: impl Into<PathChunk>) -> Self {
122        let schema_path = self.schema_path.push(chunk);
123        CompilationContext {
124            base_uri: self.base_uri.clone(),
125            config: Arc::clone(&self.config),
126            resolver: Arc::clone(&self.resolver),
127            schema_path,
128        }
129    }
130
131    /// Create a JSON Pointer from the current `schema_path` & a new chunk.
132    #[inline]
133    pub(crate) fn into_pointer(self) -> JSONPointer {
134        self.schema_path.into()
135    }
136
137    /// Create a JSON Pointer from the current `schema_path` & a new chunk.
138    #[inline]
139    pub(crate) fn as_pointer_with(&self, chunk: impl Into<PathChunk>) -> JSONPointer {
140        self.schema_path.push(chunk).into()
141    }
142
143    /// Build a new URL. Used for `ref` compilation to keep their full paths.
144    pub(crate) fn build_url(&self, reference: &str) -> Result<Url, ParseError> {
145        let cowbase: Cow<Url> = (&self.base_uri).into();
146        Url::options()
147            .base_url(Some(cowbase.as_ref()))
148            .parse(reference)
149    }
150
151    pub(crate) fn base_uri(&self) -> Option<Url> {
152        match &self.base_uri {
153            BaseUri::Known(u) => Some(u.as_ref().clone()),
154            BaseUri::Unknown => None,
155        }
156    }
