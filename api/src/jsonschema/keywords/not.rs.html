1use crate::{
2    compilation::{compile_validators, context::CompilationContext},
3    error::{error, no_error, ErrorIterator, ValidationError},

9use serde_json::{Map, Value};

11pub(crate) struct NotValidator {
12    // needed only for error representation
13    original: Value,
14    node: SchemaNode,
15    schema_path: JSONPointer,
16}
17
18impl NotValidator {
19    #[inline]
20    pub(crate) fn compile<'a>(
21        schema: &'a Value,
22        context: &CompilationContext,
23    ) -> CompilationResult<'a> {
24        let keyword_context = context.with_path("not");
25        Ok(Box::new(NotValidator {
26            original: schema.clone(),
27            node: compile_validators(schema, &keyword_context)?,
28            schema_path: keyword_context.into_pointer(),
29        }))
30    }
31}
32
33impl Validate for NotValidator {
34    fn is_valid(&self, instance: &Value) -> bool {
35        !self.node.is_valid(instance)
36    }
37
38    fn validate<'instance>(
39        &self,
40        instance: &'instance Value,
41        instance_path: &InstancePath,
42    ) -> ErrorIterator<'instance> {
43        if self.is_valid(instance) {
44            no_error()
45        } else {
46            error(ValidationError::not(
47                self.schema_path.clone(),
48                instance_path.into(),
49                instance,
50                self.original.clone(),
51            ))
52        }
53    }
54}
55
56impl core::fmt::Display for NotValidator {
57    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
58        write!(f, "not: {}", format_validators(self.node.validators()))
59    }
60}
61
62#[inline]
63pub(crate) fn compile<'a>(
64    _: &'a Map<String, Value>,
65    schema: &'a Value,
66    context: &CompilationContext,
67) -> Option<CompilationResult<'a>> {
68    Some(NotValidator::compile(schema, context))
69}
