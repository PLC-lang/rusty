167    fn error_message(schema: &Value, instance: &Value, expected: &str) {
168        let compiled = JSONSchema::compile(schema).unwrap();
169        let errors: Vec<_> = compiled
170            .validate(instance)
171            .expect_err(&format!(
172                "Validation error is expected. Schema=`{:?}` Instance=`{:?}`",
173                schema, instance
174            ))
175            .collect();
176        assert_eq!(errors[0].to_string(), expected);
177    }

254    fn required_all_properties() {
255        // See: GH-190
256        let schema = json!({"required": ["foo", "bar"]});
257        let instance = json!({});
258        let compiled = JSONSchema::compile(&schema).unwrap();
259        let errors: Vec<_> = compiled
260            .validate(&instance)
261            .expect_err("Validation errors")
262            .collect();
263        assert_eq!(errors.len(), 2);
264        assert_eq!(errors[0].to_string(), r#""foo" is a required property"#);
265        assert_eq!(errors[1].to_string(), r#""bar" is a required property"#);
