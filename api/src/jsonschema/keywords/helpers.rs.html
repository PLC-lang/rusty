9macro_rules! num_cmp {
10    ($left:expr, $right:expr) => {
11        if let Some(b) = $right.as_u64() {
12            NumCmp::num_eq($left, b)
13        } else if let Some(b) = $right.as_i64() {
14            NumCmp::num_eq($left, b)
15        } else {
16            NumCmp::num_eq($left, $right.as_f64().expect("Always valid"))
17        }
18    };
19}

21#[inline]
22pub(crate) fn equal(left: &Value, right: &Value) -> bool {
23    match (left, right) {
24        (Value::String(left), Value::String(right)) => left == right,
25        (Value::Bool(left), Value::Bool(right)) => left == right,
26        (Value::Null, Value::Null) => true,
27        (Value::Number(left), Value::Number(right)) => {
28            if let Some(a) = left.as_u64() {
29                num_cmp!(a, right)
30            } else if let Some(a) = left.as_i64() {
31                num_cmp!(a, right)
32            } else {
33                let a = left.as_f64().expect("Always valid");
34                num_cmp!(a, right)
35            }
36        }
37        (Value::Array(left), Value::Array(right)) => equal_arrays(left, right),
38        (Value::Object(left), Value::Object(right)) => equal_objects(left, right),
39        (_, _) => false,
40    }
41}

43#[inline]
44pub(crate) fn equal_arrays(left: &[Value], right: &[Value]) -> bool {
45    left.len() == right.len() && {
46        let mut idx = 0_usize;
47        while idx < left.len() {
48            if !equal(&left[idx], &right[idx]) {
49                return false;
50            }
51            idx += 1;
52        }
53        true
54    }
55}

57#[inline]
58pub(crate) fn equal_objects(left: &Map<String, Value>, right: &Map<String, Value>) -> bool {
59    left.len() == right.len()
60        && left
61            .iter()
62            .zip(right)
63            .all(|((ka, va), (kb, vb))| ka == kb && equal(va, vb))
64}

66#[inline]
67pub(crate) fn map_get_u64<'a>(
68    m: &'a Map<String, Value>,
69    context: &CompilationContext,
70    type_name: &str,
71) -> Option<Result<u64, ValidationError<'a>>> {
72    let value = m.get(type_name)?;
73    match value.as_u64() {
74        Some(n) => Some(Ok(n)),
75        None if value.is_i64() => Some(Err(ValidationError::minimum(
76            JSONPointer::default(),
77            context.clone().into_pointer(),
78            value,
79            0.into(),
80        ))),
81        None => Some(Err(ValidationError::single_type_error(
82            JSONPointer::default(),
83            context.clone().into_pointer(),
84            value,
85            PrimitiveType::Integer,
86        ))),
87    }
88}

90/// Fail if the input value is not `u64`.
91pub(crate) fn fail_on_non_positive_integer(
92    value: &Value,
93    instance_path: JSONPointer,
94) -> ValidationError<'_> {
95    if value.is_i64() {
96        ValidationError::minimum(JSONPointer::default(), instance_path, value, 0.into())
97    } else {
98        ValidationError::single_type_error(
99            JSONPointer::default(),
100            instance_path,
101            value,
102            PrimitiveType::Integer,
103        )
104    }
105}
