8/// Convert ECMA Script 262 regex to Rust regex on the best effort basiso.
9///
10/// NOTE: Patterns with look arounds and backreferecnes are not supported.
11pub(crate) fn to_rust_regex(pattern: &str) -> Result<Cow<'_, str>, ()> {
12    let mut pattern = Cow::Borrowed(pattern);
13    let mut ast = loop {
14        match Parser::new().parse(&pattern) {
15            Ok(ast) => break ast,
16            Err(error) if *error.kind() == ErrorKind::EscapeUnrecognized => {
17                let Span { start, end } = error.span();
18                let source = error.pattern();
19                if &source[start.offset..end.offset] == r"\c" {
20                    if let Some(letter) = &source[end.offset..].chars().next() {
21                        if letter.is_ascii_alphabetic() {
22                            let start = start.offset;
23                            let end = end.offset + 1;
24                            let replacement = ((*letter as u8) % 32) as char;
25                            match pattern {
26                                Cow::Borrowed(_) => {
27                                    let prefix = &source[..start];
28                                    let suffix = &source[end..];
29                                    pattern = Cow::Owned(format!("{prefix}{replacement}{suffix}"));
30                                }
31                                Cow::Owned(ref mut buffer) => {
32                                    let mut char_buffer = [0; 4];
33                                    let replacement = replacement.encode_utf8(&mut char_buffer);
34                                    buffer.replace_range(start..end, replacement);
35                                }
36                            }
37                            continue;
38                        }
39                    }
40                }
41                return Err(());
42            }
43            Err(error)
44                if matches!(
45                    error.kind(),
46                    ErrorKind::UnsupportedLookAround | ErrorKind::UnsupportedBackreference
47                ) =>
48            {
49                // Can't translate patterns with look arounds & backreferences
50                return Ok(pattern);
51            }
52            Err(_) => {
53                return Err(());
54            }
55        };
56    };
57    let mut has_changes;
58    loop {
59        let translator = Ecma262Translator::new(pattern);
60        (pattern, has_changes) = ast::visit(&ast, translator).map_err(|_| ())?;
61        if !has_changes {
62            return Ok(pattern);
63        }
64        match Parser::new().parse(&pattern) {
65            Ok(updated_ast) => {
66                ast = updated_ast;
67            }
68            Err(_) => {
69                return Err(());
70            }
71        }
72    }
73}

75struct Ecma262Translator<'a> {
76    pattern: Cow<'a, str>,
77    offset: usize,
78    has_changes: bool,
79}
80
81impl<'a> Ecma262Translator<'a> {
82    fn new(input: Cow<'a, str>) -> Self {
83        Self {
84            pattern: input,
85            offset: 0,
86            has_changes: false,
87        }
88    }
89
90    fn replace_impl(&mut self, span: &Span, replacement: &str) {
91        let Span { start, end } = span;
92        match self.pattern {
93            Cow::Borrowed(pattern) => {
94                let prefix = &pattern[..start.offset];
95                let suffix = &pattern[end.offset..];
96                self.pattern = Cow::Owned(format!("{prefix}{replacement}{suffix}"));
97            }
98            Cow::Owned(ref mut buffer) => {
99                buffer.replace_range(
100                    start.offset + self.offset..end.offset + self.offset,
101                    replacement,
102                );
103            }
104        }
105        self.offset += replacement.len() - (end.offset - start.offset);
106        self.has_changes = true;
107    }
108
109    fn replace(&mut self, cls: &ClassPerl) {
110        match cls.kind {
111            ClassPerlKind::Digit => {
112                let replacement = if cls.negated { "[^0-9]" } else { "[0-9]" };
113                self.replace_impl(&cls.span, replacement);
114            }
115            ClassPerlKind::Word => {
116                let replacement = if cls.negated {
117                    "[^A-Za-z0-9_]"
118                } else {
119                    "[A-Za-z0-9_]"
120                };
121                self.replace_impl(&cls.span, replacement);
122            }
123            ClassPerlKind::Space => {
124                let replacement = &if cls.negated {
125                    "[^ \t\n\r\u{000b}\u{000c}\u{00a0}\u{feff}\u{2003}\u{2029}]"
126                } else {
127                    "[ \t\n\r\u{000b}\u{000c}\u{00a0}\u{feff}\u{2003}\u{2029}]"
128                };
129                self.replace_impl(&cls.span, replacement);
130            }
131        }
132    }
133}
134
135impl<'a> Visitor for Ecma262Translator<'a> {
136    type Output = (Cow<'a, str>, bool);
137    type Err = ast::Error;
138
139    fn finish(self) -> Result<Self::Output, Self::Err> {
140        Ok((self.pattern, self.has_changes))
141    }
142
143    fn visit_class_set_item_pre(&mut self, item: &ast::ClassSetItem) -> Result<(), Self::Err> {
144        if let ClassSetItem::Perl(cls) = item {
145            self.replace(cls);
146        }
147        Ok(())
148    }
149    fn visit_post(&mut self, ast: &Ast) -> Result<(), Self::Err> {
150        if self.has_changes {
151            return Ok(());
152        }
153        match ast {
154            Ast::ClassPerl(perl) => {
155                self.replace(perl);
156            }
157            Ast::Literal(literal) => {
158                if let Literal {
159                    kind: LiteralKind::Special(SpecialLiteralKind::Bell),
160                    ..
161                } = literal.as_ref()
162                {
163                    // Not possible to create a custom error, hence throw an arbitrary one from a
164                    // known invalid pattern.
165                    return Parser::new().parse("[").map(|_| ());
166                }
167            }
168            _ => (),
169        }
170        Ok(())
171    }
172}

195    fn test_ecma262_to_rust_regex(input: &str, expected: &str) {
196        let result = to_rust_regex(input).unwrap();
197        assert_eq!(result, expected);
198    }
