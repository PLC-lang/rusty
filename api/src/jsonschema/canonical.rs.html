51    /// Serialize a JSON value into a deterministic canonical JSON string.
52    ///
53    /// This is intended to provide a stable representation for schema-oriented
54    /// canonicalization/normalization workflows, including deduplication of
55    /// equivalent JSON Schemas and downstream processing that relies on a
56    /// single deterministic form.
57    ///
58    /// # Rules
59    ///
60    /// - Object keys are emitted in lexicographic order.
61    /// - Integer-valued floats are emitted as integers (`1.0` becomes `1`).
62    /// - Output is always compact (no extra whitespace).
63    ///
64    /// # Examples
65    ///
66    /// ```rust
67    /// use serde_json::json;
68    ///
69    /// let schema = json!({"b": 1, "a": {"y": 2, "x": 3}});
70    /// let canonical = jsonschema::canonical::json::to_string(&schema).unwrap();
71    /// assert_eq!(canonical, r#"{"a":{"x":3,"y":2},"b":1}"#);
72    /// ```
73    ///
74    /// # Errors
75    ///
76    /// Returns an error if serialization fails, for example when the input exceeds
77    /// the canonical serializer recursion limit.
78    pub fn to_string(value: &Value) -> Result<String, Error> {
79        let mut output = Vec::with_capacity(initial_output_capacity(value));
80        let formatter = CanonicalFormatter {
81            default: serde_json::ser::CompactFormatter,
82        };
83        let scratch_pool = RefCell::new(Vec::new());
84        let mut serializer = serde_json::Serializer::with_formatter(&mut output, formatter);
85        CanonicalValue::new(value, 0, &scratch_pool)
86            .serialize(&mut serializer)
87            .map_err(Error::from)?;
88        String::from_utf8(output).map_err(|err| {
89            serde_json::Error::io(io::Error::new(io::ErrorKind::InvalidData, err)).into()
90        })
91    }
92}

94#[inline]
95fn initial_output_capacity(value: &Value) -> usize {
96    const MIN_CAPACITY: usize = 16;
97    const MAX_PREALLOC: usize = 1 << 20; // 1 MiB
98
99    let estimated = match value {
100        Value::Object(map) => map.len().saturating_mul(24).saturating_add(2),
101        Value::Array(items) => items.len().saturating_mul(12).saturating_add(2),
102        Value::String(s) => s.len().saturating_add(2),
103        Value::Number(_) => 32,
104        Value::Bool(_) => 8,
105        Value::Null => 4,
106    };
107
108    estimated.clamp(MIN_CAPACITY, MAX_PREALLOC)
109}

111/// A formatter that emits integer-valued floats as integers.
112#[derive(Default)]
113struct CanonicalFormatter {
114    default: CompactFormatter,
115}
116
117impl Formatter for CanonicalFormatter {
118    #[inline]
119    fn write_f64<W: io::Write + ?Sized>(&mut self, writer: &mut W, value: f64) -> io::Result<()> {
120        if value.fract() == 0.0 {
121            if (0.0..U64_UPPER_EXCLUSIVE_F64).contains(&value) {
122                #[allow(clippy::cast_possible_truncation, clippy::cast_sign_loss)]
123                let int = value as u64;
124                return self.default.write_u64(writer, int);
125            }
126            if (I64_LOWER_INCLUSIVE_F64..I64_UPPER_EXCLUSIVE_F64).contains(&value) {
127                #[allow(clippy::cast_possible_truncation)]
128                let int = value as i64;
129                return self.default.write_i64(writer, int);
130            }
131            let integer = format!("{value:.0}");
132            return writer.write_all(integer.as_bytes());
133        }
134
135        self.default.write_f64(writer, value)
136    }
137}

139struct CanonicalValue<'value> {

145struct ObjectEntry<'value> {
146    key: &'value str,
147    value: &'value Value,
148}
149
150struct ObjectEntryScratch<'value, 'pool> {
151    entries: Vec<ObjectEntry<'value>>,
152    pool: &'pool RefCell<Vec<Vec<ObjectEntry<'value>>>>,
153}
154
155impl<'value, 'pool> ObjectEntryScratch<'value, 'pool> {
156    fn with_capacity(pool: &'pool RefCell<Vec<Vec<ObjectEntry<'value>>>>, capacity: usize) -> Self {
157        let mut entries = pool.borrow_mut().pop().unwrap_or_default();
158        entries.clear();
159        if entries.capacity() < capacity {
160            entries.reserve(capacity - entries.capacity());
161        }
162        Self { entries, pool }
163    }
164
165    #[inline]
166    fn entries_mut(&mut self) -> &mut Vec<ObjectEntry<'value>> {
167        &mut self.entries
168    }
169
170    #[inline]
171    fn entries(&self) -> &[ObjectEntry<'value>] {
172        &self.entries
173    }
174}
175
176impl Drop for ObjectEntryScratch<'_, '_> {
177    fn drop(&mut self) {
178        self.entries.clear();
179        if self.entries.capacity() > MAX_SCRATCH_CAPACITY {
180            return;
181        }
182        let mut pool = self.pool.borrow_mut();
183        if pool.len() < MAX_SCRATCH_POOL_SIZE {
184            pool.push(mem::take(&mut self.entries));
185        }
186    }
187}

222#[cfg(feature = "arbitrary-precision")]
223fn canonical_integral_number(raw: &str) -> Option<String> {
224    let bytes = raw.as_bytes();
225    if bytes.is_empty() {
226        return None;
227    }
228
229    let mut idx = 0;
230    let negative = if bytes[idx] == b'-' {
231        idx += 1;
232        true
233    } else {
234        false
235    };
236    if idx == bytes.len() {
237        return None;
238    }
239
240    let int_start = idx;
241    if bytes[idx] == b'0' {
242        idx += 1;
243        if idx < bytes.len() && bytes[idx].is_ascii_digit() {
244            return None;
245        }
246    } else if bytes[idx].is_ascii_digit() {
247        while idx < bytes.len() && bytes[idx].is_ascii_digit() {
248            idx += 1;
249        }
250    } else {
251        return None;
252    }
253    let int_end = idx;
254
255    let mut frac_start = idx;
256    let mut frac_end = idx;
257    if idx < bytes.len() && bytes[idx] == b'.' {
258        idx += 1;
259        frac_start = idx;
260        if idx == bytes.len() || !bytes[idx].is_ascii_digit() {
261            return None;
262        }
263        while idx < bytes.len() && bytes[idx].is_ascii_digit() {
264            idx += 1;
265        }
266        frac_end = idx;
267    }
268    let has_fraction = frac_end > frac_start;
269
270    let mut exponent: i64 = 0;
271    let has_exponent = if idx < bytes.len() && matches!(bytes[idx], b'e' | b'E') {
272        idx += 1;
273        let mut exp_negative = false;
274        if idx < bytes.len() && matches!(bytes[idx], b'+' | b'-') {
275            exp_negative = bytes[idx] == b'-';
276            idx += 1;
277        }
278        if idx == bytes.len() || !bytes[idx].is_ascii_digit() {
279            return None;
280        }
281        while idx < bytes.len() && bytes[idx].is_ascii_digit() {
282            exponent = exponent
283                .saturating_mul(10)
284                .saturating_add(i64::from(bytes[idx] - b'0'));
285            idx += 1;
286        }
287        if exp_negative {
288            exponent = -exponent;
289        }
290        true
291    } else {
292        false
293    };
294
295    if idx != bytes.len() {
296        return None;
297    }
298
299    if !has_fraction && !has_exponent {
300        if negative && &raw[int_start..int_end] == "0" {
301            return Some(String::from("0"));
302        }
303        return Some(raw.to_owned());
304    }
305
306    let int_digits = &raw[int_start..int_end];
307    let frac_digits = &raw[frac_start..frac_end];
308    let mut digits = Vec::with_capacity(int_digits.len() + frac_digits.len());
309    digits.extend_from_slice(int_digits.as_bytes());
310    digits.extend_from_slice(frac_digits.as_bytes());
311
312    let frac_len = i64::try_from(frac_digits.len()).unwrap_or(i64::MAX);
313    let shift = exponent.saturating_sub(frac_len);
314
315    let kept_len = if shift >= 0 {
316        let extra_zeros = usize::try_from(shift).ok()?;
317        digits.resize(digits.len().saturating_add(extra_zeros), b'0');
318        digits.len()
319    } else {
320        let drop_len = usize::try_from(shift.unsigned_abs()).ok()?;
321        if drop_len > digits.len() {
322            if digits.iter().all(|&byte| byte == b'0') {
323                0
324            } else {
325                return None;
326            }
327        } else {
328            let kept = digits.len() - drop_len;
329            if digits[kept..].iter().all(|&byte| byte == b'0') {
330                kept
331            } else {
332                return None;
333            }
334        }
335    };
336
337    let prefix = &digits[..kept_len];
338    let first_non_zero = prefix.iter().position(|&byte| byte != b'0');
339    let Some(first_non_zero) = first_non_zero else {
340        return Some(String::from("0"));
341    };
342
343    let mut output = String::with_capacity(
344        prefix
345            .len()
346            .saturating_sub(first_non_zero)
347            .saturating_add(usize::from(negative)),
348    );
349    if negative {
350        output.push('-');
351    }
352    for byte in &prefix[first_non_zero..] {
353        output.push(char::from(*byte));
354    }
355    Some(output)
356}

358fn serialize_number<S>(number: &Number, serializer: S) -> Result<S::Ok, S::Error>
359where
360    S: Serializer,
361{
362    #[cfg(feature = "arbitrary-precision")]
363    {
364        if let Some(integer) = canonical_integral_number(number.as_str()) {
365            return serializer.serialize_some(&BorrowedNumber(&integer));
366        }
367    }
368    number.serialize(serializer)
369}

371impl Serialize for CanonicalValue<'_> {
372    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
373    where
374        S: Serializer,
375    {
376        match self.value {
377            Value::Null => serializer.serialize_unit(),
378            Value::Bool(value) => serializer.serialize_bool(*value),
379            Value::Number(number) => serialize_number(number, serializer),
380            Value::String(value) => serializer.serialize_str(value),
381            Value::Array(items) => {
382                if self.recursion_depth == RECURSION_LIMIT {
383                    return Err(ser::Error::custom("Recursion limit reached"));
384                }
385                let mut sequence = serializer.serialize_seq(Some(items.len()))?;
386                for item in items {
387                    sequence.serialize_element(&CanonicalValue::new(
388                        item,
389                        self.recursion_depth + 1,
390                        self.scratch_pool,
391                    ))?;
392                }
393                sequence.end()
394            }
395            Value::Object(map) => {
396                if self.recursion_depth == RECURSION_LIMIT {
397                    return Err(ser::Error::custom("Recursion limit reached"));
398                }
399                let mut output = serializer.serialize_map(Some(map.len()))?;
400                // Canonical output must be independent from serde_json's internal map backend.
401                // Downstream crates can enable serde_json/preserve_order transitively, so we
402                // always sort object keys here instead of depending on iteration order.
403                let mut scratch = ObjectEntryScratch::with_capacity(self.scratch_pool, map.len());
404                {
405                    let entries = scratch.entries_mut();
406                    for (key, value) in map {
407                        entries.push(ObjectEntry {
408                            key: key.as_str(),
409                            value,
410                        });
411                    }
412                    entries.sort_unstable_by(|left, right| {
413                        left.key.as_bytes().cmp(right.key.as_bytes())
414                    });
415                }
416                for entry in scratch.entries() {
417                    output.serialize_entry(
418                        entry.key,
419                        &CanonicalValue::new(
420                            entry.value,
421                            self.recursion_depth + 1,
422                            self.scratch_pool,
423                        ),
424                    )?;
425                }
426                output.end()
427            }
428        }
429    }
