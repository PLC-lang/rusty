12/// A node in the schema tree, returned by [`compile_validators`]
13#[derive(Debug)]
14pub(crate) struct SchemaNode {
15    validators: NodeValidators,
16    relative_path: JSONPointer,
17    absolute_path: Option<AbsolutePath>,
18}
19
20#[derive(Debug)]
21enum NodeValidators {
22    /// The result of compiling a boolean valued schema, e.g
23    ///
24    /// ```json
25    /// {
26    ///     "additionalProperties": false
27    /// }
28    /// ```
29    ///
30    /// Here the result of `compile_validators` called with the `false` value will return a
31    /// `SchemaNode` with a single `BooleanValidator` as it's `validators`.
32    Boolean { validator: Option<BoxedValidator> },
33    /// The result of compiling a schema which is composed of keywords (almost all schemas)
34    Keyword(Box<KeywordValidators>),
35    /// The result of compiling a schema which is "array valued", e.g the "dependencies" keyword of
36    /// draft 7 which can take values which are an array of other property names
37    Array { validators: Vec<BoxedValidator> },
38}
39
40#[derive(Debug)]
41struct KeywordValidators {
42    /// The keywords on this node which were not recognized by any vocabularies. These are
43    /// stored so we can later produce them as annotations
44    unmatched_keywords: Option<AHashMap<String, serde_json::Value>>,
45    // We should probably use AHashMap here but it breaks a bunch of test which assume
46    // validators are in a particular order
47    validators: Vec<(String, BoxedValidator)>,
48}

50impl SchemaNode {
51    pub(crate) fn new_from_boolean(
52        context: &CompilationContext<'_>,
53        validator: Option<BoxedValidator>,
54    ) -> SchemaNode {
55        SchemaNode {
56            relative_path: context.clone().into_pointer(),
57            absolute_path: context.base_uri().map(AbsolutePath::from),
58            validators: NodeValidators::Boolean { validator },
59        }
60    }
61
62    pub(crate) fn new_from_keywords(
63        context: &CompilationContext<'_>,
64        mut validators: Vec<(String, BoxedValidator)>,
65        unmatched_keywords: Option<AHashMap<String, serde_json::Value>>,
66    ) -> SchemaNode {
67        validators.shrink_to_fit();
68        SchemaNode {
69            relative_path: context.clone().into_pointer(),
70            absolute_path: context.base_uri().map(AbsolutePath::from),
71            validators: NodeValidators::Keyword(Box::new(KeywordValidators {
72                unmatched_keywords,
73                validators,
74            })),
75        }
76    }
77
78    pub(crate) fn new_from_array(
79        context: &CompilationContext<'_>,
80        mut validators: Vec<BoxedValidator>,
81    ) -> SchemaNode {
82        validators.shrink_to_fit();
83        SchemaNode {
84            relative_path: context.clone().into_pointer(),
85            absolute_path: context.base_uri().map(AbsolutePath::from),
86            validators: NodeValidators::Array { validators },
87        }
88    }
89
90    pub(crate) fn validators(&self) -> impl Iterator<Item = &BoxedValidator> + ExactSizeIterator {
91        match &self.validators {
92            NodeValidators::Boolean { validator } => {
93                if let Some(v) = validator {
94                    NodeValidatorsIter::BooleanValidators(std::iter::once(v))
95                } else {
96                    NodeValidatorsIter::NoValidator
97                }
98            }
99            NodeValidators::Keyword(kvals) => {
100                NodeValidatorsIter::KeywordValidators(kvals.validators.iter())
101            }
102            NodeValidators::Array { validators } => {
103                NodeValidatorsIter::ArrayValidators(validators.iter())
104            }
105        }
106    }
107
108    fn format_validators(&self, f: &mut fmt::Formatter) -> fmt::Result {
109        f.write_str(&format_validators(self.validators()))
110    }

112    /// This is similar to `Validate::apply` except that `SchemaNode` knows where it is in the
113    /// validator tree and so rather than returning a `PartialApplication` it is able to return a
114    /// complete `BasicOutput`. This is the mechanism which compositional validators use to combine
115    /// results from sub-schemas
116    pub(crate) fn apply_rooted(
117        &self,
118        instance: &serde_json::Value,
119        instance_path: &InstancePath,
120    ) -> BasicOutput {
121        match self.apply(instance, instance_path) {
122            PartialApplication::Valid {
123                annotations,
124                mut child_results,
125            } => {
126                if let Some(annotations) = annotations {
127                    child_results.insert(0, self.annotation_at(instance_path, annotations));
128                };
129                BasicOutput::Valid(child_results)
130            }
131            PartialApplication::Invalid {
132                errors,
133                mut child_results,
134            } => {
135                for error in errors {
136                    child_results.insert(0, self.error_at(instance_path, error));
137                }
138                BasicOutput::Invalid(child_results)
139            }
140        }
141    }

171    /// Here we return a `NodeValidatorsErrIter` to avoid allocating in some situations. This isn't
172    /// always possible but for a lot of common cases (e.g nodes with a single child) we can do it.
173    /// This is wrapped in a `Box` by `SchemaNode::validate`
174    pub(crate) fn err_iter<'a>(
175        &self,
176        instance: &'a serde_json::Value,
177        instance_path: &InstancePath,
178    ) -> NodeValidatorsErrIter<'a> {
179        match &self.validators {
180            NodeValidators::Keyword(kvs) if kvs.validators.len() == 1 => {
181                NodeValidatorsErrIter::Single(kvs.validators[0].1.validate(instance, instance_path))
182            }
183            NodeValidators::Keyword(kvs) => NodeValidatorsErrIter::Multiple(
184                kvs.validators
185                    .iter()
186                    .flat_map(|(_, v)| v.validate(instance, instance_path))
187                    .collect::<Vec<_>>()
188                    .into_iter(),
189            ),
190            NodeValidators::Boolean {
191                validator: Some(v), ..
192            } => NodeValidatorsErrIter::Single(v.validate(instance, instance_path)),
193            NodeValidators::Boolean {
194                validator: None, ..
195            } => NodeValidatorsErrIter::NoErrs,
196            NodeValidators::Array { validators } => NodeValidatorsErrIter::Multiple(
197                validators
198                    .iter()
199                    .flat_map(move |v| v.validate(instance, instance_path))
200                    .collect::<Vec<_>>()
201                    .into_iter(),
202            ),
203        }
204    }
205
206    /// Helper function to apply an iterator of `(Into<PathChunk>, Validate)` to a value. This is
207    /// useful as a keyword schemanode has a set of validators keyed by their keywords, so the
208    /// `Into<Pathchunk>` is a `String` whereas an array schemanode has an array of validators so
209    /// the `Into<PathChunk>` is a `usize`
210    fn apply_subschemas<'a, I, P>(
211        &self,
212        instance: &serde_json::Value,
213        instance_path: &InstancePath,
214        path_and_validators: I,
215        annotations: Option<Annotations<'a>>,
216    ) -> PartialApplication<'a>
217    where
218        I: Iterator<Item = (P, &'a Box<dyn Validate + Send + Sync + 'a>)> + 'a,
219        P: Into<crate::paths::PathChunk> + std::fmt::Display,
220    {
221        let mut success_results: VecDeque<OutputUnit<Annotations>> = VecDeque::new();
222        let mut error_results = VecDeque::new();
223        for (path, validator) in path_and_validators {
224            let path = self.relative_path.extend_with(&[path.into()]);
225            let absolute_path = self
226                .absolute_path
227                .clone()
228                .map(|p| p.with_path(path.to_string().as_str()));
229            match validator.apply(instance, instance_path) {
230                PartialApplication::Valid {
231                    annotations,
232                    child_results,
233                } => {
234                    if let Some(annotations) = annotations {
235                        success_results.push_front(OutputUnit::<Annotations<'a>>::annotations(
236                            path,
237                            instance_path.into(),
238                            absolute_path,
239                            annotations,
240                        ));
241                    }
242                    success_results.extend(child_results);
243                }
244                PartialApplication::Invalid {
245                    errors: these_errors,
246                    child_results,
247                } => {
248                    error_results.extend(child_results);
249                    error_results.extend(these_errors.into_iter().map(|error| {
250                        OutputUnit::<ErrorDescription>::error(
251                            path.clone(),
252                            instance_path.into(),
253                            absolute_path.clone(),
254                            error,
255                        )
256                    }));
257                }
258            }
259        }
260        if error_results.is_empty() {
261            PartialApplication::Valid {
262                annotations,
263                child_results: success_results,
264            }
265        } else {
266            PartialApplication::Invalid {
267                errors: Vec::new(),
268                child_results: error_results,
269            }
270        }
271    }
272}

280impl Validate for SchemaNode {
281    fn validate<'instance>(
282        &self,
283        instance: &'instance serde_json::Value,
284        instance_path: &InstancePath,
285    ) -> ErrorIterator<'instance> {
286        return Box::new(self.err_iter(instance, instance_path));
287    }
288
289    fn is_valid(&self, instance: &serde_json::Value) -> bool {
290        match &self.validators {
291            // If we only have one validator then calling it's `is_valid` directly does
292            // actually save the 20 or so instructions required to call the `slice::Iter::all`
293            // implementation. Validators at the leaf of a tree are all single node validators so
294            // this optimization can have significant cumulative benefits
295            NodeValidators::Keyword(kvs) if kvs.validators.len() == 1 => {
296                kvs.validators[0].1.is_valid(instance)
297            }
298            NodeValidators::Keyword(kvs) => {
299                kvs.validators.iter().all(|(_, v)| v.is_valid(instance))
300            }
301            NodeValidators::Array { validators } => validators.iter().all(|v| v.is_valid(instance)),
302            NodeValidators::Boolean { validator: Some(_) } => false,
303            NodeValidators::Boolean { validator: None } => true,
304        }
305    }
306
307    fn apply<'a>(
308        &'a self,
309        instance: &serde_json::Value,
310        instance_path: &InstancePath,
311    ) -> PartialApplication<'a> {
312        match self.validators {
313            NodeValidators::Array { ref validators } => {
314                self.apply_subschemas(instance, instance_path, validators.iter().enumerate(), None)
315            }
316            NodeValidators::Boolean { ref validator } => {
317                if let Some(validator) = validator {
318                    validator.apply(instance, instance_path)
319                } else {
320                    PartialApplication::Valid {
321                        annotations: None,
322                        child_results: VecDeque::new(),
323                    }
324                }
325            }
326            NodeValidators::Keyword(ref kvals) => {
327                let KeywordValidators {
328                    ref unmatched_keywords,
329                    ref validators,
330                } = **kvals;
331                let annotations: Option<Annotations<'a>> =
332                    unmatched_keywords.as_ref().map(Annotations::from);
333                self.apply_subschemas(
334                    instance,
335                    instance_path,
336                    validators.iter().map(|(p, v)| (p.clone(), v)),
337                    annotations,
338                )
339            }
340        }
341    }
342}

344enum NodeValidatorsIter<'a> {
345    NoValidator,
346    BooleanValidators(std::iter::Once<&'a BoxedValidator>),
347    KeywordValidators(std::slice::Iter<'a, (String, BoxedValidator)>),
348    ArrayValidators(std::slice::Iter<'a, BoxedValidator>),
349}
350
351impl<'a> Iterator for NodeValidatorsIter<'a> {
352    type Item = &'a BoxedValidator;
353
354    fn next(&mut self) -> Option<Self::Item> {
355        match self {
356            Self::NoValidator => None,
357            Self::BooleanValidators(i) => i.next(),
358            Self::KeywordValidators(v) => v.next().map(|(_, v)| v),
359            Self::ArrayValidators(v) => v.next(),
360        }
361    }
362
363    fn all<F>(&mut self, mut f: F) -> bool
364    where
365        Self: Sized,
366        F: FnMut(Self::Item) -> bool,
367    {
368        match self {
369            Self::NoValidator => true,
370            Self::BooleanValidators(i) => i.all(f),
371            Self::KeywordValidators(v) => v.all(|(_, v)| f(v)),
372            Self::ArrayValidators(v) => v.all(f),
373        }
374    }
375}
376
377impl<'a> ExactSizeIterator for NodeValidatorsIter<'a> {
378    fn len(&self) -> usize {
379        match self {
380            Self::NoValidator => 0,
381            Self::BooleanValidators(..) => 1,
382            Self::KeywordValidators(v) => v.len(),
383            Self::ArrayValidators(v) => v.len(),
384        }
385    }
386}

388pub(crate) enum NodeValidatorsErrIter<'a> {
389    NoErrs,
390    Single(ErrorIterator<'a>),
391    Multiple(std::vec::IntoIter<crate::error::ValidationError<'a>>),
392}
393
394impl<'a> Iterator for NodeValidatorsErrIter<'a> {
395    type Item = crate::error::ValidationError<'a>;
396
397    fn next(&mut self) -> Option<Self::Item> {
398        match self {
399            Self::NoErrs => None,
400            Self::Single(i) => i.next(),
401            Self::Multiple(ms) => ms.next(),
402        }
403    }
