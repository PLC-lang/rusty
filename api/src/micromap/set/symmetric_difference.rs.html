8impl<T: PartialEq, const N: usize> Set<T, N> {
9    /// Visits the values representing the symmetric difference,
10    /// i.e., the values that are in `self` or `other` but not in both.
11    ///
12    /// # Examples
13    /// ```
14    /// use micromap::Set;
15    /// let a = Set::from([1, 2, 3]);
16    /// let b = Set::from([4, 2, 3, 4]);
17    /// // Can be seen as `(a - b) ∪ (b - a)`.
18    /// for x in a.symmetric_difference(&b) {
19    ///     println!("{x}"); // Print 1, 4
20    /// }
21    /// let sym_diff: Set<_, 7> = a.symmetric_difference(&b).copied().collect();
22    /// assert_eq!(sym_diff, Set::from([1, 4]));
23    /// ```
24    #[inline]
25    pub fn symmetric_difference<'a, const M: usize>(
26        &'a self,
27        other: &'a Set<T, M>,
28    ) -> SymmetricDifference<'a, T, N, M> {
29        SymmetricDifference {
30            iter: self.difference(other).chain(other.difference(self)),
31        }
32    }
33}
34
35/// A lazy iterator producing elements in the symmetric difference of Linear `Set`s.
36///
37/// This `struct` is created by the [`symmetric_difference`] method on [`Set`].
38///
39/// [`symmetric_difference`]: Set::symmetric_difference
40///
41/// # Examples
42/// ```
43/// use micromap::Set;
44/// let a = Set::from([1, 2, 3]);
45/// let b = Set::from([4, 2, 3, 4]);
46/// let sym_diff = a.symmetric_difference(&b);
47/// assert_eq!(sym_diff.count(), 2);
48/// ```
49#[must_use = "this returns the difference as an iterator, without modifying \
50              either input set"]
51pub struct SymmetricDifference<'a, T, const N: usize, const M: usize> {
52    iter: core::iter::Chain<Difference<'a, 'a, T, M>, Difference<'a, 'a, T, N>>,
53}
54
55impl<T, const N: usize, const M: usize> Clone for SymmetricDifference<'_, T, N, M> {
56    #[inline]
57    fn clone(&self) -> Self {
58        SymmetricDifference {
59            iter: self.iter.clone(),
60        }
61    }
62}
63
64impl<'a, T: PartialEq, const N: usize, const M: usize> Iterator
65    for SymmetricDifference<'a, T, N, M>
66{
67    type Item = &'a T;
68
69    #[inline]
70    fn next(&mut self) -> Option<Self::Item> {
71        self.iter.next()
72    }
73
74    #[inline]
75    fn size_hint(&self) -> (usize, Option<usize>) {
76        self.iter.size_hint()
77    }
78
79    #[inline]
80    fn fold<B, F>(self, init: B, f: F) -> B
81    where
82        Self: Sized,
83        F: FnMut(B, Self::Item) -> B,
84    {
85        self.iter.fold(init, f)
86    }
87}

89impl<T: PartialEq + core::fmt::Debug, const N: usize, const M: usize> core::fmt::Debug

92    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {

94    }

106    #[test]
107    fn symmetric_difference_simple() {
108        let set_a = Set::from([0, 1, 2, 3, 5, 7, 9]);
109        let set_b = Set::from([2, 5, 6, 7, 8, 10]);
110        let set_c = Set::from([0, 1, 2, 3, 5, 7, 9, 2, 5, 6, 7, 8]);
111        let set_result: Set<_, 13> = set_a.symmetric_difference(&set_b).copied().collect();
112        assert_eq!(set_result, Set::from([0, 1, 3, 6, 8, 9, 10]));
113        let set_result: Set<_, 19> = set_a.symmetric_difference(&set_c).copied().collect();
114        assert_eq!(set_result, Set::from([6, 8]));
115        let set_result: Set<_, 18> = set_b.symmetric_difference(&set_c).copied().collect();
116        assert_eq!(set_result, Set::from([0, 1, 3, 9, 10]));
117    }
118
119    #[test]
120    fn symmetric_difference_with_empty_set() {
121        let set_a = Set::from([1, 2, 3]);
122        let set_b: Set<i32, 3> = Set::new();
123        let sym_diff: Set<_, 6> = set_a.symmetric_difference(&set_b).copied().collect();
124        assert_eq!(sym_diff, set_a);
125    }

151    #[test]
152    fn symmetric_difference_size_hint() {
153        let set_a = Set::from([1, 1, 2, 3]); // cap is 4, but len() is 3
154        let set_b = Set::from([4, 5, 6, 6, 6, 7, 8, 9]); // cap is 8, but len() is 6
155        let set_c = Set::from([]);
156        let set_d = Set::from([3, 4]);
157        assert_eq!(set_a.symmetric_difference(&set_b).size_hint(), (3, Some(9)));
158        assert_eq!(set_a.symmetric_difference(&set_c).size_hint(), (3, Some(3)));
159        assert_eq!(set_a.symmetric_difference(&set_d).size_hint(), (1, Some(5)));
160        assert_eq!(set_b.symmetric_difference(&set_a).size_hint(), (3, Some(9)));
161        assert_eq!(set_b.symmetric_difference(&set_d).size_hint(), (4, Some(8)));
162        assert_eq!(set_c.symmetric_difference(&set_b).size_hint(), (6, Some(6)));
163        assert_eq!(set_d.symmetric_difference(&set_a).size_hint(), (1, Some(5)));
164        assert_eq!(set_d.symmetric_difference(&set_b).size_hint(), (4, Some(8)));
165        assert_eq!(set_d.symmetric_difference(&set_c).size_hint(), (2, Some(2)));
166    }

170        let set_a = Set::from([1, 2, 3]);
171        let set_b = Set::from([3, 4, 5]);
172        let sym_diff = set_a.symmetric_difference(&set_b);
173        let cloned = sym_diff.clone();
174        let collected_original: Set<_, 6> = sym_diff.copied().collect();
175        let collected_cloned: Set<_, 6> = cloned.copied().collect();

182        let set_b = Set::from([3, 4, 5]);

184        let sum = sym_diff.fold(0, |acc, &x| acc + x);
