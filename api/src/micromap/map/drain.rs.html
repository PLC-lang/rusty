8impl<K, V, const N: usize> Map<K, V, N> {
9    /// Clears the map, returning all key-value pairs as an iterator. For reuse, the
10    /// memory of capacity `N` will be keeped.
11    ///
12    /// If the returned iterator is dropped before being fully consumed, it drops the
13    /// remaining key-value pairs. The returned iterator keeps a mutable borrow on
14    /// the map to optimize its implementation.
15    ///
16    /// # Examples
17    /// ```
18    /// use micromap::Map;
19    /// let mut a = Map::<_, _, 3>::new();
20    /// a.insert(1, "a");
21    /// a.insert(2, "b");
22    /// for (k, v) in a.drain().take(1) {
23    ///     assert!(k == 1 || k == 2);
24    ///     assert!(v == "a" || v == "b");
25    /// }
26    /// assert!(a.is_empty());
27    /// ```
28    pub fn drain(&mut self) -> Drain<'_, K, V> {
29        let drain = Drain {
30            iter: self.pairs[0..self.len].iter_mut(),
31        };
32        self.len = 0;
33        drain
34    }
35}

37/// A draining iterator over the entries of a `Map`.
38///
39/// This `struct` is created by the [`drain`][`Map::drain`] method on [`Map`].
40/// See its documentation for more.
41///
42/// # Example
43/// ```
44/// use micromap::Map;
45/// let mut map = Map::from([("a", 1), ("b", 2)]);
46/// assert_eq!(map.len(), 2);
47/// let iter = map.drain();
48/// assert_eq!(iter.len(), 2);
49/// // assert_eq!(map.len(), 0); // Drain keeps a mutable borrow on the map
50/// drop(iter);
51/// assert_eq!(map.len(), 0); // Now we can use the map's borrow
52/// ```
53#[must_use = "iterators are lazy and do nothing unless consumed"]
54pub struct Drain<'a, K, V> {
55    iter: core::slice::IterMut<'a, MaybeUninit<(K, V)>>,
56}
57
58impl<K, V> fmt::Debug for Drain<'_, K, V>
59where
60    K: fmt::Debug,
61    V: fmt::Debug,
62{
63    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
64        f.debug_list()
65            .entries(slice_iter(self.iter.as_slice()))
66            .finish()
67    }
68}

70impl<K, V> Drop for Drain<'_, K, V> {
71    fn drop(&mut self) {
72        for pair in &mut self.iter {
73            unsafe { pair.assume_init_drop() };
74        }
75    }
76}
77
78impl<K, V> Iterator for Drain<'_, K, V> {
79    type Item = (K, V);
80
81    #[inline]
82    fn next(&mut self) -> Option<(K, V)> {
83        self.iter.next().map(|p| unsafe { p.assume_init_read() })
84    }
85
86    #[inline]
87    fn size_hint(&self) -> (usize, Option<usize>) {
88        (self.iter.len(), Some(self.iter.len()))
89    }
90}
91
92impl<K, V> ExactSizeIterator for Drain<'_, K, V> {
93    #[inline]
94    fn len(&self) -> usize {
95        self.iter.len()
96    }
97}

101#[cfg(test)]
102mod tests {
103    use super::Map;
104
105    #[test]
106    fn normal_drain() {
107        let mut map = Map::<char, u8, 10>::from_iter([('a', 97), ('b', 98), ('c', 99), ('d', 100)]);
108        let mut cloned_map = map.clone();
109        let mut drain = map.drain();
110        // For ExactSizeIterator
111        assert_eq!(drain.len(), drain.size_hint().0);
112        // Consume the first two items by iterator
113        assert_eq!(drain.next(), Some(('a', 97)));
114        assert_eq!(drain.next(), Some(('b', 98)));
115        // We can fuse the drain
116        let mut fuse_it = drain.fuse();
117        assert_eq!(fuse_it.next(), Some(('c', 99)));
118        assert_eq!(fuse_it.next(), Some(('d', 100)));
119        // Further calls to next() should return None
120        assert!(fuse_it.next().is_none());
121        // Then fuse works. (It doesn't make sense in our Drain really, but it can.)
122        assert!(fuse_it.next().is_none());
123        let mut drain = cloned_map.drain();
124        assert_eq!(drain.next(), Some(('a', 97)));
125        // Three elements left for Drop
126        drop(drain);
127    }
128
129    #[test]
130    fn debug_trait_for_drain() {
131        let mut map = Map::from([('x', 120), ('y', 121), ('z', 122)]);
132        let drain = map.drain();
133        // use Debug trait by `format!` macro
134        let debug_output = format!("{:?}", drain);
135        assert!(debug_output.contains("('x', 120)"));
136        assert!(debug_output.contains("('y', 121)"));
137        assert!(debug_output.contains("('z', 122)"));
138    }
