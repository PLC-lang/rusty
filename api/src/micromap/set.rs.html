38/// A faster alternative of [`std::collections::HashSet`].
39///
40/// For example, this is how you make a set, which is allocated on stack and is capable of storing
41/// up to eight key-values pairs:
42///
43/// ```
44/// use micromap::Set;
45/// let mut set : Set<u64, 8> = Set::new();
46/// set.insert(1);
47/// set.insert(2);
48/// assert_eq!(set.len(), 2);
49/// assert_eq!(set.capacity(), 8);
50/// ```
51///
52/// It is faster because it doesn't use a hash function at all. It simply keeps
53/// all items in an array and when it's necessary to find a value, it goes through
54/// all items comparing the needle with each pair available. Also it is faster
55/// because it doesn't use heap. When a [`Set`] is being created, it allocates the
56/// necessary space on stack. That's why the maximum size of the set must be
57/// provided in compile time.
58///
59/// It is also faster because it doesn't grow in size. When a [`Set`] is created,
60/// its size is fixed on stack (or as `Box<Set<_>>` on heap if you like). If an
61/// attempt is made to [`insert()`] too many value into it, it simply panics (or
62/// you can use [`checked_insert()`]). Moreover, for method [`insert_unchecked()`]
63/// in the "release" mode, it doesn't panic, instead, **its behaviour is undefined**.
64/// In the "release" mode all boundary checks are disabled, for the sake of higher
65/// performance (But in general, you can just use [`insert()`] or [`checked_insert()`]
66/// in most cases).
67///
68/// [`insert()`]: Set::insert
69/// [`checked_insert()`]: Set::checked_insert
70/// [`insert_unchecked()`]: Set::insert_unchecked
71#[repr(transparent)]
72pub struct Set<T, const N: usize> {
73    map: Map<T, (), N>,
74}

80    /// This is a deliberately lengthy test function.
81    /// As a test under the mod root path, it is necessary to consider the impact
82    /// of previous operations on subsequent operations. Some bugs are triggered
83    /// only when they accumulate. This function is used to test whether there
84    /// are such bugs. And no blank line separation is the rule for this project.
85    #[test]
86    fn various() {
87        let mut set: Set<i32, 10> = Set::new();
88        let set_default: Set<i32, 10> = Set::default();
89        assert!(set == set_default);
90        assert_eq!(set.capacity(), 10);
91        assert_eq!(set.len(), 0);
92        assert!(set.is_empty());
93        set.insert(0);
94        assert_eq!(set.len(), 1);
95        set.insert(0);
96        assert_eq!(set.len(), 1);
97        set.insert(1);
98        assert_eq!(set.len(), 2);
99        let mut drain = set.drain();
100        assert_eq!(drain.len(), 2);
101        assert!(drain.next().is_some());
102        assert_eq!(drain.len(), 1);
103        assert_eq!(drain.len(), drain.size_hint().0);
104        assert_eq!(drain.len(), drain.size_hint().1.unwrap());
105        drop(drain);
106        set.clear();
107        for n in 0..10 {
108            set.insert(n);
109        }
110        assert_eq!(set.len(), 10);
111        assert!(set.contains(&0));
112        assert_eq!(set.take(&0), Some(0));
113        assert!(!set.contains(&10));
114        assert_eq!(set.remove(&0), false);
115        set.insert(0);
116        assert_eq!(set.remove(&0), true);
117        set.insert(10);
118        assert_eq!(set.get(&10), Some(&10));
119        assert_eq!(set.get(&11), None);
120        set.retain(|&k| k % 2 == 0);
121        let mut it = set.iter();
122        assert!(it.next().is_some());
123        assert_eq!(it.len(), 4);
124        assert_eq!(it.len(), it.size_hint().0);
125        assert_eq!(it.len(), it.size_hint().1.unwrap());
126        let mut it_into = set.into_iter();
127        assert!(it_into.next().is_some());
128        assert!(it_into.next().is_some());
129        assert!(it_into.next().is_some());
130        assert_eq!(it_into.len(), 2);
131        assert_eq!(it_into.len(), it_into.size_hint().0);
132        assert_eq!(it_into.len(), it_into.size_hint().1.unwrap());
133    }
