29/// A combined trait containing all of the necessary implementations for generating a header
30pub trait GeneratedHeader: FileHelper + TypeHelper + TemplateHelper + SymbolHelper {
31    /// Returns whether or not this generated header is empty
32    ///
33    /// ---
34    ///
35    /// This must return true if the generation process has not yet occured,
36    /// or was aborted in a valid case.
37    fn is_empty(&self) -> bool;
38
39    /// Returns the contents of the header file
40    fn get_contents(&self) -> &str;
41
42    /// Prepares the template data for this header given a compilation unit
43    ///
44    /// ---
45    ///
46    /// The outcome of this method must be a populated [TemplateData](crate::header_generator::template_helper::TemplateData) on the generated header
47    /// that contains all of the data necessary to run the templating engine.
48    fn prepare_template_data(&mut self, compilation_unit: &CompilationUnit);
49
50    /// Runs the templating engine and generates the header file contents
51    ///
52    /// ---
53    ///
54    /// The outcome of this method must be a populated "contents" (accessible via the "get_contents" method)
55    /// on the generated header.
56    fn generate_headers(&mut self) -> Result<(), Diagnostic>;
57}
58
59/// Returns a generated header given the options for generation and a compilation unit.
60///
61/// ---
62///
63/// Should the process fail to determine an output directory and path for the header file
64/// a none-generated header will be returned, this is a valid outcome as some compilation units
65/// are not related to a file.
66pub fn get_generated_header(
67    generate_header_options: &GenerateHeaderOptions,
68    compilation_unit: &CompilationUnit,
69) -> Result<Box<dyn GeneratedHeader>, Diagnostic> {
70    // Prepare the template data
71    let mut generated_header =
72        prepare_template_data_for_header_generation(generate_header_options, compilation_unit)?;
73
74    // Generate the headers
75    generated_header.generate_headers()?;
76
77    Ok(generated_header)
78}

80/// Prepares the header data for generation given the options for generation and a compilation unit.
81///
82/// ---
83///
84/// Should the process fail to determine an output directory and path for the header file
85/// a none-generated header will be returned, this is a valid outcome as some compilation units
86/// are not related to a file.
87pub fn prepare_template_data_for_header_generation(
88    generate_header_options: &GenerateHeaderOptions,
89    compilation_unit: &CompilationUnit,
90) -> Result<Box<dyn GeneratedHeader>, Diagnostic> {
91    let mut generated_header = get_empty_generated_header_from_options(generate_header_options)?;
92
93    // Determine file and directory
94    // If the directories could not be configured with an acceptable outcome, then we exit without performing generation for this compilation unit
95    if !generated_header.determine_header_file_information(generate_header_options, compilation_unit) {
96        return Ok(generated_header);
97    }
98
99    // Prepare the template data
100    generated_header.prepare_template_data(compilation_unit);
101
102    Ok(generated_header)
103}

105/// Combines several generated headers and returns a single generated header containing the contents of all of them
106pub fn combine_generated_headers(
107    generate_header_options: &GenerateHeaderOptions,
108    generated_headers: Vec<Box<dyn GeneratedHeader>>,
109    output_file: String,
110) -> Result<Box<dyn GeneratedHeader>, Diagnostic> {
111    let mut generated_header = get_empty_generated_header_from_options(generate_header_options)?;
112
113    // We assume the first generated header in the list's location is the location we would like to place the header
114    let some_generated_header = generated_headers.iter().find(|p| !p.is_empty());
115    if some_generated_header.is_none() {
116        return Ok(generated_header);
117    }
118
119    let non_empty_generated_header = some_generated_header.unwrap();
120
121    generated_header.set_directory(non_empty_generated_header.get_directory());
122    let binding = PathBuf::from(non_empty_generated_header.get_directory()).join(&output_file);
123    let path = binding.to_str().unwrap_or_default();
124    generated_header.set_path(path);
125    generated_header.set_formatted_path(&format_path(path));
126    generated_header.set_file_name(&output_file);
127
128    let mut template_data = TemplateData::new();
129    for mut t_generated_header in generated_headers {
130        if !t_generated_header.is_empty() {
131            let user_defined_types = t_generated_header.get_mutable_template_data_user_defined_types();
132            template_data.user_defined_types.aliases.append(&mut user_defined_types.aliases);
133            template_data.user_defined_types.structs.append(&mut user_defined_types.structs);
134            template_data.user_defined_types.enums.append(&mut user_defined_types.enums);
135            template_data
136                .global_variables
137                .append(t_generated_header.get_mutable_template_data_global_variables());
138            template_data.functions.append(t_generated_header.get_mutable_template_data_functions());
139        }
140    }
141
142    // Overwrite the template data in the header with the combined template data
143    generated_header.set_template_data(template_data);
144
145    // Generate the headers
146    generated_header.generate_headers()?;
147
148    Ok(generated_header)
149}

151/// Generates an appropriate header for the given header options
152pub fn get_empty_generated_header_from_options(
153    generate_header_options: &GenerateHeaderOptions,
154) -> Result<Box<dyn GeneratedHeader>, Diagnostic> {
155    let generated_header: Result<Box<dyn GeneratedHeader>, Diagnostic> = match generate_header_options
156        .language
157    {
158        GenerateLanguage::C => {
159            let generated_header = GeneratedHeaderForC::new();
160            Ok(Box::new(generated_header))
161        }
162        language => Err(Diagnostic::new(format!("This language '{:?}' is not yet implemented!", language))),
163    };
164
165    generated_header
166}

187/// Given a name and a field name override, this will return the field name if present, the name if not, or default if the name is empty.
188fn coalesce_field_name_override_with_default(
189    name: &Option<String>,
190    field_name_override: Option<&String>,
191) -> String {
192    if let Some(field_name_ovr) = field_name_override {
193        field_name_ovr.to_string()
194    } else {
195        name.clone().unwrap_or_default()
196    }

199/// Given an ast node this will extract the enum declarations an return a Vec<Variable>.
200///
201/// Will return an empty Vec if the statement of the node is not type [ExpressionList](plc_ast::ast::AstStatement::ExpressionList)
202fn extract_enum_declaration_from_elements(enum_name: &str, node: &AstNode) -> Vec<Variable> {
203    let mut enum_declarations: Vec<Variable> = Vec::new();
204
205    match &node.stmt {
206        AstStatement::ExpressionList(exp_nodes) => {
207            let mut declared_value: i128 = 0;
208            for exp_node in exp_nodes {
209                match &exp_node.stmt {
210                    AstStatement::Assignment(assignment) => {
211                        let option_left = extract_enum_field_name_from_statement(&assignment.left.stmt);
212
213                        if let Some(left) = option_left {
214                            let option_right =
215                                extract_enum_field_value_from_statement(&assignment.right.stmt);
216
217                            let name = format!("{enum_name}_{left}");
218
219                            if let Some(right) = option_right {
220                                declared_value = right;
221                            } else {
222                                declared_value += 1;
223                            };
224
225                            enum_declarations.push(Variable {
226                                data_type: enum_name.to_string(),
227                                name,
228                                variable_type: VariableType::Declaration(declared_value),
229                            });
230                        } else {
231                            log::warn!(
232                                "Unable to extract the enum field name from the given assignment {:?}!",
233                                assignment
234                            )
235                        }
236                    }
237                    _ => log::warn!(
238                        "Given node {:?} is not an Assignment, unable to extract the enum declaration!",
239                        exp_node
240                    ),
241                }
242            }
243        }
244        _ => log::warn!(
245            "Given node {:?} is not an Expression list, unable to extract the enum declaration!",
246            node
247        ),
248    }
249
250    enum_declarations
251}

253/// Given an AstStatement, this will extract the name of the enum field.
254///
255/// Will return a new string if the AstStatement type not [ReferenceExpr](plc_ast::ast::AstStatement::ReferenceExpr),
256/// the access of that expression is not type [Member](plc_ast::ast::ReferenceAccess::Member)
257/// and the statement of that member is not type [Identifier](plc_ast::ast::AstStatement::Identifier).
258fn extract_enum_field_name_from_statement(statement: &AstStatement) -> Option<String> {
259    match statement {
260        AstStatement::ReferenceExpr(reference_expression) => match &reference_expression.access {
261            ReferenceAccess::Member(member_node) => {
262                let member_statement = member_node.get_stmt();
263                match member_statement {
264                    AstStatement::Identifier(enum_field) => Some(enum_field.to_string()),
265                    _ => None,
266                }
267            }
268            _ => None,
269        },
270        _ => None,
271    }
272}

274/// Extracts the value from an AstStatement type [Literal](plc_ast::ast::AstStatement::Literal).
275///
276/// Will return a new string if the AstStatement type is not [Literal](plc_ast::ast::AstStatement::Literal).
277fn extract_enum_field_value_from_statement(statement: &AstStatement) -> Option<i128> {
278    match statement {
279        AstStatement::Literal(AstLiteral::Integer(value)) => Some(*value),
280        _ => None,
281    }
282}

284/// Creates an ExtendedTypeName from a given Option<DataTypeDeclaration>.
285///
286/// Will return the default for ExtendedTypeName if the data type declaration is not [Reference](plc_ast::ast::DataTypeDeclaration::Reference) or [Definition](plc_ast::ast::DataTypeDeclaration::Definition).
287fn get_type_from_data_type_decleration(
288    data_type_declaration: &Option<DataTypeDeclaration>,
289    enforce_no_variadic: bool,
290) -> ExtendedTypeName {
291    match data_type_declaration {
292        Some(DataTypeDeclaration::Reference { referenced_type, .. }) => ExtendedTypeName {
293            type_name: referenced_type.clone(),
294            is_variadic: false,
295            is_sized_variadic: false,
296        },
297        Some(DataTypeDeclaration::Definition { data_type, .. }) => {
298            let type_name: String = data_type.get_name().unwrap_or("").to_string();
299            let (is_variadic, is_sized_variadic) = if enforce_no_variadic {
300                (false, false)
301            } else {
302                match &**data_type {
303                    ast::DataType::VarArgs { sized, .. } => {
304                        if *sized {
305                            (false, true)
306                        } else {
307                            (true, false)
308                        }
309                    }
310                    _ => (false, false),
311                }
312            };
313
314            ExtendedTypeName { type_name, is_variadic, is_sized_variadic }
315        }
316        _ => ExtendedTypeName::new(),
317    }
318}

320/// Given a data type name and a list of user type declarations,
321/// this will extract the user type declaration with a data type name that matches the given data type name.
322///
323/// This will return None in the case that the given data type name does not match the data type name of the user type declaration.
324fn get_user_generated_type_by_name<'a>(
325    name: &'a str,
326    user_types: &'a [UserTypeDeclaration],
327) -> Option<&'a UserTypeDeclaration> {
328    for user_type in user_types {
329        if let Some(data_type_name) = user_type.data_type.get_name() {
330            if data_type_name == name {
331                return Some(user_type);
332            }
333        }
334    }
335
336    None
337}

339/// Given an Option<AstNode> containing a [Literal](plc_ast::ast::AstStatement::Literal), this will determine the [i128] size of a string.
340///
341/// This will return [i128] default in the case the AstNode is None or does not match the expected [Literal](plc_ast::ast::AstStatement::Literal).
342fn extract_string_size(size: &Option<AstNode>) -> i128 {
343    if size.is_none() {
344        return i128::default();
345    }
346
347    let size = size.clone().unwrap();
348
349    match size.stmt {
350        // The string-termination-marker needs to be accounted for
351        AstStatement::Literal(AstLiteral::Integer(value)) => value + 1,
352        _ => i128::default(),
353    }
354}

356/// Determines what type of array this is, standard or multidimensional
357fn determine_array_type(bounds: &AstNode) -> Option<VariableType> {
358    match &bounds.stmt {
359        AstStatement::RangeStatement(..) => {
360            let size = extract_array_size(bounds);
361            Some(VariableType::Array(size))
362        }
363        AstStatement::VlaRangeStatement => Some(VariableType::Array(i128::default())),
364        AstStatement::ExpressionList(nodes) => {
365            let mut sizes: Vec<i128> = Vec::new();
366
367            for node in nodes {
368                sizes.push(extract_array_size(node));
369            }
370
371            Some(VariableType::MultidimensionalArray(sizes))
372        }
373        _ => None,
374    }
375}

377/// Given an AstNode containing a [RangeStatement](plc_ast::ast::AstStatement::RangeStatement), this will determine the [i128] size of an array.
378///
379/// This will return [i128] default in the case the AstNode does not match the expected [RangeStatement](plc_ast::ast::AstStatement::RangeStatement).
380fn extract_array_size(bounds: &AstNode) -> i128 {
381    match &bounds.stmt {
382        AstStatement::RangeStatement(range_stmt) => {
383            let start_value = match range_stmt.start.get_stmt() {
384                AstStatement::Literal(AstLiteral::Integer(value)) => *value,
385                _ => i128::default(),
386            };
387
388            let end_value = match range_stmt.end.get_stmt() {
389                AstStatement::Literal(AstLiteral::Integer(value)) => *value,
390                _ => i128::default(),
391            };
392
393            end_value - start_value + 1
394        }
395        _ => i128::default(),
396    }
397}

399/// Common method for determining if the string representation of the data type name is system generated.
400///
401/// i.e. Starts with "__"
402fn data_type_is_system_generated(data_type: &str) -> bool {
403    if data_type.starts_with("__") {
404        return true;
