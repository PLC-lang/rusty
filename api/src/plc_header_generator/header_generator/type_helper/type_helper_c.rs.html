1use crate::header_generator::{
2    header_generator_c::GeneratedHeaderForC,
3    type_helper::{determine_type_attribute, extract_string_size, TypeHelper, TypeInformation},
4    ExtendedTypeName,
5};
6use plc::typesystem::{DataType, DataTypeInformation, StringEncoding, BOOL_TYPE, REAL_SIZE};
7use plc_ast::ast::TypeNature;
8
9impl TypeHelper for GeneratedHeaderForC {
10    fn get_type_name_for_type(
11        &self,
12        extended_type_name: &ExtendedTypeName,
13        builtin_types: &[DataType],
14    ) -> TypeInformation {
15        if extended_type_name.type_name.is_empty() {
16            return TypeInformation {
17                name: String::from(C_VOID),
18                attribute: determine_type_attribute(
19                    extended_type_name.is_variadic,
20                    extended_type_name.is_sized_variadic,
21                    false,
22                    None,
23                ),
24            };
25        }
26
27        let Some(builtin_type) =
28            builtin_types.iter().find(|builtin_type| builtin_type.name == extended_type_name.type_name)
29        else {
30            // This is a user-generated type
31            return TypeInformation {
32                name: extended_type_name.type_name.to_string(),
33                attribute: determine_type_attribute(
34                    extended_type_name.is_variadic,
35                    extended_type_name.is_sized_variadic,
36                    true,
37                    None,
38                ),
39            };
40        };
41
42        match &builtin_type.information {
43            DataTypeInformation::Integer { signed, size, .. } => {
44                // Booleans have their own type
45                if extended_type_name.type_name == BOOL_TYPE {
46                    return TypeInformation {
47                        name: String::from(C_BOOL),
48                        attribute: determine_type_attribute(
49                            extended_type_name.is_variadic,
50                            extended_type_name.is_sized_variadic,
51                            false,
52                            None,
53                        ),
54                    };
55                }
56
57                // Dates have their own type
58                if builtin_type.nature == TypeNature::Date || builtin_type.nature == TypeNature::Duration {
59                    return TypeInformation {
60                        name: String::from(C_TIME),
61                        attribute: determine_type_attribute(
62                            extended_type_name.is_variadic,
63                            extended_type_name.is_sized_variadic,
64                            false,
65                            None,
66                        ),
67                    };
68                }
69
70                let signed_operator = if *signed { "" } else { "u" };
71                let constructed_type_name = format!("{signed_operator}int{size}_t");
72
73                TypeInformation {
74                    name: constructed_type_name,
75                    attribute: determine_type_attribute(
76                        extended_type_name.is_variadic,
77                        extended_type_name.is_sized_variadic,
78                        false,
79                        None,
80                    ),
81                }
82            }
83            DataTypeInformation::Float { size, .. } => {
84                if *size == REAL_SIZE {
85                    TypeInformation {
86                        name: String::from(C_FLOAT),
87                        attribute: determine_type_attribute(
88                            extended_type_name.is_variadic,
89                            extended_type_name.is_sized_variadic,
90                            false,
91                            None,
92                        ),
93                    }
94                } else {
95                    TypeInformation {
96                        name: String::from(C_DOUBLE),
97                        attribute: determine_type_attribute(
98                            extended_type_name.is_variadic,
99                            extended_type_name.is_sized_variadic,
100                            false,
101                            None,
102                        ),
103                    }
104                }
105            }
106            DataTypeInformation::Alias { referenced_type, .. } => {
107                let referenced_data_type_info = ExtendedTypeName {
108                    type_name: referenced_type.to_string(),
109                    is_variadic: false,
110                    is_sized_variadic: false,
111                };
112
113                self.get_type_name_for_type(&referenced_data_type_info, builtin_types)
114            }
115            DataTypeInformation::String { encoding, size } => match encoding {
116                StringEncoding::Utf8 => TypeInformation {
117                    name: self.get_type_name_for_string(&false),
118                    attribute: determine_type_attribute(
119                        extended_type_name.is_variadic,
120                        extended_type_name.is_sized_variadic,
121                        false,
122                        Some(extract_string_size(size)),
123                    ),
124                },
125                StringEncoding::Utf16 => TypeInformation {
126                    name: self.get_type_name_for_string(&true),
127                    attribute: determine_type_attribute(
128                        extended_type_name.is_variadic,
129                        extended_type_name.is_sized_variadic,
130                        false,
131                        Some(extract_string_size(size)),
132                    ),
133                },
134            },
135            _ => {
136                log::debug!("{} this type is not yet supported!", extended_type_name.type_name);
137                TypeInformation::new()
138            }
139        }
140    }
141
142    fn get_type_name_for_string(&self, is_wide: &bool) -> String {
143        if *is_wide {
144            return String::from(C_INT16);
145        }
146
147        String::from(C_CHAR)
148    }
149}
