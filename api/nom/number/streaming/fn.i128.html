Expand description

Recognizes a signed 16 byte integer

If the parameter is nom::number::Endianness::Big, parse a big endian i128 integer, otherwise if nom::number::Endianness::Little parse a little endian i128 integer. Streaming version: Will return Err(nom::Err::Incomplete(_)) if there is not enough data.

use nom::number::streaming::i128;

let be_i128 = |s| {
  i128::<_, (_, ErrorKind)>(nom::number::Endianness::Big)(s)
};

assert_eq!(be_i128(&b"\x00\x01\x02\x03\x04\x05\x06\x07\x00\x01\x02\x03\x04\x05\x06\x07abcefg"[..]), Ok((&b"abcefg"[..], 0x00010203040506070001020304050607)));
assert_eq!(be_i128(&b"\x01"[..]), Err(Err::Incomplete(Needed::new(15))));

let le_i128 = |s| {
  i128::<_, (_, ErrorKind)>(nom::number::Endianness::Little)(s)
};

assert_eq!(le_i128(&b"\x00\x01\x02\x03\x04\x05\x06\x07\x00\x01\x02\x03\x04\x05\x06\x07abcefg"[..]), Ok((&b"abcefg"[..], 0x07060504030201000706050403020100)));
assert_eq!(le_i128(&b"\x01"[..]), Err(Err::Incomplete(Needed::new(15))));
